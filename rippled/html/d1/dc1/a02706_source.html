59 Status SstFileReader::ReadSequential(bool print_kv,
60  uint64_t read_num,
61  bool has_from,
62  const std::string& from_key,
63  bool has_to,
64  const std::string& to_key)
65 {
66  unique_ptr<TableReader> table_reader;
67  InternalKeyComparator internal_comparator_(BytewiseComparator());
68  Options table_options;
69  table_options.comparator = &internal_comparator_;
70  unique_ptr<RandomAccessFile> file;
71  Status s = table_options.env->NewRandomAccessFile(file_name_, &file,
72  soptions_);
73  if(!s.ok()) {
74  return s;
75  }
76  uint64_t file_size;
77  table_options.env->GetFileSize(file_name_, &file_size);
78  unique_ptr<TableFactory> table_factory;
79  s = table_options.table_factory->GetTableReader(table_options, soptions_,
80  std::move(file), file_size,
81  &table_reader);
82  if(!s.ok()) {
83  return s;
84  }
85 
86  Iterator* iter = table_reader->NewIterator(ReadOptions(verify_checksum_,
87  false));
88  uint64_t i = 0;
89  if (has_from) {
90  InternalKey ikey(from_key, kMaxSequenceNumber, kValueTypeForSeek);
91  iter->Seek(ikey.Encode());
92  } else {
93  iter->SeekToFirst();
94  }
95  for (; iter->Valid(); iter->Next()) {
96  Slice key = iter->key();
97  Slice value = iter->value();
98  ++i;
99  if (read_num > 0 && i > read_num)
100  break;
101 
102  ParsedInternalKey ikey;
103  if (!ParseInternalKey(key, &ikey)) {
104  std::cerr << "Internal Key ["
105  << key.ToString(true /* in hex*/)
106  << "] parse error!\n";
107  continue;
108  }
109 
110  // If end marker was specified, we stop before it
111  if (has_to && BytewiseComparator()->Compare(ikey.user_key, to_key) >= 0) {
112  break;
113  }
114 
115  if (print_kv) {
116  std::cout << ikey.DebugString(output_hex_)
117  << " => "
118  << value.ToString(output_hex_) << "\n";
119  }
120 
121  }
122 
123  read_num_ += i;
124 
125  Status ret = iter->status();
126  delete iter;
127  return ret;
128 }

143 string HexToString(const string& str) {
144  string parsed;
145  if (str[0] != '0' || str[1] != 'x') {
146  fprintf(stderr, "Invalid hex input %s. Must start with 0x\n",
147  str.c_str());
148  throw "Invalid hex input";
149  }
150 
151  for (unsigned int i = 2; i < str.length();) {
152  int c;
153  sscanf(str.c_str() + i, "%2X", &c);
154  parsed.push_back(c);
155  i += 2;
156  }
157  return parsed;
158 }

192  from_key = argv[i] + 7;
193  has_from = true;
194  } else if (strncmp(argv[i], "--to=", 5) == 0) {
195  to_key = argv[i] + 5;
196  has_to = true;
197  }else {
198  print_help();
199  exit(1);
200  }
201  }
202 
203

205  if (has_from) {

208  if (has_to) {

249  read_num,

258  }
259  }
