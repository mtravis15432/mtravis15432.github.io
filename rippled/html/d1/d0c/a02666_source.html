37 struct BytewiseLessThan {
38  bool operator()(const std::string& key1, const std::string& key2) const {
39  // smaller entries will be placed in front.
40  return comparator->Compare(key1, key2) <= 0;
41  }
42  const Comparator* comparator = BytewiseComparator();
43 };

61 static bool GoodCompressionRatio(size_t compressed_size, size_t raw_size) {
62  // Check to see if compressed less than 12.5%
63  return compressed_size < raw_size - (raw_size / 8u);

142 void BlockBasedTableBuilder::Add(const Slice& key, const Slice& value) {
143  Rep* r = rep_;
144  assert(!r->closed);
145  if (!ok()) return;
146  if (r->props.num_entries > 0) {
147  assert(r->options.comparator->Compare(key, Slice(r->last_key)) > 0);
148  }
149 
150  auto should_flush = r->flush_block_policy->Update(key, value);
151  if (should_flush) {
152  assert(!r->data_block.empty());
153  Flush();
154 
155  // Add item to index block.
156  // We do not emit the index entry for a block until we have seen the
157  // first key for the next data block. This allows us to use shorter
158  // keys in the index block. For example, consider a block boundary
159  // between the keys "the quick brown fox" and "the who". We can use
160  // "the r" as the key for the index block entry since it is >= all
161  // entries in the first block and < all entries in subsequent
162  // blocks.
163  if (ok()) {
164  r->options.comparator->FindShortestSeparator(&r->last_key, key);
165  std::string handle_encoding;
166  r->pending_handle.EncodeTo(&handle_encoding);
167  r->index_block.Add(r->last_key, Slice(handle_encoding));
168  }
169  }
170 
171  if (r->filter_block != nullptr) {
172  r->filter_block->AddKey(key);
173  }
174 
175  r->last_key.assign(key.data(), key.size());
176  r->data_block.Add(key, value);
177  r->props.num_entries++;
178  r->props.raw_key_size += key.size();
179  r->props.raw_value_size += value.size();
180 
181  for (auto collector : r->options.table_properties_collectors) {
182  Status s = collector->Add(key, value);
183  if (!s.ok()) {
184  LogPropertiesCollectionError(
185  r->options.info_log.get(),
186  "Add", /* method */
187  collector->Name()
188  );
189  }
190  }
191 }

193 void BlockBasedTableBuilder::Flush() {
194  Rep* r = rep_;
195  assert(!r->closed);
196  if (!ok()) return;
197  if (r->data_block.empty()) return;
198  WriteBlock(&r->data_block, &r->pending_handle);
199  if (ok()) {
200  r->status = r->file->Flush();
201  }
202  if (r->filter_block != nullptr) {
203  r->filter_block->StartBlock(r->offset);
204  }
205  r->props.data_size = r->offset;
206  ++r->props.num_data_blocks;
207 }

209 void BlockBasedTableBuilder::WriteBlock(BlockBuilder* block,
210  BlockHandle* handle) {
211  // File format contains a sequence of blocks where each block has:
212  // block_data: uint8[n]
213  // type: uint8
214  // crc: uint32
215  assert(ok());
216  Rep* r = rep_;
217  Slice raw = block->Finish();
218 
219  Slice block_contents;
220  std::string* compressed = &r->compressed_output;
221  CompressionType type = r->compression_type;
222  switch (type) {
223  case kNoCompression:
224  block_contents = raw;
225  break;
226 
227  case kSnappyCompression: {
228  std::string* compressed = &r->compressed_output;
229  if (port::Snappy_Compress(r->options.compression_opts, raw.data(),
230  raw.size(), compressed) &&
231  GoodCompressionRatio(compressed->size(), raw.size())) {
232  block_contents = *compressed;
233  } else {
234  // Snappy not supported, or not good compression ratio, so just
235  // store uncompressed form
236  block_contents = raw;
237  type = kNoCompression;
238  }
239  break;
240  }
241  case kZlibCompression:
242  if (port::Zlib_Compress(r->options.compression_opts, raw.data(),
243  raw.size(), compressed) &&
244  GoodCompressionRatio(compressed->size(), raw.size())) {
245  block_contents = *compressed;
246  } else {
247  // Zlib not supported, or not good compression ratio, so just
248  // store uncompressed form
249  block_contents = raw;
250  type = kNoCompression;
251  }
252  break;
253  case kBZip2Compression:
254  if (port::BZip2_Compress(r->options.compression_opts, raw.data(),
255  raw.size(), compressed) &&
256  GoodCompressionRatio(compressed->size(), raw.size())) {
257  block_contents = *compressed;
258  } else {
259  // BZip not supported, or not good compression ratio, so just
260  // store uncompressed form
261  block_contents = raw;
262  type = kNoCompression;
263  }
264  break;
265  }
266  WriteRawBlock(block_contents, type, handle);
267  r->compressed_output.clear();
268  block->Reset();
269 }

271 void BlockBasedTableBuilder::WriteRawBlock(const Slice& block_contents,
272  CompressionType type,
273  BlockHandle* handle) {
274  Rep* r = rep_;
275  StopWatch sw(r->options.env, r->options.statistics, WRITE_RAW_BLOCK_MICROS);
276  handle->set_offset(r->offset);
277  handle->set_size(block_contents.size());
278  r->status = r->file->Append(block_contents);
279  if (r->status.ok()) {
280  char trailer[kBlockTrailerSize];
281  trailer[0] = type;
282  uint32_t crc = crc32c::Value(block_contents.data(), block_contents.size());
283  crc = crc32c::Extend(crc, trailer, 1); // Extend crc to cover block type
284  EncodeFixed32(trailer+1, crc32c::Mask(crc));
285  r->status = r->file->Append(Slice(trailer, kBlockTrailerSize));
286  if (r->status.ok()) {
287  r->status = InsertBlockInCache(block_contents, type, handle);
288  }
289  if (r->status.ok()) {
290  r->offset += block_contents.size() + kBlockTrailerSize;
291  }
292  }
293 }

313 //
314 // Make a copy of the block contents and insert into compressed block cache
315 //
316 Status BlockBasedTableBuilder::InsertBlockInCache(const Slice& block_contents,
317  const CompressionType type,
318  const BlockHandle* handle) {
319  Rep* r = rep_;
320  Cache* block_cache_compressed = r->options.block_cache_compressed.get();
321 
322  if (type != kNoCompression && block_cache_compressed != nullptr) {
323 
324  Cache::Handle* cache_handle = nullptr;
325  size_t size = block_contents.size();
326 
327  char* ubuf = new char[size]; // make a new copy
328  memcpy(ubuf, block_contents.data(), size);
329 
330  BlockContents results;
331  Slice sl(ubuf, size);
332  results.data = sl;
333  results.cachable = true; // XXX
334  results.heap_allocated = true;
335  results.compression_type = type;
336 
337  Block* block = new Block(results);
338 
339  // make cache key by appending the file offset to the cache prefix id
340  char* end = EncodeVarint64(
341  r->compressed_cache_key_prefix +
342  r->compressed_cache_key_prefix_size,
343  handle->offset());
344  Slice key(r->compressed_cache_key_prefix, static_cast<size_t>
345  (end - r->compressed_cache_key_prefix));
346 
347  // Insert into compressed block cache.
348  cache_handle = block_cache_compressed->Insert(key, block, block->size(),
349  &DeleteCachedBlock);
350  block_cache_compressed->Release(cache_handle);
351 
352  // Invalidate OS cache.
353  r->file->InvalidateCache(r->offset, size);
354  }
355  return Status::OK();
356 }

358 Status BlockBasedTableBuilder::Finish() {
359  Rep* r = rep_;
360  bool empty_data_block = r->data_block.empty();
361  Flush();
362  assert(!r->closed);
363  r->closed = true;
364 
365  BlockHandle filter_block_handle,
366  metaindex_block_handle,
367  index_block_handle;
368 
369  // Write filter block
370  if (ok() && r->filter_block != nullptr) {
371  auto filter_contents = r->filter_block->Finish();
372  r->props.filter_size = filter_contents.size();
373  WriteRawBlock(filter_contents, kNoCompression, &filter_block_handle);
374  }
375 
376  // To make sure properties block is able to keep the accurate size of index
377  // block, we will finish writing all index entries here and flush them
378  // to storage after metaindex block is written.
379  if (ok() && !empty_data_block) {
380  r->options.comparator->FindShortSuccessor(&r->last_key);
381 
382  std::string handle_encoding;
383  r->pending_handle.EncodeTo(&handle_encoding);
384  r->index_block.Add(r->last_key, handle_encoding);
385  }
386 
387  // Write meta blocks and metaindex block with the following order.
388  // 1. [meta block: filter]
389  // 2. [meta block: properties]
390  // 3. [metaindex block]
391  if (ok()) {
392  // We use `BytewiseComparator` as the comparator for meta block.
393  BlockBuilder meta_index_block(
394  r->options.block_restart_interval,
395  BytewiseComparator()
396  );
397  // Key: meta block name
398  // Value: block handle to that meta block
399  BytewiseSortedMap meta_block_handles;
400 
401  // Write filter block.
402  if (r->filter_block != nullptr) {
403  // Add mapping from "<filter_block_prefix>.Name" to location
404  // of filter data.
405  std::string key = BlockBasedTable::kFilterBlockPrefix;
406  key.append(r->options.filter_policy->Name());
407  std::string handle_encoding;
408  filter_block_handle.EncodeTo(&handle_encoding);
409  meta_block_handles.insert(
410  std::make_pair(key, handle_encoding)
411  );
412  }
413 
414  // Write properties block.
415  {
416  BlockBuilder properties_block(
417  r->options.block_restart_interval,
418  BytewiseComparator()
419  );
420 
421  BytewiseSortedMap properties;
422 
423  // Add basic properties
424  AddProperties(
425  properties,
426  BlockBasedTablePropertiesNames::kRawKeySize,
427  r->props.raw_key_size
428  );
429  AddProperties(
430  properties,
431  BlockBasedTablePropertiesNames::kRawValueSize,
432  r->props.raw_value_size
433  );
434  AddProperties(
435  properties,
436  BlockBasedTablePropertiesNames::kDataSize,
437  r->props.data_size
438  );
439  r->props.index_size =
440  r->index_block.CurrentSizeEstimate() + kBlockTrailerSize;
441  AddProperties(
442  properties,
443  BlockBasedTablePropertiesNames::kIndexSize,
444  r->props.index_size
445  );
446  AddProperties(
447  properties,
448  BlockBasedTablePropertiesNames::kNumEntries,
449  r->props.num_entries
450  );
451  AddProperties(
452  properties,
453  BlockBasedTablePropertiesNames::kNumDataBlocks,
454  r->props.num_data_blocks);
455  if (r->filter_block != nullptr) {
456  properties.insert({
457  BlockBasedTablePropertiesNames::kFilterPolicy,
458  r->options.filter_policy->Name()
459  });
460  }
461  AddProperties(
462  properties,
463  BlockBasedTablePropertiesNames::kFilterSize,
464  r->props.filter_size
465  );
466 
467  for (auto collector : r->options.table_properties_collectors) {
468  TableProperties::UserCollectedProperties user_collected_properties;
469  Status s =
470  collector->Finish(&user_collected_properties);
471 
472  if (!s.ok()) {
473  LogPropertiesCollectionError(
474  r->options.info_log.get(),
475  "Finish", /* method */
476  collector->Name()
477  );
478  } else {
479  properties.insert(
480  user_collected_properties.begin(),
481  user_collected_properties.end()
482  );
483  }
484  }
485 
486  for (const auto& stat : properties) {
487  properties_block.Add(stat.first, stat.second);
488  }
489 
490  BlockHandle properties_block_handle;
491  WriteBlock(&properties_block, &properties_block_handle);
492 
493  std::string handle_encoding;
494  properties_block_handle.EncodeTo(&handle_encoding);
495  meta_block_handles.insert(
496  { BlockBasedTable::kPropertiesBlock, handle_encoding }
497  );
498  } // end of properties block writing
499 
500  for (const auto& metablock : meta_block_handles) {
501  meta_index_block.Add(metablock.first, metablock.second);
502  }
503 
504  WriteBlock(&meta_index_block, &metaindex_block_handle);
505  } // meta blocks and metaindex block.
506 
507  // Write index block
508  if (ok()) {
509  WriteBlock(&r->index_block, &index_block_handle);
510  }
511 
512  // Write footer
513  if (ok()) {
514  Footer footer;
515  footer.set_metaindex_handle(metaindex_block_handle);
516  footer.set_index_handle(index_block_handle);
517  std::string footer_encoding;
518  footer.EncodeTo(&footer_encoding);
519  r->status = r->file->Append(footer_encoding);
520  if (r->status.ok()) {
521  r->offset += footer_encoding.size();
522  }
523  }
524 
525  // Print out the table stats
526  if (ok()) {
527  // user collected properties
528  std::string user_collected;
529  user_collected.reserve(1024);
530  for (auto collector : r->options.table_properties_collectors) {
531  for (const auto& prop : collector->GetReadableProperties()) {
532  user_collected.append(prop.first);
533  user_collected.append("=");
534  user_collected.append(prop.second);
535  user_collected.append("; ");
536  }
537  }
538 
539  Log(
540  r->options.info_log,
541  "Table was constructed:\n"
542  " basic properties: %s\n"
543  " user collected properties: %s",
544  r->props.ToString().c_str(),
545  user_collected.c_str()
546  );
547  }
548 
549  return r->status;
550 }
