21 bool BlobChunk::Overlap(const BlobChunk &chunk) const {
22  return size != 0 && chunk.size != 0 && bucket_id == chunk.bucket_id &&
23  ((offset >= chunk.offset && offset < chunk.offset + chunk.size) ||
24  (chunk.offset >= offset && chunk.offset < offset + size));

50 Status FreeList::Free(const Blob& blob) {
51  // add it back to the free list
52  for (auto chunk : blob.chunks) {
53  free_blocks_ += chunk.size;
54  if (fifo_free_chunks_.size() &&
55  fifo_free_chunks_.back().ImmediatelyBefore(chunk)) {
56  fifo_free_chunks_.back().size += chunk.size;
57  } else {
58  fifo_free_chunks_.push_back(chunk);
59  }
60  }
61 
62  return Status::OK();
63 }

65 Status FreeList::Allocate(uint32_t blocks, Blob* blob) {
66  if (free_blocks_ < blocks) {
67  return Status::Incomplete("");
68  }
69 
70  blob->chunks.clear();
71  free_blocks_ -= blocks;
72 
73  while (blocks > 0) {
74  assert(fifo_free_chunks_.size() > 0);
75  auto& front = fifo_free_chunks_.front();
76  if (front.size > blocks) {
77  blob->chunks.push_back(BlobChunk(front.bucket_id, front.offset, blocks));
78  front.offset += blocks;
79  front.size -= blocks;
80  blocks = 0;
81  } else {
82  blob->chunks.push_back(front);
83  blocks -= front.size;
84  fifo_free_chunks_.pop_front();
85  }
86  }
87  assert(blocks == 0);
88 
89  return Status::OK();
90 }

132 Status BlobStore::Put(const Slice& value, Blob* blob) {
133  // convert size to number of blocks
134  Status s = Allocate((value.size() + block_size_ - 1) / block_size_, blob);
135  if (!s.ok()) {
136  return s;
137  }
138  auto size_left = (uint64_t) value.size();
139 
140  uint64_t offset = 0; // in bytes, not blocks
141  for (auto chunk : blob->chunks) {
142  uint64_t write_size = min(chunk.size * block_size_, size_left);
143  assert(chunk.bucket_id < buckets_size_);
144  s = buckets_[chunk.bucket_id].get()->Write(chunk.offset * block_size_,
145  Slice(value.data() + offset,
146  write_size));
147  if (!s.ok()) {
148  Delete(*blob);
149  return s;
150  }
151  offset += write_size;
152  size_left -= write_size;
153  if (write_size < chunk.size * block_size_) {
154  // if we have any space left in the block, fill it up with zeros
155  string zero_string(chunk.size * block_size_ - write_size, 0);
156  s = buckets_[chunk.bucket_id].get()->Write(chunk.offset * block_size_ +
157  write_size,
158  Slice(zero_string));
159  }
160  }
161 
162  if (size_left > 0) {
163  Delete(*blob);
164  return Status::IOError("Tried to write more data than fits in the blob");
165  }
166 
167  return Status::OK();
168 }

170 Status BlobStore::Get(const Blob& blob,
171  string* value) const {
172  {
173  // assert that it doesn't overlap with free list
174  // it will get compiled out for release
175  MutexLock l(&free_list_mutex_);
176  assert(!free_list_.Overlap(blob));
177  }
178 
179  value->resize(blob.Size() * block_size_);
180 
181  uint64_t offset = 0; // in bytes, not blocks
182  for (auto chunk : blob.chunks) {
183  Slice result;
184  assert(chunk.bucket_id < buckets_size_);
185  Status s;
186  s = buckets_[chunk.bucket_id].get()->Read(chunk.offset * block_size_,
187  chunk.size * block_size_,
188  &result,
189  &value->at(offset));
190  if (!s.ok() || result.size() < chunk.size * block_size_) {
191  value->clear();
192  return Status::IOError("Could not read in from file");
193  }
194  offset += chunk.size * block_size_;
195  }
196 
197  // remove the '\0's at the end of the string
198  value->erase(find(value->begin(), value->end(), '\0'), value->end());
199 
200  return Status::OK();
201 }

235 Status BlobStore::CreateNewBucket() {
236  MutexLock l(&buckets_mutex_);
237 
238  if (buckets_size_ >= max_buckets_) {
239  return Status::IOError("Max size exceeded\n");
240  }
241 
242  int new_bucket_id = buckets_size_;
243 
244  char fname[200];
245  sprintf(fname, "%s/%d.bs", directory_.c_str(), new_bucket_id);
246 
247  Status s = env_->NewRandomRWFile(string(fname),
248  &buckets_[new_bucket_id],
249  storage_options_);
250  if (!s.ok()) {
251  return s;
252  }
253 
254  // whether Allocate succeeds or not, does not affect the overall correctness
255  // of this function - calling Allocate is really optional
256  // (also, tmpfs does not support allocate)
257  buckets_[new_bucket_id].get()->Allocate(0, block_size_ * blocks_per_bucket_);
258

260 
261  return free_list_.Free(Blob(new_bucket_id, 0, blocks_per_bucket_));
