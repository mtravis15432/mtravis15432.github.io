528  // count is the number of buckets
529  template <class Container>
530  void rehash (size_type count, Container& c)
531  {
532  size_type const size (m_vec.size());
533  if (count == size)
534  return;
535  if (count > m_vec.capacity())
536  {
537  // Need two vectors otherwise we
538  // will destroy non-empty buckets.
539  vec_type vec (m_vec.get_allocator());
540  std::swap (m_vec, vec);
541  m_vec.resize (count);
542  c.rehash (bucket_traits (
543  &m_vec[0], m_vec.size()));
544  return;
545  }
546  // Rehash in place.
547  if (count > size)
548  {
549  // This should not reallocate since
550  // we checked capacity earlier.
551  m_vec.resize (count);
552  c.rehash (bucket_traits (
553  &m_vec[0], count));
554  return;
555  }
556  // Resize must happen after rehash otherwise
557  // we might destroy non-empty buckets.
558  c.rehash (bucket_traits (
559  &m_vec[0], count));
560  m_vec.resize (count);
561  }
