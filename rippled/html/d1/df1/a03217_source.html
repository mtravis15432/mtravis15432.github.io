90  void handleConnect (RPCServerImp::pointer new_connection,
91  boost::system::error_code const& error)
92  {
93  bool delay = false;
94 
95  if (!error)
96  {
97  // Restrict callers by IP
98  std::string client_ip (
99  new_connection->getRemoteEndpoint ().address ().to_string ());
100 
101  if (! isClientAllowed (client_ip))
102  {
103  startListening ();
104  return;
105  }
106 
107  new_connection->getSocket ().async_handshake (AutoSocket::ssl_socket::server,
108  boost::bind (&RPCServer::connected, new_connection));
109  }
110  else
111  {
112  if (error == boost::system::errc::too_many_files_open)
113  delay = true;
114 
115  WriteLog (lsINFO, RPCDoor) << "RPCDoorImp::handleConnect Error: " << error;
116  }
117 
118  if (delay)
119  {
120  mDelayTimer.expires_from_now (boost::posix_time::milliseconds (1000));
121  mDelayTimer.async_wait (boost::bind (&RPCDoorImp::startListening, this));
122  }
123  else
124  {
125  startListening ();
126  }
127  }
