130 // We want to make sure that with a single writer and multiple
131 // concurrent readers (with no synchronization other than when a
132 // reader's iterator is created), the reader always observes all the
133 // data that was present in the skip list when the iterator was
134 // constructor. Because insertions are happening concurrently, we may
135 // also observe new values that were inserted since the iterator was
136 // constructed, but we should never miss any values that were present
137 // at iterator construction time.
138 //
139 // We generate multi-part keys:
140 // <key,gen,hash>
141 // where:
142 // key is in range [0..K-1]
143 // gen is a generation number for key
144 // hash is hash(key,gen)
145 //
146 // The insertion code picks a random key, sets gen to be 1 + the last
147 // generation number inserted for that key, and sets hash to Hash(key,gen).
148 //
149 // At the beginning of a read, we snapshot the last inserted
150 // generation number for each key. We then iterate, including random
151 // calls to Next() and Seek(). For every key we encounter, we
152 // check that it is either expected given the initial snapshot or has
153 // been concurrently added since the iterator started.
154 class ConcurrentTest {
155  private:
156  static const uint32_t K = 4;
157 
158  static uint64_t key(Key key) { return (key >> 40); }
159  static uint64_t gen(Key key) { return (key >> 8) & 0xffffffffu; }
160  static uint64_t hash(Key key) { return key & 0xff; }
161 
162  static uint64_t HashNumbers(uint64_t k, uint64_t g) {
163  uint64_t data[2] = { k, g };
164  return Hash(reinterpret_cast<char*>(data), sizeof(data), 0);
165  }
166 
167  static Key MakeKey(uint64_t k, uint64_t g) {
168  assert(sizeof(Key) == sizeof(uint64_t));
169  assert(k <= K); // We sometimes pass K to seek to the end of the skiplist
170  assert(g <= 0xffffffffu);
171  return ((k << 40) | (g << 8) | (HashNumbers(k, g) & 0xff));
172  }
173 
174  static bool IsValidKey(Key k) {
175  return hash(k) == (HashNumbers(key(k), gen(k)) & 0xff);
176  }
177 
178  static Key RandomTarget(Random* rnd) {
179  switch (rnd->Next() % 10) {
180  case 0:
181  // Seek to beginning
182  return MakeKey(0, 0);
183  case 1:
184  // Seek to end
185  return MakeKey(K, 0);
186  default:
187  // Seek to middle
188  return MakeKey(rnd->Next() % K, 0);
189  }
190  }
191 
192  // Per-key generation
193  struct State {
194  port::AtomicPointer generation[K];
195  void Set(int k, intptr_t v) {
196  generation[k].Release_Store(reinterpret_cast<void*>(v));
197  }
198  intptr_t Get(int k) {
199  return reinterpret_cast<intptr_t>(generation[k].Acquire_Load());
200  }
201 
202  State() {
203  for (unsigned int k = 0; k < K; k++) {
204  Set(k, 0);
205  }
206  }
207  };

210  State current_;

212  ArenaImpl arena_impl_;

214  // SkipList is not protected by mu_. We just use a single writer
215  // thread to modify it.
216  SkipList<Key, TestComparator> list_;

221  // REQUIRES: External synchronization
222  void WriteStep(Random* rnd) {
223  const uint32_t k = rnd->Next() % K;
224  const intptr_t g = current_.Get(k) + 1;
225  const Key key = MakeKey(k, g);
226  list_.Insert(key);
227  current_.Set(k, g);
228  }

230  void ReadStep(Random* rnd) {
231  // Remember the initial committed state of the skiplist.
232  State initial_state;
233  for (unsigned int k = 0; k < K; k++) {
234  initial_state.Set(k, current_.Get(k));
235  }
236 
237  Key pos = RandomTarget(rnd);
238  SkipList<Key, TestComparator>::Iterator iter(&list_);
239  iter.Seek(pos);
240  while (true) {
241  Key current;
242  if (!iter.Valid()) {
243  current = MakeKey(K, 0);
244  } else {
245  current = iter.key();
246  ASSERT_TRUE(IsValidKey(current)) << current;
247  }
248  ASSERT_LE(pos, current) << "should not go backwards";
249 
250  // Verify that everything in [pos,current) was not present in
251  // initial_state.
252  while (pos < current) {
253  ASSERT_LT(key(pos), K) << pos;
254 
255  // Note that generation 0 is never inserted, so it is ok if
256  // <*,0,*> is missing.
257  ASSERT_TRUE((gen(pos) == 0U) ||
258  (gen(pos) > (uint64_t)initial_state.Get(key(pos)))
259  ) << "key: " << key(pos)
260  << "; gen: " << gen(pos)
261  << "; initgen: "
262  << initial_state.Get(key(pos));
263 
264  // Advance to next key in the valid key space
265  if (key(pos) < key(current)) {
266  pos = MakeKey(key(pos) + 1, 0);
267  } else {
268  pos = MakeKey(key(pos), gen(pos) + 1);
269  }
270  }
271 
272  if (!iter.Valid()) {
273  break;
274  }
275 
276  if (rnd->Next() % 2) {
277  iter.Next();
278  pos = MakeKey(key(pos), gen(pos) + 1);
279  } else {
280  Key new_target = RandomTarget(rnd);
281  if (new_target > pos) {
282  pos = new_target;
283  iter.Seek(new_target);
284  }
285  }
286  }
287  }

341 static void ConcurrentReader(void* arg) {
342  TestState* state = reinterpret_cast<TestState*>(arg);
343  Random rnd(state->seed_);
344  int64_t reads = 0;
345  state->Change(TestState::RUNNING);
346  while (!state->quit_flag_.Acquire_Load()) {
347  state->t_.ReadStep(&rnd);
348  ++reads;
349  }
350  state->Change(TestState::DONE);
351 }

353 static void RunConcurrent(int run) {
354  const int seed = test::RandomSeed() + (run * 100);
355  Random rnd(seed);
356  const int N = 1000;
357  const int kSize = 1000;
358  for (int i = 0; i < N; i++) {
359  if ((i % 100) == 0) {
360  fprintf(stderr, "Run %d of %d\n", i, N);
361  }
362  TestState state(seed + 1);
363  Env::Default()->Schedule(ConcurrentReader, &state);
364  state.Wait(TestState::RUNNING);
365  for (int i = 0; i < kSize; i++) {
366  state.t_.WriteStep(&rnd);
367  }
368  state.quit_flag_.Release_Store(&state); // Any non-nullptr arg will do
369  state.Wait(TestState::DONE);
370  }
371 }
