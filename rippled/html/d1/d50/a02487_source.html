212  // Called when the Checker completes a connectivity test
213  void checkComplete (beast::IP::Endpoint const& address,
214  beast::IP::Endpoint const & checkedAddress, Checker::Result const& result)
215  {
216  if (result.error == boost::asio::error::operation_aborted)
217  return;
218 
219  SharedState::Access state (m_state);
220  Slots::iterator const iter (state->slots.find (address));
221  SlotImp& slot (*iter->second);
222 
223  if (iter == state->slots.end())
224  {
225  // The slot disconnected before we finished the check
226  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
227  "Logic tested " << address <<
228  " but the connection was closed";
229  return;
230  }
231 
232  // Mark that a check for this slot is finished.
233  slot.connectivityCheckInProgress = false;
234 
235  if (! result.error)
236  {
237  slot.checked = true;
238  slot.canAccept = result.canAccept;
239 
240  if (slot.canAccept)
241  {
242  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
243  "Logic testing " << address << " succeeded";
244  }
245  else
246  {
247  if (m_journal.info) m_journal.info << beast::leftw (18) <<
248  "Logic testing " << address << " failed";
249  }
250  }
251  else
252  {
253  // VFALCO TODO Should we retry depending on the error?
254  slot.checked = true;
255  slot.canAccept = false;
256 
257  if (m_journal.error) m_journal.error << beast::leftw (18) <<
258  "Logic testing " << iter->first << " with error, " <<
259  result.error.message();
260  }
261 
262  if (! slot.canAccept)
263  state->bootcache.on_failure (address);
264  }

383  void on_handshake (SlotImp::ptr const& slot,
384  RipplePublicKey const& key, bool cluster)
385  {
386  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
387  "Logic handshake " << slot->remote_endpoint () <<
388  " with " << (cluster ? "clustered " : "") << "key " << key;
389 
390  SharedState::Access state (m_state);
391 
392  // The object must exist in our table
393  assert (state->slots.find (slot->remote_endpoint ()) !=
394  state->slots.end ());
395  // Must be accepted or connected
396  assert (slot->state() == Slot::accept ||
397  slot->state() == Slot::connected);
398 
399  // Check for duplicate connection by key
400  if (state->keys.find (key) != state->keys.end())
401  {
402  m_callback.disconnect (slot, true);
403  return;
404  }
405 
406  // See if we have an open space for this slot
407  if (state->counts.can_activate (*slot))
408  {
409  // Set key and cluster right before adding to the map
410  // otherwise we could assert later when erasing the key.
411  state->counts.remove (*slot);
412  slot->public_key (key);
413  slot->cluster (cluster);
414  state->counts.add (*slot);
415 
416  // Add the public key to the active set
417  std::pair <Keys::iterator, bool> const result (
418  state->keys.insert (key));
419  // Public key must not already exist
420  assert (result.second);
421 
422  // Change state and update counts
423  state->counts.remove (*slot);
424  slot->activate (m_clock.now ());
425  state->counts.add (*slot);
426 
427  if (! slot->inbound())
428  state->bootcache.on_success (
429  slot->remote_endpoint());
430 
431  // Mark fixed slot success
432  if (slot->fixed() && ! slot->inbound())
433  {
434  auto iter (state->fixed.find (slot->remote_endpoint()));
435  assert (iter != state->fixed.end ());
436  iter->second.success (m_clock.now ());
437  if (m_journal.trace) m_journal.trace << beast::leftw (18) <<
438  "Logic fixed " << slot->remote_endpoint () << " success";
439  }
440 
441  m_callback.activate (slot);
442  }
443  else
444  {
445  if (! slot->inbound())
446  state->bootcache.on_success (
447  slot->remote_endpoint());
448 
449  // Maybe give them some addresses to try
450  if (slot->inbound ())
451  redirect (slot, state);
452 
453  m_callback.disconnect (slot, true);
454  }
455  }

459  // Validate and clean up the list that we received from the slot.
460  void preprocess (SlotImp::ptr const& slot, Endpoints& list,
461  SharedState::Access& state)
462  {
463  bool neighbor (false);
464  for (auto iter (list.begin()); iter != list.end();)
465  {
466  Endpoint& ep (*iter);
467 
468  // Enforce hop limit
469  if (ep.hops > Tuning::maxHops)
470  {
471  if (m_journal.warning) m_journal.warning << beast::leftw (18) <<
472  "Endpoints drop " << ep.address <<
473  " for excess hops " << ep.hops;
474  iter = list.erase (iter);
475  continue;
476  }
477 
478  // See if we are directly connected
479  if (ep.hops == 0)
480  {
481  if (! neighbor)
482  {
483  // Fill in our neighbors remote address
484  neighbor = true;
485  ep.address = slot->remote_endpoint().at_port (
486  ep.address.port ());
487  }
488  else
489  {
490  if (m_journal.warning) m_journal.warning << beast::leftw (18) <<
491  "Endpoints drop " << ep.address <<
492  " for extra self";
493  iter = list.erase (iter);
494  continue;
495  }
496  }
497 
498  // Discard invalid addresses
499  if (! is_valid_address (ep.address))
500  {
501  if (m_journal.warning) m_journal.warning << beast::leftw (18) <<
502  "Endpoints drop " << ep.address <<
503  " as invalid";
504  iter = list.erase (iter);
505  continue;
506  }
507 
508  // Filter duplicates
509  if (std::any_of (list.begin(), iter,
510  [ep](Endpoints::value_type const& other)
511  {
512  return ep.address == other.address;
513  }))
514  {
515  if (m_journal.warning) m_journal.warning << beast::leftw (18) <<
516  "Endpoints drop " << ep.address <<
517  " as duplicate";
518  iter = list.erase (iter);
519  continue;
520  }
521 
522  // Increment hop count on the incoming message, so
523  // we store it at the hop count we will send it at.
524  //
525  ++ep.hops;
526 
527  ++iter;
528  }
529  }

531  void on_endpoints (SlotImp::ptr const& slot, Endpoints list)
532  {
533  if (m_journal.trace) m_journal.trace << beast::leftw (18) <<
534  "Endpoints from " << slot->remote_endpoint () <<
535  " contained " << list.size () <<
536  ((list.size() > 1) ? " entries" : " entry");
537 
538  SharedState::Access state (m_state);
539 
540  // The object must exist in our table
541  assert (state->slots.find (slot->remote_endpoint ()) !=
542  state->slots.end ());
543 
544  // Must be handshaked!
545  assert (slot->state() == Slot::active);
546 
547  preprocess (slot, list, state);
548 
549  clock_type::time_point const now (m_clock.now());
550 
551  for (auto iter (list.cbegin()); iter != list.cend(); ++iter)
552  {
553  Endpoint const& ep (*iter);
554 
555  assert (ep.hops != 0);
556 
557  slot->recent.insert (ep.address, ep.hops);
558 
559  // Note hops has been incremented, so 1
560  // means a directly connected neighbor.
561  //
562  if (ep.hops == 1)
563  {
564  if (slot->connectivityCheckInProgress)
565  {
566  if (m_journal.warning) m_journal.warning << beast::leftw (18) <<
567  "Logic testing " << ep.address << " already in progress";
568  continue;
569  }
570 
571  if (! slot->checked)
572  {
573  // Mark that a check for this slot is now in progress.
574  slot->connectivityCheckInProgress = true;
575 
576  // Test the slot's listening port before
577  // adding it to the livecache for the first time.
578  //
579  m_checker.async_test (ep.address, bind (
580  &Logic::checkComplete, this, slot->remote_endpoint (),
581  ep.address, beast::_1));
582 
583  // Note that we simply discard the first Endpoint
584  // that the neighbor sends when we perform the
585  // listening test. They will just send us another
586  // one in a few seconds.
587 
588  continue;
589  }
590 
591  // If they failed the test then skip the address
592  if (! slot->canAccept)
593  continue;
594  }
595 
596  // We only add to the livecache if the neighbor passed the
597  // listening test, else we silently drop their messsage
598  // since their listening port is misconfigured.
599  //
600  state->livecache.insert (ep);
601  state->bootcache.insert (ep.address);
602  }
603 
604  slot->whenAcceptEndpoints = now + Tuning::secondsPerMessage;
605  }

775  SharedState::Access state (m_state);
776 
777  // Count how many more outbound attempts to make
778  //
779  auto needed (state->counts.attempts_needed ());
780  if (needed == 0)
781  return;
782 
783  ConnectHandouts h (needed, m_squelches);
784 
785  // Make sure we don't connect to already-connected entries.
786  squelch_slots (state);
787 
788  // 1. Use Fixed if:
789  // Fixed active count is below fixed count AND
790  // ( There are eligible fixed addresses to try OR
791  // Any outbound attempts are in progress)
792  //
793  if (state->counts.fixed_active() < state->fixed.size ())
794  {
795  get_fixed (needed, h.list(), state);
796 
797  if (! h.list().empty ())
798  {
799  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
800  "Logic connect " << h.list().size() << " fixed";
801  m_callback.connect (h.list());
802  return;
803  }
804 
805  if (state->counts.attempts() > 0)
806  {
807  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
808  "Logic waiting on " <<
809  state->counts.attempts() << " attempts";
810  return;
811  }
812  }
813 
814  // Only proceed if auto connect is enabled and we
815  // have less than the desired number of outbound slots
816  //
817  if (! state->config.autoConnect ||
818  state->counts.out_active () >= state->counts.out_max ())
819  return;
820 
821  // 2. Use Livecache if:
822  // There are any entries in the cache OR
823  // Any outbound attempts are in progress
824  //
825  {
826  state->livecache.hops.shuffle();
827  handout (&h, (&h)+1,
828  state->livecache.hops.rbegin(),
829  state->livecache.hops.rend());
830  if (! h.list().empty ())
831  {
832  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
833  "Logic connect " << h.list().size () << " live " <<
834  ((h.list().size () > 1) ? "endpoints" : "endpoint");
835  m_callback.connect (h.list());
836  return;
837  }
838  else if (state->counts.attempts() > 0)
839  {
840  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
841  "Logic waiting on " <<
842  state->counts.attempts() << " attempts";
843  return;
844  }
845  }
846 
847  /* 3. Bootcache refill
848  If the Bootcache is empty, try to get addresses from the current
849  set of Sources and add them into the Bootstrap cache.
850 
851  Pseudocode:
852  If ( domainNames.count() > 0 AND (
853  unusedBootstrapIPs.count() == 0
854  OR activeNameResolutions.count() > 0) )
855  ForOneOrMore (DomainName that hasn't been resolved recently)
856  Contact DomainName and add entries to the unusedBootstrapIPs
857  return;
858  */
859 
860  // 4. Use Bootcache if:
861  // There are any entries we haven't tried lately
862  //
863  for (auto iter (state->bootcache.begin());
864  ! h.full() && iter != state->bootcache.end(); ++iter)
865  h.try_insert (*iter);
866 
867  if (! h.list().empty ())
868  {
869  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
870  "Logic connect " << h.list().size () << " boot " <<
871  ((h.list().size () > 1) ? "addresses" : "address");
872  m_callback.connect (h.list());
873  return;
874  }
875 
876  // If we get here we are stuck
877  }

1044  // Send mtENDPOINTS for each slot as needed
1045  void broadcast ()
1046  {
1047  SharedState::Access state (m_state);
1048 
1049  std::vector <SlotHandouts> targets;
1050 
1051  {
1052  // build list of active slots
1053  std::vector <SlotImp::ptr> slots;
1054  slots.reserve (state->slots.size());
1055  std::for_each (state->slots.cbegin(), state->slots.cend(),
1056  [&slots](Slots::value_type const& value)
1057  {
1058  if (value.second->state() == Slot::active)
1059  slots.emplace_back (value.second);
1060  });
1061  std::random_shuffle (slots.begin(), slots.end());
1062 
1063  // build target vector
1064  targets.reserve (slots.size());
1065  std::for_each (slots.cbegin(), slots.cend(),
1066  [&targets](SlotImp::ptr const& slot)
1067  {
1068  targets.emplace_back (slot);
1069  });
1070  }
1071 
1072  /* VFALCO NOTE
1073  This is a temporary measure. Once we know our own IP
1074  address, the correct solution is to put it into the Livecache
1075  at hops 0, and go through the regular handout path. This way
1076  we avoid handing our address out too frequenty, which this code
1077  suffers from.
1078  */
1079  // Add an entry for ourselves if:
1080  // 1. We want incoming
1081  // 2. We have slots
1082  // 3. We haven't failed the firewalled test
1083  //
1084  if (state->config.wantIncoming &&
1085  state->counts.inboundSlots() > 0)
1086  {
1087  Endpoint ep;
1088  ep.hops = 0;
1089  ep.address = beast::IP::Endpoint (
1090  beast::IP::AddressV4 ()).at_port (
1091  state->config.listeningPort);
1092  for (auto& t : targets)
1093  t.insert (ep);
1094  }
1095 
1096  // build sequence of endpoints by hops
1097  state->livecache.hops.shuffle();
1098  handout (targets.begin(), targets.end(),
1099  state->livecache.hops.begin(),
1100  state->livecache.hops.end());
1101 
1102  // broadcast
1103  for (auto const& t : targets)
1104  {
1105  SlotImp::ptr const& slot (t.slot());
1106  auto const& list (t.list());
1107  if (m_journal.trace) m_journal.trace << beast::leftw (18) <<
1108  "Logic sending " << slot->remote_endpoint() <<
1109  " with " << list.size() <<
1110  ((list.size() == 1) ? " endpoint" : " endpoints");
1111  m_callback.send (slot, list);
1112  }
1113  }
