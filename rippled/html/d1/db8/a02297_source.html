321 String StringArray::joinIntoString (const String& separator, int start, int numberToJoin) const
322 {
323  const int last = (numberToJoin < 0) ? size()
324  : bmin (size(), start + numberToJoin);
325 
326  if (start < 0)
327  start = 0;
328 
329  if (start >= last)
330  return String::empty;
331 
332  if (start == last - 1)
333  return strings.getReference (start);
334 
335  const size_t separatorBytes = separator.getCharPointer().sizeInBytes() - sizeof (String::CharPointerType::CharType);
336  size_t bytesNeeded = separatorBytes * (size_t) (last - start - 1);
337 
338  for (int i = start; i < last; ++i)
339  bytesNeeded += strings.getReference(i).getCharPointer().sizeInBytes() - sizeof (String::CharPointerType::CharType);
340 
341  String result;
342  result.preallocateBytes (bytesNeeded);
343 
344  String::CharPointerType dest (result.getCharPointer());
345 
346  while (start < last)
347  {
348  const String& s = strings.getReference (start);
349 
350  if (! s.isEmpty())
351  dest.writeAll (s.getCharPointer());
352 
353  if (++start < last && separatorBytes > 0)
354  dest.writeAll (separator.getCharPointer());
355  }
356 
357  dest.writeNull();
358 
359  return result;
360 }

396  bool finished = text.isEmpty();
397 
398  while (! finished)
