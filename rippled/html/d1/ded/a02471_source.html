127 void
128 Bootcache::on_success (beast::IP::Endpoint const& endpoint)
129 {
130  auto result (m_map.insert (
131  value_type (endpoint, 1)));
132  if (result.second)
133  {
134  prune ();
135  }
136  else
137  {
138  Entry entry (result.first->right);
139  if (entry.valence() < 0)
140  entry.valence() = 0;
141  ++entry.valence();
142  m_map.erase (result.first);
143  result = m_map.insert (
144  value_type (endpoint, entry));
145  assert (result.second);
146  }
147  Entry const& entry (result.first->right);
148  if (m_journal.info) m_journal.info << beast::leftw (18) <<
149  "Bootcache connect " << endpoint <<
150  " with " << entry.valence() <<
151  ((entry.valence() > 1) ? " successes" : " success");
152  flagForUpdate();

198  // Checks the cache size and prunes if its over the limit.
199 void
200 Bootcache::prune ()
201 {
202  if (size() <= Tuning::bootcacheSize)
203  return;
204 
205  // Calculate the amount to remove
206  auto count ((size() *
207  Tuning::bootcachePrunePercent) / 100);
208  decltype(count) pruned (0);
209 
210  // Work backwards because bimap doesn't handle
211  // erasing using a reverse iterator very well.
212  //
213  for (auto iter (m_map.right.end());
214  count-- > 0 && iter != m_map.right.begin(); ++pruned)
215  {
216  --iter;
217  beast::IP::Endpoint const& endpoint (iter->get_left());
218  Entry const& entry (iter->get_right());
219  if (m_journal.trace) m_journal.trace << beast::leftw (18) <<
220  "Bootcache pruned" << endpoint <<
221  " at valence " << entry.valence();
222  iter = m_map.right.erase (iter);
223  }
224 
225  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
226  "Bootcache pruned " << pruned << " entries total";
227 }

229 // Updates the Store with the current set of entries if needed.
230 void
231 Bootcache::update ()
232 {
233  if (! m_needsUpdate)
234  return;
235  std::vector <Store::Entry> list;
236  list.reserve (m_map.size());
237  for (auto const& e : m_map)
238  {
239  Store::Entry se;
240  se.endpoint = e.get_left();
241  se.valence = e.get_right().valence();
242  list.push_back (se);
243  }
244  m_store.save (list);
245  // Reset the flag and cooldown timer
246  m_needsUpdate = false;
247  m_whenUpdate = m_clock.now() + Tuning::bootcacheCooldownTime;
248 }
