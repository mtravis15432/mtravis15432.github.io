91  void Check(int min_expected, int max_expected) {
92  unsigned int next_expected = 0;
93  int missed = 0;
94  int bad_keys = 0;
95  int bad_values = 0;
96  int correct = 0;
97  std::string value_space;
98  Iterator* iter = db_->NewIterator(ReadOptions());
99  for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
100  uint64_t key;
101  Slice in(iter->key());
102  if (!ConsumeDecimalNumber(&in, &key) ||
103  !in.empty() ||
104  key < next_expected) {
105  bad_keys++;
106  continue;
107  }
108  missed += (key - next_expected);
109  next_expected = key + 1;
110  if (iter->value() != Value(key, &value_space)) {
111  bad_values++;
112  } else {
113  correct++;
114  }
115  }
116  delete iter;
117 
118  fprintf(stderr,
119  "expected=%d..%d; got=%d; bad_keys=%d; bad_values=%d; missed=%d\n",
120  min_expected, max_expected, correct, bad_keys, bad_values, missed);
121  ASSERT_LE(min_expected, correct);
122  ASSERT_GE(max_expected, correct);
123  }

125  void Corrupt(FileType filetype, int offset, int bytes_to_corrupt) {
126  // Pick file to corrupt
127  std::vector<std::string> filenames;
128  ASSERT_OK(env_.GetChildren(dbname_, &filenames));
129  uint64_t number;
130  FileType type;
131  std::string fname;
132  int picked_number = -1;
133  for (unsigned int i = 0; i < filenames.size(); i++) {
134  if (ParseFileName(filenames[i], &number, &type) &&
135  type == filetype &&
136  int(number) > picked_number) { // Pick latest file
137  fname = dbname_ + "/" + filenames[i];
138  picked_number = number;
139  }
140  }
141  ASSERT_TRUE(!fname.empty()) << filetype;
142 
143  struct stat sbuf;
144  if (stat(fname.c_str(), &sbuf) != 0) {
145  const char* msg = strerror(errno);
146  ASSERT_TRUE(false) << fname << ": " << msg;
147  }
148 
149  if (offset < 0) {
150  // Relative to end of file; make it absolute
151  if (-offset > sbuf.st_size) {
152  offset = 0;
153  } else {
154  offset = sbuf.st_size + offset;
155  }
156  }
157  if (offset > sbuf.st_size) {
158  offset = sbuf.st_size;
159  }
160  if (offset + bytes_to_corrupt > sbuf.st_size) {
161  bytes_to_corrupt = sbuf.st_size - offset;
162  }
163 
164  // Do it
165  std::string contents;
166  Status s = ReadFileToString(Env::Default(), fname, &contents);
167  ASSERT_TRUE(s.ok()) << s.ToString();
168  for (int i = 0; i < bytes_to_corrupt; i++) {
169  contents[i + offset] ^= 0x80;
170  }
171  s = WriteStringToFile(Env::Default(), contents, fname);
172  ASSERT_TRUE(s.ok()) << s.ToString();
173  }

186  // Return the ith key
187  Slice Key(int i, std::string* storage) {
188  char buf[100];
189  snprintf(buf, sizeof(buf), "%016d", i);
190  storage->assign(buf, strlen(buf));
191  return Slice(*storage);
192  }
