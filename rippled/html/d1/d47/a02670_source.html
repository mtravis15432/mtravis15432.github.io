203 Slice GetCacheKey(const char* cache_key_prefix,
204  size_t cache_key_prefix_size,
205  const BlockHandle& handle,
206  char* cache_key) {
207  assert(cache_key != nullptr);
208  assert(cache_key_prefix_size != 0);
209  assert(cache_key_prefix_size <= kMaxCacheKeyPrefixSize);
210  memcpy(cache_key, cache_key_prefix, cache_key_prefix_size);
211  char* end = EncodeVarint64(cache_key + cache_key_prefix_size,
212  handle.offset());
213  return Slice(cache_key, static_cast<size_t>(end - cache_key));
214 }

241 Status BlockBasedTable::Open(const Options& options,
242  const EnvOptions& soptions,
243  unique_ptr<RandomAccessFile> && file,
244  uint64_t size,
245  unique_ptr<TableReader>* table_reader) {
246  table_reader->reset();
247  if (size < Footer::kEncodedLength) {
248  return Status::InvalidArgument("file is too short to be an sstable");
249  }
250 
251  char footer_space[Footer::kEncodedLength];
252  Slice footer_input;
253  Status s = file->Read(size - Footer::kEncodedLength, Footer::kEncodedLength,
254  &footer_input, footer_space);
255  if (!s.ok()) return s;
256 
257  // Check that we actually read the whole footer from the file. It may be
258  // that size isn't correct.
259  if (footer_input.size() != Footer::kEncodedLength) {
260  return Status::InvalidArgument("file is too short to be an sstable");
261  }
262 
263  Footer footer;
264  s = footer.DecodeFrom(&footer_input);
265  if (!s.ok()) return s;
266 
267  // We've successfully read the footer and the index block: we're
268  // ready to serve requests.
269  Rep* rep = new BlockBasedTable::Rep(soptions);
270  rep->options = options;
271  rep->file = std::move(file);
272  rep->metaindex_handle = footer.metaindex_handle();
273  rep->index_handle = footer.index_handle();
274  SetupCacheKeyPrefix(rep);
275  unique_ptr<BlockBasedTable> new_table(new BlockBasedTable(rep));
276 
277  // Read meta index
278  std::unique_ptr<Block> meta;
279  std::unique_ptr<Iterator> meta_iter;
280  s = ReadMetaBlock(rep, &meta, &meta_iter);
281 
282  // Read the properties
283  meta_iter->Seek(kPropertiesBlock);
284  if (meta_iter->Valid() && meta_iter->key() == Slice(kPropertiesBlock)) {
285  s = meta_iter->status();
286  if (s.ok()) {
287  s = ReadProperties(meta_iter->value(), rep, &rep->table_properties);
288  }
289 
290  if (!s.ok()) {
291  auto err_msg =
292  "[Warning] Encountered error while reading data from properties "
293  "block " + s.ToString();
294  Log(rep->options.info_log, "%s", err_msg.c_str());
295  }
296  }
297 
298  // Initialize index/filter blocks. If block cache is not specified,
299  // these blocks will be kept in member variables in Rep, which will
300  // reside in the memory as long as this table object is alive; otherwise
301  // they will be added to block cache.
302  if (!options.block_cache) {
303  Block* index_block = nullptr;
304  // TODO: we never really verify check sum for index block
305  s = ReadBlockFromFile(
306  rep->file.get(),
307  ReadOptions(),
308  footer.index_handle(),
309  &index_block,
310  options.env
311  );
312 
313  if (s.ok()) {
314  assert(index_block->compressionType() == kNoCompression);
315  rep->index_block.reset(index_block);
316 
317  // Set index block
318  if (rep->options.filter_policy) {
319  std::string key = kFilterBlockPrefix;
320  key.append(rep->options.filter_policy->Name());
321  meta_iter->Seek(key);
322 
323  if (meta_iter->Valid() && meta_iter->key() == Slice(key)) {
324  rep->filter.reset(ReadFilter(meta_iter->value(), rep));
325  }
326  }
327  } else {
328  delete index_block;
329  }
330  } else {
331  // Call IndexBlockReader() to implicitly add index to the block_cache
332  unique_ptr<Iterator> iter(
333  new_table->IndexBlockReader(ReadOptions())
334  );
335  s = iter->status();
336 
337  if (s.ok()) {
338  // Call GetFilter() to implicitly add filter to the block_cache
339  auto filter_entry = new_table->GetFilter();
340  filter_entry.Release(options.block_cache.get());
341  }
342  }
343 
344  if (s.ok()) {
345  *table_reader = std::move(new_table);
346  }
347 
348  return s;
349 }

409  const Slice& filter_handle_value,
410  BlockBasedTable::Rep* rep,
411  size_t* filter_size) {
412  Slice v = filter_handle_value;
413  BlockHandle filter_handle;
414  if (!filter_handle.DecodeFrom(&v).ok()) {
415  return nullptr;
416  }

421  BlockContents block;
422  if (!ReadBlockContents(rep->file.get(), opt, filter_handle, &block,
423  rep->options.env, false).ok()) {
424  return nullptr;
425  }
426 
427  if (filter_size) {
428  *filter_size = block.data.size();
429  }
430 
431  return new FilterBlockReader(
432  rep->options, block.data, block.heap_allocated);

435 Status BlockBasedTable::ReadProperties(
436  const Slice& handle_value, Rep* rep, TableProperties* table_properties) {
437  assert(table_properties);
438 
439  Slice v = handle_value;
440  BlockHandle handle;
441  if (!handle.DecodeFrom(&v).ok()) {
442  return Status::InvalidArgument("Failed to decode properties block handle");
443  }
444 
445  BlockContents block_contents;
446  Status s = ReadBlockContents(
447  rep->file.get(),
448  ReadOptions(),
449  handle,
450  &block_contents,
451  rep->options.env,
452  false
453  );
454 
455  if (!s.ok()) {
456  return s;
457  }
458 
459  Block properties_block(block_contents);
460  std::unique_ptr<Iterator> iter(
461  properties_block.NewIterator(BytewiseComparator())
462  );
463 
464  // All pre-defined properties of type uint64_t
465  std::unordered_map<std::string, uint64_t*> predefined_uint64_properties = {
466  { BlockBasedTablePropertiesNames::kDataSize,
467  &table_properties->data_size },
468  { BlockBasedTablePropertiesNames::kIndexSize,
469  &table_properties->index_size },
470  { BlockBasedTablePropertiesNames::kFilterSize,
471  &table_properties->filter_size },
472  { BlockBasedTablePropertiesNames::kRawKeySize,
473  &table_properties->raw_key_size },
474  { BlockBasedTablePropertiesNames::kRawValueSize,
475  &table_properties->raw_value_size },
476  { BlockBasedTablePropertiesNames::kNumDataBlocks,
477  &table_properties->num_data_blocks },
478  { BlockBasedTablePropertiesNames::kNumEntries,
479  &table_properties->num_entries },
480  };
481 
482  std::string last_key;
483  for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
484  s = iter->status();
485  if (!s.ok()) {
486  break;
487  }
488 
489  auto key = iter->key().ToString();
490  // properties block is strictly sorted with no duplicate key.
491  assert(
492  last_key.empty() ||
493  BytewiseComparator()->Compare(key, last_key) > 0
494  );
495  last_key = key;
496 
497  auto raw_val = iter->value();
498  auto pos = predefined_uint64_properties.find(key);
499 
500  if (pos != predefined_uint64_properties.end()) {
501  // handle predefined rocksdb properties
502  uint64_t val;
503  if (!GetVarint64(&raw_val, &val)) {
504  // skip malformed value
505  auto error_msg =
506  "[Warning] detect malformed value in properties meta-block:"
507  "\tkey: " + key + "\tval: " + raw_val.ToString();
508  Log(rep->options.info_log, "%s", error_msg.c_str());
509  continue;
510  }
511  *(pos->second) = val;
512  } else if (key == BlockBasedTablePropertiesNames::kFilterPolicy) {
513  table_properties->filter_policy_name = raw_val.ToString();
514  } else {
515  // handle user-collected
516  table_properties->user_collected_properties.insert(
517  std::make_pair(key, raw_val.ToString())
518  );
519  }
520  }
521 
522  return s;
523 }

525 Status BlockBasedTable::GetBlock(
526  const BlockBasedTable* table,
527  const BlockHandle& handle,
528  const ReadOptions& options,
529  const bool for_compaction,
530  const Tickers block_cache_miss_ticker,
531  const Tickers block_cache_hit_ticker,
532  bool* didIO,
533  CachableEntry<Block>* entry) {
534  bool no_io = options.read_tier == kBlockCacheTier;
535  Cache* block_cache = table->rep_->options.block_cache.get();
536  auto statistics = table->rep_->options.statistics;
537  Status s;
538 
539  if (block_cache != nullptr) {
540  char cache_key[kMaxCacheKeyPrefixSize + kMaxVarint64Length];
541  auto key = GetCacheKey(
542  table->rep_->cache_key_prefix,
543  table->rep_->cache_key_prefix_size,
544  handle,
545  cache_key
546  );
547 
548  entry->cache_handle = GetFromBlockCache(
549  block_cache,
550  key,
551  block_cache_miss_ticker,
552  block_cache_hit_ticker,
553  table->rep_->options.statistics
554  );
555 
556  if (entry->cache_handle != nullptr) {
557  entry->value =
558  reinterpret_cast<Block*>(block_cache->Value(entry->cache_handle));
559  } else if (no_io) {
560  // Did not find in block_cache and can't do IO
561  return Status::Incomplete("no blocking io");
562  } else {
563  Histograms histogram = for_compaction ?
564  READ_BLOCK_COMPACTION_MICROS : READ_BLOCK_GET_MICROS;
565  {
566  // block for stop watch
567  StopWatch sw(table->rep_->options.env, statistics, histogram);
568  s = ReadBlockFromFile(
569  table->rep_->file.get(),
570  options,
571  handle,
572  &entry->value,
573  table->rep_->options.env,
574  didIO
575  );
576  }
577  if (s.ok()) {
578  if (options.fill_cache && entry->value->isCachable()) {
579  entry->cache_handle = block_cache->Insert(
580  key, entry->value, entry->value->size(), &rocksdb::DeleteCachedBlock);
581  RecordTick(statistics, BLOCK_CACHE_ADD);
582  }
583  }
584  }
585  } else if (no_io) {
586  // Could not read from block_cache and can't do IO
587  return Status::Incomplete("no blocking io");
588  } else {
589  s = ReadBlockFromFile(
590  table->rep_->file.get(),
591  options,
592  handle,
593  &entry->value,
594  table->rep_->options.env,
595  didIO
596  );
597  }
598 
599  return s;
600 }

602 // Convert an index iterator value (i.e., an encoded BlockHandle)
603 // into an iterator over the contents of the corresponding block.
604 Iterator* BlockBasedTable::BlockReader(void* arg,
605  const ReadOptions& options,
606  const Slice& index_value,
607  bool* didIO,
608  bool for_compaction) {
609  const bool no_io = (options.read_tier == kBlockCacheTier);
610  BlockBasedTable* table = reinterpret_cast<BlockBasedTable*>(arg);
611  Cache* block_cache = table->rep_->options.block_cache.get();
612  Cache* block_cache_compressed = table->rep_->options.
613  block_cache_compressed.get();
614  std::shared_ptr<Statistics> statistics = table->rep_->options.statistics;
615  Block* block = nullptr;
616  Block* cblock = nullptr;
617  Cache::Handle* cache_handle = nullptr;
618  Cache::Handle* compressed_cache_handle = nullptr;
619 
620  BlockHandle handle;
621  Slice input = index_value;
622  Status s = handle.DecodeFrom(&input);
623  // We intentionally allow extra stuff in index_value so that we
624  // can add more features in the future.
625 
626  if (!s.ok()) {
627  return NewErrorIterator(s);
628  }
629 
630  if (block_cache != nullptr || block_cache_compressed != nullptr) {
631  char cache_key[kMaxCacheKeyPrefixSize + kMaxVarint64Length];
632  char compressed_cache_key[kMaxCacheKeyPrefixSize + kMaxVarint64Length];
633  Slice key, /* key to the block cache */
634  ckey /* key to the compressed block cache */ ;
635 
636  // create key for block cache
637  if (block_cache != nullptr) {
638  key = GetCacheKey(
639  table->rep_->cache_key_prefix,
640  table->rep_->cache_key_prefix_size,
641  handle,
642  cache_key
643  );
644  }
645 
646  if (block_cache_compressed != nullptr) {
647  ckey = GetCacheKey(
648  table->rep_->compressed_cache_key_prefix,
649  table->rep_->compressed_cache_key_prefix_size,
650  handle,
651  compressed_cache_key
652  );
653  }
654 
655  // Lookup uncompressed cache first
656  if (block_cache != nullptr) {
657  assert(!key.empty());
658  cache_handle = block_cache->Lookup(key);
659  if (cache_handle != nullptr) {
660  block = reinterpret_cast<Block*>(block_cache->Value(cache_handle));
661  RecordTick(statistics, BLOCK_CACHE_HIT);
662  RecordTick(statistics, BLOCK_CACHE_DATA_HIT);
663  } else {
664  RecordTick(statistics, BLOCK_CACHE_MISS);
665  RecordTick(statistics, BLOCK_CACHE_DATA_MISS);
666  }
667  }
668 
669  // If not found in uncompressed cache, lookup compressed cache
670  if (block == nullptr && block_cache_compressed != nullptr) {
671  assert(!ckey.empty());
672  compressed_cache_handle = block_cache_compressed->Lookup(ckey);
673 
674  // if we found in the compressed cache, then uncompress and
675  // insert into uncompressed cache
676  if (compressed_cache_handle != nullptr) {
677  // found compressed block
678  cblock = reinterpret_cast<Block*>(block_cache_compressed->
679  Value(compressed_cache_handle));
680  assert(cblock->compressionType() != kNoCompression);
681 
682  // Retrieve the uncompressed contents into a new buffer
683  BlockContents contents;
684  s = UncompressBlockContents(cblock->data(), cblock->size(),
685  &contents);
686 
687  // Insert uncompressed block into block cache
688  if (s.ok()) {
689  block = new Block(contents); // uncompressed block
690  assert(block->compressionType() == kNoCompression);
691  if (block_cache != nullptr && block->isCachable() &&
692  options.fill_cache) {
693  cache_handle = block_cache->Insert(key, block, block->size(),
694  &rocksdb::DeleteCachedBlock);
695  assert(reinterpret_cast<Block*>(block_cache->Value(cache_handle))
696  == block);
697  }
698  }
699  // Release hold on compressed cache entry
700  block_cache_compressed->Release(compressed_cache_handle);
701  RecordTick(statistics, BLOCK_CACHE_COMPRESSED_HIT);
702  }
703  }
704 
705  if (block != nullptr) {
706  BumpPerfCount(&perf_context.block_cache_hit_count);
707  } else if (no_io) {
708  // Did not find in block_cache and can't do IO
709  return NewErrorIterator(Status::Incomplete("no blocking io"));
710  } else {
711  Histograms histogram = for_compaction ?
712  READ_BLOCK_COMPACTION_MICROS : READ_BLOCK_GET_MICROS;
713  { // block for stop watch
714  StopWatch sw(table->rep_->options.env, statistics, histogram);
715  s = ReadBlockFromFile(
716  table->rep_->file.get(),
717  options,
718  handle,
719  &cblock,
720  table->rep_->options.env,
721  didIO,
722  block_cache_compressed == nullptr
723  );
724  }
725  if (s.ok()) {
726  assert(cblock->compressionType() == kNoCompression ||
727  block_cache_compressed != nullptr);
728 
729  // Retrieve the uncompressed contents into a new buffer
730  BlockContents contents;
731  if (cblock->compressionType() != kNoCompression) {
732  s = UncompressBlockContents(cblock->data(), cblock->size(),
733  &contents);
734  }
735  if (s.ok()) {
736  if (cblock->compressionType() != kNoCompression) {
737  block = new Block(contents); // uncompressed block
738  } else {
739  block = cblock;
740  cblock = nullptr;
741  }
742  if (block->isCachable() && options.fill_cache) {
743  // Insert compressed block into compressed block cache.
744  // Release the hold on the compressed cache entry immediately.
745  if (block_cache_compressed != nullptr && cblock != nullptr) {
746  compressed_cache_handle = block_cache_compressed->Insert(
747  ckey, cblock, cblock->size(), &rocksdb::DeleteCachedBlock);
748  block_cache_compressed->Release(compressed_cache_handle);
749  RecordTick(statistics, BLOCK_CACHE_COMPRESSED_MISS);
750  cblock = nullptr;
751  }
752  // insert into uncompressed block cache
753  assert((block->compressionType() == kNoCompression));
754  if (block_cache != nullptr) {
755  cache_handle = block_cache->Insert(
756  key, block, block->size(), &rocksdb::DeleteCachedBlock);
757  RecordTick(statistics, BLOCK_CACHE_ADD);
758  assert(reinterpret_cast<Block*>(block_cache->Value(
759  cache_handle))== block);
760  }
761  }
762  }
763  }
764  if (cblock != nullptr) {
765  delete cblock;
766  }
767  }
768  } else if (no_io) {
769  // Could not read from block_cache and can't do IO
770  return NewErrorIterator(Status::Incomplete("no blocking io"));
771  } else {
772  s = ReadBlockFromFile(
773  table->rep_->file.get(),
774  options,
775  handle,
776  &block,
777  table->rep_->options.env,
778  didIO
779  );
780  }
781 
782  Iterator* iter;
783  if (block != nullptr) {
784  iter = block->NewIterator(table->rep_->options.comparator);
785  if (cache_handle != nullptr) {
786  iter->RegisterCleanup(&ReleaseBlock, block_cache, cache_handle);
787  } else {
788  iter->RegisterCleanup(&DeleteBlock, block, nullptr);
789  }
790  } else {
791  iter = NewErrorIterator(s);
792  }
793  return iter;
794 }

797 BlockBasedTable::GetFilter(bool no_io) const {
798  if (!rep_->options.filter_policy || !rep_->options.block_cache) {
799  return {rep_->filter.get(), nullptr};
800  }
801 
802  // Fetching from the cache
803  Cache* block_cache = rep_->options.block_cache.get();
804  char cache_key[kMaxCacheKeyPrefixSize + kMaxVarint64Length];
805  auto key = GetCacheKey(
806  rep_->cache_key_prefix,
807  rep_->cache_key_prefix_size,
808  rep_->metaindex_handle,
809  cache_key
810  );
811 
812  auto cache_handle = GetFromBlockCache(
813  block_cache,
814  key,
815  BLOCK_CACHE_FILTER_MISS,
816  BLOCK_CACHE_FILTER_HIT,
817  rep_->options.statistics
818  );
819 
820  FilterBlockReader* filter = nullptr;
821  if (cache_handle != nullptr) {
822  filter = reinterpret_cast<FilterBlockReader*>(
823  block_cache->Value(cache_handle));
824  } else if (no_io) {
825  // Do not invoke any io.
826  return CachableEntry<FilterBlockReader>();
827  } else {
828  size_t filter_size = 0;
829  std::unique_ptr<Block> meta;
830  std::unique_ptr<Iterator> iter;
831  auto s = ReadMetaBlock(rep_, &meta, &iter);
832 
833  if (s.ok()) {
834  std::string filter_block_key = kFilterBlockPrefix;
835  filter_block_key.append(rep_->options.filter_policy->Name());
836  iter->Seek(filter_block_key);
837 
838  if (iter->Valid() && iter->key() == Slice(filter_block_key)) {
839  filter = ReadFilter(iter->value(), rep_, &filter_size);
840  assert(filter);
841  assert(filter_size > 0);
842 
843  cache_handle = block_cache->Insert(
844  key, filter, filter_size, &DeleteCachedFilter);
845  RecordTick(rep_->options.statistics, BLOCK_CACHE_ADD);
846  }
847  }
848  }
849 
850  return { filter, cache_handle };

900 // This will be broken if the user specifies an unusual implementation
901 // of Options.comparator, or if the user specifies an unusual
902 // definition of prefixes in Options.filter_policy. In particular, we
903 // require the following three properties:
904 //
905 // 1) key.starts_with(prefix(key))
906 // 2) Compare(prefix(key), key) <= 0.
907 // 3) If Compare(key1, key2) <= 0, then Compare(prefix(key1), prefix(key2)) <= 0
908 //
909 // Otherwise, this method guarantees no I/O will be incurred.
910 //
911 // REQUIRES: this method shouldn't be called while the DB lock is held.
912 bool BlockBasedTable::PrefixMayMatch(const Slice& internal_prefix) {
913  bool may_match = true;
914  Status s;
915 
916  if (!rep_->options.filter_policy) {
917  return true;
918  }
919 
920  // To prevent any io operation in this method, we set `read_tier` to make
921  // sure we always read index or filter only when they have already been
922  // loaded to memory.
923  ReadOptions no_io_read_options;
924  no_io_read_options.read_tier = kBlockCacheTier;
925  unique_ptr<Iterator> iiter(
926  IndexBlockReader(no_io_read_options)
927  );
928  iiter->Seek(internal_prefix);
929 
930  if (!iiter->Valid()) {
931  // we're past end of file
932  // if it's incomplete, it means that we avoided I/O
933  // and we're not really sure that we're past the end
934  // of the file
935  may_match = iiter->status().IsIncomplete();
936  } else if (ExtractUserKey(iiter->key()).starts_with(
937  ExtractUserKey(internal_prefix))) {
938  // we need to check for this subtle case because our only
939  // guarantee is that "the key is a string >= last key in that data
940  // block" according to the doc/table_format.txt spec.
941  //
942  // Suppose iiter->key() starts with the desired prefix; it is not
943  // necessarily the case that the corresponding data block will
944  // contain the prefix, since iiter->key() need not be in the
945  // block. However, the next data block may contain the prefix, so
946  // we return true to play it safe.
947  may_match = true;
948  } else {
949  // iiter->key() does NOT start with the desired prefix. Because
950  // Seek() finds the first key that is >= the seek target, this
951  // means that iiter->key() > prefix. Thus, any data blocks coming
952  // after the data block corresponding to iiter->key() cannot
953  // possibly contain the key. Thus, the corresponding data block
954  // is the only one which could potentially contain the prefix.
955  Slice handle_value = iiter->value();
956  BlockHandle handle;
957  s = handle.DecodeFrom(&handle_value);
958  assert(s.ok());
959  auto filter_entry = GetFilter(true /* no io */);
960  may_match =
961  filter_entry.value == nullptr ||
962  filter_entry.value->PrefixMayMatch(handle.offset(), internal_prefix);
963  filter_entry.Release(rep_->options.block_cache.get());
964  }
965 
966  RecordTick(rep_->options.statistics, BLOOM_FILTER_PREFIX_CHECKED);
967  if (!may_match) {
968  RecordTick(rep_->options.statistics, BLOOM_FILTER_PREFIX_USEFUL);
969  }
970 
971  return may_match;
972 }

993 Status BlockBasedTable::Get(
994  const ReadOptions& readOptions,
995  const Slice& key,
996  void* handle_context,
997  bool (*result_handler)(void* handle_context, const Slice& k,
998  const Slice& v, bool didIO),
999  void (*mark_key_may_exist_handler)(void* handle_context)) {
1000  Status s;
1001  Iterator* iiter = IndexBlockReader(readOptions);
1002  auto filter_entry = GetFilter(readOptions.read_tier == kBlockCacheTier);
1003  FilterBlockReader* filter = filter_entry.value;
1004  bool done = false;
1005  for (iiter->Seek(key); iiter->Valid() && !done; iiter->Next()) {
1006  Slice handle_value = iiter->value();
1007 
1008  BlockHandle handle;
1009  bool may_not_exist_in_filter =
1010  filter != nullptr &&
1011  handle.DecodeFrom(&handle_value).ok() &&
1012  !filter->KeyMayMatch(handle.offset(), key);
1013 
1014  if (may_not_exist_in_filter) {
1015  // Not found
1016  // TODO: think about interaction with Merge. If a user key cannot
1017  // cross one data block, we should be fine.
1018  RecordTick(rep_->options.statistics, BLOOM_FILTER_USEFUL);
1019  break;
1020  } else {
1021  bool didIO = false;
1022  unique_ptr<Iterator> block_iter(
1023  BlockReader(this, readOptions, iiter->value(), &didIO));
1024 
1025  if (readOptions.read_tier && block_iter->status().IsIncomplete()) {
1026  // couldn't get block from block_cache
1027  // Update Saver.state to Found because we are only looking for whether
1028  // we can guarantee the key is not there when "no_io" is set
1029  (*mark_key_may_exist_handler)(handle_context);
1030  break;
1031  }
1032 
1033  // Call the *saver function on each entry/block until it returns false
1034  for (block_iter->Seek(key); block_iter->Valid(); block_iter->Next()) {
1035  if (!(*result_handler)(handle_context, block_iter->key(),
1036  block_iter->value(), didIO)) {
1037  done = true;
1038  break;
1039  }
1040  }
1041  s = block_iter->status();
1042  }
1043  }
1044 
1045  filter_entry.Release(rep_->options.block_cache.get());
1046  if (s.ok()) {
1047  s = iiter->status();
1048  }
1049  delete iiter;
1050  return s;
1051 }

1067 uint64_t BlockBasedTable::ApproximateOffsetOf(const Slice& key) {
1068  Iterator* index_iter = IndexBlockReader(ReadOptions());
1069 
1070  index_iter->Seek(key);
1071  uint64_t result;
1072  if (index_iter->Valid()) {
1073  BlockHandle handle;
1074  Slice input = index_iter->value();
1075  Status s = handle.DecodeFrom(&input);
1076  if (s.ok()) {
1077  result = handle.offset();
1078  } else {
1079  // Strange: we can't decode the block handle in the index block.
1080  // We'll just return the offset of the metaindex block, which is
1081  // close to the whole file size for this case.
1082  result = rep_->metaindex_handle.offset();
1083  }
1084  } else {
1085  // key is past the last key in the file. Approximate the offset
1086  // by returning the offset of the metaindex block (which is
1087  // right near the end of the file).
1088  result = rep_->metaindex_handle.offset();
1089  }
1090  delete index_iter;
1091  return result;
