36  // Merge entries until we hit
37  // - a corrupted key
38  // - a Put/Delete,
39  // - a different user key,
40  // - a specific sequence number (snapshot boundary),
41  // or - the end of iteration
42  // iter: (IN) points to the first merge type entry
43  // (OUT) points to the first entry not included in the merge process
44  // stop_before: (IN) a sequence number that merge should not cross.
45  // 0 means no restriction
46  // at_bottom: (IN) true if the iterator covers the bottem level, which means
47  // we could reach the start of the history of this user key.
48  void MergeUntil(Iterator* iter, SequenceNumber stop_before = 0,
49  bool at_bottom = false, shared_ptr<Statistics> stats=nullptr);

51  // Query the merge result
52  // These are valid until the next MergeUntil call
53  // If the merging was successful:
54  // - IsSuccess() will be true
55  // - key() will have the latest sequence number of the merges.
56  // The type will be Put or Merge. See IMPORTANT 1 note, below.
57  // - value() will be the result of merging all the operands together
58  // - The user should ignore keys() and values().
59  //
60  // IMPORTANT 1: the key type could change after the MergeUntil call.
61  // Put/Delete + Merge + ... + Merge => Put
62  // Merge + ... + Merge => Merge
63  //
64  // If the merge operator is not associative, and if a Put/Delete is not found
65  // then the merging will be unsuccessful. In this case:
66  // - IsSuccess() will be false
67  // - keys() contains the list of internal keys seen in order of iteration.
68  // - values() contains the list of values (merges) seen in the same order.
69  // values() is parallel to keys() so that the first entry in
70  // keys() is the key associated with the first entry in values()
71  // and so on. These lists will be the same length.
72  // All of these pairs will be merges over the same user key.
73  // See IMPORTANT 2 note below.
74  // - The user should ignore key() and value().
75  //
76  // IMPORTANT 2: The entries were traversed in order from BACK to FRONT.
77  // So keys().back() was the first key seen by iterator.
78  // TODO: Re-style this comment to be like the first one
79  bool IsSuccess() { return success_; }
80  Slice key() { assert(success_); return Slice(keys_.back()); }
81  Slice value() { assert(success_); return Slice(operands_.back()); }
82  const std::deque<std::string>& keys() { assert(!success_); return keys_; }
83  const std::deque<std::string>& values() {

85  }
