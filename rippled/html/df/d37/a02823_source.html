25 // Why base-58 instead of standard base-64 encoding?
26 // - Don't want 0OIl characters that look the same in some fonts and
27 // could be used to create visually identical looking account numbers.
28 // - A string with non-alphanumeric characters is not as easily accepted as an account number.
29 // - E-mail usually won't line-break if there's no punctuation to break at.
30 // - Doubleclicking selects the whole number as one word if it's all alphanumeric.
31 //

85  template <class InputIt>
86  static std::string encode (InputIt first, InputIt last,
87  Alphabet const& alphabet, bool withCheck)
88  {
89  typedef typename std::iterator_traits<InputIt>::value_type value_type;
90  std::vector <typename std::remove_const <value_type>::type> v;
91  std::size_t const size (std::distance (first, last));
92  if (withCheck)
93  {
94  v.reserve (size + 1 + 4);
95  v.insert (v.begin(), first, last);
96  unsigned char hash [4];
97  fourbyte_hash256 (hash, &v.front(), v.size());
98  v.resize (0);
99  // Place the hash reversed in the front
100  std::copy (std::reverse_iterator <unsigned char const*> (hash+4),
101  std::reverse_iterator <unsigned char const*> (hash),
102  std::back_inserter (v));
103  }
104  else
105  {
106  v.reserve (size + 1);
107  }
108  // Append input little endian
109  std::copy (std::reverse_iterator <InputIt> (last),
110  std::reverse_iterator <InputIt> (first),
111  std::back_inserter (v));
112  // Pad zero to make the BIGNUM positive
113  v.push_back (0);
114  return raw_encode (&v.front(), &v.back()+1, alphabet, withCheck);
115  }
