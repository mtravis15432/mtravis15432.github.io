88 std::vector<RippleAddress> TransactionMetaSet::getAffectedAccounts ()
89 {
90  std::vector<RippleAddress> accounts;
91  accounts.reserve (10);
92 
93  // This code should match the behavior of the JS method:
94  // Meta#getAffectedAccounts
95  BOOST_FOREACH (const STObject & it, mNodes)
96  {
97  int index = it.getFieldIndex ((it.getFName () == sfCreatedNode) ? sfNewFields : sfFinalFields);
98 
99  if (index != -1)
100  {
101  const STObject* inner = dynamic_cast<const STObject*> (&it.peekAtIndex (index));
102 
103  if (inner)
104  {
105  BOOST_FOREACH (const SerializedType & field, inner->peekData ())
106  {
107  const STAccount* sa = dynamic_cast<const STAccount*> (&field);
108 
109  if (sa)
110  addIfUnique (accounts, sa->getValueNCA ());
111  else if ((field.getFName () == sfLowLimit) || (field.getFName () == sfHighLimit) ||
112  (field.getFName () == sfTakerPays) || (field.getFName () == sfTakerGets))
113  {
114  const STAmount* lim = dynamic_cast<const STAmount*> (&field);
115 
116  if (lim != nullptr)
117  {
118  uint160 issuer = lim->getIssuer ();
119 
120  if (issuer.isNonZero ())
121  {
122  RippleAddress na;
123  na.setAccountID (issuer);
124  addIfUnique (accounts, na);
125  }
126  }
127  else
128  {
129  WriteLog (lsFATAL, TransactionMetaSet) << "limit is not amount " << field.getJson (0);
130  }
131  }
132  }
133  }
134  else assert (false);
135  }
136  }
137 
138  return accounts;
139 }
