16 // PRE: iter points to the first merge type entry
17 // POST: iter points to the first entry beyond the merge process (or the end)
18 // keys_, operands_ are updated to reflect the merge result.
19 // keys_ stores the list of keys encountered while merging.
20 // operands_ stores the list of merge operands encountered while merging.
21 // keys_[i] corresponds to operands_[i] for each i.
22 void MergeHelper::MergeUntil(Iterator* iter, SequenceNumber stop_before,
23  bool at_bottom, shared_ptr<Statistics> stats) {
24  // Get a copy of the internal key, before it's invalidated by iter->Next()
25  // Also maintain the list of merge operands seen.
26  keys_.clear();
27  operands_.clear();
28  keys_.push_front(iter->key().ToString());
29  operands_.push_front(iter->value().ToString());
30 
31  success_ = false; // Will become true if we hit Put/Delete or bottom
32 
33  // We need to parse the internal key again as the parsed key is
34  // backed by the internal key!
35  // Assume no internal key corruption as it has been successfully parsed
36  // by the caller.
37  // Invariant: keys_.back() will not change. Hence, orig_ikey is always valid.
38  ParsedInternalKey orig_ikey;
39  ParseInternalKey(keys_.back(), &orig_ikey);
40 
41  bool hit_the_next_user_key = false;
42  ParsedInternalKey ikey;
43  std::string merge_result; // Temporary value for merge results
44  for (iter->Next(); iter->Valid(); iter->Next()) {
45  assert(operands_.size() >= 1); // Should be invariants!
46  assert(keys_.size() == operands_.size());
47 
48  if (!ParseInternalKey(iter->key(), &ikey)) {
49  // stop at corrupted key
50  if (assert_valid_internal_key_) {
51  assert(!"corrupted internal key is not expected");
52  }
53  break;
54  }
55 
56  if (user_comparator_->Compare(ikey.user_key, orig_ikey.user_key) != 0) {
57  // hit a different user key, stop right here
58  hit_the_next_user_key = true;
59  break;
60  }
61 
62  if (stop_before && ikey.sequence <= stop_before) {
63  // hit an entry that's visible by the previous snapshot, can't touch that
64  break;
65  }
66 
67  // At this point we are guaranteed that we need to process this key.
68 
69  if (kTypeDeletion == ikey.type) {
70  // hit a delete
71  // => merge nullptr with operands_
72  // => store result in operands_.back() (and update keys_.back())
73  // => change the entry type to kTypeValue for keys_.back()
74  // We are done! Return a success if the merge passes.
75  success_ = user_merge_operator_->FullMerge(ikey.user_key, nullptr,
76  operands_, &merge_result,
77  logger_);
78 
79  // We store the result in keys_.back() and operands_.back()
80  // if nothing went wrong (i.e.: no operand corruption on disk)
81  if (success_) {
82  std::string& key = keys_.back(); // The original key encountered
83  orig_ikey.type = kTypeValue;
84  UpdateInternalKey(&key[0], key.size(),
85  orig_ikey.sequence, orig_ikey.type);
86  swap(operands_.back(), merge_result);
87  } else {
88  RecordTick(stats, NUMBER_MERGE_FAILURES);
89  }
90 
91  // move iter to the next entry (before doing anything else)
92  iter->Next();
93  return;
94  }
95 
96  if (kTypeValue == ikey.type) {
97  // hit a put
98  // => merge the put value with operands_
99  // => store result in operands_.back() (and update keys_.back())
100  // => change the entry type to kTypeValue for keys_.back()
101  // We are done! Success!
102  const Slice value = iter->value();
103  success_ = user_merge_operator_->FullMerge(ikey.user_key, &value,
104  operands_, &merge_result,
105  logger_);
106 
107  // We store the result in keys_.back() and operands_.back()
108  // if nothing went wrong (i.e.: no operand corruption on disk)
109  if (success_) {
110  std::string& key = keys_.back(); // The original key encountered
111  orig_ikey.type = kTypeValue;
112  UpdateInternalKey(&key[0], key.size(),
113  orig_ikey.sequence, orig_ikey.type);
114  swap(operands_.back(), merge_result);
115  } else {
116  RecordTick(stats, NUMBER_MERGE_FAILURES);
117  }
118 
119  // move iter to the next entry
120  iter->Next();
121  return;
122  }
123 
124  if (kTypeMerge == ikey.type) {
125  // hit a merge
126  // => merge the operand into the front of the operands_ list
127  // => use the user's associative merge function to determine how.
128  // => then continue because we haven't yet seen a Put/Delete.
129  assert(!operands_.empty()); // Should have at least one element in it
130 
131  keys_.push_front(iter->key().ToString());
132  operands_.push_front(iter->value().ToString());
133  while (operands_.size() >= 2) {
134  // Returns false when the merge_operator can no longer process it
135  if (user_merge_operator_->PartialMerge(ikey.user_key,
136  Slice(operands_[0]),
137  Slice(operands_[1]),
138  &merge_result,
139  logger_)) {
140  // Merging of operands (associative merge) was successful.
141  // Replace these frontmost two operands with the merge result
142  keys_.pop_front();
143  operands_.pop_front();
144  swap(operands_.front(), merge_result);
145  } else {
146  // Merging of operands (associative merge) returned false.
147  // The user merge_operator does not know how to merge these operands.
148  // So we just stack them up until we find a Put/Delete or end of key.
149  break;
150  }
151  }
152  continue;
153  }
154  }
155 
156  // We are sure we have seen this key's entire history if we are at the
157  // last level and exhausted all internal keys of this user key.
158  // NOTE: !iter->Valid() does not necessarily mean we hit the
159  // beginning of a user key, as versions of a user key might be
160  // split into multiple files (even files on the same level)
161  // and some files might not be included in the compaction/merge.
162  //
163  // There are also cases where we have seen the root of history of this
164  // key without being sure of it. Then, we simply miss the opportunity
165  // to combine the keys. Since VersionSet::SetupOtherInputs() always makes
166  // sure that all merge-operands on the same level get compacted together,
167  // this will simply lead to these merge operands moving to the next level.
168  //
169  // So, we only perform the following logic (to merge all operands together
170  // without a Put/Delete) if we are certain that we have seen the end of key.
171  bool surely_seen_the_beginning = hit_the_next_user_key && at_bottom;
172  if (surely_seen_the_beginning) {
173  // do a final merge with nullptr as the existing value and say
174  // bye to the merge type (it's now converted to a Put)
175  assert(kTypeMerge == orig_ikey.type);
176  assert(operands_.size() >= 1);
177  assert(operands_.size() == keys_.size());
178  success_ = user_merge_operator_->FullMerge(orig_ikey.user_key, nullptr,
179  operands_, &merge_result,
180  logger_);
181 
182  if (success_) {
183  std::string& key = keys_.back(); // The original key encountered
184  orig_ikey.type = kTypeValue;
185  UpdateInternalKey(&key[0], key.size(),
186  orig_ikey.sequence, orig_ikey.type);
187 
188  // The final value() is always stored in operands_.back()
189  swap(operands_.back(),merge_result);
190  } else {
191  RecordTick(stats, NUMBER_MERGE_FAILURES);
192  // Do nothing if not success_. Leave keys() and operands() as they are.
193  }
194  }
195 }
