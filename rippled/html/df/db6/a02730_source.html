40 // Used for reading a file from HDFS. It implements both sequential-read
41 // access methods as well as random read access methods.
42 class HdfsReadableFile: virtual public SequentialFile, virtual public RandomAccessFile {

49  HdfsReadableFile(hdfsFS fileSys, const std::string& fname)

71  // sequential access, read data at current offset in file
72  virtual Status Read(size_t n, Slice* result, char* scratch) {
73  Status s;
74  Log(mylog, "[hdfs] HdfsReadableFile reading %s %ld\n",
75  filename_.c_str(), n);
76  size_t bytes_read = hdfsRead(fileSys_, hfile_, scratch, (tSize)n);
77  Log(mylog, "[hdfs] HdfsReadableFile read %s\n", filename_.c_str());
78  *result = Slice(scratch, bytes_read);
79  if (bytes_read < n) {
80  if (feof()) {
81  // We leave status as ok if we hit the end of the file
82  } else {
83  // A partial read with an error: return a non-ok status
84  s = IOError(filename_, errno);
85  }
86  }
87  return s;
88  }
89 
90  // random access, read data from specified offset in file
91  virtual Status Read(uint64_t offset, size_t n, Slice* result,
92  char* scratch) const {
93  Status s;
94  Log(mylog, "[hdfs] HdfsReadableFile preading %s\n", filename_.c_str());
95  ssize_t bytes_read = hdfsPread(fileSys_, hfile_, offset,
96  (void*)scratch, (tSize)n);
97  Log(mylog, "[hdfs] HdfsReadableFile pread %s\n", filename_.c_str());
98  *result = Slice(scratch, (bytes_read < 0) ? 0 : bytes_read);
99  if (bytes_read < 0) {
100  // An error: return a non-ok status
101  s = IOError(filename_, errno);
102  }
103  return s;
104  }
105 
106  virtual Status Skip(uint64_t n) {
107  Log(mylog, "[hdfs] HdfsReadableFile skip %s\n", filename_.c_str());
108  // get current offset from file
109  tOffset current = hdfsTell(fileSys_, hfile_);
110  if (current < 0) {
111  return IOError(filename_, errno);
112  }
113  // seek to new offset in file
114  tOffset newoffset = current + n;
115  int val = hdfsSeek(fileSys_, hfile_, newoffset);
116  if (val < 0) {
117  return IOError(filename_, errno);
118  }
119  return Status::OK();
120  }
121 
122  private:
123 
124  // returns true if we are at the end of file, false otherwise
125  bool feof() {
126  Log(mylog, "[hdfs] HdfsReadableFile feof %s\n", filename_.c_str());
127  if (hdfsTell(fileSys_, hfile_) == fileSize()) {
128  return true;
129  }
130  return false;
131  }
132 
133  // the current size of the file
134  tOffset fileSize() {
135  Log(mylog, "[hdfs] HdfsReadableFile fileSize %s\n", filename_.c_str());
136  hdfsFileInfo* pFileInfo = hdfsGetPathInfo(fileSys_, filename_.c_str());
137  tOffset size = 0L;
138  if (pFileInfo != nullptr) {
139  size = pFileInfo->mSize;
140  hdfsFreeFileInfo(pFileInfo, 1);
141  } else {
142  throw rocksdb::HdfsFatalException("fileSize on unknown file " +
143  filename_);
144  }
145  return size;
146  }

203  if (hdfsFlush(fileSys_, hfile_) == -1) {

254  virtual void Logv(const char* format, va_list ap) {
255  const uint64_t thread_id = (*gettid_)();
256 
257  // We try twice: the first time with a fixed-size stack allocated buffer,
258  // and the second time with a much larger dynamically allocated buffer.
259  char buffer[500];
260  for (int iter = 0; iter < 2; iter++) {
261  char* base;
262  int bufsize;
263  if (iter == 0) {
264  bufsize = sizeof(buffer);
265  base = buffer;
266  } else {
267  bufsize = 30000;
268  base = new char[bufsize];
269  }
270  char* p = base;
271  char* limit = base + bufsize;
272 
273  struct timeval now_tv;
274  gettimeofday(&now_tv, nullptr);
275  const time_t seconds = now_tv.tv_sec;
276  struct tm t;
277  localtime_r(&seconds, &t);
278  p += snprintf(p, limit - p,
279  "%04d/%02d/%02d-%02d:%02d:%02d.%06d %llx ",
280  t.tm_year + 1900,
281  t.tm_mon + 1,
282  t.tm_mday,
283  t.tm_hour,
284  t.tm_min,
285  t.tm_sec,
286  static_cast<int>(now_tv.tv_usec),
287  static_cast<long long unsigned int>(thread_id));
288 
289  // Print the message
290  if (p < limit) {
291  va_list backup_ap;
292  va_copy(backup_ap, ap);
293  p += vsnprintf(p, limit - p, format, backup_ap);
294  va_end(backup_ap);
295  }
296 
297  // Truncate to available space if necessary
298  if (p >= limit) {
299  if (iter == 0) {
300  continue; // Try again with larger buffer
301  } else {
302  p = limit - 1;
303  }
304  }
305 
306  // Add newline if necessary
307  if (p == base || p[-1] != '\n') {
308  *p++ = '\n';
309  }
310 
311  assert(p <= limit);
312  file_->Append(base, p-base);
313  file_->Flush();
314  if (base != buffer) {
315  delete[] base;
316  }
317  break;
318  }
319  }

377 Status HdfsEnv::GetChildren(const std::string& path,
378  std::vector<std::string>* result) {
379  int value = hdfsExists(fileSys_, path.c_str());
380  switch (value) {
381  case 0: {
382  int numEntries = 0;
383  hdfsFileInfo* pHdfsFileInfo = 0;
384  pHdfsFileInfo = hdfsListDirectory(fileSys_, path.c_str(), &numEntries);
385  if (numEntries >= 0) {
386  for(int i = 0; i < numEntries; i++) {
387  char* pathname = pHdfsFileInfo[i].mName;
388  char* filename = rindex(pathname, '/');
389  if (filename != nullptr) {
390  result->push_back(filename+1);
391  }
392  }
393  if (pHdfsFileInfo != nullptr) {
394  hdfsFreeFileInfo(pHdfsFileInfo, numEntries);
395  }
396  } else {
397  // numEntries < 0 indicates error
398  Log(mylog, "hdfsListDirectory call failed with error ");
399  throw HdfsFatalException("hdfsListDirectory call failed negative error.\n");
400  }
401  break;
402  }
403  case 1: // directory does not exist, exit
404  break;
405  default: // anything else should be an error
406  Log(mylog, "hdfsListDirectory call failed with error ");
407  throw HdfsFatalException("hdfsListDirectory call failed with error.\n");
408  }
409  return Status::OK();
410 }
