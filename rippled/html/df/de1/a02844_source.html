53 std::string Base58::raw_encode (
54  unsigned char const* begin, unsigned char const* end,
55  Alphabet const& alphabet, bool withCheck)
56 {
57  CAutoBN_CTX pctx;
58  CBigNum bn58 = 58;
59  CBigNum bn0 = 0;
60 
61  // Convert little endian data to bignum
62  CBigNum bn (begin, end);
63  std::size_t const size (std::distance (begin, end));
64 
65  // Convert bignum to std::string
66  std::string str;
67  // Expected size increase from base58 conversion is approximately 137%
68  // use 138% to be safe
69  str.reserve (size * 138 / 100 + 1);
70  CBigNum dv;
71  CBigNum rem;
72 
73  while (bn > bn0)
74  {
75  if (!BN_div (&dv, &rem, &bn, &bn58, pctx))
76  throw bignum_error ("EncodeBase58 : BN_div failed");
77 
78  bn = dv;
79  unsigned int c = rem.getuint ();
80  str += alphabet [c];
81  }
82 
83  for (const unsigned char* p = end-2; p >= begin && *p == 0; p--)
84  str += alphabet [0];
85 
86  // Convert little endian std::string to big endian
87  reverse (str.begin (), str.end ());
88  return str;
89 }

103 bool Base58::raw_decode (char const* first, char const* last, void* dest,
104  std::size_t size, bool checked, Alphabet const& alphabet)
105 {
106  CAutoBN_CTX pctx;
107  CBigNum bn58 = 58;
108  CBigNum bn = 0;
109  CBigNum bnChar;
110 
111  // Convert big endian string to bignum
112  for (char const* p = first; p != last; ++p)
113  {
114  int i (alphabet.from_char (*p));
115  if (i == -1)
116  return false;
117  bnChar.setuint ((unsigned int) i);
118 
119  int const success (BN_mul (&bn, &bn, &bn58, pctx));
120  assert (success);
121 
122  bn += bnChar;
123  }
124 
125  // Get bignum as little endian data
126  Blob vchTmp = bn.getvch ();
127 
128  // Trim off sign byte if present
129  if (vchTmp.size () >= 2 && vchTmp.end ()[-1] == 0 && vchTmp.end ()[-2] >= 0x80)
130  vchTmp.erase (vchTmp.end () - 1);
131 
132  char* const out (static_cast <char*> (dest));
133 
134  // Count leading zeros
135  int nLeadingZeros = 0;
136  for (char const* p = first; p!=last && *p==alphabet[0]; p++)
137  nLeadingZeros++;
138 
139  // Verify that the size is correct
140  if (vchTmp.size() + nLeadingZeros != size)
141  return false;
142 
143  // Fill the leading zeros
144  memset (out, 0, nLeadingZeros);
145 
146  // Copy little endian data to big endian
147  std::reverse_copy (vchTmp.begin (), vchTmp.end (),
148  out + nLeadingZeros);
149 
150  if (checked)
151  {
152  char hash4 [4];
153  fourbyte_hash256 (hash4, out, size - 4);
154  if (memcmp (hash4, out + size - 4, 4) != 0)
155  return false;
156  }
157 
158  return true;
159 }

161 bool Base58::decode (const char* psz, Blob& vchRet, Alphabet const& alphabet)
162 {
163  CAutoBN_CTX pctx;
164  vchRet.clear ();
165  CBigNum bn58 = 58;
166  CBigNum bn = 0;
167  CBigNum bnChar;
168 
169  while (isspace (*psz))
170  psz++;
171 
172  // Convert big endian string to bignum
173  for (const char* p = psz; *p; p++)
174  {
175  // VFALCO TODO Make this use the inverse table!
176  // Or better yet ditch this and call raw_decode
177  //
178  const char* p1 = strchr (alphabet.chars(), *p);
179 
180  if (p1 == nullptr)
181  {
182  while (isspace (*p))
183  p++;
184 
185  if (*p != '\0')
186  return false;
187 
188  break;
189  }
190 
191  bnChar.setuint (p1 - alphabet.chars());
192 
193  if (!BN_mul (&bn, &bn, &bn58, pctx))
194  throw bignum_error ("DecodeBase58 : BN_mul failed");
195 
196  bn += bnChar;
197  }
198 
199  // Get bignum as little endian data
200  Blob vchTmp = bn.getvch ();
201 
202  // Trim off sign byte if present
203  if (vchTmp.size () >= 2 && vchTmp.end ()[-1] == 0 && vchTmp.end ()[-2] >= 0x80)
204  vchTmp.erase (vchTmp.end () - 1);
205 
206  // Restore leading zeros
207  int nLeadingZeros = 0;
208 
209  for (const char* p = psz; *p == alphabet.chars()[0]; p++)
210  nLeadingZeros++;
211 
212  vchRet.assign (nLeadingZeros + vchTmp.size (), 0);
213 
214  // Convert little endian data to big endian
215  std::reverse_copy (vchTmp.begin (), vchTmp.end (), vchRet.end () - vchTmp.size ());
216  return true;
217 }

224 bool Base58::decodeWithCheck (const char* psz, Blob& vchRet, Alphabet const& alphabet)
225 {
226  if (!decode (psz, vchRet, alphabet))
227  return false;
228

232  return false;

236

240  return false;
241  }

244  return true;
