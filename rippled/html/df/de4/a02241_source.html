107 void FileSearchPath::removeRedundantPaths()
108 {
109  for (int i = directories.size(); --i >= 0;)
110  {
111  const File d1 (directories[i]);
112 
113  for (int j = directories.size(); --j >= 0;)
114  {
115  const File d2 (directories[j]);
116 
117  if ((i != j) && (d1.isAChildOf (d2) || d1 == d2))
118  {
119  directories.remove (i);
120  break;
121  }
122  }
123  }
