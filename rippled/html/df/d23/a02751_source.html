294 //
295 // The goal of this method is to create a configuration that
296 // allows an application to write all files into L0 and
297 // then do a single compaction to output all files into L1.
298 Options*
299 Options::PrepareForBulkLoad()
300 {
301  // never slowdown ingest.
302  level0_file_num_compaction_trigger = (1<<30);
303  level0_slowdown_writes_trigger = (1<<30);
304  level0_stop_writes_trigger = (1<<30);
305 
306  // no auto compactions please. The application should issue a
307  // manual compaction after all data is loaded into L0.
308  disable_auto_compactions = true;
309  disable_seek_compaction = true;
310  disableDataSync = true;
311 
312  // A manual compaction run should pick all files in L0 in
313  // a single compaction run.
314  source_compaction_factor = (1<<30);
315 
316  // It is better to have only 2 levels, otherwise a manual
317  // compaction would compact at every possible level, thereby
318  // increasing the total time needed for compactions.
319  num_levels = 2;
320 
321  // Prevent a memtable flush to automatically promote files
322  // to L1. This is helpful so that all files that are
323  // input to the manual compaction are all at L0.
324  max_background_compactions = 2;
325 
326  // The compaction would create large files in L1.
327  target_file_size_base = 256 * 1024 * 1024;
328  return this;
329 }
