95 bool SerializedValidation::isValid (uint256 const& signingHash) const
96 {
97  try
98  {
99  const ECDSA fullyCanonical = getFlags () & vfFullyCanonicalSig ?
100  ECDSA::strict : ECDSA::not_strict;
101  RippleAddress raPublicKey = RippleAddress::createNodePublic (getFieldVL (sfSigningPubKey));
102  return raPublicKey.isValid () &&
103  raPublicKey.verifyNodePublic (signingHash, getFieldVL (sfSignature), fullyCanonical);
104  }
105  catch (...)
106  {
107  Log (lsINFO) << "exception validating validation";
108  return false;
109  }
110 }
