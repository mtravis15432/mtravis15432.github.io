129 bool OutputStream::writeCompressedInt (int value)
130 {
131  unsigned int un = (value < 0) ? (unsigned int) -value
132  : (unsigned int) value;
133 
134  std::uint8_t data[5];
135  int num = 0;
136 
137  while (un > 0)
138  {
139  data[++num] = (std::uint8_t) un;
140  un >>= 8;
141  }
142 
143  data[0] = (std::uint8_t) num;
144 
145  if (value < 0)
146  data[0] |= 0x80;
147 
148  return write (data, (size_t) num + 1);
149 }

191 bool OutputStream::writeString (const String& text)
192 {
193  // (This avoids using toUTF8() to prevent the memory bloat that it would leave behind
194  // if lots of large, persistent strings were to be written to streams).
195  const size_t numBytes = text.getNumBytesAsUTF8() + 1;
196  HeapBlock<char> temp (numBytes);
197  text.copyToUTF8 (temp, numBytes);
198  return write (temp, numBytes);

201 bool OutputStream::writeText (const String& text, const bool asUTF16,
202  const bool writeUTF16ByteOrderMark)
203 {
204  if (asUTF16)
205  {
206  if (writeUTF16ByteOrderMark)
207  write ("\x0ff\x0fe", 2);
208 
209  String::CharPointerType src (text.getCharPointer());
210  bool lastCharWasReturn = false;
211 
212  for (;;)
213  {
214  const beast_wchar c = src.getAndAdvance();
215 
216  if (c == 0)
217  break;
218 
219  if (c == '\n' && ! lastCharWasReturn)
220  writeShort ((short) '\r');
221 
222  lastCharWasReturn = (c == L'\r');
223 
224  if (! writeShort ((short) c))
225  return false;
226  }
227  }
228  else
229  {
230  const char* src = text.toUTF8();
231  const char* t = src;
232 
233  for (;;)
234  {
235  if (*t == '\n')
236  {
237  if (t > src)
238  if (! write (src, (size_t) (t - src)))
239  return false;
240 
241  if (! write ("\r\n", 2))
242  return false;
243 
244  src = t + 1;
245  }
246  else if (*t == '\r')
247  {
248  if (t[1] == '\n')
249  ++t;
250  }
251  else if (*t == 0)
252  {
253  if (t > src)
254  if (! write (src, (size_t) (t - src)))
255  return false;
256 
257  break;
258  }
259 
260  ++t;
261  }
262  }
263 
264  return true;
265 }

267 int OutputStream::writeFromInputStream (InputStream& source, std::int64_t numBytesToWrite)
268 {
269  if (numBytesToWrite < 0)
270  numBytesToWrite = std::numeric_limits<std::int64_t>::max();
271 
272  int numWritten = 0;
273 
274  while (numBytesToWrite > 0)
275  {
276  char buffer [8192];
277  const int num = source.read (buffer, (int) bmin (numBytesToWrite, (std::int64_t) sizeof (buffer)));
278 
279  if (num <= 0)
280  break;
281 
282  write (buffer, (size_t) num);
283 
284  numBytesToWrite -= num;
285  numWritten += num;
286  }
287 
288  return numWritten;
289 }

421 OutputStream& operator<< (OutputStream& stream, const String& text)
422 {
423  const size_t numBytes = text.getNumBytesAsUTF8();
424
