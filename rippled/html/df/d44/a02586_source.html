78 // Encode a suitable internal key target for "target" and return it.
79 // Uses *scratch as scratch space, and the returned pointer will point
80 // into this scratch space.
81 static const char* EncodeKey(std::string* scratch, const Slice& target) {
82  scratch->clear();
83  PutVarint32(scratch, target.size());
84  scratch->append(target.data(), target.size());
85  return scratch->data();
86 }

134 void MemTable::Add(SequenceNumber s, ValueType type,
135  const Slice& key,
136  const Slice& value) {
137  // Format of an entry is concatenation of:
138  // key_size : varint32 of internal_key.size()
139  // key bytes : char[internal_key.size()]
140  // value_size : varint32 of value.size()
141  // value bytes : char[value.size()]
142  size_t key_size = key.size();
143  size_t val_size = value.size();
144  size_t internal_key_size = key_size + 8;
145  const size_t encoded_len =
146  VarintLength(internal_key_size) + internal_key_size +
147  VarintLength(val_size) + val_size;
148  char* buf = arena_impl_.Allocate(encoded_len);
149  char* p = EncodeVarint32(buf, internal_key_size);
150  memcpy(p, key.data(), key_size);
151  p += key_size;
152  EncodeFixed64(p, (s << 8) | type);
153  p += 8;
154  p = EncodeVarint32(p, val_size);
155  memcpy(p, value.data(), val_size);
156  assert((p + val_size) - buf == (unsigned)encoded_len);
157  table_->Insert(buf);
158 
159  // The first sequence number inserted into the memtable
160  assert(first_seqno_ == 0 || s > first_seqno_);
161  if (first_seqno_ == 0) {
162  first_seqno_ = s;
163  }
164 }

166 bool MemTable::Get(const LookupKey& key, std::string* value, Status* s,
167  std::deque<std::string>* operands, const Options& options) {
168  Slice memkey = key.memtable_key();
169  std::shared_ptr<MemTableRep::Iterator> iter(
170  table_->GetIterator(key.user_key()));
171  iter->Seek(memkey.data());
172 
173  // It is the caller's responsibility to allocate/delete operands list
174  assert(operands != nullptr);
175 
176  bool merge_in_progress = s->IsMergeInProgress();
177  auto merge_operator = options.merge_operator.get();
178  auto logger = options.info_log;
179  std::string merge_result;
180 
181  for (; iter->Valid(); iter->Next()) {
182  // entry format is:
183  // klength varint32
184  // userkey char[klength-8]
185  // tag uint64
186  // vlength varint32
187  // value char[vlength]
188  // Check that it belongs to same user key. We do not check the
189  // sequence number since the Seek() call above should have skipped
190  // all entries with overly large sequence numbers.
191  const char* entry = iter->key();
192  uint32_t key_length;
193  const char* key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
194  if (comparator_.comparator.user_comparator()->Compare(
195  Slice(key_ptr, key_length - 8), key.user_key()) == 0) {
196  // Correct user key
197  const uint64_t tag = DecodeFixed64(key_ptr + key_length - 8);
198  switch (static_cast<ValueType>(tag & 0xff)) {
199  case kTypeValue: {
200  if (options.inplace_update_support) {
201  GetLock(key.user_key())->ReadLock();
202  }
203  Slice v = GetLengthPrefixedSlice(key_ptr + key_length);
204  *s = Status::OK();
205  if (merge_in_progress) {
206  assert(merge_operator);
207  if (!merge_operator->FullMerge(key.user_key(), &v, *operands,
208  value, logger.get())) {
209  RecordTick(options.statistics, NUMBER_MERGE_FAILURES);
210  *s = Status::Corruption("Error: Could not perform merge.");
211  }
212  } else {
213  value->assign(v.data(), v.size());
214  }
215  if (options.inplace_update_support) {
216  GetLock(key.user_key())->Unlock();
217  }
218  return true;
219  }
220  case kTypeDeletion: {
221  if (merge_in_progress) {
222  assert(merge_operator);
223  *s = Status::OK();
224  if (!merge_operator->FullMerge(key.user_key(), nullptr, *operands,
225  value, logger.get())) {
226  RecordTick(options.statistics, NUMBER_MERGE_FAILURES);
227  *s = Status::Corruption("Error: Could not perform merge.");
228  }
229  } else {
230  *s = Status::NotFound(Slice());
231  }
232  return true;
233  }
234  case kTypeMerge: {
235  Slice v = GetLengthPrefixedSlice(key_ptr + key_length);
236  merge_in_progress = true;
237  operands->push_front(v.ToString());
238  while(operands->size() >= 2) {
239  // Attempt to associative merge. (Returns true if successful)
240  if (merge_operator->PartialMerge(key.user_key(),
241  Slice((*operands)[0]),
242  Slice((*operands)[1]),
243  &merge_result,
244  logger.get())) {
245  operands->pop_front();
246  swap(operands->front(), merge_result);
247  } else {
248  // Stack them because user can't associative merge
249  break;
250  }
251  }
252  break;
253  }
254  case kTypeLogData:
255  assert(false);
256  break;
257  }
258  } else {
259  // exit loop if user key does not match
260  break;
261  }
262  }
263 
264  // No change to value, since we have not yet found a Put/Delete
265 
266  if (merge_in_progress) {
267  *s = Status::MergeInProgress("");
268  }
269  return false;
270 }

272 bool MemTable::Update(SequenceNumber seq, ValueType type,
273  const Slice& key,
274  const Slice& value) {
275  LookupKey lkey(key, seq);
276  Slice memkey = lkey.memtable_key();
277 
278  std::shared_ptr<MemTableRep::Iterator> iter(
279  table_.get()->GetIterator(lkey.user_key()));
280  iter->Seek(memkey.data());
281 
282  if (iter->Valid()) {
283  // entry format is:
284  // klength varint32
285  // userkey char[klength-8]
286  // tag uint64
287  // vlength varint32
288  // value char[vlength]
289  // Check that it belongs to same user key. We do not check the
290  // sequence number since the Seek() call above should have skipped
291  // all entries with overly large sequence numbers.
292  const char* entry = iter->key();
293  uint32_t key_length;
294  const char* key_ptr = GetVarint32Ptr(entry, entry + 5, &key_length);
295  if (comparator_.comparator.user_comparator()->Compare(
296  Slice(key_ptr, key_length - 8), lkey.user_key()) == 0) {
297  // Correct user key
298  const uint64_t tag = DecodeFixed64(key_ptr + key_length - 8);
299  switch (static_cast<ValueType>(tag & 0xff)) {
300  case kTypeValue: {
301  uint32_t vlength;
302  GetVarint32Ptr(key_ptr + key_length,
303  key_ptr + key_length+5, &vlength);
304  // Update value, if newValue size <= curValue size
305  if (value.size() <= vlength) {
306  char* p = EncodeVarint32(const_cast<char*>(key_ptr) + key_length,
307  value.size());
308  WriteLock wl(GetLock(lkey.user_key()));
309  memcpy(p, value.data(), value.size());
310  assert(
311  (p + value.size()) - entry ==
312  (unsigned) (VarintLength(key_length) +
313  key_length +
314  VarintLength(value.size()) +
315  value.size())
316  );
317  return true;
318  }
319  }
320  default:
321  // If the latest value is kTypeDeletion, kTypeMerge or kTypeLogData
322  // then we probably don't have enough space to update in-place
323  // Maybe do something later
324  // Return false, and do normal Add()
325  return false;
326  }
327  }
328  }
329 
330  // Key doesn't exist
331  return false;
332 }
