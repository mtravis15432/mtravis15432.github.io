73  void send (const Json::Value& jvObj, bool broadcast)
74  {
75  ScopedLockType sl (mLock);
76 
77  if (mDeque.size () >= eventQueueMax)
78  {
79  // Drop the previous event.
80  WriteLog (lsWARNING, RPCSub) << "RPCCall::fromNetwork drop";
81  mDeque.pop_back ();
82  }
83 
84  WriteLog (broadcast ? lsDEBUG : lsINFO, RPCSub) <<
85  "RPCCall::fromNetwork push: " << jvObj;
86 
87  mDeque.push_back (std::make_pair (mSeq++, jvObj));
88 
89  if (!mSending)
90  {
91  // Start a sending thread.
92  mSending = true;
93 
94  WriteLog (lsINFO, RPCSub) << "RPCCall::fromNetwork start";
95 
96  m_jobQueue.addJob (
97  jtCLIENT, "RPCSub::sendThread", BIND_TYPE (&RPCSubImp::sendThread, this));
98  }
99  }

115 private:
116  // XXX Could probably create a bunch of send jobs in a single get of the lock.
117  void sendThread ()
118  {
119  Json::Value jvEvent;
120  bool bSend;
121 
122  do
123  {
124  {
125  // Obtain the lock to manipulate the queue and change sending.
126  ScopedLockType sl (mLock);
127 
128  if (mDeque.empty ())
129  {
130  mSending = false;
131  bSend = false;
132  }
133  else
134  {
135  std::pair<int, Json::Value> pEvent = mDeque.front ();
136 
137  mDeque.pop_front ();
138 
139  jvEvent = pEvent.second;
140  jvEvent["seq"] = pEvent.first;
141 
142  bSend = true;
143  }
144  }
145 
146  // Send outside of the lock.
147  if (bSend)
148  {
149  // XXX Might not need this in a try.
150  try
151  {
152  WriteLog (lsINFO, RPCSub) << "RPCCall::fromNetwork: " << mIp;
153 
154  RPCCall::fromNetwork (
155  m_io_service,
156  mIp, mPort,
157  mUsername, mPassword,
158  mPath, "event",
159  jvEvent,
160  mSSL);
161  }
162  catch (const std::exception& e)
163  {
164  WriteLog (lsINFO, RPCSub) << "RPCCall::fromNetwork exception: " << e.what ();
165  }
166  }
167  }
168  while (bSend);
169  }
