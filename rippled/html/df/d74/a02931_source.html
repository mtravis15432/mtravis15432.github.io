31 bool BookDirIterator::nextDirectory (LedgerEntrySet& les)
32 {
33  WriteLog (lsTRACE, Ledger) << "BookDirectoryIterator:: nextDirectory";
34 
35  // Are we already at the end?
36  if (mIndex.isZero ())
37  return false;
38 
39  // Get the ledger index of the next directory
40  mIndex = les.getNextLedgerIndex (mIndex, mEnd);
41 
42  if (mIndex.isZero ())
43  {
44  // We ran off the end of the book
45  WriteLog (lsTRACE, Ledger) << "BookDirectoryIterator:: no next ledger index";
46  return false;
47  }
48  assert (mIndex < mEnd);
49 
50  WriteLog (lsTRACE, Ledger) << "BookDirectoryIterator:: index " << mIndex.GetHex();
51 
52  // Retrieve the SLE from the LES
53  mOfferDir = les.entryCache (ltDIR_NODE, mIndex);
54  assert (mOfferDir);
55 
56  return !!mOfferDir;
57 }

170 bool OrderBookIterator::nextOffer ()
171 {
172  WriteLog (lsTRACE, Ledger) << "OrderBookIterator: next offer";
173  do
174  {
175 
176  // Is there a next offer in the current directory
177  if (mOfferIterator.nextEntry (mEntrySet))
178  {
179  WriteLog (lsTRACE, Ledger) << "OrderBookIterator: there is a next offer in this directory";
180  return true;
181  }
182 
183  // Is there a next directory
184  if (!mDirectoryIterator.nextDirectory (mEntrySet))
185  {
186  WriteLog (lsTRACE, Ledger) << "OrderBookIterator: there is no next directory";
187  return false;
188  }
189  WriteLog (lsTRACE, Ledger) << "OrderBookIterator: going to next directory";
190 
191  // Set to before its first offer
192  mOfferIterator = mDirectoryIterator.getOfferIterator ();
193  }
194  while (1);
195 }
