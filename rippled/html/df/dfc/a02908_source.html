114 bool InboundLedger::tryLocal ()
115 {
116  // return value: true = no more work to do
117 
118  if (!mHaveBase)
119  {
120  // Nothing we can do without the ledger base
121  NodeObject::pointer node = getApp().getNodeStore ().fetch (mHash);
122 
123  if (!node)
124  {
125  Blob data;
126 
127  if (!getApp().getOPs ().getFetchPack (mHash, data))
128  return false;
129 
130  if (m_journal.trace) m_journal.trace <<
131  "Ledger base found in fetch pack";
132  mLedger = boost::make_shared<Ledger> (data, true);
133  getApp().getNodeStore ().store (hotLEDGER,
134  mLedger->getLedgerSeq (), data, mHash);
135  }
136  else
137  {
138  mLedger = boost::make_shared<Ledger> (
139  strCopy (node->getData ()), true);
140  }
141 
142  if (mLedger->getHash () != mHash)
143  {
144  // We know for a fact the ledger can never be acquired
145  if (m_journal.warning) m_journal.warning <<
146  mHash << " cannot be a ledger";
147  mFailed = true;
148  return true;
149  }
150 
151  mHaveBase = true;
152  }
153 
154  if (!mHaveTransactions)
155  {
156  if (mLedger->getTransHash ().isZero ())
157  {
158  if (m_journal.trace) m_journal.trace <<
159  "No TXNs to fetch";
160  mHaveTransactions = true;
161  }
162  else
163  {
164  TransactionStateSF filter (mLedger->getLedgerSeq ());
165 
166  if (mLedger->peekTransactionMap ()->fetchRoot (
167  mLedger->getTransHash (), &filter))
168  {
169  std::vector<uint256> h (mLedger->getNeededTransactionHashes (
170  1, &filter));
171 
172  if (h.empty ())
173  {
174  if (m_journal.trace) m_journal.trace <<
175  "Had full txn map locally";
176  mHaveTransactions = true;
177  }
178  }
179  }
180  }
181 
182  if (!mHaveState)
183  {
184  if (mLedger->getAccountHash ().isZero ())
185  {
186  if (m_journal.fatal) m_journal.fatal <<
187  "We are acquiring a ledger with a zero account hash";
188  mFailed = true;
189  return true;
190  }
191  else
192  {
193  AccountStateSF filter (mLedger->getLedgerSeq ());
194 
195  if (mLedger->peekAccountStateMap ()->fetchRoot (
196  mLedger->getAccountHash (), &filter))
197  {
198  std::vector<uint256> h (mLedger->getNeededAccountStateHashes (
199  1, &filter));
200 
201  if (h.empty ())
202  {
203  if (m_journal.trace) m_journal.trace <<
204  "Had full AS map locally";
205  mHaveState = true;
206  }
207  }
208  }
209  }
210 
211  if (mHaveTransactions && mHaveState)
212  {
213  if (m_journal.debug) m_journal.debug <<
214  "Had everything locally";
215  mComplete = true;
216  mLedger->setClosed ();
217  mLedger->setImmutable ();
218  }
219 
220  return mComplete;
221 }

225 void InboundLedger::onTimer (bool wasProgress, ScopedLockType&)
226 {
227  mRecentTXNodes.clear ();
228  mRecentASNodes.clear ();
229 
230  if (isDone())
231  {
232  if (m_journal.info) m_journal.info <<
233  "Already done " << mHash;
234  return;
235  }
236 
237  if (getTimeouts () > ledgerTimeoutRetriesMax)
238  {
239  if (mSeq != 0)
240  {
241  if (m_journal.warning) m_journal.warning <<
242  getTimeouts() << " timeouts for ledger " << mSeq;
243  }
244  else
245  {
246  if (m_journal.warning) m_journal.warning <<
247  getTimeouts() << " timeouts for ledger " << mHash;
248  }
249  setFailed ();
250  done ();
251  return;
252  }
253 
254  if (!wasProgress)
255  {
256  checkLocal();
257 
258  mAggressive = true;
259  mByHash = true;
260 
261  std::size_t pc = getPeerCount ();
262  WriteLog (lsDEBUG, InboundLedger) <<
263  "No progress(" << pc <<
264  ") for ledger " << mHash;
265 
266  trigger (Peer::ptr ());
267  if (pc < 4)

273 void InboundLedger::addPeers ()
274 {
275  Overlay::PeerSequence peerList = getApp().overlay ().getActivePeers ();
276 
277  int vSize = peerList.size ();
278 
279  if (vSize == 0)
280  {
281  WriteLog (lsERROR, InboundLedger) <<
282  "No peers to add for ledger acquisition";
283  return;
284  }
285 
286  // FIXME-NIKB why are we doing this convoluted thing here instead of simply
287  // shuffling this vector and then pulling however many entries we need?
288 
289  // We traverse the peer list in random order so as not to favor
290  // any particular peer
291  //
292  // VFALCO Use random_shuffle
293  // http://en.cppreference.com/w/cpp/algorithm/random_shuffle
294  //
295  int firstPeer = rand () % vSize;
296 
297  int found = 0;
298 
299  // First look for peers that are likely to have this ledger
300  for (int i = 0; i < vSize; ++i)
301  {
302  Peer::ptr const& peer = peerList[ (i + firstPeer) % vSize];
303 
304  if (peer->hasLedger (getHash (), mSeq))
305  {
306  if (peerHas (peer) && (++found > 6))
307  break;
308  }
309  }
310 
311  if (!found)
312  { // Oh well, try some random peers
313  for (int i = 0; (i < 6) && (i < vSize); ++i)
314  {
315  if (peerHas (peerList[ (i + firstPeer) % vSize]))
316  ++found;
317  }
318  if (mSeq != 0)
319  {
320  if (m_journal.debug) m_journal.debug <<
321  "Chose " << found << " peer(s) for ledger " << mSeq;
322  }
323  else
324  {
325  if (m_journal.debug) m_journal.debug <<
326  "Chose " << found << " peer(s) for ledger " <<
327  getHash ().GetHex();
328  }
329  }
330  else if (mSeq != 0)
331  {
332  if (m_journal.debug) m_journal.debug <<
333  "Found " << found << " peer(s) with ledger " << mSeq;
334  }
335  else
336  {
337  if (m_journal.debug) m_journal.debug <<
338  "Found " << found << " peer(s) with ledger " <<
339  getHash ().GetHex();
340  }
341 }

364 void InboundLedger::done ()
365 {
366  if (mSignaled)
367  return;
368 
369  mSignaled = true;
370  touch ();
371 
372  if (m_journal.trace) m_journal.trace <<
373  "Done acquiring ledger " << mHash;
374 
375  assert (isComplete () || isFailed ());
376 
377  std::vector< std::function<void (InboundLedger::pointer)> > triggers;
378  {
379  ScopedLockType sl (mLock);
380  triggers.swap (mOnComplete);
381  }
382 
383  if (isComplete () && !isFailed () && mLedger)
384  {
385  mLedger->setClosed ();
386  mLedger->setImmutable ();
387  getApp().getLedgerMaster ().storeLedger (mLedger);
388  }
389  else
390  getApp().getInboundLedgers ().logFailure (mHash);
391 
392  // We hold the PeerSet lock, so must dispatch
393  getApp().getJobQueue ().addJob (jtLEDGER_DATA, "triggers",
394  BIND_TYPE (LADispatch, P_1, shared_from_this (), triggers));
395 }

411 void InboundLedger::trigger (Peer::ptr const& peer)
412 {
413  ScopedLockType sl (mLock);
414 
415  if (isDone ())
416  {
417  if (m_journal.debug) m_journal.debug <<
418  "Trigger on ledger: " << mHash << (mAborted ? " aborted" : "") <<
419  (mComplete ? " completed" : "") << (mFailed ? " failed" : "");
420  return;
421  }
422 
423  if (m_journal.trace)
424  {
425  if (peer)
426  m_journal.trace <<
427  "Trigger acquiring ledger " << mHash << " from " << peer;
428  else
429  m_journal.trace <<
430  "Trigger acquiring ledger " << mHash;
431 
432  if (mComplete || mFailed)
433  m_journal.trace <<
434  "complete=" << mComplete << " failed=" << mFailed;
435  else
436  m_journal.trace <<
437  "base=" << mHaveBase << " tx=" << mHaveTransactions <<
438  " as=" << mHaveState;
439  }
440 
441  if (!mHaveBase)
442  {
443  tryLocal ();
444 
445  if (mFailed)
446  {
447  if (m_journal.warning) m_journal.warning <<
448  " failed local for " << mHash;
449  return;
450  }
451  }
452 
453  protocol::TMGetLedger tmGL;
454  tmGL.set_ledgerhash (mHash.begin (), mHash.size ());
455 
456  if (getTimeouts () != 0)
457  { // Be more aggressive if we've timed out at least once
458  tmGL.set_querytype (protocol::qtINDIRECT);
459 
460  if (!isProgress () && !mFailed && mByHash && (
461  getTimeouts () > ledgerBecomeAggressiveThreshold))
462  {
463  std::vector<neededHash_t> need = getNeededHashes ();
464 
465  if (!need.empty ())
466  {
467  protocol::TMGetObjectByHash tmBH;
468  tmBH.set_query (true);
469  tmBH.set_ledgerhash (mHash.begin (), mHash.size ());
470  bool typeSet = false;
471  BOOST_FOREACH (neededHash_t & p, need)
472  {
473  if (m_journal.warning) m_journal.warning
474  << "Want: " << p.second;
475 
476  if (!typeSet)
477  {
478  tmBH.set_type (p.first);
479  typeSet = true;
480  }
481 
482  if (p.first == tmBH.type ())
483  {
484  protocol::TMIndexedObject* io = tmBH.add_objects ();
485  io->set_hash (p.second.begin (), p.second.size ());
486  }
487  }
488 
489  Message::pointer packet (boost::make_shared <Message> (
490  tmBH, protocol::mtGET_OBJECTS));
491  {
492  ScopedLockType sl (mLock);
493 
494  for (PeerSetMap::iterator it = mPeers.begin (), end = mPeers.end ();
495  it != end; ++it)
496  {
497  Peer::ptr iPeer (
498  getApp().overlay ().findPeerByShortID (it->first));
499 
500  if (iPeer)
501  {
502  mByHash = false;
503  iPeer->sendPacket (packet, false);
504  }
505  }
506  }
507  if (m_journal.info) m_journal.info <<
508  "Attempting by hash fetch for ledger " << mHash;
509  }
510  else
511  {
512  if (m_journal.info) m_journal.info <<
513  "getNeededHashes says acquire is complete";
514  mHaveBase = true;
515  mHaveTransactions = true;
516  mHaveState = true;
517  mComplete = true;
518  }
519  }
520  }
521 
522  // We can't do much without the base data because we don't know the
523  // state or transaction root hashes.
524  if (!mHaveBase && !mFailed)
525  {
526  tmGL.set_itype (protocol::liBASE);
527  if (m_journal.trace) m_journal.trace <<
528  "Sending base request to " << (peer ? "selected peer" : "all peers");
529  sendRequest (tmGL, peer);
530  return;
531  }
532 
533  if (mLedger)
534  tmGL.set_ledgerseq (mLedger->getLedgerSeq ());
535 
536  // Get the state data first because it's the most likely to be useful
537  // if we wind up abandoning this fetch.
538  if (mHaveBase && !mHaveState && !mFailed)
539  {
540  assert (mLedger);
541 
542  if (mLedger->peekAccountStateMap ()->getHash ().isZero ())
543  {
544  // we need the root node
545  tmGL.set_itype (protocol::liAS_NODE);
546  * (tmGL.add_nodeids ()) = SHAMapNode ().getRawString ();
547  if (m_journal.trace) m_journal.trace <<
548  "Sending AS root request to " << (peer ? "selected peer" : "all peers");
549  sendRequest (tmGL, peer);
550  return;
551  }
552  else
553  {
554  std::vector<SHAMapNode> nodeIDs;
555  std::vector<uint256> nodeHashes;
556  // VFALCO Why 256? Make this a constant
557  nodeIDs.reserve (256);
558  nodeHashes.reserve (256);
559  AccountStateSF filter (mSeq);
560 
561  // Release the lock while we process the large state map
562  sl.unlock();
563  mLedger->peekAccountStateMap ()->getMissingNodes (
564  nodeIDs, nodeHashes, 256, &filter);
565  sl.lock();
566 
567  // Make sure nothing happened while we released the lock
568  if (!mFailed && !mComplete && !mHaveState)
569  {
570  if (nodeIDs.empty ())
571  {
572  if (!mLedger->peekAccountStateMap ()->isValid ())
573  mFailed = true;
574  else
575  {
576  mHaveState = true;
577 
578  if (mHaveTransactions)
579  mComplete = true;
580  }
581  }
582  else
583  {
584  // VFALCO Why 128? Make this a constant
585  if (!mAggressive)
586  filterNodes (nodeIDs, nodeHashes, mRecentASNodes,
587  128, !isProgress ());
588 
589  if (!nodeIDs.empty ())
590  {
591  tmGL.set_itype (protocol::liAS_NODE);
592  BOOST_FOREACH (SHAMapNode const& it, nodeIDs)
593  {
594  * (tmGL.add_nodeids ()) = it.getRawString ();
595  }
596  if (m_journal.trace) m_journal.trace <<
597  "Sending AS node " << nodeIDs.size () <<
598  " request to " << (
599  peer ? "selected peer" : "all peers");
600  if (nodeIDs.size () == 1 && m_journal.trace) m_journal.trace <<
601  "AS node: " << nodeIDs[0];
602  sendRequest (tmGL, peer);
603  return;
604  }
605  else
606  if (m_journal.trace) m_journal.trace <<
607  "All AS nodes filtered";
608  }
609  }
610  }
611  }
612 
613  if (mHaveBase && !mHaveTransactions && !mFailed)
614  {
615  assert (mLedger);
616 
617  if (mLedger->peekTransactionMap ()->getHash ().isZero ())
618  {
619  // we need the root node
620  tmGL.set_itype (protocol::liTX_NODE);
621  * (tmGL.add_nodeids ()) = SHAMapNode ().getRawString ();
622  if (m_journal.trace) m_journal.trace <<
623  "Sending TX root request to " << (
624  peer ? "selected peer" : "all peers");
625  sendRequest (tmGL, peer);
626  return;
627  }
628  else
629  {
630  std::vector<SHAMapNode> nodeIDs;
631  std::vector<uint256> nodeHashes;
632  nodeIDs.reserve (256);
633  nodeHashes.reserve (256);
634  TransactionStateSF filter (mSeq);
635  mLedger->peekTransactionMap ()->getMissingNodes (
636  nodeIDs, nodeHashes, 256, &filter);
637 
638  if (nodeIDs.empty ())
639  {
640  if (!mLedger->peekTransactionMap ()->isValid ())
641  mFailed = true;
642  else
643  {
644  mHaveTransactions = true;
645 
646  if (mHaveState)
647  mComplete = true;
648  }
649  }
650  else
651  {
652  if (!mAggressive)
653  filterNodes (nodeIDs, nodeHashes, mRecentTXNodes,
654  128, !isProgress ());
655 
656  if (!nodeIDs.empty ())
657  {
658  tmGL.set_itype (protocol::liTX_NODE);
659  BOOST_FOREACH (SHAMapNode const& it, nodeIDs)
660  {
661  * (tmGL.add_nodeids ()) = it.getRawString ();
662  }
663  if (m_journal.trace) m_journal.trace <<
664  "Sending TX node " << nodeIDs.size () <<
665  " request to " << (
666  peer ? "selected peer" : "all peers");
667  sendRequest (tmGL, peer);
668  return;
669  }
670  else
671  if (m_journal.trace) m_journal.trace <<
672  "All TX nodes filtered";
673  }
674  }
675  }
676 
677  if (mComplete || mFailed)
678  {
679  if (m_journal.debug) m_journal.debug <<
680  "Done:" << (mComplete ? " complete" : "") <<
681  (mFailed ? " failed " : " ") <<
682  mLedger->getLedgerSeq ();
683  sl.unlock ();
684  done ();
685  }
686 }

688 void InboundLedger::filterNodes (std::vector<SHAMapNode>& nodeIDs,
689  std::vector<uint256>& nodeHashes, std::set<SHAMapNode>& recentNodes,
690  int max, bool aggressive)
691 {
692  // ask for new nodes in preference to ones we've already asked for
693  assert (nodeIDs.size () == nodeHashes.size ());
694 
695  std::vector<bool> duplicates;
696  duplicates.reserve (nodeIDs.size ());
697 
698  int dupCount = 0;
699 
700  BOOST_FOREACH(SHAMapNode const& nodeID, nodeIDs)
701  {
702  if (recentNodes.count (nodeID) != 0)
703  {
704  duplicates.push_back (true);
705  ++dupCount;
706  }
707  else
708  duplicates.push_back (false);
709  }
710 
711  if (dupCount == nodeIDs.size ())
712  {
713  // all duplicates
714  if (!aggressive)
715  {
716  nodeIDs.clear ();
717  nodeHashes.clear ();
718  if (m_journal.trace) m_journal.trace <<
719  "filterNodes: all are duplicates";
720  return;
721  }
722  }
723  else if (dupCount > 0)
724  {
725  // some, but not all, duplicates
726  int insertPoint = 0;
727 
728  for (unsigned int i = 0; i < nodeIDs.size (); ++i)
729  if (!duplicates[i])
730  {
731  // Keep this node
732  if (insertPoint != i)
733  {
734  nodeIDs[insertPoint] = nodeIDs[i];
735  nodeHashes[insertPoint] = nodeHashes[i];
736  }
737 
738  ++insertPoint;
739  }
740 
741  if (m_journal.trace) m_journal.trace <<
742  "filterNodes " << nodeIDs.size () << " to " << insertPoint;
743  nodeIDs.resize (insertPoint);
744  nodeHashes.resize (insertPoint);
745  }
746 
747  if (nodeIDs.size () > max)
748  {
749  nodeIDs.resize (max);
750  nodeHashes.resize (max);
751  }
752 
753  BOOST_FOREACH (const SHAMapNode & n, nodeIDs)
754  {
755  recentNodes.insert (n);
756  }
757 }

762 // data must not have hash prefix
763 bool InboundLedger::takeBase (const std::string& data)
764 {
765  // Return value: true=normal, false=bad data
766  if (m_journal.trace) m_journal.trace <<
767  "got base acquiring ledger " << mHash;
768 
769  if (mComplete || mFailed || mHaveBase)
770  return true;
771 
772  mLedger = boost::make_shared<Ledger> (data, false);
773 
774  if (mLedger->getHash () != mHash)
775  {

1056 int InboundLedger::processData (boost::shared_ptr<Peer> peer,
1057  protocol::TMLedgerData& packet)
1058 {

1060 
1061  if (packet.type () == protocol::liBASE)
1062  {
1063  if (packet.nodes_size () < 1)
1064  {
1065  if (m_journal.warning) m_journal.warning <<
1066  "Got empty base data";
1067  peer->charge (Resource::feeInvalidRequest);
1068  return -1;
1069  }
1070 
1071  SHAMapAddNode san;
1072 
1073  if (!mHaveBase)
1074  {
1075  if (takeBase (packet.nodes (0).nodedata ()))
1076  san.incUseful ();
1077  else
1078  {
1079  if (m_journal.warning) m_journal.warning <<
1080  "Got invalid base data";
1081  peer->charge (Resource::feeInvalidRequest);
1082  return -1;
1083  }
1084  }
1085 
1086 
1087  if (!mHaveState && (packet.nodes ().size () > 1) &&
1088  !takeAsRootNode (strCopy (packet.nodes (1).nodedata ()), san))
1089  {

1092  }

1097  if (m_journal.warning) m_journal.warning <<

1102  progress ();
1103  else
1104  if (m_journal.debug) m_journal.debug <<
1105  "Peer sends invalid base data";
1106

1108  }
1109

1114  std::list< Blob > nodeData;

1117  {
1118  if (m_journal.info) m_journal.info <<
1119  "Got response with no nodes";
1120  peer->charge (Resource::feeInvalidRequest);
1121  return -1;
1122  }
1123 
1124  for (int i = 0; i < packet.nodes ().size (); ++i)
1125  {
1126  const protocol::TMLedgerNode& node = packet.nodes (i);
1127

1129  {

1137  node.nodeid ().size ()));

1172 void InboundLedger::runData ()
1173 {
1174  boost::shared_ptr<Peer> chosenPeer;
1175  int chosenPeerCount = -1;
1176 
1177  std::vector <PeerDataPairType> data;
1178  do
1179  {
1180  data.clear();
1181  {
1182  ScopedLockType sl (mReceivedDataLock);
1183 
1184  if (mReceivedData.empty ())
1185  {
1186  mReceiveDispatched = false;
1187  break;
1188  }
1189  data.swap(mReceivedData);
1190  }
1191 
1192  // Select the peer that gives us the most nodes that are useful,
1193  // breaking ties in favor of the peer that responded first.
1194  BOOST_FOREACH (PeerDataPairType& entry, data)
1195  {
1196  Peer::ptr peer = entry.first.lock();
1197  if (peer)
1198  {
1199  int count = processData (peer, *(entry.second));
1200  if (count > chosenPeerCount)
1201  {
1202  chosenPeer = peer;
1203  chosenPeerCount = count;
1204  }
1205  }
1206  }
1207 
1208  } while (1);
1209 
1210  if (chosenPeer)
1211  trigger (chosenPeer);
1212 }
