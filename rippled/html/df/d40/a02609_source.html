17 // Tag numbers for serialized VersionEdit. These numbers are written to
18 // disk and should not be changed.
19 enum Tag {
20  kComparator = 1,
21  kLogNumber = 2,
22  kNextFileNumber = 3,
23  kLastSequence = 4,
24  kCompactPointer = 5,
25  kDeletedFile = 6,
26  kNewFile = 7,
27  // 8 was used for large value refs
28  kPrevLogNumber = 9,
29 
30  // these are new formats divergent from open source leveldb
31  kNewFile2 = 100 // store smallest & largest seqno
32 };

49 void VersionEdit::EncodeTo(std::string* dst) const {
50  if (has_comparator_) {
51  PutVarint32(dst, kComparator);
52  PutLengthPrefixedSlice(dst, comparator_);
53  }
54  if (has_log_number_) {
55  PutVarint32(dst, kLogNumber);
56  PutVarint64(dst, log_number_);
57  }
58  if (has_prev_log_number_) {
59  PutVarint32(dst, kPrevLogNumber);
60  PutVarint64(dst, prev_log_number_);
61  }
62  if (has_next_file_number_) {
63  PutVarint32(dst, kNextFileNumber);
64  PutVarint64(dst, next_file_number_);
65  }
66  if (has_last_sequence_) {
67  PutVarint32(dst, kLastSequence);
68  PutVarint64(dst, last_sequence_);
69  }
70 
71  for (size_t i = 0; i < compact_pointers_.size(); i++) {
72  PutVarint32(dst, kCompactPointer);
73  PutVarint32(dst, compact_pointers_[i].first); // level
74  PutLengthPrefixedSlice(dst, compact_pointers_[i].second.Encode());
75  }
76 
77  for (DeletedFileSet::const_iterator iter = deleted_files_.begin();
78  iter != deleted_files_.end();
79  ++iter) {
80  PutVarint32(dst, kDeletedFile);
81  PutVarint32(dst, iter->first); // level
82  PutVarint64(dst, iter->second); // file number
83  }
84 
85  for (size_t i = 0; i < new_files_.size(); i++) {
86  const FileMetaData& f = new_files_[i].second;
87  PutVarint32(dst, kNewFile2);
88  PutVarint32(dst, new_files_[i].first); // level
89  PutVarint64(dst, f.number);
90  PutVarint64(dst, f.file_size);
91  PutLengthPrefixedSlice(dst, f.smallest.Encode());
92  PutLengthPrefixedSlice(dst, f.largest.Encode());
93  PutVarint64(dst, f.smallest_seqno);
94  PutVarint64(dst, f.largest_seqno);
95  }
96 }

108 bool VersionEdit::GetLevel(Slice* input, int* level, const char** msg) {
109  uint32_t v;
110  if (GetVarint32(input, &v) &&
111  (int)v < number_levels_) {
112  *level = v;
113  return true;
114  } else {
115  if ((int)v >= number_levels_) {
116  *msg = "db already has more levels than options.num_levels";
117  }
118  return false;
119  }

122 Status VersionEdit::DecodeFrom(const Slice& src) {
123  Clear();
124  Slice input = src;
125  const char* msg = nullptr;
126  uint32_t tag;
127 
128  // Temporary storage for parsing
129  int level;
130  uint64_t number;
131  FileMetaData f;
132  Slice str;
133  InternalKey key;
134 
135  while (msg == nullptr && GetVarint32(&input, &tag)) {
136  switch (tag) {
137  case kComparator:
138  if (GetLengthPrefixedSlice(&input, &str)) {
139  comparator_ = str.ToString();
140  has_comparator_ = true;
141  } else {
142  msg = "comparator name";
143  }
144  break;
145 
146  case kLogNumber:
147  if (GetVarint64(&input, &log_number_)) {
148  has_log_number_ = true;
149  } else {
150  msg = "log number";
151  }
152  break;
153 
154  case kPrevLogNumber:
155  if (GetVarint64(&input, &prev_log_number_)) {
156  has_prev_log_number_ = true;
157  } else {
158  msg = "previous log number";
159  }
160  break;
161 
162  case kNextFileNumber:
163  if (GetVarint64(&input, &next_file_number_)) {
164  has_next_file_number_ = true;
165  } else {
166  msg = "next file number";
167  }
168  break;
169 
170  case kLastSequence:
171  if (GetVarint64(&input, &last_sequence_)) {
172  has_last_sequence_ = true;
173  } else {
174  msg = "last sequence number";
175  }
176  break;
177 
178  case kCompactPointer:
179  if (GetLevel(&input, &level, &msg) &&
180  GetInternalKey(&input, &key)) {
181  compact_pointers_.push_back(std::make_pair(level, key));
182  } else {
183  if (!msg) {
184  msg = "compaction pointer";
185  }
186  }
187  break;
188 
189  case kDeletedFile:
190  if (GetLevel(&input, &level, &msg) &&
191  GetVarint64(&input, &number)) {
192  deleted_files_.insert(std::make_pair(level, number));
193  } else {
194  if (!msg) {
195  msg = "deleted file";
196  }
197  }
198  break;
199 
200  case kNewFile:
201  if (GetLevel(&input, &level, &msg) &&
202  GetVarint64(&input, &f.number) &&
203  GetVarint64(&input, &f.file_size) &&
204  GetInternalKey(&input, &f.smallest) &&
205  GetInternalKey(&input, &f.largest)) {
206  new_files_.push_back(std::make_pair(level, f));
207  } else {
208  if (!msg) {
209  msg = "new-file entry";
210  }
211  }
212  break;
213 
214  case kNewFile2:
215  if (GetLevel(&input, &level, &msg) &&
216  GetVarint64(&input, &f.number) &&
217  GetVarint64(&input, &f.file_size) &&
218  GetInternalKey(&input, &f.smallest) &&
219  GetInternalKey(&input, &f.largest) &&
220  GetVarint64(&input, &f.smallest_seqno) &&
221  GetVarint64(&input, &f.largest_seqno) ) {
222  new_files_.push_back(std::make_pair(level, f));
223  } else {
224  if (!msg) {
225  msg = "new-file2 entry";
226  }
227  }
228  break;
229 
230  default:
231  msg = "unknown tag";
232  break;
233  }
234  }
235 
236  if (msg == nullptr && !input.empty()) {
237  msg = "invalid tag";
238  }
239 
240  Status result;
241  if (msg != nullptr) {
242  result = Status::Corruption("VersionEdit", msg);
243  }
244  return result;
245 }

247 std::string VersionEdit::DebugString(bool hex_key) const {
248  std::string r;
249  r.append("VersionEdit {");
250  if (has_comparator_) {
251  r.append("\n Comparator: ");
252  r.append(comparator_);
253  }
254  if (has_log_number_) {
255  r.append("\n LogNumber: ");
256  AppendNumberTo(&r, log_number_);
257  }
258  if (has_prev_log_number_) {
259  r.append("\n PrevLogNumber: ");
260  AppendNumberTo(&r, prev_log_number_);
261  }
262  if (has_next_file_number_) {
263  r.append("\n NextFile: ");
264  AppendNumberTo(&r, next_file_number_);
265  }
266  if (has_last_sequence_) {
267  r.append("\n LastSeq: ");
268  AppendNumberTo(&r, last_sequence_);
269  }
270  for (size_t i = 0; i < compact_pointers_.size(); i++) {
271  r.append("\n CompactPointer: ");
272  AppendNumberTo(&r, compact_pointers_[i].first);
273  r.append(" ");
274  r.append(compact_pointers_[i].second.DebugString(hex_key));
275  }
276  for (DeletedFileSet::const_iterator iter = deleted_files_.begin();
277  iter != deleted_files_.end();
278  ++iter) {
279  r.append("\n DeleteFile: ");
280  AppendNumberTo(&r, iter->first);
281  r.append(" ");
282  AppendNumberTo(&r, iter->second);
283  }
284  for (size_t i = 0; i < new_files_.size(); i++) {
285  const FileMetaData& f = new_files_[i].second;
286  r.append("\n AddFile: ");
287  AppendNumberTo(&r, new_files_[i].first);
288  r.append(" ");
289  AppendNumberTo(&r, f.number);
290  r.append(" ");
291  AppendNumberTo(&r, f.file_size);
292  r.append(" ");
293  r.append(f.smallest.DebugString(hex_key));
294  r.append(" .. ");
295  r.append(f.largest.DebugString(hex_key));
296  }
297  r.append("\n}\n");
298  return r;
299 }
