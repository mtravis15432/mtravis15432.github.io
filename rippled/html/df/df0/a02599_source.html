10 // Thread safety
11 // -------------
12 //
13 // Writes require external synchronization, most likely a mutex.
14 // Reads require a guarantee that the SkipList will not be destroyed
15 // while the read is in progress. Apart from that, reads progress
16 // without any internal locking or synchronization.
17 //
18 // Invariants:
19 //
20 // (1) Allocated nodes are never deleted until the SkipList is
21 // destroyed. This is trivially guaranteed by the code since we
22 // never delete any skip list nodes.
23 //
24 // (2) The contents of a Node except for the next/prev pointers are
25 // immutable after the Node has been linked into the SkipList.
26 // Only Insert() modifies the list, and it is careful to initialize
27 // a node and use release-stores to publish the nodes in one or
28 // more lists.
29 //
30 // ... prev vs. next pointer ordering ...
31 //

156 template<typename Key, class Comparator>
157 struct SkipList<Key,Comparator>::Node {
158  explicit Node(const Key& k) : key(k) { }
159 
160  Key const key;
161 
162  // Accessors/mutators for links. Wrapped in methods so we can
163  // add the appropriate barriers as necessary.
164  Node* Next(int n) {
165  assert(n >= 0);
166  // Use an 'acquire load' so that we observe a fully initialized
167  // version of the returned Node.
168  return reinterpret_cast<Node*>(next_[n].Acquire_Load());
169  }
170  void SetNext(int n, Node* x) {
171  assert(n >= 0);
172  // Use a 'release store' so that anybody who reads through this
173  // pointer observes a fully initialized version of the inserted node.
174  next_[n].Release_Store(x);
175  }
176 
177  // No-barrier variants that can be safely used in a few locations.
178  Node* NoBarrier_Next(int n) {
179  assert(n >= 0);
180  return reinterpret_cast<Node*>(next_[n].NoBarrier_Load());
181  }
182  void NoBarrier_SetNext(int n, Node* x) {
183  assert(n >= 0);
184  next_[n].NoBarrier_Store(x);
185  }
186 
187  private:
188  // Array of length equal to the node height. next_[0] is lowest level link.
189  port::AtomicPointer next_[1];
190 };

257 template<typename Key, class Comparator>
258 int SkipList<Key,Comparator>::RandomHeight() {
259  // Increase height with probability 1 in kBranching
260  static const unsigned int kBranching = 4;
261  int height = 1;
262  while (height < kMaxHeight && ((rnd_.Next() % kBranching) == 0)) {
263  height++;
264  }
265  assert(height > 0);
266  assert(height <= kMaxHeight);
267  return height;
268 }

276 template<typename Key, class Comparator>
277 typename SkipList<Key,Comparator>::Node* SkipList<Key,Comparator>::FindGreaterOrEqual(const Key& key, Node** prev)
278  const {
279  // Use prev as an optimization hint and fallback to slow path
280  if (prev && !KeyIsAfterNode(key, prev[0]->Next(0))) {
281  Node* x = prev[0];
282  Node* next = x->Next(0);
283  if ((x == head_) || KeyIsAfterNode(key, x)) {
284  // Adjust all relevant insertion points to the previous entry
285  for (int i = 1; i < prev_height_; i++) {
286  prev[i] = x;
287  }
288  return next;
289  }
290  }
291  // Normal lookup
292  Node* x = head_;
293  int level = GetMaxHeight() - 1;
294  while (true) {
295  Node* next = x->Next(level);
296  // Make sure the lists are sorted.
297  // If x points to head_ or next points nullptr, it is trivially satisfied.
298  assert((x == head_) || (next == nullptr) || KeyIsAfterNode(next->key, x));
299  if (KeyIsAfterNode(key, next)) {
300  // Keep searching in this list
301  x = next;
302  } else {
303  if (prev != nullptr) prev[level] = x;
304  if (level == 0) {
305  return next;
306  } else {
307  // Switch to next list
308  level--;
309  }
310  }
311  }
312 }

314 template<typename Key, class Comparator>
315 typename SkipList<Key,Comparator>::Node*
316 SkipList<Key,Comparator>::FindLessThan(const Key& key) const {
317  Node* x = head_;
318  int level = GetMaxHeight() - 1;
319  while (true) {
320  assert(x == head_ || compare_(x->key, key) < 0);
321  Node* next = x->Next(level);
322  if (next == nullptr || compare_(next->key, key) >= 0) {
323  if (level == 0) {
324  return x;
325  } else {
326  // Switch to next list
327  level--;
328  }
329  } else {
330  x = next;
331  }
332  }
333 }

369 template<typename Key, class Comparator>
370 void SkipList<Key,Comparator>::Insert(const Key& key) {
371  // TODO(opt): We can use a barrier-free variant of FindGreaterOrEqual()
372  // here since Insert() is externally synchronized.
373  Node* x = FindGreaterOrEqual(key, prev_);
374 
375  // Our data structure does not allow duplicate insertion
376  assert(x == nullptr || !Equal(key, x->key));
377 
378  int height = RandomHeight();
379  if (height > GetMaxHeight()) {
380  for (int i = GetMaxHeight(); i < height; i++) {
381  prev_[i] = head_;
382  }
383  //fprintf(stderr, "Change height from %d to %d\n", max_height_, height);
384 
385  // It is ok to mutate max_height_ without any synchronization
386  // with concurrent readers. A concurrent reader that observes
387  // the new value of max_height_ will see either the old value of
388  // new level pointers from head_ (nullptr), or a new value set in
389  // the loop below. In the former case the reader will
390  // immediately drop to the next level since nullptr sorts after all
391  // keys. In the latter case the reader will use the new node.
392  max_height_.NoBarrier_Store(reinterpret_cast<void*>(height));
393  }
394 
395  x = NewNode(key, height);
396  for (int i = 0; i < height; i++) {
397  // NoBarrier_SetNext() suffices since we will add a barrier when
398  // we publish a pointer to "x" in prev[i].
399  x->NoBarrier_SetNext(i, prev_[i]->NoBarrier_Next(i));
400  prev_[i]->SetNext(i, x);
401  }
402  prev_[0] = x;
403  prev_height_ = height;
404 }
