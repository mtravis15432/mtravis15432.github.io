37 Status TransactionLogIteratorImpl::OpenLogFile(
38  const LogFile* logFile,
39  unique_ptr<SequentialFile>* file) {
40  Env* env = options_->env;
41  if (logFile->Type() == kArchivedLogFile) {
42  std::string fname = ArchivedLogFileName(dir_, logFile->LogNumber());
43  return env->NewSequentialFile(fname, file, soptions_);
44  } else {
45  std::string fname = LogFileName(dir_, logFile->LogNumber());
46  Status status = env->NewSequentialFile(fname, file, soptions_);
47  if (!status.ok()) {
48  // If cannot open file in DB directory.
49  // Try the archive dir, as it could have moved in the meanwhile.
50  fname = ArchivedLogFileName(dir_, logFile->LogNumber());
51  status = env->NewSequentialFile(fname, file, soptions_);
52  if (!status.ok()) {
53  return Status::IOError("Requested file not present in the dir");
54  }
55  }
56  return status;
57  }

86 void TransactionLogIteratorImpl::SeekToStartSequence(
87  uint64_t startFileIndex,
88  bool strict) {
89  std::string scratch;
90  Slice record;
91  started_ = false;
92  isValid_ = false;
93  if (files_->size() <= startFileIndex) {
94  return;
95  }
96  Status s = OpenLogReader(files_->at(startFileIndex).get());
97  if (!s.ok()) {
98  currentStatus_ = s;
99  return;
100  }
101  while (RestrictedRead(&record, &scratch)) {
102  if (record.size() < 12) {
103  reporter_.Corruption(
104  record.size(), Status::Corruption("very small log record"));
105  continue;
106  }
107  UpdateCurrentWriteBatch(record);
108  if (currentLastSeq_ >= startingSequenceNumber_) {
109  if (strict && currentBatchSeq_ != startingSequenceNumber_) {
110  currentStatus_ = Status::Corruption("Gap in sequence number. Could not "
111  "seek to required sequence number");
112  reporter_.Info(currentStatus_.ToString().c_str());
113  return;
114  } else if (strict) {
115  reporter_.Info("Could seek required sequence number. Iterator will "
116  "continue.");
117  }
118  isValid_ = true;
119  started_ = true; // set started_ as we could seek till starting sequence
120  return;
121  } else {
122  isValid_ = false;
123  }
124  }
125 
126  // Could not find start sequence in first file. Normally this must be the
127  // only file. Otherwise log the error and let the iterator return next entry
128  // If strict is set, we want to seek exactly till the start sequence and it
129  // should have been present in the file we scanned above
130  if (strict) {
131  currentStatus_ = Status::Corruption("Gap in sequence number. Could not "
132  "seek to required sequence number");
133  reporter_.Info(currentStatus_.ToString().c_str());
134  } else if (files_->size() != 1) {
135  currentStatus_ = Status::Corruption("Start sequence was not found, "
136  "skipping to the next available");
137  reporter_.Info(currentStatus_.ToString().c_str());
138  // Let NextImpl find the next available entry. started_ remains false
139  // because we don't want to check for gaps while moving to start sequence
140  NextImpl(true);
141  }
142 }

148 void TransactionLogIteratorImpl::NextImpl(bool internal) {
149  std::string scratch;
150  Slice record;
151  isValid_ = false;
152  if (!internal && !started_) {
153  // Runs every time until we can seek to the start sequence
154  return SeekToStartSequence();
155  }
156  while(true) {
157  assert(currentLogReader_);
158  if (currentLogReader_->IsEOF()) {
159  currentLogReader_->UnmarkEOF();
160  }
161  while (RestrictedRead(&record, &scratch)) {
162  if (record.size() < 12) {
163  reporter_.Corruption(
164  record.size(), Status::Corruption("very small log record"));
165  continue;
166  } else {
167  // started_ should be true if called by application
168  assert(internal || started_);
169  // started_ should be false if called internally
170  assert(!internal || !started_);
171  UpdateCurrentWriteBatch(record);
172  if (internal && !started_) {
173  started_ = true;
174  }
175  return;
176  }
177  }
178 
179  // Open the next file
180  if (currentFileIndex_ < files_->size() - 1) {
181  ++currentFileIndex_;
182  Status status =OpenLogReader(files_->at(currentFileIndex_).get());
183  if (!status.ok()) {
184  isValid_ = false;
185  currentStatus_ = status;
186  return;
187  }
188  } else {
189  isValid_ = false;
190  if (currentLastSeq_ == dbimpl_->GetLatestSequenceNumber()) {
191  currentStatus_ = Status::OK();
192  } else {
193  currentStatus_ = Status::IOError("NO MORE DATA LEFT");
194  }
195  return;
196  }
197  }
198 }

200 bool TransactionLogIteratorImpl::IsBatchExpected(
201  const WriteBatch* batch,
202  const SequenceNumber expectedSeq) {
203  assert(batch);
204  SequenceNumber batchSeq = WriteBatchInternal::Sequence(batch);
205  if (batchSeq != expectedSeq) {
206  char buf[200];
207  snprintf(buf, sizeof(buf),
208  "Discontinuity in log records. Got seq=%lu, Expected seq=%lu, "
209  "Last flushed seq=%lu.Log iterator will reseek the correct "
210  "batch.",
211  (unsigned long)batchSeq,
212  (unsigned long)expectedSeq,
213  (unsigned long)dbimpl_->GetLatestSequenceNumber());
214  reporter_.Info(buf);
215  return false;
216  }
217  return true;
218 }

220 void TransactionLogIteratorImpl::UpdateCurrentWriteBatch(const Slice& record) {
221  std::unique_ptr<WriteBatch> batch(new WriteBatch());
222  WriteBatchInternal::SetContents(batch.get(), record);
223 
224  SequenceNumber expectedSeq = currentLastSeq_ + 1;
225  // If the iterator has started, then confirm that we get continuous batches
226  if (started_ && !IsBatchExpected(batch.get(), expectedSeq)) {
227  // Seek to the batch having expected sequence number
228  if (expectedSeq < files_->at(currentFileIndex_)->StartSequence()) {
229  // Expected batch must lie in the previous log file
230  // Avoid underflow.
231  if (currentFileIndex_ != 0) {
232  currentFileIndex_--;
233  }
234  }
235  startingSequenceNumber_ = expectedSeq;
236  // currentStatus_ will be set to Ok if reseek succeeds
237  currentStatus_ = Status::NotFound("Gap in sequence numbers");
238  return SeekToStartSequence(currentFileIndex_, true);
239  }
240 
241  currentBatchSeq_ = WriteBatchInternal::Sequence(batch.get());
242  currentLastSeq_ = currentBatchSeq_ +
243  WriteBatchInternal::Count(batch.get()) - 1;
244  // currentBatchSeq_ can only change here
245  assert(currentLastSeq_ <= dbimpl_->GetLatestSequenceNumber());
246 
247  currentBatch_ = move(batch);
248  isValid_ = true;
249  currentStatus_ = Status::OK();
250 }

252 Status TransactionLogIteratorImpl::OpenLogReader(const LogFile* logFile) {
253  unique_ptr<SequentialFile> file;
