28  template <typename Integer>
29  class VotableInteger
30  {
31  public:
32  VotableInteger (Integer current, Integer target)

36  // Add our vote
37  ++mVoteMap[mTarget];
38  }

40  bool
41  mayVote () const
42  {
43  // If we love the current setting, we will not vote
44  return mCurrent != mTarget;
45  }
46 
47  void
48  addVote (Integer vote)
49  {
50  ++mVoteMap[vote];
51  }
52 
53  void
54  noVote ()
55  {
56  addVote (mCurrent);
57  }
58 
59  Integer
60  getVotes ()
61  {
62  Integer ourVote = mCurrent;
63  int weight = 0;
64 
65  typedef typename std::map<Integer, int>::value_type mapVType;
66  for (auto const& e : mVoteMap)
67  {
68  // Take most voted value between current and target, inclusive
69  if ((e.first <= std::max (mTarget, mCurrent)) &&
70  (e.first >= std::min (mTarget, mCurrent)) &&
71  (e.second > weight))
72  {
73  ourVote = e.first;
74  weight = e.second;
75  }
76  }
77 
78  return ourVote;
79  }

81  private:
82  Integer mCurrent; // The current setting
83  Integer mTarget; // The setting we want
84  std::map<Integer, int> mVoteMap;
85  };

129  void
130  doVoting (Ledger::ref lastClosedLedger, SHAMap::ref initialPosition) override
131  {
132  // LCL must be flag ledger
133  assert ((lastClosedLedger->getLedgerSeq () % 256) == 0);
134 
135  VotableInteger<std::uint64_t> baseFeeVote (lastClosedLedger->getBaseFee (), mTargetBaseFee);
136  VotableInteger<std::uint32_t> baseReserveVote (lastClosedLedger->getReserve (0), mTargetReserveBase);
137  VotableInteger<std::uint32_t> incReserveVote (lastClosedLedger->getReserveInc (), mTargetReserveIncrement);
138 
139  // get validations for ledger before flag
140  ValidationSet set = getApp().getValidations ().getValidations (lastClosedLedger->getParentHash ());
141  for (auto const& e : set)
142  {
143  SerializedValidation const& val = *e.second;
144 
145  if (val.isTrusted ())
146  {
147  if (val.isFieldPresent (sfBaseFee))
148  {
149  baseFeeVote.addVote (val.getFieldU64 (sfBaseFee));
150  }
151  else
152  {
153  baseFeeVote.noVote ();
154  }
155 
156  if (val.isFieldPresent (sfReserveBase))
157  {
158  baseReserveVote.addVote (val.getFieldU32 (sfReserveBase));
159  }
160  else
161  {
162  baseReserveVote.noVote ();
163  }
164 
165  if (val.isFieldPresent (sfReserveIncrement))
166  {
167  incReserveVote.addVote (val.getFieldU32 (sfReserveIncrement));
168  }
169  else
170  {
171  incReserveVote.noVote ();
172  }
173  }
174  }
175 
176  // choose our positions
177  std::uint64_t baseFee = baseFeeVote.getVotes ();
178  std::uint32_t baseReserve = baseReserveVote.getVotes ();
179  std::uint32_t incReserve = incReserveVote.getVotes ();
180 
181  // add transactions to our position
182  if ((baseFee != lastClosedLedger->getBaseFee ()) ||
183  (baseReserve != lastClosedLedger->getReserve (0)) ||
184  (incReserve != lastClosedLedger->getReserveInc ()))
185  {
186  if (m_journal.warning) m_journal.warning <<
187  "We are voting for a fee change: " << baseFee <<
188  "/" << baseReserve <<
189  "/" << incReserve;
190 
191  SerializedTransaction trans (ttFEE);
192  trans.setFieldAccount (sfAccount, uint160 ());
193  trans.setFieldU64 (sfBaseFee, baseFee);
194  trans.setFieldU32 (sfReferenceFeeUnits, 10);
195  trans.setFieldU32 (sfReserveBase, baseReserve);
196  trans.setFieldU32 (sfReserveIncrement, incReserve);
197 
198  uint256 txID = trans.getTransactionID ();
199 
200  if (m_journal.warning) m_journal.warning <<
201  "Vote: " << txID;
202 
203  Serializer s;
204  trans.add (s, true);
205 
206  SHAMapItem::pointer tItem = boost::make_shared<SHAMapItem> (txID, s.peekData ());
207 
208  if (!initialPosition->addGiveItem (tItem, true, false))
209  {
210  if (m_journal.warning) m_journal.warning <<
211  "Ledger already had fee change";
212  }
213  }
214  }
