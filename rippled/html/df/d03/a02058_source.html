241  // Swap ownership of the currently referenced object.
242  // The caller receives a pointer to the original object,
243  // and this container is left with the passed object. No
244  // reference counts are changed.
245  //
246  template <class U>
247  T* swap (U* u)
248  {
249  T* const t (m_p);
250  m_p = u;
251  return t;
252  }

254  // Acquire ownership of u.
255  // Any previous reference is released.
256  //
257  template <class U>
258  SharedPtr& assign (U* u)
259  {
260  if (m_p != u)
