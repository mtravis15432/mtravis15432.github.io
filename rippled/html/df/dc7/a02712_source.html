64 void AutoRollLoggerTest::RollLogFileBySizeTest(AutoRollLogger* logger,
65  size_t log_max_size,
66  const string& log_message) {
67  // measure the size of each message, which is supposed
68  // to be equal or greater than log_message.size()
69  LogMessage(logger, log_message.c_str());
70  size_t message_size = logger->GetLogFileSize();
71  size_t current_log_size = message_size;
72 
73  // Test the cases when the log file will not be rolled.
74  while (current_log_size + message_size < log_max_size) {
75  LogMessage(logger, log_message.c_str());
76  current_log_size += message_size;
77  ASSERT_EQ(current_log_size, logger->GetLogFileSize());
78  }
79 
80  // Now the log file will be rolled
81  LogMessage(logger, log_message.c_str());
82  // Since rotation is checked before actual logging, we need to
83  // trigger the rotation by logging another message.
84  LogMessage(logger, log_message.c_str());
85 
86  ASSERT_TRUE(message_size == logger->GetLogFileSize());
87 }

89 uint64_t AutoRollLoggerTest::RollLogFileByTimeTest(
90  AutoRollLogger* logger, size_t time, const string& log_message) {
91  uint64_t expected_create_time;
92  uint64_t actual_create_time;
93  uint64_t total_log_size;
94  ASSERT_OK(env->GetFileSize(kLogFile, &total_log_size));
95  GetFileCreateTime(kLogFile, &expected_create_time);
96  logger->SetCallNowMicrosEveryNRecords(0);
97 
98  // -- Write to the log for several times, which is supposed
99  // to be finished before time.
100  for (int i = 0; i < 10; ++i) {
101  LogMessage(logger, log_message.c_str());
102  ASSERT_OK(logger->GetStatus());
103  // Make sure we always write to the same log file (by
104  // checking the create time);
105  GetFileCreateTime(kLogFile, &actual_create_time);
106 
107  // Also make sure the log size is increasing.
108  ASSERT_EQ(expected_create_time, actual_create_time);
109  ASSERT_GT(logger->GetLogFileSize(), total_log_size);
110  total_log_size = logger->GetLogFileSize();
111  }
112 
113  // -- Make the log file expire
114  sleep(time);
115  LogMessage(logger, log_message.c_str());
116 
117  // At this time, the new log file should be created.
118  GetFileCreateTime(kLogFile, &actual_create_time);
119  ASSERT_GT(actual_create_time, expected_create_time);
120  ASSERT_LT(logger->GetLogFileSize(), total_log_size);
121  expected_create_time = actual_create_time;
122 
123  return expected_create_time;
124 }

239  std::vector<std::string> files;
240  Env::Default()->GetChildren(dir, &files);

242 
243  for (std::vector<std::string>::iterator it = files.begin();
244  it != files.end(); ++it) {
245  uint64_t create_time;
246  FileType type;
247  if (!ParseFileName(*it, &create_time, &type)) {
248  continue;
249  }

251  ++log_file_count;
252  }
