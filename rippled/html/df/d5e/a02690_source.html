87 static void WorkerThreadBody(void* arg) {
88  WorkerThread* t = reinterpret_cast<WorkerThread*>(arg);
89  Random rnd(5);
90  string buf;
91  vector<pair<Blob, uint64_t>> blobs;
92  vector<string> random_strings;
93 
94  for (int i = 0; i < 10; ++i) {
95  random_strings.push_back(RandomString(&rnd, t->data_size_to));
96  }
97 
98  uint64_t total_size = 0;
99 
100  uint64_t start_micros = env->NowMicros();
101  while (env->NowMicros() - start_micros < timeout_sec * 1000 * 1000) {
102  if (blobs.size() && rand() < RAND_MAX * t->read_ratio) {
103  // read
104  int bi = rand() % blobs.size();
105  Status s = bs->Get(blobs[bi].first, &buf);
106  assert(s.ok());
107  t->result.data_read += buf.size();
108  t->result.reads++;
109  } else {
110  // write
111  uint64_t size = rand() % (t->data_size_to - t->data_size_from) +
112  t->data_size_from;
113  total_size += size;
114  string put_str = random_strings[rand() % random_strings.size()];
115  blobs.push_back(make_pair(Blob(), size));
116  Status s = bs->Put(Slice(put_str.data(), size), &blobs.back().first);
117  assert(s.ok());
118  t->result.data_written += size;
119  t->result.writes++;
120  }
121 
122  while (total_size >= t->working_set_size) {
123  // delete random
124  int bi = rand() % blobs.size();
125  total_size -= blobs[bi].second;
126  bs->Delete(blobs[bi].first);
127  blobs.erase(blobs.begin() + bi);
128  t->result.deletes++;
129  }
130  }
131  t->stopped.store(true);
132 }
