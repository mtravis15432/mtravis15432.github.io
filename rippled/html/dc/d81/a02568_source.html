42 // Memtables and sstables that make the DB representation contain
43 // (userkey,seq,type) => uservalue entries. DBIter
44 // combines multiple entries for the same userkey found in the DB
45 // representation into a single entry while accounting for sequence
46 // numbers, deletion markers, overwrites, etc.
47 class DBIter: public Iterator {
48  public:
49  // The following is grossly complicated. TODO: clean it up
50  // Which direction is the iterator currently moving?
51  // (1) When moving forward, the internal iterator is positioned at
52  // the exact entry that yields this->key(), this->value()
53  // (2) When moving backwards, the internal iterator is positioned
54  // just before all entries whose user key == this->key().
55  enum Direction {
56  kForward,
57  kReverse
58  };
59

157 void DBIter::Next() {
158  assert(valid_);
159 
160  if (direction_ == kReverse) { // Switch directions?
161  direction_ = kForward;
162  // iter_ is pointing just before the entries for this->key(),
163  // so advance into the range of entries for this->key() and then
164  // use the normal skipping code below.
165  if (!iter_->Valid()) {
166  iter_->SeekToFirst();
167  } else {
168  iter_->Next();
169  }
170  if (!iter_->Valid()) {
171  valid_ = false;
172  saved_key_.clear();
173  return;
174  }
175  }
176 
177  // If the current value is merged, we might already hit end of iter_
178  if (!iter_->Valid()) {
179  valid_ = false;
180  return;
181  }
182  FindNextUserEntry(true /* skipping the current user key */);
183 }
184 
185 
186 // PRE: saved_key_ has the current user key if skipping
187 // POST: saved_key_ should have the next user key if valid_,
188 // if the current entry is a result of merge
189 // current_entry_is_merged_ => true
190 // saved_value_ => the merged value
191 //
192 // NOTE: In between, saved_key_ can point to a user key that has
193 // a delete marker
194 void DBIter::FindNextUserEntry(bool skipping) {
195  // Loop until we hit an acceptable entry to yield
196  assert(iter_->Valid());
197  assert(direction_ == kForward);
198  current_entry_is_merged_ = false;
199  uint64_t num_skipped = 0;
200  do {
201  ParsedInternalKey ikey;
202  if (ParseKey(&ikey) && ikey.sequence <= sequence_) {
203  if (skipping &&
204  user_comparator_->Compare(ikey.user_key, saved_key_) <= 0) {
205  num_skipped++; // skip this entry
206  BumpPerfCount(&perf_context.internal_key_skipped_count);
207  } else {
208  skipping = false;
209  switch (ikey.type) {
210  case kTypeDeletion:
211  // Arrange to skip all upcoming entries for this key since
212  // they are hidden by this deletion.
213  SaveKey(ikey.user_key, &saved_key_);
214  skipping = true;
215  num_skipped = 0;
216  BumpPerfCount(&perf_context.internal_delete_skipped_count);
217  break;
218  case kTypeValue:
219  valid_ = true;
220  SaveKey(ikey.user_key, &saved_key_);
221  return;
222  case kTypeMerge:
223  // By now, we are sure the current ikey is going to yield a value
224  SaveKey(ikey.user_key, &saved_key_);
225  current_entry_is_merged_ = true;
226  valid_ = true;
227  MergeValuesNewToOld(); // Go to a different state machine
228  return;
229  case kTypeLogData:
230  assert(false);
231  break;
232  }
233  }
234  }
235  // If we have sequentially iterated via numerous keys and still not
236  // found the next user-key, then it is better to seek so that we can
237  // avoid too many key comparisons. We seek to the last occurence of
238  // our current key by looking for sequence number 0.
239  if (skipping && num_skipped > max_skip_) {
240  num_skipped = 0;
241  std::string last_key;
242  AppendInternalKey(&last_key,
243  ParsedInternalKey(Slice(saved_key_), 0, kValueTypeForSeek));
244  iter_->Seek(last_key);
245  RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
246  } else {
247  iter_->Next();
248  }
249  } while (iter_->Valid());
250  valid_ = false;
251 }
252 
253 // Merge values of the same user key starting from the current iter_ position
254 // Scan from the newer entries to older entries.
255 // PRE: iter_->key() points to the first merge type entry
256 // saved_key_ stores the user key
257 // POST: saved_value_ has the merged value for the user key
258 // iter_ points to the next entry (or invalid)
259 void DBIter::MergeValuesNewToOld() {
260  if (!user_merge_operator_) {
261  Log(logger_, "Options::merge_operator is null.");
262  throw std::logic_error("DBIter::MergeValuesNewToOld() with"
263  " Options::merge_operator null");
264  }
265 
266  // Start the merge process by pushing the first operand
267  std::deque<std::string> operands;
268  operands.push_front(iter_->value().ToString());
269 
270  std::string merge_result; // Temporary string to hold merge result later
271  ParsedInternalKey ikey;
272  for (iter_->Next(); iter_->Valid(); iter_->Next()) {
273  if (!ParseKey(&ikey)) {
274  // skip corrupted key
275  continue;
276  }
277 
278  if (user_comparator_->Compare(ikey.user_key, saved_key_) != 0) {
279  // hit the next user key, stop right here
280  break;
281  }
282 
283  if (kTypeDeletion == ikey.type) {
284  // hit a delete with the same user key, stop right here
285  // iter_ is positioned after delete
286  iter_->Next();
287  break;
288  }
289 
290  if (kTypeValue == ikey.type) {
291  // hit a put, merge the put value with operands and store the
292  // final result in saved_value_. We are done!
293  // ignore corruption if there is any.
294  const Slice value = iter_->value();
295  user_merge_operator_->FullMerge(ikey.user_key, &value, operands,
296  &saved_value_, logger_.get());
297  // iter_ is positioned after put
298  iter_->Next();
299  return;
300  }
301 
302  if (kTypeMerge == ikey.type) {
303  // hit a merge, add the value as an operand and run associative merge.
304  // when complete, add result to operands and continue.
305  const Slice& value = iter_->value();
306  operands.push_front(value.ToString());
307  while(operands.size() >= 2) {
308  // Call user associative-merge until it returns false
309  if (user_merge_operator_->PartialMerge(ikey.user_key,
310  Slice(operands[0]),
311  Slice(operands[1]),
312  &merge_result,
313  logger_.get())) {
314  operands.pop_front();
315  swap(operands.front(), merge_result);
316  } else {
317  // Associative merge returns false ==> stack the operands
318  break;
319  }
320  }
321 
322  }
323  }
324 
325  // we either exhausted all internal keys under this user key, or hit
326  // a deletion marker.
327  // feed null as the existing value to the merge operator, such that
328  // client can differentiate this scenario and do things accordingly.
329  user_merge_operator_->FullMerge(saved_key_, nullptr, operands,
330  &saved_value_, logger_.get());
331 }

333 void DBIter::Prev() {
334  assert(valid_);
335 
336  // Throw an exception now if merge_operator is provided
337  // TODO: support backward iteration
338  if (user_merge_operator_) {
339  Log(logger_, "Prev not supported yet if merge_operator is provided");
340  throw std::logic_error("DBIter::Prev backward iteration not supported"
341  " if merge_operator is provided");
342  }
343 
344  if (direction_ == kForward) { // Switch directions?
345  // iter_ is pointing at the current entry. Scan backwards until
346  // the key changes so we can use the normal reverse scanning code.
347  assert(iter_->Valid()); // Otherwise valid_ would have been false
348  SaveKey(ExtractUserKey(iter_->key()), &saved_key_);
349  while (true) {
350  iter_->Prev();
351  if (!iter_->Valid()) {
352  valid_ = false;
353  saved_key_.clear();
354  ClearSavedValue();
355  return;
356  }
357  if (user_comparator_->Compare(ExtractUserKey(iter_->key()),
358  saved_key_) < 0) {
359  break;
360  }
361  }
362  direction_ = kReverse;
363  }
364 
365  FindPrevUserEntry();
366 }

368 void DBIter::FindPrevUserEntry() {
369  assert(direction_ == kReverse);
370  uint64_t num_skipped = 0;
371 
372  ValueType value_type = kTypeDeletion;
373  if (iter_->Valid()) {
374  do {
375  ParsedInternalKey ikey;
376  bool saved_key_cleared = false;
377  if (ParseKey(&ikey) && ikey.sequence <= sequence_) {
378  if ((value_type != kTypeDeletion) &&
379  user_comparator_->Compare(ikey.user_key, saved_key_) < 0) {
380  // We encountered a non-deleted value in entries for previous keys,
381  break;
382  }
383  value_type = ikey.type;
384  if (value_type == kTypeDeletion) {
385  saved_key_.clear();
386  ClearSavedValue();
387  saved_key_cleared = true;
388  } else {
389  Slice raw_value = iter_->value();
390  if (saved_value_.capacity() > raw_value.size() + 1048576) {
391  std::string empty;
392  swap(empty, saved_value_);
393  }
394  SaveKey(ExtractUserKey(iter_->key()), &saved_key_);
395  saved_value_.assign(raw_value.data(), raw_value.size());
396  }
397  }
398  num_skipped++;
399  // If we have sequentially iterated via numerous keys and still not
400  // found the prev user-key, then it is better to seek so that we can
401  // avoid too many key comparisons. We seek to the first occurence of
402  // our current key by looking for max sequence number.
403  if (!saved_key_cleared && num_skipped > max_skip_) {
404  num_skipped = 0;
405  std::string last_key;
406  AppendInternalKey(&last_key,
407  ParsedInternalKey(Slice(saved_key_), kMaxSequenceNumber,
408  kValueTypeForSeek));
409  iter_->Seek(last_key);
410  RecordTick(statistics_, NUMBER_OF_RESEEKS_IN_ITERATION);
411  } else {
412  iter_->Prev();
413  }
414  } while (iter_->Valid());
415  }
416 
417  if (value_type == kTypeDeletion) {
418  // End
419  valid_ = false;
420  saved_key_.clear();
421  ClearSavedValue();
422  direction_ = kForward;
423  } else {
424  valid_ = true;
425  }
426 }

428 void DBIter::Seek(const Slice& target) {
429  direction_ = kForward;
430  ClearSavedValue();
431  saved_key_.clear();
432  AppendInternalKey(
433  &saved_key_, ParsedInternalKey(target, sequence_, kValueTypeForSeek));
434  iter_->Seek(saved_key_);
435  if (iter_->Valid()) {
436  FindNextUserEntry(false /*not skipping */);
437  } else {
438  valid_ = false;
439  }
440 }
