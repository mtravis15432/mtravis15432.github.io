27 struct RegistryKeyWrapper : public Uncopyable
28 {
29  RegistryKeyWrapper (String name, const bool createForWriting, const DWORD wow64Flags)
30  : key (0), wideCharValueName (nullptr)
31  {
32  HKEY rootKey = 0;
33 
34  if (name.startsWithIgnoreCase ("HKEY_CURRENT_USER\\")) rootKey = HKEY_CURRENT_USER;
35  else if (name.startsWithIgnoreCase ("HKEY_LOCAL_MACHINE\\")) rootKey = HKEY_LOCAL_MACHINE;
36  else if (name.startsWithIgnoreCase ("HKEY_CLASSES_ROOT\\")) rootKey = HKEY_CLASSES_ROOT;
37 
38  if (rootKey != 0)
39  {
40  name = name.substring (name.indexOfChar ('\\') + 1);
41 
42  const int lastSlash = name.lastIndexOfChar ('\\');
43  valueName = name.substring (lastSlash + 1);
44  wideCharValueName = valueName.toWideCharPointer();
45 
46  name = name.substring (0, lastSlash);
47  const wchar_t* const wideCharName = name.toWideCharPointer();
48  DWORD result;
49 
50  if (createForWriting)
51  RegCreateKeyEx (rootKey, wideCharName, 0, 0, REG_OPTION_NON_VOLATILE,
52  KEY_WRITE | KEY_QUERY_VALUE | wow64Flags, 0, &key, &result);
53  else
54  RegOpenKeyEx (rootKey, wideCharName, 0, KEY_READ | wow64Flags, &key);
55  }
56  }
57 
58  ~RegistryKeyWrapper()
59  {
60  if (key != 0)
61  RegCloseKey (key);
62  }

75  static std::uint32_t getBinaryValue (const String& regValuePath, MemoryBlock& result, DWORD wow64Flags)
76  {
77  const RegistryKeyWrapper key (regValuePath, false, wow64Flags);
78 
79  if (key.key != 0)
80  {
81  for (unsigned long bufferSize = 1024; ; bufferSize *= 2)
82  {
83  result.setSize (bufferSize, false);
84  DWORD type = REG_NONE;
85 
86  const LONG err = RegQueryValueEx (key.key, key.wideCharValueName, 0, &type,
87  (LPBYTE) result.getData(), &bufferSize);
88 
89  if (err == ERROR_SUCCESS)
90  {
91  result.setSize (bufferSize, false);
92  return type;
93  }
94 
95  if (err != ERROR_MORE_DATA)
96  break;
97  }
98  }
99 
100  return REG_NONE;
101  }

103  static String getValue (const String& regValuePath, const String& defaultValue, DWORD wow64Flags)
104  {
105  MemoryBlock buffer;
106  switch (getBinaryValue (regValuePath, buffer, wow64Flags))
107  {
108  case REG_SZ: return static_cast <const WCHAR*> (buffer.getData());
109  case REG_DWORD: return String ((int) *reinterpret_cast<const DWORD*> (buffer.getData()));
110  default: break;
111  }
112 
113  return defaultValue;

117  {

121  return false;
