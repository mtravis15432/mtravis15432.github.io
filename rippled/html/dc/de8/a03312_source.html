116  ErrorInfo const& get (error_code_i code) const
117  {
118  Map::const_iterator const iter (m_map.find (code));
119  if (iter != m_map.end())
120  return iter->second;
121  return m_unknown;
122  }

125  void add (error_code_i code, std::string const& token,
126  std::string const& message)
127  {
128  std::pair <Map::iterator, bool> result (
129  m_map.emplace (std::piecewise_construct,
130  std::forward_as_tuple (code), std::forward_as_tuple (
131  code, token, message)));
132  if (! result.second)
133  throw std::invalid_argument ("duplicate error code");
134  }
