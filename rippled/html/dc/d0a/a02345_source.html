125 Error session::open(String fileName, std::string options)
126 {
127  Error err;
128 
129  // can't open twice
130  assert (! m_connection);
131 
132  int mode = 0;
133  int flags = 0;
134  int timeout = 0;
135 
136  std::stringstream ssconn(options);
137 
138  while (!err && !ssconn.eof() &&
139  ssconn.str().find('=') != std::string::npos)
140  {
141  std::string key, val;
142  std::getline(ssconn, key, '=');
143  std::getline(ssconn, val, '|');
144 
145  if ("timeout" == key)
146  {
147  if ("infinite" == val)
148  {
149  //timeout = -1;
150  timeout = 0x7fffffff;
151  }
152  else
153  {
154  std::istringstream converter(val);
155  converter >> timeout;
156 
157  if (timeout < 1)
158  timeout = 1;
159  }
160  }
161  else if ("mode" == key)
162  {
163  if (!(mode & (SQLITE_OPEN_READONLY | SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE)))
164  {
165  if ("read" == val)
166  {
167  mode = SQLITE_OPEN_READONLY;
168  }
169  else if ("write" == val)
170  {
171  mode = SQLITE_OPEN_READWRITE;
172  }
173  else if ("create" == val)
174  {
175  mode = SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE;
176  }
177  else
178  {
179  throw std::invalid_argument ("bad parameter");
180  }
181  }
182  else
183  {
184  throw std::invalid_argument ("duplicate parameter");
185  }
186  }
187 
188  /* Most native SQLite libraries don't have these experimental features.
189  */
190 #if ! VF_USE_NATIVE_SQLITE
191  else if ("cache" == key)
192  {
193  if (!(flags & (SQLITE_OPEN_SHAREDCACHE | SQLITE_OPEN_PRIVATECACHE)))
194  {
195  if ("shared" == val)
196  {
197  flags |= SQLITE_OPEN_SHAREDCACHE;
198  }
199  else if ("private" == val)
200  {
201  flags |= SQLITE_OPEN_PRIVATECACHE;
202  }
203  else
204  {
205  throw std::invalid_argument ("bad parameter");
206  }
207  }
208  else
209  {
210  throw std::invalid_argument ("duplicate parameter");
211  }
212  }
213 
214 #endif
215  else if ("threads" == key)
216  {
217  if (!(flags & (SQLITE_OPEN_NOMUTEX | SQLITE_OPEN_FULLMUTEX)))
218  {
219  if ("single" == val)
220  {
221  flags |= SQLITE_OPEN_FULLMUTEX;
222  }
223  else if ("multi" == val)
224  {
225  flags |= SQLITE_OPEN_NOMUTEX;
226  }
227  else
228  {
229  throw std::invalid_argument ("bad parameter");
230  }
231  }
232  else
233  {
234  throw std::invalid_argument ("duplicate parameter");
235  }
236  }
237  else
238  {
239  throw std::invalid_argument ("unknown parameter");
240  }
241  }
242 
243  if (!err)
244  {
245  if (! mode)
246  mode = SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE;
247 
248  flags |= mode;
249 
250  err = detail::sqliteError(__FILE__, __LINE__,
251  sqlite3_open_v2(fileName.toUTF8(), &m_connection, flags, 0));
252 
253  if (!err)
254  {
255  if (timeout > 0)
256  {
257  err = detail::sqliteError(__FILE__, __LINE__,
258  sqlite3_busy_timeout(m_connection, timeout));
259  }
260 
261  /*
262  else
263  {
264  err = detail::sqliteError (__FILE__, __LINE__,
265  sqlite3_busy_handler(m_connection, infiniteBusyHandler, 0));
266  }
267  */
268  }
269 
270  if (!err)
271  {
272  m_fileName = fileName;
273  m_connectString = options;
274  }
275 
276  if (err)
277  {
278  close();
279  }
280  }
281 
282  return err;
283 }

355 Error session::hard_exec(std::string const& query)
356 {
357  Error error;
358  sqlite3_stmt* stmt;
359  char const* tail = 0;
360 
361  int result = sqlite3_prepare_v2(
362  m_connection,
363  query.c_str(),
364  static_cast<int>(query.size()),
365  &stmt,
366  &tail);
367 
368  if (result == SQLITE_OK)
369  {
370  result = sqlite3_step(stmt);
371 
372  sqlite3_finalize(stmt);
373  }
374 
375  if (result != SQLITE_DONE)
376  error = detail::sqliteError(__FILE__, __LINE__, result);
377 
378  return error;
379 }
