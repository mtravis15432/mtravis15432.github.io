96 void standard_use_type::do_use()
97 {
98  int result = SQLITE_OK;
99 
100  convert_to_base();
101 
102  switch (m_type)
103  {
104  case x_null:
105  result = sqlite3_bind_null(m_st->m_stmt, m_iParam);
106  break;
107 
108  case x_bool:
109  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <bool> (m_data) ? 1 : 0);
110  break;
111 
112  case x_char:
113  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <char> (m_data));
114  break;
115 
116  case x_uchar:
117  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <unsigned char> (m_data));
118  break;
119 
120  case x_short:
121  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <short> (m_data));
122  break;
123 
124  case x_ushort:
125  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <unsigned short> (m_data));
126  break;
127 
128  case x_int:
129  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <int> (m_data));
130  break;
131 
132  case x_uint:
133  result = sqlite3_bind_int64(m_st->m_stmt, m_iParam, as <unsigned int> (m_data));
134  break;
135 
136  case x_long:
137  result = sqlite3_bind_int(m_st->m_stmt, m_iParam, as <long> (m_data));
138  break;
139 
140  case x_ulong:
141  result = sqlite3_bind_int64(m_st->m_stmt, m_iParam, as <unsigned long> (m_data));
142  break;
143 
144  case x_longlong:
145  result = sqlite3_bind_int64(m_st->m_stmt, m_iParam, as <long long> (m_data));
146  break;
147 
148  case x_ulonglong:
149  result = sqlite3_bind_int64(m_st->m_stmt, m_iParam, as <unsigned long long> (m_data));
150  break;
151 
152  case x_float:
153  result = sqlite3_bind_double(m_st->m_stmt, m_iParam, as <float> (m_data));
154  break;
155 
156  case x_double:
157  result = sqlite3_bind_double(m_st->m_stmt, m_iParam, as <double> (m_data));
158  break;
159 
160 #if 0
161  case x_longdouble:
162  result = sqlite3_bind_double(m_st->m_stmt, m_iParam, as <long double> (m_data));
163  break;
164 #endif
165 
166  case x_cstring:
167  result = sqlite3_bind_text(m_st->m_stmt, m_iParam, as <char*> (m_data), -1, SQLITE_STATIC);
168  break;
169 
170  case x_cwstring:
171  result = sqlite3_bind_text16(m_st->m_stmt, m_iParam, as <wchar_t*> (m_data), -1, SQLITE_STATIC);
172  break;
173 
174  case x_stdstring:
175  {
176  std::string const& s = as <std::string> (m_data);
177  result = sqlite3_bind_text(m_st->m_stmt,
178  m_iParam,
179  s.c_str(),
180  int(s.size() * sizeof(s[0])),
181  SQLITE_STATIC);
182  }
183  break;
184 
185  case x_stdwstring:
186  {
187  std::wstring const& s = as <std::wstring> (m_data);
188  result = sqlite3_bind_text16(m_st->m_stmt,
189  m_iParam,
190  s.c_str(),
191  int(s.size() * sizeof(s[0])),
192  SQLITE_STATIC);
193  }
194  break;
195 
196  case x_beastString:
197  {
198  String const& s = as <String> (m_data);
199  result = sqlite3_bind_text(m_st->m_stmt,
200  m_iParam,
201  s.toUTF8(),
202  -1,
203  SQLITE_STATIC);
204  }
205  break;
206 
207  case x_stdtm:
208  case x_blob:
209  default:
210  throw std::invalid_argument ("bad parameter");
211  }
212 
213  if (result != SQLITE_OK)
214  throw detail::sqliteError(__FILE__, __LINE__, result);
215 }
