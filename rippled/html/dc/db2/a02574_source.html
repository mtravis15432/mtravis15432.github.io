25 // Value types encoded as the last component of internal keys.
26 // DO NOT CHANGE THESE ENUM VALUES: they are embedded in the on-disk
27 // data structures.
28 enum ValueType {
29  kTypeDeletion = 0x0,
30  kTypeValue = 0x1,
31  kTypeMerge = 0x2,
32  kTypeLogData = 0x3
33 };
34 // kValueTypeForSeek defines the ValueType that should be passed when
35 // constructing a ParsedInternalKey object for seeking to a particular
36 // sequence number (since we sort sequence numbers in decreasing order
37 // and the value type is embedded as the low 8 bits in the sequence
38 // number in internal keys, we need to use the highest-numbered
39 // ValueType, not the lowest).
40 static const ValueType kValueTypeForSeek = kTypeMerge;

42 // We leave eight bits empty at the bottom so a type and sequence#
43 // can be packed together into 64-bits.

80 inline ValueType ExtractValueType(const Slice& internal_key) {
81  assert(internal_key.size() >= 8);
82  const size_t n = internal_key.size();
83  uint64_t num = DecodeFixed64(internal_key.data() + n - 8);

158 inline bool ParseInternalKey(const Slice& internal_key,
159  ParsedInternalKey* result) {
160  const size_t n = internal_key.size();
161  if (n < 8) return false;
162  uint64_t num = DecodeFixed64(internal_key.data() + n - 8);
163  unsigned char c = num & 0xff;
164  result->sequence = num >> 8;
165  result->type = static_cast<ValueType>(c);
166  result->user_key = Slice(internal_key.data(), n - 8);
167  return (c <= static_cast<unsigned char>(kValueTypeForSeek));
168 }

170 // Update the sequence number in the internal key
171 inline void UpdateInternalKey(char* internal_key,
172  const size_t internal_key_size,
173  uint64_t seq, ValueType t) {
174  assert(internal_key_size >= 8);
175  char* seqtype = internal_key + internal_key_size - 8;
176  uint64_t newval = (seq << 8) | t;
177  EncodeFixed64(seqtype, newval);
178 }
