30  void Add(const char* smallest, const char* largest,
31  SequenceNumber smallest_seq = 100,
32  SequenceNumber largest_seq = 100) {
33  FileMetaData* f = new FileMetaData;
34  f->number = files_.size() + 1;
35  f->smallest = InternalKey(smallest, smallest_seq, kTypeValue);
36  f->largest = InternalKey(largest, largest_seq, kTypeValue);
37  files_.push_back(f);
38  }

40  int Find(const char* key) {
41  InternalKey target(key, 100, kTypeValue);
42  InternalKeyComparator cmp(BytewiseComparator());
43  return FindFile(cmp, files_, target.Encode());
44  }

46  bool Overlaps(const char* smallest, const char* largest) {
47  InternalKeyComparator cmp(BytewiseComparator());
48  Slice s(smallest != nullptr ? smallest : "");
49  Slice l(largest != nullptr ? largest : "");
50  return SomeFileOverlapsRange(cmp, disjoint_sorted_files_, files_,
51  (smallest != nullptr ? &s : nullptr),

53  }
