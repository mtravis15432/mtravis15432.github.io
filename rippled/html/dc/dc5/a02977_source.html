64 std::uint64_t ProofOfWork::getDifficulty (uint256 const& target, int iterations)
65 {
66  // calculate the approximate number of hashes required to solve this proof of work
67  if ((iterations > kMaxIterations) || (target < sMinTarget))
68  {
69  WriteLog (lsINFO, ProofOfWork) << "Iterations:" << iterations;
70  WriteLog (lsINFO, ProofOfWork) << "MaxIterat: " << kMaxIterations;
71  WriteLog (lsINFO, ProofOfWork) << "Target: " << target;
72  WriteLog (lsINFO, ProofOfWork) << "MinTarget: " << sMinTarget;
73  throw std::runtime_error ("invalid proof of work target/iteration");
74  }
75 
76  // more iterations means more hashes per iteration but also a larger final hash
77  std::uint64_t difficulty = iterations + (iterations / 8);
78 
79  // Multiply the number of hashes needed by 256 for each leading zero byte in the difficulty
80  const unsigned char* ptr = target.begin ();
81 
82  while (*ptr == 0)
83  {
84  difficulty *= 256;
85  ++ptr;
86  }
87 
88  difficulty = (difficulty * 256) / (*ptr + 1);
89 
90  return difficulty;
91 }

98 uint256 ProofOfWork::solve (int maxIterations) const
99 {
100  if (!isValid ())
101  throw std::runtime_error ("invalid proof of work target/iteration");
102 
103  uint256 nonce;
104  RandomNumbers::getInstance ().fill (&nonce);
105 
106  std::vector<uint256> buf2;
107  buf2.resize (mIterations);
108 
109  std::vector<uint256> buf1;
110  buf1.resize (3);
111  buf1[0] = mChallenge;
112 
113  while (maxIterations > 0)
114  {
115  buf1[1] = nonce;
116  buf1[2].zero ();
117 
118  for (int i = (mIterations - 1); i >= 0; --i)
119  {
120  buf1[2] = getSHA512Half (buf1);
121  buf2[i] = buf1[2];
122  }
123 
124  if (getSHA512Half (buf2) <= mTarget)
125  return nonce;
126 
127  ++nonce;
128  --maxIterations;
129  }
130 
131  return uint256 ();
132 }
