377 template <class Allocator>
378 void
379 Livecache <Allocator>::expire()
380 {
381  std::size_t n (0);
382  typename cache_type::time_point const expired (
383  m_cache.clock().now() - Tuning::liveCacheSecondsToLive);
384  for (auto iter (m_cache.chronological.begin());
385  iter != m_cache.chronological.end() && iter.when() <= expired;)
386  {
387  Element& e (iter->second);
388  hops.remove (e);
389  iter = m_cache.erase (iter);
390  ++n;
391  }
392  if (n > 0)
393  {
394  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
395  "Livecache expired " << n <<
396  ((n > 1) ? " entries" : " entry");
397  }

400 template <class Allocator>
401 void Livecache <Allocator>::insert (Endpoint const& ep)
402 {
403  // The caller already incremented hop, so if we got a
404  // message at maxHops we will store it at maxHops + 1.
405  // This means we won't give out the address to other peers
406  // but we will use it to make connections and hand it out
407  // when redirecting.
408  //
409  assert (ep.hops <= (Tuning::maxHops + 1));
410  std::pair <typename cache_type::iterator, bool> result (
411  m_cache.emplace (ep.address, ep));
412  Element& e (result.first->second);
413  if (result.second)
414  {
415  hops.insert (e);
416  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
417  "Livecache insert " << ep.address <<
418  " at hops " << ep.hops;
419  return;
420  }
421  else if (! result.second && (ep.hops > e.endpoint.hops))
422  {
423  // Drop duplicates at higher hops
424  std::size_t const excess (
425  ep.hops - e.endpoint.hops);
426  if (m_journal.trace) m_journal.trace << beast::leftw(18) <<
427  "Livecache drop " << ep.address <<
428  " at hops +" << excess;
429  return;
430  }
431 
432  m_cache.touch (result.first);
433 
434  // Address already in the cache so update metadata
435  if (ep.hops < e.endpoint.hops)
436  {
437  hops.reinsert (e, ep.hops);
438  if (m_journal.debug) m_journal.debug << beast::leftw (18) <<
439  "Livecache update " << ep.address <<
440  " at hops " << ep.hops;
441  }
442  else
443  {
444  if (m_journal.trace) m_journal.trace << beast::leftw (18) <<
445  "Livecache refresh " << ep.address <<
446  " at hops " << ep.hops;
447  }
448 }

488 template <class Allocator>
489 void
490 Livecache <Allocator>::hops_t::shuffle()
491 {
492  for (auto& list : m_lists)
493  {
494  std::vector <std::reference_wrapper <Element>> v;
495  v.reserve (list.size());
496  std::copy (list.begin(), list.end(),
497  std::back_inserter (v));
498  std::random_shuffle (v.begin(), v.end());
499  list.clear();
500  for (auto& e : v)
501  list.push_back (e);
502  }
503 }
