608  Cont::is_unordered::value,

610  Cont::is_map::value
611  > Traits;
612  typedef typename Cont::size_type size_type;
613  auto const hash (c.hash_function());
614  auto const key_eq (c.key_eq());
615  for (size_type i (0); i < c.bucket_count(); ++i)
616  {
617  auto const last (c.end(i));
618  for (auto iter (c.begin (i)); iter != last; ++iter)
619  {
620  auto const match (std::find_if (v.begin(), v.end(),
621  [iter](typename Values::value_type const& e)
622  {
623  return Traits::extract (*iter) ==
624  Traits::extract (e);
625  }));
626  expect (match != v.end());
627  expect (key_eq (Traits::extract (*iter),
628  Traits::extract (*match)));
629  expect (hash (Traits::extract (*iter)) ==
630  hash (Traits::extract (*match)));
631  }
632  }
