99 // Return true, iff lhs has less priority than rhs.
100 bool PathState::lessPriority (PathState& lhs, PathState& rhs)
101 {
102  // First rank is quality.
103  if (lhs.uQuality != rhs.uQuality)
104  return lhs.uQuality > rhs.uQuality; // Bigger is worse.
105 
106  // Second rank is best quantity.
107  if (lhs.saOutPass != rhs.saOutPass)
108  return lhs.saOutPass < rhs.saOutPass; // Smaller is worse.
109 
110  // Third rank is path index.
111  return lhs.mIndex > rhs.mIndex; // Bigger is worse.
112 }

114 // Make sure last path node delivers to uAccountID: uCurrencyID from uIssuerID.
115 //
116 // If the unadded next node as specified by arguments would not work as is, then add the necessary nodes so it would work.
117 //
118 // Rules:
119 // - Currencies must be converted via an offer.
120 // - A node names it's output.
121 // - A ripple nodes output issuer must be the node's account or the next node's account.
122 // - Offers can only go directly to another offer if the currency and issuer are an exact match.
123 // - Real issuers must be specified for non-XRP.
124 TER PathState::pushImply (
125  const uint160& uAccountID, // --> Delivering to this account.
126  const uint160& uCurrencyID, // --> Delivering this currency.
127  const uint160& uIssuerID) // --> Delivering this issuer.
128 {
129  const Node& pnPrv = vpnNodes.back ();
130  TER terResult = tesSUCCESS;
131 
132  WriteLog (lsTRACE, RippleCalc) << "pushImply>" <<
133  " " << RippleAddress::createHumanAccountID (uAccountID) <<
134  " " << STAmount::createHumanCurrency (uCurrencyID) <<
135  " " << RippleAddress::createHumanAccountID (uIssuerID);
136 
137  if (pnPrv.uCurrencyID != uCurrencyID)
138  {
139  // Currency is different, need to convert via an offer.
140 
141  terResult = pushNode ( // Offer.
142  !!uCurrencyID
143  ? STPathElement::typeCurrency | STPathElement::typeIssuer
144  : STPathElement::typeCurrency,
145  ACCOUNT_XRP, // Placeholder for offers.
146  uCurrencyID, // The offer's output is what is now wanted.
147  uIssuerID);
148  }
149 
150  const Node& pnBck = vpnNodes.back ();
151 
152  // For ripple, non-XRP, ensure the issuer is on at least one side of the transaction.
153  if (tesSUCCESS == terResult
154  && !!uCurrencyID // Not XRP.
155  && (pnBck.uAccountID != uIssuerID // Previous is not issuing own IOUs.
156  && uAccountID != uIssuerID)) // Current is not receiving own IOUs.
157  {
158  // Need to ripple through uIssuerID's account.
159 
160  terResult = pushNode (
161  STPathElement::typeAccount | STPathElement::typeCurrency | STPathElement::typeIssuer,
162  uIssuerID, // Intermediate account is the needed issuer.
163  uCurrencyID,
164  uIssuerID);
165  }
166 
167  WriteLog (lsTRACE, RippleCalc) << "pushImply< : " << transToken (terResult);
168 
169  return terResult;
170 }

172 // Append a node and insert before it any implied nodes.
173 // Offers may go back to back.
174 // <-- terResult: tesSUCCESS, temBAD_PATH, terNO_ACCOUNT, terNO_AUTH, terNO_LINE, tecPATH_DRY
175 TER PathState::pushNode (
176  const int iType,
177  const uint160& uAccountID,
178  const uint160& uCurrencyID,
179  const uint160& uIssuerID)
180 {
181  Node pnCur;
182  const bool bFirst = vpnNodes.empty ();
183  const Node& pnPrv = bFirst ? Node () : vpnNodes.back ();
184  // true, iff node is a ripple account. false, iff node is an offer node.
185  const bool bAccount = is_bit_set (iType, STPathElement::typeAccount);
186  // true, iff currency supplied.
187  // Currency is specified for the output of the current node.
188  const bool bCurrency = is_bit_set (iType, STPathElement::typeCurrency);
189  // Issuer is specified for the output of the current node.
190  const bool bIssuer = is_bit_set (iType, STPathElement::typeIssuer);
191  TER terResult = tesSUCCESS;
192 
193  WriteLog (lsTRACE, RippleCalc) << "pushNode> " <<
194  iType <<
195  ": " << (bAccount ? RippleAddress::createHumanAccountID (uAccountID) : "-") <<
196  " " << (bCurrency ? STAmount::createHumanCurrency (uCurrencyID) : "-") <<
197  "/" << (bIssuer ? RippleAddress::createHumanAccountID (uIssuerID) : "-");
198 
199  pnCur.uFlags = iType;
200  pnCur.uCurrencyID = bCurrency ? uCurrencyID : pnPrv.uCurrencyID;
201 
202  if (iType & ~STPathElement::typeValidBits)
203  {
204  WriteLog (lsDEBUG, RippleCalc) << "pushNode: bad bits.";
205 
206  terResult = temBAD_PATH;
207  }
208  else if (bIssuer && !pnCur.uCurrencyID)
209  {
210  WriteLog (lsDEBUG, RippleCalc) << "pushNode: issuer specified for XRP.";
211 
212  terResult = temBAD_PATH;
213  }
214  else if (bIssuer && !uIssuerID)
215  {
216  WriteLog (lsDEBUG, RippleCalc) << "pushNode: specified bad issuer.";
217 
218  terResult = temBAD_PATH;
219  }
220  else if (!bAccount && !bCurrency && !bIssuer)
221  {
222  WriteLog (lsDEBUG, RippleCalc) << "pushNode: offer must specify at least currency or issuer.";
223 
224  terResult = temBAD_PATH;
225  }
226  else if (bAccount)
227  {
228  // Account link
229 
230  pnCur.uAccountID = uAccountID;
231  pnCur.uIssuerID = bIssuer
232  ? uIssuerID
233  : !!pnCur.uCurrencyID
234  ? uAccountID
235  : ACCOUNT_XRP;
236  pnCur.saRevRedeem = STAmount (pnCur.uCurrencyID, uAccountID);
237  pnCur.saRevIssue = STAmount (pnCur.uCurrencyID, uAccountID);
238  pnCur.saRevDeliver = STAmount (pnCur.uCurrencyID, pnCur.uIssuerID);
239  pnCur.saFwdDeliver = pnCur.saRevDeliver;
240 
241  if (bFirst)
242  {
243  // The first node is always correct as is.
244 
245  nothing ();
246  }
247  else if (!uAccountID)
248  {
249  WriteLog (lsDEBUG, RippleCalc) << "pushNode: specified bad account.";
250 
251  terResult = temBAD_PATH;
252  }
253  else
254  {
255  // Add required intermediate nodes to deliver to current account.
256  WriteLog (lsTRACE, RippleCalc) << "pushNode: imply for account.";
257 
258  terResult = pushImply (
259  pnCur.uAccountID, // Current account.
260  pnCur.uCurrencyID, // Wanted currency.
261  !!pnCur.uCurrencyID ? uAccountID : ACCOUNT_XRP); // Account as wanted issuer.
262 
263  // Note: pnPrv may no longer be the immediately previous node.
264  }
265 
266  if (tesSUCCESS == terResult && !vpnNodes.empty ())
267  {
268  const Node& pnBck = vpnNodes.back ();
269  bool bBckAccount = is_bit_set (pnBck.uFlags, STPathElement::typeAccount);
270 
271  if (bBckAccount)
272  {
273  SLE::pointer sleRippleState = lesEntries.entryCache (ltRIPPLE_STATE, Ledger::getRippleStateIndex (pnBck.uAccountID, pnCur.uAccountID, pnPrv.uCurrencyID));
274 
275  if (!sleRippleState)
276  {
277  WriteLog (lsTRACE, RippleCalc) << "pushNode: No credit line between "
278  << RippleAddress::createHumanAccountID (pnBck.uAccountID)
279  << " and "
280  << RippleAddress::createHumanAccountID (pnCur.uAccountID)
281  << " for "
282  << STAmount::createHumanCurrency (pnCur.uCurrencyID)
283  << "." ;
284 
285  WriteLog (lsTRACE, RippleCalc) << getJson ();
286 
287  terResult = terNO_LINE;
288  }
289  else
290  {
291  WriteLog (lsTRACE, RippleCalc) << "pushNode: Credit line found between "
292  << RippleAddress::createHumanAccountID (pnBck.uAccountID)
293  << " and "
294  << RippleAddress::createHumanAccountID (pnCur.uAccountID)
295  << " for "
296  << STAmount::createHumanCurrency (pnCur.uCurrencyID)
297  << "." ;
298 
299  SLE::pointer sleBck = lesEntries.entryCache (ltACCOUNT_ROOT, Ledger::getAccountRootIndex (pnBck.uAccountID));
300  bool bHigh = pnBck.uAccountID > pnCur.uAccountID;
301 
302  if (!sleBck)
303  {
304  WriteLog (lsWARNING, RippleCalc) << "pushNode: delay: can't receive IOUs from non-existent issuer: " << RippleAddress::createHumanAccountID (pnBck.uAccountID);
305 
306  terResult = terNO_ACCOUNT;
307  }
308  else if ((is_bit_set (sleBck->getFieldU32 (sfFlags), lsfRequireAuth)
309  && !is_bit_set (sleRippleState->getFieldU32 (sfFlags), (bHigh ? lsfHighAuth : lsfLowAuth)))
310  && sleRippleState->getFieldAmount(sfBalance) == zero) // CHECKME
311  {
312  WriteLog (lsWARNING, RippleCalc) << "pushNode: delay: can't receive IOUs from issuer without auth.";
313 
314  terResult = terNO_AUTH;
315  }
316 
317  if (tesSUCCESS == terResult)
318  {
319  STAmount saOwed = lesEntries.rippleOwed (pnCur.uAccountID, pnBck.uAccountID, pnCur.uCurrencyID);
320  STAmount saLimit;
321 
322  if (saOwed <= zero
323  && -saOwed >= (saLimit = lesEntries.rippleLimit (pnCur.uAccountID, pnBck.uAccountID, pnCur.uCurrencyID)))
324  {
325  WriteLog (lsWARNING, RippleCalc) <<
326  "pushNode: dry:" <<
327  " saOwed=" << saOwed <<
328  " saLimit=" << saLimit;
329 
330  terResult = tecPATH_DRY;
331  }
332  }
333  }
334  }
335  }
336 
337  if (tesSUCCESS == terResult)
338  {
339  vpnNodes.push_back (pnCur);
340  }
341  }
342  else
343  {
344  // Offer link
345  // Offers bridge a change in currency & issuer or just a change in issuer.
346  pnCur.uIssuerID = bIssuer
347  ? uIssuerID
348  : !!pnCur.uCurrencyID
349  ? !!pnPrv.uIssuerID
350  ? pnPrv.uIssuerID // Default to previous issuer
351  : pnPrv.uAccountID // Or previous account if no previous issuer.
352  : ACCOUNT_XRP;
353  pnCur.saRateMax = saZero;
354  pnCur.saRevDeliver = STAmount (pnCur.uCurrencyID, pnCur.uIssuerID);
355  pnCur.saFwdDeliver = pnCur.saRevDeliver;
356 
357  if (!!pnCur.uCurrencyID != !!pnCur.uIssuerID)
358  {
359  WriteLog (lsDEBUG, RippleCalc) << "pushNode: currency is inconsistent with issuer.";
360 
361  terResult = temBAD_PATH;
362  }
363  else if (!!pnPrv.uAccountID)
364  {
365  // Previous is an account.
366  WriteLog (lsTRACE, RippleCalc) << "pushNode: imply for offer.";
367 
368  // Insert intermediary issuer account if needed.
369  terResult = pushImply (
370  ACCOUNT_XRP, // Rippling, but offers don't have an account.
371  pnPrv.uCurrencyID,
372  pnPrv.uIssuerID);
373  }
374 
375  if (tesSUCCESS == terResult)
376  {
377  vpnNodes.push_back (pnCur);
378  }
379  }
380 
381  WriteLog (lsTRACE, RippleCalc) << "pushNode< : " << transToken (terResult);
382 
383  return terResult;
384 }

599  uint160 uCurrencyID = uMaxCurrencyID;
600  uint160 uIssuerID = uMaxIssuerID;
601 
602  // Node 0 is a composite of the sending account and saInAct.
603  ++uNode; // skip node 0
604 
605  // Last node is implied: Always skip last node
606  --uEnd; // skip last node
607 
608  // saInAct
609  // - currency is always the same as vpnNodes[0].
610 #if 1
611 
612  if (uNode != uEnd && uMaxIssuerID != uAccountID)
613  {
614  // saInAct issuer is not the sender. This forces an implied node.
615  // WriteLog (lsDEBUG, RippleCalc) << boost::str(boost::format("setCanonical: in diff: uNode=%d uEnd=%d") % uNode % uEnd);
616 
617  // skip node 1
618 
619  uIssuerID = psExpanded.vpnNodes[uNode].uIssuerID;
620 
621  ++uNode;
622  }
623 
624 #else
625 
626  if (uNode != uEnd)
627  {
628  // Have another node
629  bool bKeep = false;
630 
631  if (uMaxIssuerID != uAccountID)
632  {
633  }
634 
635  if (uMaxCurrencyID) // Not sending XRP.
636  {
637  // Node 1 must be an account.
638 
639  if (uMaxIssuerID != uAccountID)
640  {
641  // Node 1 is required to specify issuer.
642 
643  bKeep = true;
644  }
645  else
646  {
647  // Node 1 must be an account
648  }
649  }
650  else
651  {
652  // Node 1 must be an order book.
653 
654  bKeep = true;
655  }
656 
657  if (bKeep)
658  {
659  uCurrencyID = psExpanded.vpnNodes[uNode].uCurrencyID;
660  uIssuerID = psExpanded.vpnNodes[uNode].uIssuerID;
661  ++uNode; // Keep it.
662  }
663  }
664 
665 #endif
666 
667  if (uNode != uEnd && !!uOutCurrencyID && uOutIssuerID != uDstAccountID)
668  {
669  // WriteLog (lsDEBUG, RippleCalc) << boost::str(boost::format("setCanonical: out diff: uNode=%d uEnd=%d") % uNode % uEnd);
670  // The next to last node is saOutAct if an issuer different from receiver is supplied.
671  // The next to last node can be implied.
672 
673  --uEnd;
674  }
675 
676  const Node& pnEnd = psExpanded.vpnNodes[uEnd];
677 
678  if (uNode != uEnd
679  && !pnEnd.uAccountID && pnEnd.uCurrencyID == uOutCurrencyID && pnEnd.uIssuerID == uOutIssuerID)
680  {
681  // The current end node is an offer converting to saOutAct's currency and issuer and can be implied.
682  // WriteLog (lsDEBUG, RippleCalc) << boost::str(boost::format("setCanonical: out offer: uNode=%d uEnd=%d") % uNode % uEnd);
683 
684  --uEnd;
685  }
686 
687  // Do not include uEnd.
688  for (; uNode != uEnd; ++uNode)
689  {
690  // WriteLog (lsDEBUG, RippleCalc) << boost::str(boost::format("setCanonical: loop: uNode=%d uEnd=%d") % uNode % uEnd);
691  const Node& pnPrv = psExpanded.vpnNodes[uNode - 1];
692  const Node& pnCur = psExpanded.vpnNodes[uNode];
693  const Node& pnNxt = psExpanded.vpnNodes[uNode + 1];
694 
695  const bool bCurAccount = is_bit_set (pnCur.uFlags, STPathElement::typeAccount);
696 
697  bool bSkip = false;
698 
699  if (bCurAccount)
700  {
701  // Currently at an account.
702 
703  // Output is non-XRP and issuer is account.
704  if (!!pnCur.uCurrencyID && pnCur.uIssuerID == pnCur.uAccountID)
705  {
706  // Account issues itself.
707  // XXX Not good enough. Previous account must mention it.
708 
709  bSkip = true;
710  }
711  }
712  else
713  {
714  // Currently at an offer.
715  const bool bPrvAccount = is_bit_set (pnPrv.uFlags, STPathElement::typeAccount);
716  const bool bNxtAccount = is_bit_set (pnNxt.uFlags, STPathElement::typeAccount);
717 
718  if (bPrvAccount && bNxtAccount // Offer surrounded by accounts.
719  && pnPrv.uCurrencyID != pnNxt.uCurrencyID)
720  {
721  // Offer can be implied by currency change.
722  // XXX What about issuer?
723 
724  bSkip = true;
725  }
726  }
727 
728  if (!bSkip)
729  {
730  // Copy node
731  Node pnNew;
732 
733  bool bSetAccount = bCurAccount;
734  bool bSetCurrency = uCurrencyID != pnCur.uCurrencyID;
735  // XXX What if we need the next account because we want to skip it?
736  bool bSetIssuer = !uCurrencyID && uIssuerID != pnCur.uIssuerID;
737 
738  pnNew.uFlags = (bSetAccount ? STPathElement::typeAccount : 0)
739  | (bSetCurrency ? STPathElement::typeCurrency : 0)
740  | (bSetIssuer ? STPathElement::typeIssuer : 0);
741 
742  if (bSetAccount)
743  pnNew.uAccountID = pnCur.uAccountID;
744 
745  if (bSetCurrency)
746  {
747  pnNew.uCurrencyID = pnCur.uCurrencyID;
748  uCurrencyID = pnNew.uCurrencyID;
749  }
750 
751  if (bSetIssuer)
752  pnNew.uIssuerID = pnCur.uIssuerID;
753 
754  // XXX ^^^ What about setting uIssuerID?
755 
756  if (bSetCurrency && !uCurrencyID)
757  uIssuerID.zero ();
758 
759  vpnNodes.push_back (pnNew);
760  }
761  }
762 
763  WriteLog (lsDEBUG, RippleCalc) << "setCanonical:" <<
764  " in=" << STAmount::createHumanCurrency (uMaxCurrencyID) <<
765  "/" << RippleAddress::createHumanAccountID (uMaxIssuerID) <<
766  " out=" << STAmount::createHumanCurrency (uOutCurrencyID) <<
767  "/" << RippleAddress::createHumanAccountID (uOutIssuerID) <<
768  ": " << getJson ();
769 }

775 void PathState::checkNoRipple (
776  uint160 const& firstAccount,
777  uint160 const& secondAccount, // This is the account whose constraints we are checking
778  uint160 const& thirdAccount,
779  uint160 const& currency)
780 {
781  // fetch the ripple lines into and out of this node
782  SLE::pointer sleIn = lesEntries.entryCache (ltRIPPLE_STATE,
783  Ledger::getRippleStateIndex (firstAccount, secondAccount, currency));
784  SLE::pointer sleOut = lesEntries.entryCache (ltRIPPLE_STATE,
785  Ledger::getRippleStateIndex (secondAccount, thirdAccount, currency));
786 
787  if (!sleIn || !sleOut)
788  {
789  terStatus = terNO_LINE;
790  }
791  else if (
792  is_bit_set (sleIn->getFieldU32 (sfFlags),
793  (secondAccount > firstAccount) ? lsfHighNoRipple : lsfLowNoRipple) &&
794  is_bit_set (sleOut->getFieldU32 (sfFlags),
795  (secondAccount > thirdAccount) ? lsfHighNoRipple : lsfLowNoRipple))
796  {
797  WriteLog (lsINFO, RippleCalc) << "Path violates noRipple constraint between " <<
798  RippleAddress::createHumanAccountID (firstAccount) << ", " <<
799  RippleAddress::createHumanAccountID (secondAccount) << " and " <<
800  RippleAddress::createHumanAccountID (thirdAccount);
801 
802  terStatus = terNO_RIPPLE;
803  }
804 }

806 // Check a fully-expanded path to make sure it doesn't violate no-Ripple settings
807 void PathState::checkNoRipple (uint160 const& uDstAccountID, uint160 const& uSrcAccountID)
808 {
809 
810  // There must be at least one node for there to be two consecutive ripple lines
811  if (vpnNodes.size() == 0)
812  return;
813 
814  if (vpnNodes.size() == 1)
815  {
816  // There's just one link in the path
817  // We only need to check source-node-dest
818  if (is_bit_set (vpnNodes[0].uFlags, STPathElement::typeAccount) &&
819  (vpnNodes[0].uAccountID != uSrcAccountID) &&
820  (vpnNodes[0].uAccountID != uDstAccountID))
821  {
822  if (saInReq.getCurrency() != saOutReq.getCurrency())
823  terStatus = terNO_LINE;
824  else
825  checkNoRipple (uSrcAccountID, vpnNodes[0].uAccountID, uDstAccountID,
826  vpnNodes[0].uCurrencyID);
827  }
828  return;
829  }
830 
831  // Check source <-> first <-> second
832  if (is_bit_set (vpnNodes[0].uFlags, STPathElement::typeAccount) &&
833  is_bit_set (vpnNodes[1].uFlags, STPathElement::typeAccount) &&
834  (vpnNodes[0].uAccountID != uSrcAccountID))
835  {
836  if ((vpnNodes[0].uCurrencyID != vpnNodes[1].uCurrencyID))
837  {
838  terStatus = terNO_LINE;
839  return;
840  }
841  else
842  {
843  checkNoRipple (uSrcAccountID, vpnNodes[0].uAccountID, vpnNodes[1].uAccountID,
844  vpnNodes[0].uCurrencyID);
845  if (tesSUCCESS != terStatus)
846  return;
847  }
848  }
849 
850  // Check second_from_last <-> last <-> destination
851  size_t s = vpnNodes.size() - 2;
852  if (is_bit_set (vpnNodes[s].uFlags, STPathElement::typeAccount) &&
853  is_bit_set (vpnNodes[s+1].uFlags, STPathElement::typeAccount) &&
854  (uDstAccountID != vpnNodes[s+1].uAccountID))
855  {
856  if ((vpnNodes[s].uCurrencyID != vpnNodes[s+1].uCurrencyID))
857  {
858  terStatus = terNO_LINE;
859  return;
860  }
861  else
862  {
863  checkNoRipple (vpnNodes[s].uAccountID, vpnNodes[s+1].uAccountID, uDstAccountID,
864  vpnNodes[s].uCurrencyID);
865  if (tesSUCCESS != terStatus)
866  return;
867  }
868  }
869 
870 
871  // Loop through all nodes that have a prior node and successor nodes
872  // These are the nodes whose no ripple constratints could be violated
873  for (int i = 1; i < (vpnNodes.size() - 1); ++i)
874  {
875 
876  if (is_bit_set (vpnNodes[i-1].uFlags, STPathElement::typeAccount) &&
877  is_bit_set (vpnNodes[i].uFlags, STPathElement::typeAccount) &&
878  is_bit_set (vpnNodes[i+1].uFlags, STPathElement::typeAccount))
879  { // two consecutive account-to-account links
880 
881  uint160 const& currencyID = vpnNodes[i].uCurrencyID;
882  if ((vpnNodes[i-1].uCurrencyID != currencyID) ||
883  (vpnNodes[i+1].uCurrencyID != currencyID))
884  {
885  terStatus = temBAD_PATH;
886  return;
887  }
888  checkNoRipple (
889  vpnNodes[i-1].uAccountID, vpnNodes[i].uAccountID, vpnNodes[i+1].uAccountID,
890  currencyID);
891  if (terStatus != tesSUCCESS)
892  return;
893  }
894 
895  }
896 }
