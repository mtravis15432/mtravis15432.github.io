165  void sweep ()
166  {
167  int cacheRemovals = 0;
168  int mapRemovals = 0;
169  int cc = 0;
170 
171  // Keep references to all the stuff we sweep
172  // so that we can destroy them outside the lock.
173  //
174  std::vector <mapped_ptr> stuffToSweep;
175 
176  {
177  clock_type::time_point const now (m_clock.now());
178  clock_type::time_point when_expire;
179 
180  lock_guard lock (m_mutex);
181 
182  if (m_target_size == 0 ||
183  (static_cast<int> (m_cache.size ()) <= m_target_size))
184  {
185  when_expire = now - m_target_age;
186  }
187  else
188  {
189  when_expire = now - clock_type::duration (
190  m_target_age.count() * m_target_size / m_cache.size ());
191 
192  clock_type::duration const minimumAge (
193  std::chrono::seconds (1));
194  if (when_expire > (now - minimumAge))
195  when_expire = now - minimumAge;
196 
197  if (m_journal.trace) m_journal.trace <<
198  m_name << " is growing fast " << m_cache.size () << " of " << m_target_size <<
199  " aging at " << (now - when_expire) << " of " << m_target_age;
200  }
201 
202  stuffToSweep.reserve (m_cache.size ());
203 
204  cache_iterator cit = m_cache.begin ();
205 
206  while (cit != m_cache.end ())
207  {
208  if (cit->second.isWeak ())
209  {
210  // weak
211  if (cit->second.isExpired ())
212  {
213  ++mapRemovals;
214  cit = m_cache.erase (cit);
215  }
216  else
217  {
218  ++cit;
219  }
220  }
221  else if (cit->second.last_access <= when_expire)
222  {
223  // strong, expired
224  --m_cache_count;
225  ++cacheRemovals;
226  if (cit->second.ptr.unique ())
227  {
228  stuffToSweep.push_back (cit->second.ptr);
229  ++mapRemovals;
230  cit = m_cache.erase (cit);
231  }
232  else
233  {
234  // remains weakly cached
235  cit->second.ptr.reset ();
236  ++cit;
237  }
238  }
239  else
240  {
241  // strong, not expired
242  ++cc;
243  ++cit;
244  }
245  }
246  }
247 
248  if (m_journal.trace && (mapRemovals || cacheRemovals)) m_journal.trace <<
249  m_name << ": cache = " << m_cache.size () << "-" << cacheRemovals <<
250  ", map-=" << mapRemovals;
251 
252  // At this point stuffToSweep will go out of scope outside the lock
253  // and decrement the reference count on each strong pointer.
254  }

296  bool canonicalize (const key_type& key, boost::shared_ptr<T>& data, bool replace = false)
297  {
298  // Return canonical value, store if needed, refresh in cache
299  // Return values: true=we had the data already
300  lock_guard lock (m_mutex);
301 
302  cache_iterator cit = m_cache.find (key);
303 
304  if (cit == m_cache.end ())
305  {
306  m_cache.insert (cache_pair (key, Entry (m_clock.now(), data)));
307  ++m_cache_count;
308  return false;
309  }
310

312  entry.touch (m_clock.now());
313 
314  if (entry.isCached ())
315  {
316  if (replace)

355  boost::shared_ptr<T> fetch (const key_type& key)
356  {
357  // fetch us a shared pointer to the stored data object
358  lock_guard lock (m_mutex);
359 
360  cache_iterator cit = m_cache.find (key);
361 
362  if (cit == m_cache.end ())
363  {
364  ++m_misses;
365  return mapped_ptr ();
366  }
367 
368  Entry& entry = cit->second;
369  entry.touch (m_clock.now());
370 
371  if (entry.isCached ())
372  {
373  ++m_hits;
374  return entry.ptr;
375  }
376 
377  entry.ptr = entry.lock ();
378 
379  if (entry.isCached ())
380  {
381  // independent of cache size, so not counted as a hit
382  ++m_cache_count;
383  return entry.ptr;
384  }
385 
386  m_cache.erase (cit);
387  ++m_misses;
388  return mapped_ptr ();

424  bool refreshIfPresent (const key_type& key)
425  {
426  bool found = false;
427 
428  // If present, make current in cache
429  lock_guard lock (m_mutex);
430 
431  cache_iterator cit = m_cache.find (key);
432 
433  if (cit != m_cache.end ())

436 
437  if (! entry.isCached ())
438  {
439  // Convert weak to strong.

445  ++m_cache_count;
