43 // Provide the JSON-RPC "result" value.
44 //
45 // JSON-RPC provides a method and an array of params. JSON-RPC is used as a transport for a command and a request object. The
46 // command is the method. The request object is supplied as the first element of the params.
47 Json::Value RPCHandler::doRpcCommand (const std::string& strMethod, Json::Value const& jvParams, int iRole, Resource::Charge& loadType)
48 {
49  WriteLog (lsTRACE, RPCHandler) << "doRpcCommand:" << strMethod << ":" << jvParams;
50 
51  if (!jvParams.isArray () || jvParams.size () > 1)
52  return logRPCError (rpcError (rpcINVALID_PARAMS));
53 
54  Json::Value params = jvParams.size () ? jvParams[0u] : Json::Value (Json::objectValue);
55 
56  if (!params.isObject ())
57  return logRPCError (rpcError (rpcINVALID_PARAMS));
58 
59  // Provide the JSON-RPC method as the field "command" in the request.
60  params["command"] = strMethod;
61 
62  Json::Value jvResult = doCommand (params, iRole, loadType);
63 
64  // Always report "status". On an error report the request as received.
65  if (jvResult.isMember ("error"))
66  {
67  jvResult["status"] = "error";
68  jvResult["request"] = params;
69 
70  }
71  else
72  {
73  jvResult["status"] = "success";
74  }
75 
76  return logRPCError (jvResult);
77 }

261 Json::Value RPCInternalHandler::runHandler (const std::string& name, const Json::Value& params)
262 {
263  RPCInternalHandler* h = sHeadHandler;
264 
265  while (h != nullptr)
266  {
267  if (name == h->mName)
268  {
269  WriteLog (lsWARNING, RPCHandler) << "Internal command " << name << ": " << params;
270  Json::Value ret = h->mHandler (params);
271  WriteLog (lsWARNING, RPCHandler) << "Internal command returns: " << ret;
272  return ret;
273  }
274 
275  h = h->mNextHandler;
276  }
277 
278  return rpcError (rpcBAD_SYNTAX);
279 }
