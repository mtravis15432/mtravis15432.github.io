68 // Appends the current timestamp to the string.
69 // Returns false if could not get the current_time, true if append succeeds
70 Status DBWithTTL::AppendTS(const Slice& val, std::string& val_with_ts) {
71  val_with_ts.reserve(kTSLength + val.size());
72  char ts_string[kTSLength];
73  int64_t curtime;
74  Status st = GetCurrentTime(curtime);
75  if (!st.ok()) {
76  return st;
77  }
78  EncodeFixed32(ts_string, (int32_t)curtime);
79  val_with_ts.append(val.data(), val.size());
80  val_with_ts.append(ts_string, kTSLength);
81  return st;
82 }

84 // Returns corruption if the length of the string is lesser than timestamp, or
85 // timestamp refers to a time lesser than ttl-feature release time
86 Status DBWithTTL::SanityCheckTimestamp(const Slice& str) {
87  if (str.size() < kTSLength) {
88  return Status::Corruption("Error: value's length less than timestamp's\n");
89  }
90  // Checks that TS is not lesser than kMinTimestamp
91  // Gaurds against corruption & normal database opened incorrectly in ttl mode
92  int32_t timestamp_value =
93  DecodeFixed32(str.data() + str.size() - kTSLength);
94  if (timestamp_value < kMinTimestamp){
95  return Status::Corruption("Error: Timestamp < ttl feature release time!\n");
96  }
97  return Status::OK();
98 }

100 // Checks if the string is stale or not according to TTl provided
101 bool DBWithTTL::IsStale(const Slice& value, int32_t ttl) {
102  if (ttl <= 0) { // Data is fresh if TTL is non-positive
103  return false;
104  }
105  int64_t curtime;
106  if (!GetCurrentTime(curtime).ok()) {
107  return false; // Treat the data as fresh if could not get current time
108  }
109  int32_t timestamp_value =
110  DecodeFixed32(value.data() + value.size() - kTSLength);
111  return (timestamp_value + ttl) < curtime;
112 }

114 // Strips the TS from the end of the string
115 Status DBWithTTL::StripTS(std::string* str) {
116  Status st;
117  if (str->length() < kTSLength) {
118  return Status::Corruption("Bad timestamp in key-value");
119  }
120  // Erasing characters which hold the TS
121  str->erase(str->length() - kTSLength, kTSLength);
122  return st;
123 }

181 Status DBWithTTL::Write(const WriteOptions& opts, WriteBatch* updates) {
182  class Handler : public WriteBatch::Handler {
183  public:
184  WriteBatch updates_ttl;
185  Status batch_rewrite_status;
186  virtual void Put(const Slice& key, const Slice& value) {
187  std::string value_with_ts;
188  Status st = AppendTS(value, value_with_ts);
189  if (!st.ok()) {
190  batch_rewrite_status = st;
191  } else {
192  updates_ttl.Put(key, value_with_ts);
193  }
194  }
195  virtual void Merge(const Slice& key, const Slice& value) {
196  std::string value_with_ts;
197  Status st = AppendTS(value, value_with_ts);
198  if (!st.ok()) {
199  batch_rewrite_status = st;
200  } else {
201  updates_ttl.Merge(key, value_with_ts);
202  }
203  }
204  virtual void Delete(const Slice& key) {
205  updates_ttl.Delete(key);
206  }
207  virtual void LogData(const Slice& blob) {
208  updates_ttl.PutLogData(blob);
209  }
210  };
211  Handler handler;
212  updates->Iterate(&handler);
213  if (!handler.batch_rewrite_status.ok()) {
214  return handler.batch_rewrite_status;
215  } else {
216  return db_->Write(opts, &(handler.updates_ttl));
217  }
218 }
