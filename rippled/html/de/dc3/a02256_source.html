83  , private SharedObject

85 public:

88  static Ptr get (SingletonLifetime::Lifetime lifetime
89  = SingletonLifetime::persistAfterCreation)
90  {
91  StaticData& staticData (getStaticData ());
92  SharedSingleton* instance = staticData.instance;
93  if (instance == nullptr)
94  {
95  std::lock_guard <LockType> lock (staticData.mutex);
96  instance = staticData.instance;
97  if (instance == nullptr)
98  {
99  bassert (lifetime == SingletonLifetime::createOnDemand || ! staticData.destructorCalled);
100  staticData.instance = &staticData.object;
101  new (staticData.instance) SharedSingleton (lifetime);
102  memoryBarrier();
103  instance = staticData.instance;
104  }
105  }
106  return instance;
107  }

136  void destroy () const
137  {
138  bool callDestructor;
139 
140  // Handle the condition where one thread is releasing the last
141  // reference just as another thread is trying to acquire it.
142  //
143  {
144  StaticData& staticData (getStaticData ());
145  std::lock_guard <LockType> lock (staticData.mutex);
146 
147  if (this->getReferenceCount() != 0)
148  {
149  callDestructor = false;
150  }
151  else
152  {
153  callDestructor = true;
154  staticData.instance = nullptr;
155  staticData.destructorCalled = true;
156  }
157  }
158 
159  if (callDestructor)
160  {
161  bassert (m_lifetime != SingletonLifetime::neverDestroyed);
162 
163  this->~SharedSingleton();
164  }
