25 Json::Value RPCHandler::doRipplePathFind (Json::Value params, Resource::Charge& loadType, Application::ScopedLockType& masterLockHolder)
26 {
27  masterLockHolder.unlock ();
28 
29  RPC::LegacyPathFind lpf (mRole == Config::ADMIN);
30  if (!lpf.isOkay ())
31  return rpcError (rpcTOO_BUSY);
32 
33  loadType = Resource::feeHighBurdenRPC;
34 
35  RippleAddress raSrc;
36  RippleAddress raDst;
37  STAmount saDstAmount;
38  Ledger::pointer lpLedger;
39 
40  Json::Value jvResult;
41 
42  if (getConfig().RUN_STANDALONE || params.isMember("ledger") || params.isMember("ledger_index") || params.isMember("ledger_hash"))
43  { // The caller specified a ledger
44  jvResult = RPC::lookupLedger (params, lpLedger, *mNetOps);
45  if (!lpLedger)
46  return jvResult;
47  }
48 
49  if (!params.isMember ("source_account"))
50  {
51  jvResult = rpcError (rpcSRC_ACT_MISSING);
52  }
53  else if (!params["source_account"].isString ()
54  || !raSrc.setAccountID (params["source_account"].asString ()))
55  {
56  jvResult = rpcError (rpcSRC_ACT_MALFORMED);
57  }
58  else if (!params.isMember ("destination_account"))
59  {
60  jvResult = rpcError (rpcDST_ACT_MISSING);
61  }
62  else if (!params["destination_account"].isString ()
63  || !raDst.setAccountID (params["destination_account"].asString ()))
64  {
65  jvResult = rpcError (rpcDST_ACT_MALFORMED);
66  }
67  else if (
68  // Parse saDstAmount.
69  !params.isMember ("destination_amount")
70  || !saDstAmount.bSetJson (params["destination_amount"])
71  || saDstAmount <= zero
72  || (!!saDstAmount.getCurrency () && (!saDstAmount.getIssuer () || ACCOUNT_ONE == saDstAmount.getIssuer ())))
73  {
74  WriteLog (lsINFO, RPCHandler) << "Bad destination_amount.";
75  jvResult = rpcError (rpcINVALID_PARAMS);
76  }
77  else if (
78  // Checks on source_currencies.
79  params.isMember ("source_currencies")
80  && (!params["source_currencies"].isArray ()
81  || !params["source_currencies"].size ()) // Don't allow empty currencies.
82  )
83  {
84  WriteLog (lsINFO, RPCHandler) << "Bad source_currencies.";
85  jvResult = rpcError (rpcINVALID_PARAMS);
86  }
87  else
88  {
89  loadType = Resource::feeHighBurdenRPC;
90  RippleLineCache::pointer cache;
91 
92  if (lpLedger)
93  { // The caller specified a ledger
94  lpLedger = boost::make_shared<Ledger> (boost::ref (*lpLedger), false);
95  cache = boost::make_shared<RippleLineCache>(lpLedger);
96  }
97  else
98  { // Use the default ledger and cache
99  lpLedger = mNetOps->getValidatedLedger();
100  cache = getApp().getPathRequests().getLineCache(lpLedger, false);
101  }
102 
103  Json::Value jvSrcCurrencies;
104 
105  if (params.isMember ("source_currencies"))
106  {
107  jvSrcCurrencies = params["source_currencies"];
108  }
109  else
110  {
111  boost::unordered_set<uint160> usCurrencies = usAccountSourceCurrencies (raSrc, cache, true);
112 
113  jvSrcCurrencies = Json::Value (Json::arrayValue);
114 
115  BOOST_FOREACH (const uint160 & uCurrency, usCurrencies)
116  {
117  Json::Value jvCurrency (Json::objectValue);
118 
119  jvCurrency["currency"] = STAmount::createHumanCurrency (uCurrency);
120 
121  jvSrcCurrencies.append (jvCurrency);
122  }
123  }
124 
125  // Fill in currencies destination will accept
126  Json::Value jvDestCur (Json::arrayValue);
127 
128  boost::unordered_set<uint160> usDestCurrID = usAccountDestCurrencies (raDst, cache, true);
129  BOOST_FOREACH (const uint160 & uCurrency, usDestCurrID)
130  jvDestCur.append (STAmount::createHumanCurrency (uCurrency));
131 
132  jvResult["destination_currencies"] = jvDestCur;
133  jvResult["destination_account"] = raDst.humanAccountID ();
134 
135  Json::Value jvArray (Json::arrayValue);
136 
137  for (unsigned int i = 0; i != jvSrcCurrencies.size (); ++i)
138  {
139  Json::Value jvSource = jvSrcCurrencies[i];
140 
141  uint160 uSrcCurrencyID;
142  uint160 uSrcIssuerID;
143 
144  if (!jvSource.isObject ())
145  return rpcError (rpcINVALID_PARAMS);
146 
147  // Parse mandatory currency.
148  if (!jvSource.isMember ("currency")
149  || !STAmount::currencyFromString (uSrcCurrencyID, jvSource["currency"].asString ()))
150  {
151  WriteLog (lsINFO, RPCHandler) << "Bad currency.";
152 
153  return rpcError (rpcSRC_CUR_MALFORMED);
154  }
155 
156  if (uSrcCurrencyID.isNonZero ())
157  uSrcIssuerID = raSrc.getAccountID ();
158 
159  // Parse optional issuer.
160  if (jvSource.isMember ("issuer") &&
161  ((!jvSource["issuer"].isString () ||
162  !STAmount::issuerFromString (uSrcIssuerID, jvSource["issuer"].asString ())) ||
163  (uSrcIssuerID.isZero () != uSrcCurrencyID.isZero ()) ||
164  (ACCOUNT_ONE == uSrcIssuerID)))
165  {
166  WriteLog (lsINFO, RPCHandler) << "Bad issuer.";
167 
168  return rpcError (rpcSRC_ISR_MALFORMED);
169  }
170 
171  STPathSet spsComputed;
172  bool bValid;
173  Pathfinder pf (cache, raSrc, raDst, uSrcCurrencyID, uSrcIssuerID, saDstAmount, bValid);
174 
175  int level = getConfig().PATH_SEARCH_OLD;
176  if ((getConfig().PATH_SEARCH_MAX > level) && !getApp().getFeeTrack().isLoadedLocal())
177  ++level;
178  if (params.isMember("depth") && params["depth"].isIntegral())
179  {
180  int rLev = params["search_depth"].asInt ();
181  if ((rLev < level) || (mRole == Config::ADMIN))
182  level = rLev;
183  }
184 
185  if (params.isMember("paths"))
186  {
187  STParsedJSON paths ("paths", params["paths"]);
188  if (paths.object.get() == nullptr)
189  return paths.error;
190  else
191  spsComputed = paths.object.get()->downcast<STPathSet> ();
192  }
193 
194  STPath extraPath;
195  if (!bValid || !pf.findPaths (level, 4, spsComputed, extraPath))
196  {
197  WriteLog (lsWARNING, RPCHandler) << "ripple_path_find: No paths found.";
198  }
199  else
200  {
201  std::vector<PathState::pointer> vpsExpanded;
202  STAmount saMaxAmountAct;
203  STAmount saDstAmountAct;
204  STAmount saMaxAmount (
205  uSrcCurrencyID,
206  !!uSrcIssuerID
207  ? uSrcIssuerID // Use specifed issuer.
208  : !!uSrcCurrencyID // Default to source account.
209  ? raSrc.getAccountID ()
210  : ACCOUNT_XRP,
211  1);
212  saMaxAmount.negate ();
213 
214  LedgerEntrySet lesSandbox (lpLedger, tapNONE);
215 
216  TER terResult =
217  RippleCalc::rippleCalc (
218  lesSandbox,
219  saMaxAmountAct, // <--
220  saDstAmountAct, // <--
221  vpsExpanded, // <--
222  saMaxAmount, // --> Amount to send is unlimited to get an estimate.
223  saDstAmount, // --> Amount to deliver.
224  raDst.getAccountID (), // --> Account to deliver to.
225  raSrc.getAccountID (), // --> Account sending from.
226  spsComputed, // --> Path set.
227  false, // --> Don't allow partial payment. This is for normal fill or kill payments.
228  // Must achieve delivery goal.
229  false, // --> Don't limit quality. Average quality is wanted for normal payments.
230  false, // --> Allow direct ripple to be added to path set. to path set.
231  true); // --> Stand alone mode, no point in deleting unfundeds.
232 
233  // WriteLog (lsDEBUG, RPCHandler) << "ripple_path_find: PATHS IN: " << spsComputed.size() << " : " << spsComputed.getJson(0);
234  // WriteLog (lsDEBUG, RPCHandler) << "ripple_path_find: PATHS EXP: " << vpsExpanded.size();
235 
236  WriteLog (lsWARNING, RPCHandler)
237  << boost::str (boost::format ("ripple_path_find: saMaxAmount=%s saDstAmount=%s saMaxAmountAct=%s saDstAmountAct=%s")
238  % saMaxAmount
239  % saDstAmount
240  % saMaxAmountAct
241  % saDstAmountAct);
242 
243  if ((extraPath.size() > 0) && ((terResult == terNO_LINE) || (terResult == tecPATH_PARTIAL)))
244  {
245  WriteLog (lsDEBUG, PathRequest) << "Trying with an extra path element";
246  spsComputed.addPath(extraPath);
247  vpsExpanded.clear ();
248  lesSandbox.clear ();
249  terResult = RippleCalc::rippleCalc (lesSandbox, saMaxAmountAct, saDstAmountAct,
250  vpsExpanded, saMaxAmount, saDstAmount,
251  raDst.getAccountID (), raSrc.getAccountID (),
252  spsComputed, false, false, false, true);
253  WriteLog (lsDEBUG, PathRequest) << "Extra path element gives " << transHuman (terResult);
254  }
255 
256  if (tesSUCCESS == terResult)
257  {
258  Json::Value jvEntry (Json::objectValue);
259 
260  STPathSet spsCanonical;
261 
262  // Reuse the expanded as it would need to be calcuated anyway to produce the canonical.
263  // (At least unless we make a direct canonical.)
264  // RippleCalc::setCanonical(spsCanonical, vpsExpanded, false);
265 
266  jvEntry["source_amount"] = saMaxAmountAct.getJson (0);
267  // jvEntry["paths_expanded"] = vpsExpanded.getJson(0);
268  jvEntry["paths_canonical"] = Json::arrayValue; // spsCanonical.getJson(0);
269  jvEntry["paths_computed"] = spsComputed.getJson (0);
270 
271  jvArray.append (jvEntry);
272  }
273  else
274  {
275  std::string strToken;
276  std::string strHuman;
277 
278  transResultInfo (terResult, strToken, strHuman);
279 
280  WriteLog (lsDEBUG, RPCHandler)
281  << boost::str (boost::format ("ripple_path_find: %s %s %s")
282  % strToken
283  % strHuman
284  % spsComputed.getJson (0));
285  }
286  }
287  }
288 
289  // Each alternative differs by source currency.
290  jvResult["alternatives"] = jvArray;
291  }
292 
293  WriteLog (lsDEBUG, RPCHandler)
294  << boost::str (boost::format ("ripple_path_find< %s")
295  % jvResult);
296 
297  return jvResult;
298 }
