10 // We recover the contents of the descriptor from the other files we find.
11 // (1) Any log files are first converted to tables
12 // (2) We scan every table to compute
13 // (a) smallest/largest for the table
14 // (b) largest sequence number in the table
15 // (3) We generate descriptor contents:
16 // - log number is set to zero
17 // - next-file-number is set to 1 + largest file number we found
18 // - last-sequence-number is set to largest sequence# found across
19 // all tables (see 2c)
20 // - compaction pointers are cleared
21 // - every table file is added at level 0
22 //
23 // Possible optimization 1:
24 // (a) Compute total size and use to pick appropriate max-level M
25 // (b) Sort tables by largest sequence# in the table
26 // (c) For each table: if it overlaps earlier table, place in level-0,
27 // else place in level-M.
28 // Possible optimization 2:
29 // Store per-table metadata (smallest, largest, largest-seq#, ...)
30 // in the table's meta section to speed up ScanTable.

69  Status Run() {
70  Status status = FindFiles();
71  if (status.ok()) {
72  ConvertLogFilesToTables();
73  ExtractMetaData();
74  status = WriteDescriptor();
75  }
76  if (status.ok()) {
77  unsigned long long bytes = 0;
78  for (size_t i = 0; i < tables_.size(); i++) {
79  bytes += tables_[i].meta.file_size;
80  }
81  Log(options_.info_log,
82  "**** Repaired rocksdb %s; "
83  "recovered %d files; %llu bytes. "
84  "Some data may have been lost. "
85  "****",
86  dbname_.c_str(),
87  static_cast<int>(tables_.size()),
88  bytes);
89  }
90  return status;
91  }

115  Status FindFiles() {
116  std::vector<std::string> filenames;
117  Status status = env_->GetChildren(dbname_, &filenames);
118  if (!status.ok()) {
119  return status;
120  }
121  if (filenames.empty()) {
122  return Status::IOError(dbname_, "repair found no files");
123  }
124 
125  uint64_t number;
126  FileType type;
127  for (size_t i = 0; i < filenames.size(); i++) {
128  if (ParseFileName(filenames[i], &number, &type)) {
129  if (type == kDescriptorFile) {
130  manifests_.push_back(filenames[i]);
131  } else {
132  if (number + 1 > next_file_number_) {
133  next_file_number_ = number + 1;
134  }
135  if (type == kLogFile) {
136  logs_.push_back(number);
137  } else if (type == kTableFile) {
138  table_numbers_.push_back(number);
139  } else {
140  // Ignore other files
141  }
142  }
143  }
144  }
145  return status;
146  }

161  Status ConvertLogToTable(uint64_t log) {
162  struct LogReporter : public log::Reader::Reporter {
163  Env* env;
164  std::shared_ptr<Logger> info_log;
165  uint64_t lognum;
166  virtual void Corruption(size_t bytes, const Status& s) {
167  // We print error messages for corruption, but continue repairing.
168  Log(info_log, "Log #%llu: dropping %d bytes; %s",
169  (unsigned long long) lognum,
170  static_cast<int>(bytes),
171  s.ToString().c_str());
172  }
173  };
174 
175  // Open the log file
176  std::string logname = LogFileName(dbname_, log);
177  unique_ptr<SequentialFile> lfile;
178  Status status = env_->NewSequentialFile(logname, &lfile, storage_options_);
179  if (!status.ok()) {
180  return status;
181  }
182 
183  // Create the log reader.
184  LogReporter reporter;
185  reporter.env = env_;
186  reporter.info_log = options_.info_log;
187  reporter.lognum = log;
188  // We intentially make log::Reader do checksumming so that
189  // corruptions cause entire commits to be skipped instead of
190  // propagating bad information (like overly large sequence
191  // numbers).
192  log::Reader reader(std::move(lfile), &reporter, false/*do not checksum*/,
193  0/*initial_offset*/);
194 
195  // Read all the records and add to a memtable
196  std::string scratch;
197  Slice record;
198  WriteBatch batch;
199  MemTable* mem = new MemTable(icmp_, options_.memtable_factory,
200  options_.num_levels);
201  mem->Ref();
202  int counter = 0;
203  while (reader.ReadRecord(&record, &scratch)) {
204  if (record.size() < 12) {
205  reporter.Corruption(
206  record.size(), Status::Corruption("log record too small"));
207  continue;
208  }
209  WriteBatchInternal::SetContents(&batch, record);
210  status = WriteBatchInternal::InsertInto(&batch, mem, &options_);
211  if (status.ok()) {
212  counter += WriteBatchInternal::Count(&batch);
213  } else {
214  Log(options_.info_log, "Log #%llu: ignoring %s",
215  (unsigned long long) log,
216  status.ToString().c_str());
217  status = Status::OK(); // Keep going with rest of file
218  }
219  }
220 
221  // Do not record a version edit for this conversion to a Table
222  // since ExtractMetaData() will also generate edits.
223  FileMetaData meta;
224  meta.number = next_file_number_++;
225  Iterator* iter = mem->NewIterator();
226  status = BuildTable(dbname_, env_, options_, storage_options_,
227  table_cache_, iter, &meta,
228  icmp_.user_comparator(), 0, 0, true);
229  delete iter;
230  mem->Unref();
231  mem = nullptr;
232  if (status.ok()) {
233  if (meta.file_size > 0) {
234  table_numbers_.push_back(meta.number);
235  }
236  }
237  Log(options_.info_log, "Log #%llu: %d ops saved to Table #%llu %s",
238  (unsigned long long) log,
239  counter,
240  (unsigned long long) meta.number,
241  status.ToString().c_str());
242  return status;
243  }

245  void ExtractMetaData() {
246  std::vector<TableInfo> kept;
247  for (size_t i = 0; i < table_numbers_.size(); i++) {
248  TableInfo t;
249  t.meta.number = table_numbers_[i];

257  } else {
258  tables_.push_back(t);

263  Status ScanTable(TableInfo* t) {
264  std::string fname = TableFileName(dbname_, t->meta.number);
265  int counter = 0;
266  Status status = env_->GetFileSize(fname, &t->meta.file_size);
267  if (status.ok()) {
268  Iterator* iter = table_cache_->NewIterator(
269  ReadOptions(), storage_options_, t->meta.number, t->meta.file_size);
270  bool empty = true;
271  ParsedInternalKey parsed;
272  t->min_sequence = 0;
273  t->max_sequence = 0;
274  for (iter->SeekToFirst(); iter->Valid(); iter->Next()) {
275  Slice key = iter->key();
276  if (!ParseInternalKey(key, &parsed)) {
277  Log(options_.info_log, "Table #%llu: unparsable key %s",
278  (unsigned long long) t->meta.number,
279  EscapeString(key).c_str());
280  continue;
281  }
282 
283  counter++;
284  if (empty) {
285  empty = false;
286  t->meta.smallest.DecodeFrom(key);
287  }
288  t->meta.largest.DecodeFrom(key);
289  if (parsed.sequence < t->min_sequence) {
290  t->min_sequence = parsed.sequence;
291  }
292  if (parsed.sequence > t->max_sequence) {
293  t->max_sequence = parsed.sequence;
294  }
295  }
296  if (!iter->status().ok()) {
297  status = iter->status();
298  }
299  delete iter;
300  }
301  Log(options_.info_log, "Table #%llu: %d entries %s",
302  (unsigned long long) t->meta.number,
303  counter,
304  status.ToString().c_str());
305  return status;
306  }

308  Status WriteDescriptor() {
309  std::string tmp = TempFileName(dbname_, 1);
310  unique_ptr<WritableFile> file;
311  Status status = env_->NewWritableFile(tmp, &file, storage_options_);
312  if (!status.ok()) {
313  return status;
314  }
315 
316  SequenceNumber max_sequence = 0;
317  for (size_t i = 0; i < tables_.size(); i++) {
318  if (max_sequence < tables_[i].max_sequence) {
319  max_sequence = tables_[i].max_sequence;
320  }
321  }
322 
323  edit_->SetComparatorName(icmp_.user_comparator()->Name());
324  edit_->SetLogNumber(0);
325  edit_->SetNextFile(next_file_number_);
326  edit_->SetLastSequence(max_sequence);
327 
328  for (size_t i = 0; i < tables_.size(); i++) {
329  // TODO(opt): separate out into multiple levels
330  const TableInfo& t = tables_[i];
331  edit_->AddFile(0, t.meta.number, t.meta.file_size,
332  t.meta.smallest, t.meta.largest,
333  t.min_sequence, t.max_sequence);
334  }
335 
336  //fprintf(stderr, "NewDescriptor:\n%s\n", edit_.DebugString().c_str());
337  {
338  log::Writer log(std::move(file));
339  std::string record;
340  edit_->EncodeTo(&record);
341  status = log.AddRecord(record);
342  }
343 
344  if (!status.ok()) {
345  env_->DeleteFile(tmp);
346  } else {
347  // Discard older manifests
348  for (size_t i = 0; i < manifests_.size(); i++) {
349  ArchiveFile(dbname_ + "/" + manifests_[i]);
350  }
351 
352  // Install new manifest
353  status = env_->RenameFile(tmp, DescriptorFileName(dbname_, 1));
354  if (status.ok()) {
355  status = SetCurrentFile(env_, dbname_, 1);
356  } else {
357  env_->DeleteFile(tmp);
358  }
359  }
360  return status;
361  }

363  void ArchiveFile(const std::string& fname) {
364  // Move into another directory. E.g., for
365  // dir/foo
366  // rename to
367  // dir/lost/foo
368  const char* slash = strrchr(fname.c_str(), '/');
369  std::string new_dir;
370  if (slash != nullptr) {
371  new_dir.assign(fname.data(), slash - fname.data());
372  }
373  new_dir.append("/lost");
374  env_->CreateDir(new_dir); // Ignore error
375  std::string new_file = new_dir;
376  new_file.append("/");
377  new_file.append((slash == nullptr) ? fname.c_str() : slash + 1);
378  Status s = env_->RenameFile(fname, new_file);
379  Log(options_.info_log, "Archiving %s: %s\n",
380  fname.c_str(), s.ToString().c_str());
381  }
