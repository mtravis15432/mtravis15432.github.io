111 int InputStream::readCompressedInt()
112 {
113  const std::uint8_t sizeByte = (std::uint8_t) readByte();
114  if (sizeByte == 0)
115  return 0;
116 
117  const int numBytes = (sizeByte & 0x7f);
118  if (numBytes > 4)
119  {
120  bassertfalse; // trying to read corrupt data - this method must only be used
121  // to read data that was written by OutputStream::writeCompressedInt()
122  return 0;
123  }
124 
125  char bytes[4] = { 0, 0, 0, 0 };
126  if (read (bytes, numBytes) != numBytes)
127  return 0;
128 
129  const int num = (int) ByteOrder::littleEndianInt (bytes);
130  return (sizeByte >> 7) ? -num : num;
131 }

184 {
185  MemoryBlock buffer (256);
186  char* data = static_cast<char*> (buffer.getData());
187  size_t i = 0;
188 
189  while ((data[i] = readByte()) != 0)

195  }
196  }
197 
198  return String (CharPointer_UTF8 (data),
199  CharPointer_UTF8 (data + i));

202 String InputStream::readNextLine()
203 {
204  MemoryBlock buffer (256);
205  char* data = static_cast<char*> (buffer.getData());
206  size_t i = 0;
207 
208  while ((data[i] = readByte()) != 0)
209  {
210  if (data[i] == '\n')
211  break;
212 
213  if (data[i] == '\r')
214  {
215  const std::int64_t lastPos = getPosition();
216 
217  if (readByte() != '\n')
218  setPosition (lastPos);
219 
220  break;
221  }
222 
223  if (++i >= buffer.getSize())
224  {
225  buffer.setSize (buffer.getSize() + 512);
226  data = static_cast<char*> (buffer.getData());
227  }
228  }
229 
230  return String::fromUTF8 (data, (int) i);
231 }

247 void InputStream::skipNextBytes (std::int64_t numBytesToSkip)
248 {
249  if (numBytesToSkip > 0)
250  {
251  const int skipBufferSize = (int) bmin (numBytesToSkip, (std::int64_t) 16384);
252  HeapBlock<char> temp ((size_t) skipBufferSize);
253 
254  while (numBytesToSkip > 0 && ! isExhausted())
255  numBytesToSkip -= read (temp, (int) bmin (numBytesToSkip, (std::int64_t) skipBufferSize));
256  }
257 }
