9 // An Env is an interface used by the rocksdb implementation to access
10 // operating system functionality like the filesystem etc. Callers
11 // may wish to provide a custom Env object when opening a database to
12 // get fine gain control; e.g., to rate limit file system operations.
13 //
14 // All Env implementations are safe for concurrent access from
15 // multiple threads without any external synchronization.

381  return 0; // Default implementation to prevent issues with backwards

384  // Remove any kind of caching of data from the offset to offset+length
385  // of this file. If the length is 0, then it refers to the end of file.
386  // If the system is not caching the file contents, then this is a noop.
387  // This call has no effect on dirty pages in the cache.
388  virtual Status InvalidateCache(size_t offset, size_t length) {
389  return Status::NotSupported("InvalidateCache not supported.");
390  }
391 
392  protected:
393  // PrepareWrite performs any necessary preparation for a write
394  // before the write actually occurs. This allows for pre-allocation
395  // of space on devices where it can result in less file
396  // fragmentation and/or less waste from over-zealous filesystem
397  // pre-allocation.
398  void PrepareWrite(size_t offset, size_t len) {
399  if (preallocation_block_size_ == 0) {
400  return;
401  }
402  // If this write would cross one or more preallocation blocks,
403  // determine what the last preallocation block necesessary to
404  // cover this write would be and Allocate to that point.
405  const auto block_size = preallocation_block_size_;
406  size_t new_last_preallocated_block =
407  (offset + len + block_size - 1) / block_size;
408  if (new_last_preallocated_block > last_preallocated_block_) {
409  size_t num_spanned_blocks =
410  new_last_preallocated_block - last_preallocated_block_;
411  Allocate(block_size * last_preallocated_block_,
412  block_size * num_spanned_blocks);
413  last_preallocated_block_ = new_last_preallocated_block;
414  }
415  }
416 
417  /*
418  * Pre-allocate space for a file.
419  */
420  virtual Status Allocate(off_t offset, off_t len) {
421  return Status::OK();
422  }

426  // nbytes specifies the length of the range to be synchronized.
427  // This asks the OS to initiate flushing the cached data to disk,
428  // without waiting for completion.
429  // Default implementation does nothing.
430  virtual Status RangeSync(off_t offset, off_t nbytes) {
431  return Status::OK();
432  }
