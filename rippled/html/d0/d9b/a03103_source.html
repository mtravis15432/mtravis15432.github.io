30 Section ParseSection (const std::string& strInput, const bool bTrim)
31 {
32  std::string strData (strInput);
33  std::vector<std::string> vLines;
34  Section secResult;
35 
36  // Convert DOS format to unix.
37  boost::algorithm::replace_all (strData, "\r\n", "\n");
38 
39  // Convert MacOS format to unix.
40  boost::algorithm::replace_all (strData, "\r", "\n");
41 
42  boost::algorithm::split (vLines, strData, boost::algorithm::is_any_of ("\n"));
43 
44  // Set the default Section name.
45  std::string strSection = SECTION_DEFAULT_NAME;
46 
47  // Initialize the default Section.
48  secResult[strSection] = Section::mapped_type ();
49 
50  // Parse each line.
51  BOOST_FOREACH (std::string & strValue, vLines)
52  {
53  if (strValue.empty () || strValue[0] == '#')
54  {
55  // Blank line or comment, do nothing.
56 
57  nothing ();
58  }
59  else if (strValue[0] == '[' && strValue[strValue.length () - 1] == ']')
60  {
61  // New Section.
62 
63  strSection = strValue.substr (1, strValue.length () - 2);
64  secResult[strSection] = Section::mapped_type ();
65  }
66  else
67  {
68  // Another line for Section.
69  if (bTrim)
70  boost::algorithm::trim (strValue);
71 
72  if (!strValue.empty ())
73  secResult[strSection].push_back (strValue);
74  }
75  }
76 
77  return secResult;
78 }
