77 static int lastProcessPriority = -1;
78 
79 // called by WindowDriver because Windows does weird things to process priority
80 // when you swap apps, and this forces an update when the app is brought to the front.
81 void beast_repeatLastProcessPriority()
82 {
83  if (lastProcessPriority >= 0) // (avoid changing this if it's not been explicitly set by the app..)
84  {
85  DWORD p;
86 
87  switch (lastProcessPriority)
88  {
89  case Process::LowPriority: p = IDLE_PRIORITY_CLASS; break;
90  case Process::NormalPriority: p = NORMAL_PRIORITY_CLASS; break;
91  case Process::HighPriority: p = HIGH_PRIORITY_CLASS; break;
92  case Process::RealtimePriority: p = REALTIME_PRIORITY_CLASS; break;
93  default: bassertfalse; return; // bad priority value
94  }
95 
96  SetPriorityClass (GetCurrentProcess(), p);
97  }
98 }
