28  virtual void CreateFilter(const Slice* keys, int n, std::string* dst) const {
29  for (int i = 0; i < n; i++) {
30  uint32_t h = Hash(keys[i].data(), keys[i].size(), 1);
31  PutFixed32(dst, h);
32  }
33  }

35  virtual bool KeyMayMatch(const Slice& key, const Slice& filter) const {
36  uint32_t h = Hash(key.data(), key.size(), 1);
37  for (unsigned int i = 0; i + 4 <= filter.size(); i += 4) {
38  if (h == DecodeFixed32(filter.data() + i)) {
39  return true;
40  }
41  }
42  return false;
43  }
44 };
