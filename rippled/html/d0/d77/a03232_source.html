84 std::string getHTTPHeaderTimestamp ()
85 {
86  // CHECKME This is probably called often enough that optimizing it makes
87  // sense. There's no point in doing all this work if this function
88  // gets called multiple times a second.
89  char buffer[96];
90  time_t now;
91  time (&now);
92  struct tm* now_gmt = gmtime (&now);
93  std::string locale (setlocale (LC_TIME, nullptr));
94  setlocale (LC_TIME, "C"); // we want posix (aka "C") weekday/month strings
95  strftime (buffer, sizeof (buffer),
96  "Date: %a, %d %b %Y %H:%M:%S +0000\r\n",
97  now_gmt);
98  setlocale (LC_TIME, locale.c_str ());
99  return std::string (buffer);

102 std::string HTTPReply (int nStatus, const std::string& strMsg)
103 {
104  if (ShouldLog (lsTRACE, RPCLog))
105  {
106  WriteLog (lsTRACE, RPCLog) << "HTTP Reply " << nStatus << " " << strMsg;
107  }
108 
109  std::string ret;
110 
111  if (nStatus == 401)
112  {
113  ret.reserve (512);
114 
115  ret.append ("HTTP/1.0 401 Authorization Required\r\n");
116  ret.append (getHTTPHeaderTimestamp ());
117 
118  // CHECKME this returns a different version than the replies below. Is
119  // this by design or an accident or should it be using
120  // BuildInfo::getFullVersionString () as well?
121  ret.append ("Server: " SYSTEM_NAME "-json-rpc/");
122  ret.append (FormatFullVersion ());
123  ret.append ("\r\n");
124 
125  // Be careful in modifying this! If you change the contents you MUST
126  // update the Content-Length header as well to indicate the correct
127  // size of the data.
128  ret.append ("WWW-Authenticate: Basic realm=\"jsonrpc\"\r\n"
129  "Content-Type: text/html\r\n"
130  "Content-Length: 296\r\n"
131  "\r\n"
132  "<!DOCTYPE HTML PUBLIC \"-//W3C//DTD HTML 4.01 Transitional//EN\"\r\n"
133  "\"http://www.w3.org/TR/1999/REC-html401-19991224/loose.dtd\">\r\n"
134  "<HTML>\r\n"
135  "<HEAD>\r\n"
136  "<TITLE>Error</TITLE>\r\n"
137  "<META HTTP-EQUIV='Content-Type' CONTENT='text/html; charset=ISO-8859-1'>\r\n"
138  "</HEAD>\r\n"
139  "<BODY><H1>401 Unauthorized.</H1></BODY>\r\n");
140 
141  return ret;
142  }
143 
144  ret.reserve(256 + strMsg.length());
145 
146  switch (nStatus)
147  {
148  case 200: ret.append ("HTTP/1.1 200 OK\r\n"); break;
149  case 400: ret.append ("HTTP/1.1 400 Bad Request\r\n"); break;
150  case 403: ret.append ("HTTP/1.1 403 Forbidden\r\n"); break;
151  case 404: ret.append ("HTTP/1.1 404 Not Found\r\n"); break;
152  case 500: ret.append ("HTTP/1.1 500 Internal Server Error\r\n"); break;
153  }
154 
155  ret.append (getHTTPHeaderTimestamp ());
156 
157  ret.append ("Connection: Keep-Alive\r\n");
158 
159  if (getConfig ().RPC_ALLOW_REMOTE)
160  ret.append ("Access-Control-Allow-Origin: *\r\n");
161 
162  ret.append ("Content-Length: ");
163  ret.append (std::to_string(strMsg.size () + 2));
164  ret.append ("\r\n");
165 
166  ret.append ("Content-Type: application/json; charset=UTF-8\r\n");
167 
168  ret.append ("Server: " SYSTEM_NAME "-json-rpc/");
169  ret.append (BuildInfo::getFullVersionString ());
170  ret.append ("\r\n");
171 
172  ret.append ("\r\n");
173  ret.append (strMsg);
174  ret.append ("\r\n");
175 
176  return ret;
177 }

192 int ReadHTTPHeader (std::basic_istream<char>& stream, std::map<std::string, std::string>& mapHeadersRet)
193 {
194  int nLen = 0;
195 
196  for (;;)
197  {
198  std::string str;
199  std::getline (stream, str);
200 
201  if (str.empty () || str == "\r")
202  break;
203 
204  std::string::size_type nColon = str.find (":");
205 
206  if (nColon != std::string::npos)
207  {
208  std::string strHeader = str.substr (0, nColon);
209  boost::trim (strHeader);
210  boost::to_lower (strHeader);
211  std::string strValue = str.substr (nColon + 1);
212  boost::trim (strValue);
213  mapHeadersRet[strHeader] = strValue;
214 
215  if (strHeader == "content-length")
216  nLen = atoi (strValue.c_str ());
217  }
218  }
219 
220  return nLen;
221 }

223 int ReadHTTP (std::basic_istream<char>& stream, std::map<std::string, std::string>& mapHeadersRet,
224  std::string& strMessageRet)
225 {
226  mapHeadersRet.clear ();
227  strMessageRet = "";
228 
229  // Read status
230  int nStatus = ReadHTTPStatus (stream);
231 
232  // Read header
233  int nLen = ReadHTTPHeader (stream, mapHeadersRet);
234 
235  if (nLen < 0 || nLen > gMaxHTTPHeaderSize)
236  return 500;
237 
238  // Read message
239  if (nLen > 0)
240  {
241  std::vector<char> vch (nLen);
242  stream.read (&vch[0], nLen);
243  strMessageRet = std::string (vch.begin (), vch.end ());
244  }
245 
246  return nStatus;
247 }

268 bool HTTPAuthorized (const std::map<std::string, std::string>& mapHeaders)
269 {
270  std::map<std::string, std::string>::const_iterator it = mapHeaders.find ("authorization");
271 
272  if ((it == mapHeaders.end ()) || (it->second.substr (0, 6) != "Basic "))
273  return getConfig ().RPC_USER.empty () && getConfig ().RPC_PASSWORD.empty ();
274 
275  std::string strUserPass64 = it->second.substr (6);
276  boost::trim (strUserPass64);
277  std::string strUserPass = DecodeBase64 (strUserPass64);
278  std::string::size_type nColon = strUserPass.find (":");
279 
280  if (nColon == std::string::npos)
281  return false;
282 
283  std::string strUser = strUserPass.substr (0, nColon);
284  std::string strPassword = strUserPass.substr (nColon + 1);
285  return (strUser == getConfig ().RPC_USER) && (strPassword == getConfig ().RPC_PASSWORD);
286 }
