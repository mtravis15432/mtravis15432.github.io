89  Status fetch (void const* key, NodeObject::Ptr* pObject)
90  {
91  pObject->reset ();
92 
93  Status status (ok);
94 
95  leveldb::ReadOptions const options;
96  leveldb::Slice const slice (static_cast <char const*> (key), m_keyBytes);
97  std::string string;
98 
99  leveldb::Status getStatus = m_db->Get (options, slice, &string);
100 
101  if (getStatus.ok ())
102  {
103  DecodedBlob decoded (key, string.data (), string.size ());
104 
105  if (decoded.wasOk ())
106  {
107  *pObject = decoded.createObject ();
108  }
109  else
110  {
111  // Decoding failed, probably corrupted!
112  //
113  status = dataCorrupt;
114  }
115  }
116  else
117  {
118  if (getStatus.IsCorruption ())
119  {
120  status = dataCorrupt;
121  }
122  else if (getStatus.IsNotFound ())
123  {
124  status = notFound;
125  }
126  else
127  {
128  status = unknown;
129  }
130  }
131 
132  return status;

140  void storeBatch (Batch const& batch)
141  {
142  leveldb::WriteBatch wb;

144  EncodedBlob encoded;

146  BOOST_FOREACH (NodeObject::ref object, batch)

162  void visitAll (VisitCallback& callback)
163  {
164  leveldb::ReadOptions const options;
165 
166  std::unique_ptr <leveldb::Iterator> it (m_db->NewIterator (options));
167 
168  for (it->SeekToFirst (); it->Valid (); it->Next ())
169  {
170  if (it->key ().size () == m_keyBytes)
171  {
172  DecodedBlob decoded (it->key ().data (),
173  it->value ().data (),
174  it->value ().size ());
175 
176  if (decoded.wasOk ())
177  {
178  NodeObject::Ptr object (decoded.createObject ());
179 
180  callback.visitObject (object);
181  }
182  else
183  {
184  // Uh oh, corrupted data!
185  WriteLog (lsFATAL, NodeObject) << "Corrupt NodeObject #" << uint256 (it->key ().data ());
186  }
187  }
188  else
189  {
190  // VFALCO NOTE What does it mean to find an
191  // incorrectly sized key? Corruption?
192  WriteLog (lsFATAL, NodeObject) << "Bad key size = " << it->key ().size ();
193  }
194  }
