25 // Given a seed and a source account get the regular public and private key for authorizing transactions.
26 // - Make sure the source account can pay.
27 // --> naRegularSeed : To find the generator
28 // --> naSrcAccountID : Account we want the public and private regular keys to.
29 // <-- naAccountPublic : Regular public key for naSrcAccountID
30 // <-- naAccountPrivate : Regular private key for naSrcAccountID
31 // <-- saSrcBalance: Balance minus fee.
32 // --> naVerifyGenerator : If provided, the found master public generator must match.
33 // XXX Be more lenient, allow use of master generator on claimed accounts.
34 Json::Value authorize (Ledger::ref lrLedger,
35  const RippleAddress& naRegularSeed, const RippleAddress& naSrcAccountID,
36  RippleAddress& naAccountPublic, RippleAddress& naAccountPrivate,
37  STAmount& saSrcBalance, const STAmount& saFee, AccountState::pointer& asSrc,
38  const RippleAddress& naVerifyGenerator, NetworkOPs& netOps)
39 {
40  // Source/paying account must exist.
41  asSrc = netOps.getAccountState (lrLedger, naSrcAccountID);
42 
43  if (!asSrc)
44  {
45  return rpcError (rpcSRC_ACT_NOT_FOUND);
46  }
47 
48  RippleAddress naMasterGenerator;
49 
50  if (asSrc->haveAuthorizedKey ())
51  {
52  Json::Value obj = getMasterGenerator (lrLedger, naRegularSeed, naMasterGenerator, netOps);
53 
54  if (!obj.empty ())
55  return obj;
56  }
57  else
58  {
59  // Try the seed as a master seed.
60  naMasterGenerator = RippleAddress::createGeneratorPublic (naRegularSeed);
61  }
62 
63  // If naVerifyGenerator is provided, make sure it is the master generator.
64  if (naVerifyGenerator.isValid () && naMasterGenerator != naVerifyGenerator)
65  {
66  return rpcError (rpcWRONG_SEED);
67  }
68 
69  // Find the index of the account from the master generator, so we can generate the public and private keys.
70  RippleAddress naMasterAccountPublic;
71  unsigned int iIndex = 0;
72  bool bFound = false;
73 
74  // Don't look at ledger entries to determine if the account exists. Don't want to leak to thin server that these accounts are
75  // related.
76  while (!bFound && iIndex != getConfig ().ACCOUNT_PROBE_MAX)
77  {
78  naMasterAccountPublic.setAccountPublic (naMasterGenerator, iIndex);
79 
80  WriteLog (lsDEBUG, RPCHandler) << "authorize: " << iIndex << " : " << naMasterAccountPublic.humanAccountID () << " : " << naSrcAccountID.humanAccountID ();
81 
82  bFound = naSrcAccountID.getAccountID () == naMasterAccountPublic.getAccountID ();
83 
84  if (!bFound)

86  }
87

91  }
92

94  RippleAddress naGenerator = RippleAddress::createGeneratorPublic (naRegularSeed);
95

98 
99  if (asSrc->haveAuthorizedKey () && (asSrc->getAuthorizedKey ().getAccountID () != naAccountPublic.getAccountID ()))

103  // Log::out() << "naAccountPublic: " << strHex(naAccountPublic.getAccountID());
104 
105  return rpcError (rpcPASSWD_CHANGED);
106  }
107 
108  saSrcBalance = asSrc->getBalance ();
109 
110  if (saSrcBalance < saFee)
111  {
112  WriteLog (lsINFO, RPCHandler) << "authorize: Insufficient funds for fees: fee=" << saFee.getText () << " balance=" << saSrcBalance.getText ();
113 
114  return rpcError (rpcINSUF_FUNDS);
115  }
116  else
117  {
118  saSrcBalance -= saFee;
119  }
120 
121  return Json::Value ();
122 }
