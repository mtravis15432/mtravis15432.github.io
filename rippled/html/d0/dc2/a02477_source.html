44 template <class OutputSequence, class InputIterator>
45 void parseAddresses (OutputSequence& out, InputIterator first, InputIterator last,
46  beast::Journal::Stream stream = beast::Journal::Stream ())
47 {
48  while (first != last)
49  {
50  auto const str (*first);
51  ++first;
52  {
53  beast::IP::Endpoint const addr (beast::IP::Endpoint::from_string (str));
54  if (! is_unspecified (addr))
55  {
56  out.push_back (addr);
57  continue;
58  }
59  }
60  {
61  beast::IP::Endpoint const addr (beast::IP::Endpoint::from_string_altform (str));
62  if (! is_unspecified (addr))
63  {
64  out.push_back (addr);
65  continue;
66  }
67  }
68  if (stream) stream <<
69  "Config: \"" << str << "\" is not a valid IP address.";
70  }
71 }

161  //
162  // Determine the config and data directories.
163  // If the config file is found in the current working directory, use the current working directory as the config directory and

181  if (!strConf.empty ())
182  {
183  // --conf=<path> : everything is relative that file.
184  CONFIG_FILE = strConfFile;
185  CONFIG_DIR = boost::filesystem::absolute (CONFIG_FILE);
186  CONFIG_DIR.remove_filename ();
187  DATA_DIR = CONFIG_DIR / strDbPath;
188  }
189  else
190  {
191  CONFIG_DIR = boost::filesystem::current_path ();
192  CONFIG_FILE = CONFIG_DIR / strConfFile;
193  DATA_DIR = CONFIG_DIR / strDbPath;
194 
195  if (exists (CONFIG_FILE)
196  // Can we figure out XDG dirs?
197  || (!getenv ("HOME") && (!getenv ("XDG_CONFIG_HOME") || !getenv ("XDG_DATA_HOME"))))
198  {
199  // Current working directory is fine, put dbs in a subdir.
200  nothing ();
201  }
202  else
203  {
204  // Construct XDG config and data home.
205  // http://standards.freedesktop.org/basedir-spec/basedir-spec-latest.html
206  std::string strHome = strGetEnv ("HOME");
207  std::string strXdgConfigHome = strGetEnv ("XDG_CONFIG_HOME");
208  std::string strXdgDataHome = strGetEnv ("XDG_DATA_HOME");
209 
210  if (strXdgConfigHome.empty ())
211  {
212  // $XDG_CONFIG_HOME was not set, use default based on $HOME.
213  strXdgConfigHome = boost::str (boost::format ("%s/.config") % strHome);
214  }
215 
216  if (strXdgDataHome.empty ())
217  {
218  // $XDG_DATA_HOME was not set, use default based on $HOME.
219  strXdgDataHome = boost::str (boost::format ("%s/.local/share") % strHome);
220  }
221 
222  CONFIG_DIR = boost::str (boost::format ("%s/" SYSTEM_NAME) % strXdgConfigHome);
223  CONFIG_FILE = CONFIG_DIR / strConfFile;
224  DATA_DIR = boost::str (boost::format ("%s/" SYSTEM_NAME) % strXdgDataHome);
225 
226  boost::filesystem::create_directories (CONFIG_DIR, ec);
227 
228  if (ec)
229  throw std::runtime_error (boost::str (boost::format ("Can not create %s") % CONFIG_DIR));
230  }
231  }
232 
233  HTTPClient::initializeSSLContext();
234 
235  // Update default values
236  load ();
237 
238  // Log::out() << "CONFIG FILE: " << CONFIG_FILE;
239  // Log::out() << "CONFIG DIR: " << CONFIG_DIR;
240  // Log::out() << "DATA DIR: " << DATA_DIR;

710 Config::Role Config::getAdminRole (Json::Value const& params, beast::IP::Endpoint const& remoteIp) const
711 {
712  Config::Role role (Config::FORBID);
713 
714  bool const bPasswordSupplied =
715  params.isMember ("admin_user") ||
716  params.isMember ("admin_password");
717 
718  bool const bPasswordRequired =
719  ! this->RPC_ADMIN_USER.empty () ||
720  ! this->RPC_ADMIN_PASSWORD.empty ();
721 
722  bool bPasswordWrong;
723 
724  if (bPasswordSupplied)
725  {
726  if (bPasswordRequired)
727  {
728  // Required, and supplied, check match
729  bPasswordWrong =
730  (this->RPC_ADMIN_USER !=
731  (params.isMember ("admin_user") ? params["admin_user"].asString () : ""))
732  ||
733  (this->RPC_ADMIN_PASSWORD !=
734  (params.isMember ("admin_user") ? params["admin_password"].asString () : ""));
735  }
736  else
737  {
738  // Not required, but supplied
739  bPasswordWrong = false;
740  }
741  }
742  else
743  {
744  // Required but not supplied,
745  bPasswordWrong = bPasswordRequired;
746  }
747 
748  // Meets IP restriction for admin.
749  beast::IP::Endpoint const remote_addr (remoteIp.at_port (0));
750  bool bAdminIP = false;
751 
752  for (auto const& allow_addr : RPC_ADMIN_ALLOW)
753  {
754  if (allow_addr == remote_addr)
755  {
756  bAdminIP = true;
757  break;
758  }
759  }
760 
761  if (bPasswordWrong // Wrong
762  || (bPasswordSupplied && !bAdminIP)) // Supplied and doesn't meet IP filter.
763  {
764  role = Config::FORBID;
765  }
766  // If supplied, password is correct.
767  else
768  {
769  // Allow admin, if from admin IP and no password is required or it was supplied and correct.
770  role = bAdminIP && (!bPasswordRequired || bPasswordSupplied) ? Config::ADMIN : Config::GUEST;
771  }
772 
773  return role;
774 }
