51  template <typename Conversion, std::size_t Bytes>
52  static std::string encode (UnsignedInteger <Bytes> v, Conversion c = Conversion ())
53  {
54  // bi is destroyed in this process
55  typename UnsignedInteger <Bytes>::CalcType bi (v.toCalcType ());
56  std::size_t const radix (Conversion::radix);

61  std::size_t const m (bi % radix);
62  bi /= radix;
63  s.push_back (c.map (m));
64  }

66  return s;
67  }

73  template <typename Conversion, std::size_t Bytes>
74  static bool decode (UnsignedInteger <Bytes>& rv,
75  std::string const& s, Conversion c = Conversion ())
76  {
77  typename UnsignedInteger <Bytes>::CalcType bi (rv.toCalcType (false));
78  std::size_t const radix (Conversion::radix);
79  bi.clear ();
80  for (std::string::const_iterator iter (s.begin()); iter != s.end(); ++iter)
81  {
82  int const v (c.invert (*iter));
83  if (v == -1)
84  return false;
85  bi *= radix;
86  bi += v;
87  }
88  bi.toCanonical();
89  return true;
90  }

134 {
135 public:
136  template <std::size_t Bytes>
137  static std::string encode (UnsignedInteger <Bytes> const& v)
138  {
139  std::string s;
140  std::uint8_t const* src (v.cbegin()-1);
141  char const* const tab (alphabet().c_str());
142  s.reserve (Bytes * 2);
143  for (std::size_t bytes (v.size);bytes--;)
144  {
145  std::uint8_t const v (*++src);
146  s.push_back (tab [v>>4]);
147  s.push_back (tab [v&0x0f]);
148  }
149  return s;
150  }

152  template <std::size_t Bytes>
153  static bool decode (UnsignedInteger <Bytes>& rv,
154  std::string const& s)
155  {
156  // can't have an odd size
157  if (s.size() & 1)
158  return false;
159  std::uint8_t* dest (rv.begin()-1);
160  int const* const tab (&inverse_alphabet().front());
161  for (std::string::const_iterator iter (s.begin()); iter != s.end();)
162  {
163  int const n1 (tab [*iter++]);
164  if (n1 == -1)
165  return false;
166  int const n2 (tab [*iter++]);
167  if (n2 == -1)
168  return false;
169  *++dest = ((std::uint8_t)((n1<<4)|n2));
170  }
171  return true;
172  }

177  return s;
