162  class DynamicPrefixIterator : public MemTableRep::Iterator {
163  private:
164  // the underlying memtable rep
165  const TransformRep& memtable_rep_;
166  // the result of a prefix seek
167  std::unique_ptr<MemTableRep::Iterator> bucket_iterator_;
168 
169  public:
170  explicit DynamicPrefixIterator(const TransformRep& memtable_rep)
171  : memtable_rep_(memtable_rep) {}
172 
173  virtual ~DynamicPrefixIterator() { };
174 
175  // Returns true iff the iterator is positioned at a valid node.
176  virtual bool Valid() const {
177  return bucket_iterator_ && bucket_iterator_->Valid();
178  }
179 
180  // Returns the key at the current position.
181  // REQUIRES: Valid()
182  virtual const char* key() const {
183  assert(Valid());
184  return bucket_iterator_->key();
185  }
186 
187  // Advances to the next position.
188  // REQUIRES: Valid()
189  virtual void Next() {
190  assert(Valid());
191  bucket_iterator_->Next();
192  }
193 
194  // Advances to the previous position.
195  // REQUIRES: Valid()
196  virtual void Prev() {
197  assert(Valid());
198  bucket_iterator_->Prev();
199  }
200 
201  // Advance to the first entry with a key >= target within the
202  // same bucket as target
203  virtual void Seek(const char* target) {
204  Slice prefix = memtable_rep_.transform_->Transform(
205  memtable_rep_.UserKey(target));
206 
207  ReadLock l(&memtable_rep_.rwlock_);
208  auto bucket = memtable_rep_.buckets_.find(prefix);
209  if (bucket == memtable_rep_.buckets_.end()) {
210  bucket_iterator_.reset(nullptr);
211  } else {
212  bucket_iterator_.reset(
213  new TransformIterator(bucket->second, memtable_rep_.GetLock(prefix)));
214  bucket_iterator_->Seek(target);
215  }
216  }
217 
218  // Position at the first entry in collection.
219  // Final state of iterator is Valid() iff collection is not empty.
220  virtual void SeekToFirst() {
221  // Prefix iterator does not support total order.
222  // We simply set the iterator to invalid state
223  bucket_iterator_.reset(nullptr);
224  }
225 
226  // Position at the last entry in collection.
227  // Final state of iterator is Valid() iff collection is not empty.
228  virtual void SeekToLast() {
229  // Prefix iterator does not support total order.
230  // We simply set the iterator to invalid state
231  bucket_iterator_.reset(nullptr);
232  }
233  };
234 };

248 std::shared_ptr<TransformRep::Bucket>& TransformRep::GetBucket(
249  const Slice& transformed) {
250  WriteLock l(&rwlock_);
251  auto& bucket = buckets_[transformed];
252  if (!bucket) {
253  bucket.reset(
254  new decltype(buckets_)::mapped_type::element_type(Compare(compare_)));
255  // To memory_usage_ we add the size of the std::set and the size of the
256  // std::pair (decltype(buckets_)::value_type) which includes the
257  // Slice and the std::shared_ptr
258  memory_usage_ += sizeof(*bucket) +
259  sizeof(decltype(buckets_)::value_type);
260  }
261  return bucket;
262 }

276 void TransformRep::Insert(const char* key) {
277  assert(!Contains(key));
278  auto transformed = transform_->Transform(UserKey(key));
279  auto& bucket = GetBucket(transformed);
280  WriteLock bl(GetLock(transformed));
281  bucket->insert(key);
282  memory_usage_ += sizeof(key);

285 bool TransformRep::Contains(const char* key) const {
286  ReadLock l(&rwlock_);
287  auto transformed = transform_->Transform(UserKey(key));
288  auto bucket = buckets_.find(transformed);
289  if (bucket == buckets_.end()) {
290  return false;
291  }
292  ReadLock bl(GetLock(transformed));
293  return bucket->second->count(key) != 0;
294 }

336 void TransformRep::Iterator::Prev() {
337  assert(Valid());
338  if (cit_ == items_->begin()) {
339  // If you try to go back from the first element, the iterator should be
340  // invalidated. So we set it to past-the-end. This means that you can

370 std::shared_ptr<MemTableRep::Iterator> TransformRep::GetIterator() {
371  auto items = std::make_shared<Bucket>(Compare(compare_));
372  // Hold read locks on all locks
373  ReadLock l(&rwlock_);
374  std::for_each(locks_.begin(), locks_.end(), [] (port::RWMutex& lock) {
375  lock.ReadLock();
376  });
377  for (auto& bucket : buckets_) {
378  items->insert(bucket.second->begin(), bucket.second->end());
379  }
380  std::for_each(locks_.begin(), locks_.end(), [] (port::RWMutex& lock) {
381  lock.Unlock();
382  });
383  return std::make_shared<Iterator>(std::move(items));
384 }

386 std::shared_ptr<MemTableRep::Iterator> TransformRep::GetTransformIterator(
387  const Slice& transformed) {
388  ReadLock l(&rwlock_);
