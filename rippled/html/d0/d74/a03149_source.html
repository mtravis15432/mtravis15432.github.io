100  bool asyncFetch (uint256 const& hash, NodeObject::pointer& object)
101  {
102  // See if the object is in cache
103  object = m_cache.fetch (hash);
104  if (object || m_negCache.touch_if_exists (hash))
105  return true;
106 
107  {
108  // No. Post a read

114  return false;

137  NodeObject::Ptr fetch (uint256 const& hash)
138  {
139  // See if the object already exists in the cache
140  //
141  NodeObject::Ptr obj = m_cache.fetch (hash);
142 
143  if (obj != nullptr)
144  return obj;
145 
146  if (m_negCache.touch_if_exists (hash))
147  return obj;
148 
149  // Check the database(s).
150 
151  bool foundInFastBackend = false;
152 
153  // Check the fast backend database if we have one
154  //
155  if (m_fastBackend != nullptr)
156  {
157  obj = fetchInternal (*m_fastBackend, hash);
158 
159  // If we found the object, avoid storing it again later.
160  if (obj != nullptr)
161  foundInFastBackend = true;
162  }
163 
164  // Are we still without an object?
165  //
166  if (obj == nullptr)
167  {
168  // Yes so at last we will try the main database.
169  //
170  obj = fetchInternal (*m_backend, hash);
171  }
172 
173  if (obj == nullptr)
174  {
175 
176  // Just in case a write occurred
177  obj = m_cache.fetch (hash);
178 
179  if (obj == nullptr)
180  {
181  // We give up
182  m_negCache.insert (hash);
183  }
184  }
185  else
186  {
187  // Ensure all threads get the same object
188  //
189  m_cache.canonicalize (hash, obj);
190 
191  if (! foundInFastBackend)
192  {
193  // If we have a fast back end, store it there for later.
194  //
195  if (m_fastBackend != nullptr)
196  m_fastBackend->store (obj);
197 
198  // Since this was a 'hard' fetch, we will log it.
199  //
200  WriteLog (lsTRACE, NodeObject) << "HOS: " << hash << " fetch: in db";
201  }
202  }
203 
204  return obj;
205  }

207  NodeObject::Ptr fetchInternal (Backend& backend,
208  uint256 const& hash)
209  {
210  NodeObject::Ptr object;
211 
212  Status const status = backend.fetch (hash.begin (), &object);
213 
214  switch (status)
215  {
216  case ok:
217  case notFound:
218  break;
219 
220  case dataCorrupt:
221  // VFALCO TODO Deal with encountering corrupt data!

223  WriteLog (lsFATAL, NodeObject) << "Corrupt NodeObject #" << hash;

285  // Entry point for async read threads
286  void threadEntry ()
287  {
288  beast::Thread::setCurrentThreadName ("prefetch");
289  while (1)
290  {
291  uint256 hash;
292 
293  {
294  std::unique_lock <std::mutex> lock (m_readLock);
295 
296  while (!m_readShut && m_readSet.empty ())
297  {
298  // all work is done
299  m_readGenCondVar.notify_all ();
300  m_readCondVar.wait (lock);
301  }
302 
303  if (m_readShut)
304  break;
305 
306  // Read in key order to make the back end more efficient
307  std::set <uint256>::iterator it = m_readSet.lower_bound (m_readLast);
308  if (it == m_readSet.end ())
309  {
310  it = m_readSet.begin ();
311 
312  // A generation has completed
313  ++m_readGen;
314  m_readGenCondVar.notify_all ();
315  }
316 
317  hash = *it;
318  m_readSet.erase (it);
319  m_readLast = hash;
320  }
321 
322  // Perform the read
323  fetch (hash);
324  }
325  }
