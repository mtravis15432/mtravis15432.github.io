34 char* ArenaImpl::AllocateFallback(size_t bytes) {
35  if (bytes > block_size_ / 4) {
36  // Object is more than a quarter of our block size. Allocate it separately
37  // to avoid wasting too much space in leftover bytes.
38  char* result = AllocateNewBlock(bytes);
39  return result;
40  }
41 
42  // We waste the remaining space in the current block.
43  alloc_ptr_ = AllocateNewBlock(block_size_);
44  alloc_bytes_remaining_ = block_size_;
45 
46  char* result = alloc_ptr_;
47  alloc_ptr_ += bytes;
48  alloc_bytes_remaining_ -= bytes;
49  return result;
50 }

52 char* ArenaImpl::AllocateAligned(size_t bytes) {
53  const int align = sizeof(void*); // We'll align to pointer size
54  assert((align & (align-1)) == 0); // Pointer size should be a power of 2
55  size_t current_mod = reinterpret_cast<uintptr_t>(alloc_ptr_) & (align-1);
56  size_t slop = (current_mod == 0 ? 0 : align - current_mod);
57  size_t needed = bytes + slop;
58  char* result;
59  if (needed <= alloc_bytes_remaining_) {
60  result = alloc_ptr_ + slop;
61  alloc_ptr_ += needed;
62  alloc_bytes_remaining_ -= needed;
63  } else {
64  // AllocateFallback always returned aligned memory
65  result = AllocateFallback(bytes);
66  }
67  assert((reinterpret_cast<uintptr_t>(result) & (align-1)) == 0);
68  return result;
69 }
