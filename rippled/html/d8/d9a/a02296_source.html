31  // strict string to integer parser
32  template <class IntType, class InputIterator>
33  static inline bool parseSigned (IntType& result, InputIterator begin, InputIterator end)
34  {
35  if (0 == std::distance (begin, end))
36  return false;
37 
38  std::uint64_t accum = 0;
39  InputIterator it = begin;
40 
41  // process sign
42  bool negative = false;
43  if ('+' == *it)
44  {
45  ++it;
46  }
47  else if ('-' == *it)
48  {
49  ++it;
50  negative = true;
51  }
52  if (end == it)
53  return false;
54 
55  // calc max of abs value
56  std::uint64_t max;
57  if (negative)
58  max = static_cast <std::uint64_t> (
59  -(static_cast <std::int64_t> (std::numeric_limits <IntType>::min ())));
60  else
61  max = std::numeric_limits <IntType>::max ();
62 
63  // process digits
64  while (end != it)
65  {
66  std::uint64_t const digit = static_cast <IntType> (
67  s_digitTable [static_cast <unsigned int> (*it++)]);
68 
69  if (0xFF == digit)
70  return false;
71 
72  std::uint64_t const overflow = (max - digit) / 10;
73 
74  if (accum > overflow)
75  return false;
76 
77  accum = (10 * accum) + digit;
78  }
79 
80  if (negative)
81  {
82  result = -static_cast <IntType> (accum);
83  }
84  else
85  {
86  result = static_cast <IntType> (accum);
87  }
88 
89  return true;
90  }
91 
92  template <class IntType, class InputIterator>
93  static inline bool parseUnsigned (IntType& result, InputIterator begin, InputIterator end)
94  {
95  if (0 == std::distance (begin, end))
96  return false;
97 
98  std::uint64_t accum = 0;
99  InputIterator it = begin;
100  std::uint64_t const max = std::numeric_limits <IntType>::max ();
101 
102  // process digits
103  while (end != it)
104  {
105  std::uint64_t const digit = static_cast <IntType> (
106  s_digitTable [static_cast <unsigned int> (*it++)]);
107 
108  if (0xFF == digit)
109  return false;
110 
111  std::uint64_t const overflow = (max - digit) / 10;
112 
113  if (accum > overflow)
114  return false;
115 
116  accum = (10 * accum) + digit;
117  }
118 
119  result = static_cast <IntType> (accum);
120 
121  return true;
122  }
