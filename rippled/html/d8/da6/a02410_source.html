246  void sweep ()
247  {
248  clock_type::time_point const now (m_clock.now ());
249  clock_type::time_point when_expire;
250 
251  lock_guard lock (m_mutex);
252 
253  if (m_target_size == 0 ||
254  (m_map.size () <= m_target_size))
255  {
256  when_expire = now - m_target_age;
257  }
258  else
259  {
260  when_expire = now - clock_type::duration (
261  m_target_age.count() * m_target_size / m_map.size ());
262 
263  clock_type::duration const minimumAge (
264  std::chrono::seconds (1));
265  if (when_expire > (now - minimumAge))
266  when_expire = now - minimumAge;
267  }
268 
269  iterator it = m_map.begin ();
270 
271  while (it != m_map.end ())
272  {
273  if (it->second.last_access > now)
274  {
275  it->second.last_access = now;
276  ++it;
277  }
278  else if (it->second.last_access <= when_expire)
279  {
280  it = m_map.erase (it);
281  }
282  else
283  {
284  ++it;
285  }
286  }
