27 void ConsensusTransSetSF::gotNode (bool fromFilter, const SHAMapNode& id, uint256 const& nodeHash,
28  Blob& nodeData, SHAMapTreeNode::TNType type)
29 {
30  if (fromFilter)
31  return;
32 
33  m_nodeCache.insert (nodeHash, nodeData);
34 
35  if ((type == SHAMapTreeNode::tnTRANSACTION_NM) && (nodeData.size () > 16))
36  {
37  // this is a transaction, and we didn't have it
38  WriteLog (lsDEBUG, TransactionAcquire) << "Node on our acquiring TX set is TXN we may not have";
39 
40  try
41  {
42  Serializer s (nodeData.begin () + 4, nodeData.end ()); // skip prefix
43  SerializerIterator sit (s);
44  SerializedTransaction::pointer stx = boost::make_shared<SerializedTransaction> (boost::ref (sit));
45  assert (stx->getTransactionID () == nodeHash);
46  getApp().getJobQueue ().addJob (jtTRANSACTION, "TXS->TXN",
47  BIND_TYPE (&NetworkOPs::submitTransaction, &getApp().getOPs (), P_1, stx, NetworkOPs::stCallback ()));
48  }
49  catch (...)
50  {
51  WriteLog (lsWARNING, TransactionAcquire) << "Fetched invalid transaction in proposed set";
52  }
53  }
54 }

56 bool ConsensusTransSetSF::haveNode (const SHAMapNode& id, uint256 const& nodeHash,
57  Blob& nodeData)
58 {
59  if (m_nodeCache.retrieve (nodeHash, nodeData))
60  return true;
61 
62  // VFALCO TODO Use a dependency injection here
63  Transaction::pointer txn = getApp().getMasterTransaction().fetch(nodeHash, false);
64 
65  if (txn)
66  {
67  // this is a transaction, and we have it
68  WriteLog (lsDEBUG, TransactionAcquire) << "Node in our acquiring TX set is TXN we have";
69  Serializer s;
70  s.add32 (HashPrefix::transactionID);
71  txn->getSTransaction ()->add (s, true);
72  assert (s.getSHA512Half () == nodeHash);
73  nodeData = s.peekData ();
74  return true;
75  }
76 
77  return false;
78 }
