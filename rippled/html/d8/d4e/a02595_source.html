6 // Wrap an underlying iterator, but exclude any results not starting
7 // with a given prefix. Seeking to keys not beginning with the prefix
8 // is invalid, and SeekToLast is not implemented (that would be
9 // non-trivial), but otherwise this iterator will behave just like the
10 // underlying iterator would if there happened to be no non-matching
11 // keys in the dataset.

14 #include "../include/rocksdb/iterator.h"

16 namespace rocksdb {

18 class PrefixFilterIterator : public Iterator {
19  private:
20  Iterator* iter_;
21  const Slice &prefix_;
22  const SliceTransform *prefix_extractor_;
23  Status status_;
24 
25  public:
26  PrefixFilterIterator(Iterator* iter,
27  const Slice &prefix,
28  const SliceTransform* prefix_extractor)
29  : iter_(iter), prefix_(prefix),
30  prefix_extractor_(prefix_extractor),
31  status_(Status::OK()) {
32  if (prefix_extractor == nullptr) {
33  status_ = Status::InvalidArgument("A prefix filter may not be used "
34  "unless a function is also defined "
35  "for extracting prefixes");
36  } else if (!prefix_extractor_->InRange(prefix)) {
37  status_ = Status::InvalidArgument("Must provide a slice for prefix which"
38  "is a prefix for some key");
39  }
40  }
41  ~PrefixFilterIterator() {
42  delete iter_;
43  }
44  Slice key() const { return iter_->key(); }
45  Slice value() const { return iter_->value(); }
46  Status status() const {
47  if (!status_.ok()) {
48  return status_;
49  }
50  return iter_->status();
51  }
52  void Next() { iter_->Next(); }
53  void Prev() { iter_->Prev(); }
54  void Seek(const Slice& k) {
55  if (prefix_extractor_->Transform(k) == prefix_) {
56  iter_->Seek(k);
57  } else {
58  status_ = Status::InvalidArgument("Seek must begin with target prefix");
59  }
60  }
61  void SeekToFirst() {
62  Seek(prefix_);
63  }
64  void SeekToLast() {
65  status_ = Status::NotSupported("SeekToLast is incompatible with prefixes");
66  }
67  bool Valid() const {
68  return (status_.ok() && iter_->Valid() &&
69  prefix_extractor_->Transform(iter_->key()) == prefix_);
70  }
71 };
