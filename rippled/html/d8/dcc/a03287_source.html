23 Json::Value RPCHandler::doSubscribe (Json::Value params, Resource::Charge& loadType, Application::ScopedLockType& masterLockHolder)
24 {
25  // FIXME: This needs to release the master lock immediately
26  // Subscriptions need to be protected by their own lock
27 
28  InfoSub::pointer ispSub;
29  Json::Value jvResult (Json::objectValue);
30  std::uint32_t uLedgerIndex = params.isMember ("ledger_index") && params["ledger_index"].isNumeric ()
31  ? params["ledger_index"].asUInt ()
32  : 0;
33 
34  if (!mInfoSub && !params.isMember ("url"))
35  {
36  // Must be a JSON-RPC call.
37  WriteLog (lsINFO, RPCHandler) << boost::str (boost::format ("doSubscribe: RPC subscribe requires a url"));
38 
39  return rpcError (rpcINVALID_PARAMS);
40  }
41 
42  if (params.isMember ("url"))
43  {
44  if (mRole != Config::ADMIN)
45  return rpcError (rpcNO_PERMISSION);
46 
47  std::string strUrl = params["url"].asString ();
48  std::string strUsername = params.isMember ("url_username") ? params["url_username"].asString () : "";
49  std::string strPassword = params.isMember ("url_password") ? params["url_password"].asString () : "";
50 
51  // DEPRECATED
52  if (params.isMember ("username"))
53  strUsername = params["username"].asString ();
54 
55  // DEPRECATED
56  if (params.isMember ("password"))
57  strPassword = params["password"].asString ();
58 
59  ispSub = mNetOps->findRpcSub (strUrl);
60 
61  if (!ispSub)
62  {
63  WriteLog (lsDEBUG, RPCHandler) << boost::str (boost::format ("doSubscribe: building: %s") % strUrl);
64 
65  RPCSub::pointer rspSub = RPCSub::New (getApp ().getOPs (),
66  getApp ().getIOService (), getApp ().getJobQueue (),
67  strUrl, strUsername, strPassword);
68  ispSub = mNetOps->addRpcSub (strUrl, boost::dynamic_pointer_cast<InfoSub> (rspSub));
69  }
70  else
71  {
72  WriteLog (lsTRACE, RPCHandler) << boost::str (boost::format ("doSubscribe: reusing: %s") % strUrl);
73 
74  if (params.isMember ("username"))
75  dynamic_cast<RPCSub*> (&*ispSub)->setUsername (strUsername);
76 
77  if (params.isMember ("password"))
78  dynamic_cast<RPCSub*> (&*ispSub)->setPassword (strPassword);
79  }
80  }
81  else
82  {
83  ispSub = mInfoSub;
84  }
85 
86  if (!params.isMember ("streams"))
87  {
88  nothing ();
89  }
90  else if (!params["streams"].isArray ())
91  {
92  WriteLog (lsINFO, RPCHandler) << boost::str (boost::format ("doSubscribe: streams requires an array."));
93 
94  return rpcError (rpcINVALID_PARAMS);
95  }
96  else
97  {
98  for (Json::Value::iterator it = params["streams"].begin (); it != params["streams"].end (); it++)
99  {
100  if ((*it).isString ())
101  {
102  std::string streamName = (*it).asString ();
103 
104  if (streamName == "server")
105  {
106  mNetOps->subServer (ispSub, jvResult);
107  }
108  else if (streamName == "ledger")
109  {
110  mNetOps->subLedger (ispSub, jvResult);
111  }
112  else if (streamName == "transactions")
113  {
114  mNetOps->subTransactions (ispSub);
115  }
116  else if (streamName == "transactions_proposed"
117  || streamName == "rt_transactions") // DEPRECATED
118  {
119  mNetOps->subRTTransactions (ispSub);
120  }
121  else
122  {
123  jvResult["error"] = "unknownStream";
124  }
125  }
126  else
127  {
128  jvResult["error"] = "malformedStream";
129  }
130  }
131  }
132 
133  std::string strAccountsProposed = params.isMember ("accounts_proposed")
134  ? "accounts_proposed"
135  : "rt_accounts"; // DEPRECATED
136 
137  if (!params.isMember (strAccountsProposed))
138  {
139  nothing ();
140  }
141  else if (!params[strAccountsProposed].isArray ())
142  {
143  return rpcError (rpcINVALID_PARAMS);
144  }
145  else
146  {
147  boost::unordered_set<RippleAddress> usnaAccoundIds = RPC::parseAccountIds (params[strAccountsProposed]);
148 
149  if (usnaAccoundIds.empty ())
150  {
151  jvResult["error"] = "malformedAccount";
152  }
153  else
154  {
155  mNetOps->subAccount (ispSub, usnaAccoundIds, uLedgerIndex, true);
156  }
157  }
158 
159  if (!params.isMember ("accounts"))
160  {
161  nothing ();
162 
163  }
164  else if (!params["accounts"].isArray ())
165  {
166  return rpcError (rpcINVALID_PARAMS);
167  }
168  else
169  {
170  boost::unordered_set<RippleAddress> usnaAccoundIds = RPC::parseAccountIds (params["accounts"]);
171 
172  if (usnaAccoundIds.empty ())
173  {
174  jvResult["error"] = "malformedAccount";
175  }
176  else
177  {
178  mNetOps->subAccount (ispSub, usnaAccoundIds, uLedgerIndex, false);
179 
180  WriteLog (lsDEBUG, RPCHandler) << boost::str (boost::format ("doSubscribe: accounts: %d") % usnaAccoundIds.size ());
181  }
182  }
183 
184  bool bHaveMasterLock = true;
185  if (!params.isMember ("books"))
186  {
187  nothing ();
188  }
189  else if (!params["books"].isArray ())
190  {
191  return rpcError (rpcINVALID_PARAMS);
192  }
193  else
194  {
195  for (Json::Value::iterator it = params["books"].begin (); it != params["books"].end (); it++)
196  {
197  Json::Value& jvSubRequest = *it;
198 
199  if (!jvSubRequest.isObject ()
200  || !jvSubRequest.isMember ("taker_pays")
201  || !jvSubRequest.isMember ("taker_gets")
202  || !jvSubRequest["taker_pays"].isObject ()
203  || !jvSubRequest["taker_gets"].isObject ())
204  return rpcError (rpcINVALID_PARAMS);
205 
206  // VFALCO TODO Use RippleAsset here
207  RippleCurrency pay_currency;
208  RippleIssuer pay_issuer;
209  RippleCurrency get_currency;
210  RippleIssuer get_issuer;
211 
212  bool bBoth = (jvSubRequest.isMember ("both") && jvSubRequest["both"].asBool ())
213  || (jvSubRequest.isMember ("both_sides") && jvSubRequest["both_sides"].asBool ()); // DEPRECATED
214  bool bSnapshot = (jvSubRequest.isMember ("snapshot") && jvSubRequest["snapshot"].asBool ())
215  || (jvSubRequest.isMember ("state_now") && jvSubRequest["state_now"].asBool ()); // DEPRECATED
216 
217  Json::Value taker_pays = jvSubRequest["taker_pays"];
218  Json::Value taker_gets = jvSubRequest["taker_gets"];
219 
220  // Parse mandatory currency.
221  if (!taker_pays.isMember ("currency")
222  || !STAmount::currencyFromString (pay_currency, taker_pays["currency"].asString ()))
223  {
224  WriteLog (lsINFO, RPCHandler) << "Bad taker_pays currency.";
225 
226  return rpcError (rpcSRC_CUR_MALFORMED);
227  }
228  // Parse optional issuer.
229  else if (((taker_pays.isMember ("issuer"))
230  && (!taker_pays["issuer"].isString ()
231  || !STAmount::issuerFromString (pay_issuer, taker_pays["issuer"].asString ())))
232  // Don't allow illegal issuers.
233  || (!pay_currency != !pay_issuer)
234  || ACCOUNT_ONE == pay_issuer)
235  {
236  WriteLog (lsINFO, RPCHandler) << "Bad taker_pays issuer.";
237 
238  return rpcError (rpcSRC_ISR_MALFORMED);
239  }
240 
241  // Parse mandatory currency.
242  if (!taker_gets.isMember ("currency")
243  || !STAmount::currencyFromString (get_currency, taker_gets["currency"].asString ()))
244  {
245  WriteLog (lsINFO, RPCHandler) << "Bad taker_pays currency.";
246 
247  return rpcError (rpcSRC_CUR_MALFORMED);
248  }
249  // Parse optional issuer.
250  else if (((taker_gets.isMember ("issuer"))
251  && (!taker_gets["issuer"].isString ()
252  || !STAmount::issuerFromString (get_issuer, taker_gets["issuer"].asString ())))
253  // Don't allow illegal issuers.
254  || (!get_currency != !get_issuer)
255  || ACCOUNT_ONE == get_issuer)
256  {
257  WriteLog (lsINFO, RPCHandler) << "Bad taker_gets issuer.";
258 
259  return rpcError (rpcDST_ISR_MALFORMED);
260  }
261 
262  if (pay_currency == get_currency
263  && pay_issuer == get_issuer)
264  {
265  WriteLog (lsINFO, RPCHandler) << "taker_gets same as taker_pays.";
266 
267  return rpcError (rpcBAD_MARKET);
268  }
269 
270  RippleAddress raTakerID;
271 
272  if (!jvSubRequest.isMember ("taker"))
273  {
274  raTakerID.setAccountID (ACCOUNT_ONE);
275  }
276  else if (!raTakerID.setAccountID (jvSubRequest["taker"].asString ()))
277  {
278  return rpcError (rpcBAD_ISSUER);
279  }
280 
281  if (!Ledger::isValidBook (pay_currency, pay_issuer, get_currency, get_issuer))
282  {
283  WriteLog (lsWARNING, RPCHandler) << "Bad market: " <<
284  pay_currency << ":" << pay_issuer << " -> " <<
285  get_currency << ":" << get_issuer;
286  return rpcError (rpcBAD_MARKET);
287  }
288 
289  mNetOps->subBook (ispSub, pay_currency, get_currency, pay_issuer, get_issuer);
290 
291  if (bBoth) mNetOps->subBook (ispSub, get_currency, pay_currency, get_issuer, pay_issuer);
292 
293  if (bSnapshot)
294  {
295  if (bHaveMasterLock)
296  {
297  masterLockHolder.unlock ();
298  bHaveMasterLock = false;
299  }
300 
301  loadType = Resource::feeMediumBurdenRPC;
302  Ledger::pointer lpLedger = getApp().getLedgerMaster ().getPublishedLedger ();
303  if (lpLedger)
304  {
305  const Json::Value jvMarker = Json::Value (Json::nullValue);
306 
307  if (bBoth)
308  {
309  Json::Value jvBids (Json::objectValue);
310  Json::Value jvAsks (Json::objectValue);
311 
312  mNetOps->getBookPage (lpLedger, pay_currency, pay_issuer, get_currency, get_issuer, raTakerID.getAccountID (), false, 0, jvMarker, jvBids);
313 
314  if (jvBids.isMember ("offers")) jvResult["bids"] = jvBids["offers"];
315 
316  mNetOps->getBookPage (lpLedger, get_currency, get_issuer, pay_currency, pay_issuer, raTakerID.getAccountID (), false, 0, jvMarker, jvAsks);
317 
318  if (jvAsks.isMember ("offers")) jvResult["asks"] = jvAsks["offers"];
319  }
320  else
321  {
322  mNetOps->getBookPage (lpLedger, pay_currency, pay_issuer, get_currency, get_issuer, raTakerID.getAccountID (), false, 0, jvMarker, jvResult);
323  }
324  }
325  }
326  }
327  }
328 
329  return jvResult;
330 }
