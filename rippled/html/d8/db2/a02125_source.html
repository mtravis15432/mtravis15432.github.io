133 String getFileNameFromPath (const char* sourceFileName, int numberOfParents)
134 {
135  String fullPath (sourceFileName);
136 
137 #if BEAST_WINDOWS
138  // Convert everything to forward slash
139  fullPath = fullPath.replaceCharacter ('\\', '/');
140 #endif
141 
142  String path;
143 
144  int chopPoint = fullPath.lastIndexOfChar ('/');
145  path = fullPath.substring (chopPoint + 1);
146 
147  while (chopPoint >= 0 && numberOfParents > 0)
148  {
149  --numberOfParents;
150  fullPath = fullPath.substring (0, chopPoint);
151  chopPoint = fullPath.lastIndexOfChar ('/');
152  path = fullPath.substring (chopPoint + 1) + '/' + path;
153  }
154 
155  return path;
