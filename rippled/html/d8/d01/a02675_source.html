52 void FilterBlockBuilder::AddKey(const Slice& key) {
53  // get slice for most recently added entry
54  Slice prev;
55  size_t added_to_start = 0;
56 
57  // add key to filter if needed
58  if (whole_key_filtering_) {
59  start_.push_back(entries_.size());
60  ++added_to_start;
61  entries_.append(key.data(), key.size());
62  }
63 
64  if (start_.size() > added_to_start) {
65  size_t prev_start = start_[start_.size() - 1 - added_to_start];
66  const char* base = entries_.data() + prev_start;
67  size_t length = entries_.size() - prev_start;
68  prev = Slice(base, length);
69  }
70 
71  // add prefix to filter if needed
72  if (prefix_extractor_ && prefix_extractor_->InDomain(ExtractUserKey(key))) {
73  // If prefix_extractor_, this filter_block layer assumes we only
74  // operate on internal keys.
75  Slice user_key = ExtractUserKey(key);
76  // this assumes prefix(prefix(key)) == prefix(key), as the last
77  // entry in entries_ may be either a key or prefix, and we use
78  // prefix(last entry) to get the prefix of the last key.
79  if (prev.size() == 0 ||
80  !SamePrefix(user_key, ExtractUserKey(prev))) {
81  Slice prefix = prefix_extractor_->Transform(user_key);
82  InternalKey internal_prefix_tmp(prefix, 0, kTypeValue);
83  Slice internal_prefix = internal_prefix_tmp.Encode();
84  assert(comparator_->Compare(internal_prefix, key) <= 0);
85  start_.push_back(entries_.size());
86  entries_.append(internal_prefix.data(), internal_prefix.size());
87  }
88  }
89 }

91 Slice FilterBlockBuilder::Finish() {
92  if (!start_.empty()) {
93  GenerateFilter();
94  }
95 
96  // Append array of per-filter offsets
97  const uint32_t array_offset = result_.size();
98  for (size_t i = 0; i < filter_offsets_.size(); i++) {
99  PutFixed32(&result_, filter_offsets_[i]);
100  }
101 
102  PutFixed32(&result_, array_offset);
103  result_.push_back(kFilterBaseLg); // Save encoding parameter in result
104  return Slice(result_);
105 }

107 void FilterBlockBuilder::GenerateFilter() {
108  const size_t num_entries = start_.size();
109  if (num_entries == 0) {
110  // Fast path if there are no keys for this filter
111  filter_offsets_.push_back(result_.size());
112  return;
113  }
114 
115  // Make list of keys from flattened key structure
116  start_.push_back(entries_.size()); // Simplify length computation
117  tmp_entries_.resize(num_entries);
118  for (size_t i = 0; i < num_entries; i++) {
119  const char* base = entries_.data() + start_[i];
120  size_t length = start_[i+1] - start_[i];
121  tmp_entries_[i] = Slice(base, length);
122  }
123 
124  // Generate filter for current set of keys and append to result_.
125  filter_offsets_.push_back(result_.size());
126  policy_->CreateFilter(&tmp_entries_[0], num_entries, &result_);
127 
128  tmp_entries_.clear();
129  entries_.clear();
130  start_.clear();
131 }

133 FilterBlockReader::FilterBlockReader(
134  const Options& opt, const Slice& contents, bool delete_contents_after_use)
135  : policy_(opt.filter_policy),
136  prefix_extractor_(opt.prefix_extractor),
137  whole_key_filtering_(opt.whole_key_filtering),
138  data_(nullptr),
139  offset_(nullptr),
140  num_(0),
141  base_lg_(0) {
142  size_t n = contents.size();
143  if (n < 5) return; // 1 byte for base_lg_ and 4 for start of offset array
144  base_lg_ = contents[n-1];
145  uint32_t last_word = DecodeFixed32(contents.data() + n - 5);
146  if (last_word > n - 5) return;
147  data_ = contents.data();
148  offset_ = data_ + last_word;
149  num_ = (n - 5 - last_word) / 4;
150  if (delete_contents_after_use) {
151  filter_data.reset(contents.data());
152  }
153 }

171 bool FilterBlockReader::MayMatch(uint64_t block_offset, const Slice& entry) {
172  uint64_t index = block_offset >> base_lg_;
173  if (index < num_) {
174  uint32_t start = DecodeFixed32(offset_ + index*4);
175  uint32_t limit = DecodeFixed32(offset_ + index*4 + 4);
176  if (start <= limit && limit <= (offset_ - data_)) {
177  Slice filter = Slice(data_ + start, limit - start);
178  return policy_->KeyMayMatch(entry, filter);
179  } else if (start == limit) {
180  // Empty filters do not match any entries
181  return false;
182  }
183  }
184  return true; // Errors are treated as potential matches
185 }
