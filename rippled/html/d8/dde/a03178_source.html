74 SField::ref SField::getField (int code)
75 {
76  int type = code >> 16;
77  int field = code % 0xffff;
78 
79  if ((type <= 0) || (field <= 0))
80  return sfInvalid;
81 
82  StaticScopedLockType sl (getMutex ());
83 
84  std::map<int, SField::ptr>::iterator it = codeToField.find (code);
85 
86  if (it != codeToField.end ())
87  return * (it->second);
88 
89  if (field > 255) // don't dynamically extend types that have no binary encoding
90  return sfInvalid;
91 
92  switch (type)
93  {
94  // types we are willing to dynamically extend
95 
96 #define FIELD(name, type, index)
97 #define TYPE(name, type, index) case STI_##type:
98 #include "../protocol/SerializeDeclarations.h"
99 #undef FIELD
100 #undef TYPE
101 
102  break;
103 
104  default:
105  return sfInvalid;
106  }
107 
108  return * (new SField (static_cast<SerializedTypeID> (type), field));

111 int SField::compare (SField::ref f1, SField::ref f2)
112 {
113  // -1 = f1 comes before f2, 0 = illegal combination, 1 = f1 comes after f2
114  if ((f1.fieldCode <= 0) || (f2.fieldCode <= 0))
115  return 0;
116 
117  if (f1.fieldCode < f2.fieldCode)
118  return -1;
119 
120  if (f2.fieldCode < f1.fieldCode)
121  return 1;
122 
123  return 0;
