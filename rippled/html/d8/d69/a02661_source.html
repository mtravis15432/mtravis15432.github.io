171 inline bool Zlib_Compress(const CompressionOptions& opts, const char* input,
172  size_t length, ::std::string* output) {
173 #ifdef ZLIB
174  // The memLevel parameter specifies how much memory should be allocated for
175  // the internal compression state.
176  // memLevel=1 uses minimum memory but is slow and reduces compression ratio.
177  // memLevel=9 uses maximum memory for optimal speed.
178  // The default value is 8. See zconf.h for more details.
179  static const int memLevel = 8;
180  z_stream _stream;
181  memset(&_stream, 0, sizeof(z_stream));
182  int st = deflateInit2(&_stream, opts.level, Z_DEFLATED, opts.window_bits,
183  memLevel, opts.strategy);
184  if (st != Z_OK) {
185  return false;
186  }
187 
188  // Resize output to be the plain data length.
189  // This may not be big enough if the compression actually expands data.
190  output->resize(length);
191 
192  // Compress the input, and put compressed data in output.
193  _stream.next_in = (Bytef *)input;
194  _stream.avail_in = length;
195 
196  // Initialize the output size.
197  _stream.avail_out = length;
198  _stream.next_out = (Bytef *)&(*output)[0];
199 
200  int old_sz =0, new_sz =0, new_sz_delta =0;
201  bool done = false;
202  while (!done) {
203  int st = deflate(&_stream, Z_FINISH);
204  switch (st) {
205  case Z_STREAM_END:
206  done = true;
207  break;
208  case Z_OK:
209  // No output space. Increase the output space by 20%.
210  // (Should we fail the compression since it expands the size?)
211  old_sz = output->size();
212  new_sz_delta = (int)(output->size() * 0.2);
213  new_sz = output->size() + (new_sz_delta < 10 ? 10 : new_sz_delta);
214  output->resize(new_sz);
215  // Set more output.
216  _stream.next_out = (Bytef *)&(*output)[old_sz];
217  _stream.avail_out = new_sz - old_sz;
218  break;
219  case Z_BUF_ERROR:
220  default:
221  deflateEnd(&_stream);
222  return false;
223  }
224  }
225 
226  output->resize(output->size() - _stream.avail_out);
227  deflateEnd(&_stream);
228  return true;
229 #endif
230  return false;
231 }

233 inline char* Zlib_Uncompress(const char* input_data, size_t input_length,
234  int* decompress_size, int windowBits = -14) {
235 #ifdef ZLIB
236  z_stream _stream;
237  memset(&_stream, 0, sizeof(z_stream));
238 
239  // For raw inflate, the windowBits should be -8..-15.
240  // If windowBits is bigger than zero, it will use either zlib
241  // header or gzip header. Adding 32 to it will do automatic detection.
242  int st = inflateInit2(&_stream,
243  windowBits > 0 ? windowBits + 32 : windowBits);
244  if (st != Z_OK) {
245  return nullptr;
246  }
247 
248  _stream.next_in = (Bytef *)input_data;
249  _stream.avail_in = input_length;
250 
251  // Assume the decompressed data size will 5x of compressed size.
252  int output_len = input_length * 5;
253  char* output = new char[output_len];
254  int old_sz = output_len;
255 
256  _stream.next_out = (Bytef *)output;
257  _stream.avail_out = output_len;
258 
259  char* tmp = nullptr;
260  int output_len_delta;
261  bool done = false;
262 
263  //while(_stream.next_in != nullptr && _stream.avail_in != 0) {
264  while (!done) {
265  int st = inflate(&_stream, Z_SYNC_FLUSH);
266  switch (st) {
267  case Z_STREAM_END:
268  done = true;
269  break;
270  case Z_OK:
271  // No output space. Increase the output space by 20%.
272  old_sz = output_len;
273  output_len_delta = (int)(output_len * 0.2);
274  output_len += output_len_delta < 10 ? 10 : output_len_delta;
275  tmp = new char[output_len];
276  memcpy(tmp, output, old_sz);
277  delete[] output;
278  output = tmp;
279 
280  // Set more output.
281  _stream.next_out = (Bytef *)(output + old_sz);
282  _stream.avail_out = output_len - old_sz;
283  break;
284  case Z_BUF_ERROR:
285  default:
286  delete[] output;
287  inflateEnd(&_stream);
288  return nullptr;
289  }
290  }
291 
292  *decompress_size = output_len - _stream.avail_out;
293  inflateEnd(&_stream);
294  return output;
295 #endif
296 
297  return nullptr;
298 }

300 inline bool BZip2_Compress(const CompressionOptions& opts, const char* input,
301  size_t length, ::std::string* output) {
302 #ifdef BZIP2
303  bz_stream _stream;
304  memset(&_stream, 0, sizeof(bz_stream));
305 
306  // Block size 1 is 100K.
307  // 0 is for silent.
308  // 30 is the default workFactor
309  int st = BZ2_bzCompressInit(&_stream, 1, 0, 30);
310  if (st != BZ_OK) {
311  return false;
312  }
313 
314  // Resize output to be the plain data length.
315  // This may not be big enough if the compression actually expands data.
316  output->resize(length);
317 
318  // Compress the input, and put compressed data in output.
319  _stream.next_in = (char *)input;
320  _stream.avail_in = length;
321 
322  // Initialize the output size.
323  _stream.next_out = (char *)&(*output)[0];
324  _stream.avail_out = length;
325 
326  int old_sz =0, new_sz =0;
327  while(_stream.next_in != nullptr && _stream.avail_in != 0) {
328  int st = BZ2_bzCompress(&_stream, BZ_FINISH);
329  switch (st) {
330  case BZ_STREAM_END:
331  break;
332  case BZ_FINISH_OK:
333  // No output space. Increase the output space by 20%.
334  // (Should we fail the compression since it expands the size?)
335  old_sz = output->size();
336  new_sz = (int)(output->size() * 1.2);
337  output->resize(new_sz);
338  // Set more output.
339  _stream.next_out = (char *)&(*output)[old_sz];
340  _stream.avail_out = new_sz - old_sz;
341  break;
342  case BZ_SEQUENCE_ERROR:
343  default:
344  BZ2_bzCompressEnd(&_stream);
345  return false;
346  }
347  }
348 
349  output->resize(output->size() - _stream.avail_out);
350  BZ2_bzCompressEnd(&_stream);
351  return true;
352  return output;
353 #endif
354  return false;
355 }
