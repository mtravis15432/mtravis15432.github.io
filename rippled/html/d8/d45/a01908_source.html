115 inline std::uint16_t ByteOrder::swap (std::uint16_t n)
116 {
117  #if BEAST_USE_INTRINSICSxxx // agh - the MS compiler has an internal error when you try to use this intrinsic!
118  return static_cast <std::uint16_t> (_byteswap_ushort (n));
119  #else
120  return static_cast <std::uint16_t> ((n << 8) | (n >> 8));
121  #endif
122 }

124 inline std::uint32_t ByteOrder::swap (std::uint32_t n)
125 {
126  #if BEAST_MAC || BEAST_IOS
127  return OSSwapInt32 (n);
128  #elif BEAST_GCC && BEAST_INTEL && ! BEAST_NO_INLINE_ASM
129  asm("bswap %%eax" : "=a"(n) : "a"(n));
130  return n;
131  #elif BEAST_USE_INTRINSICS
132  return _byteswap_ulong (n);
133  #elif BEAST_MSVC && ! BEAST_NO_INLINE_ASM
134  __asm {
135  mov eax, n
136  bswap eax
137  mov n, eax
138  }
139  return n;
140  #elif BEAST_ANDROID
141  return bswap_32 (n);
142  #else
143  return (n << 24) | (n >> 24) | ((n & 0xff00) << 8) | ((n & 0xff0000) >> 8);
144  #endif
145 }

147 inline std::uint64_t ByteOrder::swap (std::uint64_t value)
148 {
149  #if BEAST_MAC || BEAST_IOS
150  return OSSwapInt64 (value);
151  #elif BEAST_USE_INTRINSICS
152  return _byteswap_uint64 (value);
