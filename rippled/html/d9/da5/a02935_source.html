1088 bool ApplicationImp::loadOldLedger (const std::string& l, bool bReplay)

1091  {

1093 
1094  if (l.empty () || (l == "latest"))
1095  loadLedger = Ledger::getLastFullLedger ();
1096  else if (l.length () == 64)
1097  {

1099  uint256 hash;

1102  }
1103  else // assume by sequence
1104  loadLedger = Ledger::loadByIndex (beast::lexicalCastThrow <std::uint32_t> (l));
1105 
1106  if (!loadLedger)
1107  {
1108  m_journal.fatal << "No Ledger found?" << std::endl;
1109  return false;
1110  }
1111 
1112  if (bReplay)
1113  { // Replay a ledger close with same prior ledger and transactions
1114  replayLedger = loadLedger; // this ledger holds the transactions we want to replay
1115  loadLedger = Ledger::loadByIndex (replayLedger->getLedgerSeq() - 1); // this is the prior ledger
1116  if (!loadLedger || (replayLedger->getParentHash() != loadLedger->getHash()))
1117  {
1118  m_journal.fatal << "Replay ledger missing/damaged";
1119  assert (false);
1120  return false;
1121  }
1122  }

1124  loadLedger->setClosed ();
1125 
1126  m_journal.info << "Loading ledger " << loadLedger->getHash () << " seq:" << loadLedger->getLedgerSeq ();
1127

1129  {
1130  m_journal.fatal << "Ledger is empty.";
1131  assert (false);
1132  return false;
1133  }
1134 
1135  if (!loadLedger->walkLedger ())
1136  {
1137  m_journal.fatal << "Ledger is missing nodes.";
1138  return false;
1139  }
1140

1143  m_journal.fatal << "Ledger is not sane.";
1144  return false;
1145  }
1146

1148

1150  m_ledgerMaster->switchLedgers (loadLedger, openLedger);
1151  m_ledgerMaster->forceValid(loadLedger);

1158

1160  {
1161  Transaction::pointer txn = replayLedger->getTransaction(it->getTag());
1162  m_journal.info << txn->getJson(0);
1163  Serializer s;
1164  txn->getSTransaction()->add(s);
1165  if (!cur->addTransaction(it->getTag(), s))
1166  {
1167  m_journal.warning << "Unable to add transaction " << it->getTag();

1169  }
1170  }
1171  }
1172  catch (SHAMapMissingNode&)

1174  m_journal.fatal << "Data is missing for selected ledger";

1176  }
1177  catch (boost::bad_lexical_cast&)

1259 static void addTxnSeqField ()
1260 {

1262  return;
1263 
1264  Log (lsWARNING) << "Transaction sequence field is missing";
1265 
1266  Database* db = getApp().getTxnDB ()->getDB ();
1267 
1268  std::vector< std::pair<uint256, int> > txIDs;
1269  txIDs.reserve (300000);
1270 
1271  Log (lsINFO) << "Parsing transactions";
1272  int i = 0;
1273  uint256 transID;
1274  SQL_FOREACH (db, "SELECT TransID,TxnMeta FROM Transactions;")
1275  {
1276  Blob rawMeta;
1277  int metaSize = 2048;
1278  rawMeta.resize (metaSize);
1279  metaSize = db->getBinary ("TxnMeta", &*rawMeta.begin (), rawMeta.size ());
1280 
1281  if (metaSize > static_cast<int> (rawMeta.size ()))
1282  {
1283  rawMeta.resize (metaSize);
1284  db->getBinary ("TxnMeta", &*rawMeta.begin (), rawMeta.size ());
1285  }
1286  else rawMeta.resize (metaSize);
1287 
1288  std::string tid;
1289  db->getStr ("TransID", tid);
1290  transID.SetHex (tid, true);
1291 
1292  if (rawMeta.size () == 0)
1293  {
1294  txIDs.push_back (std::make_pair (transID, -1));
1295  Log (lsINFO) << "No metadata for " << transID;
1296  }
1297  else
1298  {
1299  TransactionMetaSet m (transID, 0, rawMeta);
1300  txIDs.push_back (std::make_pair (transID, m.getIndex ()));
1301  }
1302 
1303  if ((++i % 1000) == 0)
1304  Log (lsINFO) << i << " transactions read";
1305  }
1306 
1307  Log (lsINFO) << "All " << i << " transactions read";
1308 
1309  db->executeSQL ("BEGIN TRANSACTION;");
1310 
1311  Log (lsINFO) << "Dropping old index";
1312  db->executeSQL ("DROP INDEX AcctTxIndex;");
1313 
1314  Log (lsINFO) << "Altering table";
1315  db->executeSQL ("ALTER TABLE AccountTransactions ADD COLUMN TxnSeq INTEGER;");
1316 
1317  typedef std::pair<uint256, int> u256_int_pair_t;
1318  boost::format fmt ("UPDATE AccountTransactions SET TxnSeq = %d WHERE TransID = '%s';");
1319  i = 0;
1320  BOOST_FOREACH (u256_int_pair_t & t, txIDs)
1321  {
1322  db->executeSQL (boost::str (fmt % t.second % t.first.GetHex ()));
1323 
1324  if ((++i % 1000) == 0)
1325  Log (lsINFO) << i << " transactions updated";
1326  }
1327 
1328  Log (lsINFO) << "Building new index";
