331 {
332  Blob data = u.getVL ();
333  Blob ::iterator begin = data.begin ();
334 
335  std::unique_ptr<STVector256> vec (new STVector256 (name));
336 
337  int count = data.size () / (256 / 8);

340  unsigned int uStart = 0;
341

343  {

345 
346  // This next line could be optimized to construct a default uint256 in the vector and then copy into it

348  uStart = uEnd;
349  }
350 
351  return vec.release ();
352 }

421 STPathSet* STPathSet::construct (SerializerIterator& s, SField::ref name)
422 {
423  std::vector<STPath> paths;
424  std::vector<STPathElement> path;
425 
426  do
427  {
428  int iType = s.get8 ();
429 
430  if (iType == STPathElement::typeEnd || iType == STPathElement::typeBoundary)
431  {
432  if (path.empty ())
433  {
434  WriteLog (lsINFO, SerializedType) << "STPathSet: Empty path.";
435 
436  throw std::runtime_error ("empty path");
437  }
438 
439  paths.push_back (path);
440  path.clear ();
441 
442  if (iType == STPathElement::typeEnd)
443  {
444  return new STPathSet (name, paths);
445  }
446  }
447  else if (iType & ~STPathElement::typeValidBits)
448  {
449  WriteLog (lsINFO, SerializedType) << "STPathSet: Bad path element: " << iType;
450 
451  throw std::runtime_error ("bad path element");
452  }
453  else
454  {
455  const bool bAccount = !! (iType & STPathElement::typeAccount);
456  const bool bCurrency = !! (iType & STPathElement::typeCurrency);
457  const bool bIssuer = !! (iType & STPathElement::typeIssuer);
458 
459  uint160 uAccountID;
460  uint160 uCurrency;
461  uint160 uIssuerID;
462 
463  if (bAccount)
464  uAccountID = s.get160 ();
465 
466  if (bCurrency)
467  uCurrency = s.get160 ();
468 
469  if (bIssuer)
470  uIssuerID = s.get160 ();
471 
472  path.push_back (STPathElement (uAccountID, uCurrency, uIssuerID, bCurrency));
473  }
474  }
475  while (1);
476 }
