36 class Atomic_test : public unit_test::suite

39  template <typename Type>
40  void testFloat ()
41  {
42  Atomic<Type> a, b;
43  a = (Type) 21;
44  memoryBarrier();
45 
46  /* These are some simple test cases to check the atomics - let me know
47  if any of these assertions fail on your system!
48  */
49  expect (a.get() == (Type) 21);
50  expect (a.compareAndSetValue ((Type) 100, (Type) 50) == (Type) 21);
51  expect (a.get() == (Type) 21);
52  expect (a.compareAndSetValue ((Type) 101, a.get()) == (Type) 21);
53  expect (a.get() == (Type) 101);
54  expect (! a.compareAndSetBool ((Type) 300, (Type) 200));
55  expect (a.get() == (Type) 101);
56  expect (a.compareAndSetBool ((Type) 200, a.get()));
57  expect (a.get() == (Type) 200);
58 
59  expect (a.exchange ((Type) 300) == (Type) 200);
60  expect (a.get() == (Type) 300);
61 
62  b = a;
63  expect (b.get() == a.get());
64  }

66  template <typename Type>
67  void testInteger ()
68  {
69  Atomic<Type> a, b;
70  a.set ((Type) 10);
71  expect (a.value == (Type) 10);
72  expect (a.get() == (Type) 10);
73  a += (Type) 15;
74  expect (a.get() == (Type) 25);
75  memoryBarrier();
76  a -= (Type) 5;
77  expect (a.get() == (Type) 20);

79  ++a;
80  expect (--a == (Type) 21);

82  memoryBarrier();
