41 void AutoRollLogger::Logv(const char* format, va_list ap) {
42  assert(GetStatus().ok());
43 
44  std::shared_ptr<Logger> logger;
45  {
46  MutexLock l(&mutex_);
47  if ((kLogFileTimeToRoll > 0 && LogExpired()) ||
48  (kMaxLogFileSize > 0 && logger_->GetLogFileSize() >= kMaxLogFileSize)) {
49  RollLogFile();
50  ResetLogger();
51  }
52 
53  // pin down the current logger_ instance before releasing the mutex.
54  logger = logger_;
55  }
56 
57  // Another thread could have put a new Logger instance into logger_ by now.
58  // However, since logger is still hanging on to the previous instance
59  // (reference count is not zero), we don't have to worry about it being
60  // deleted while we are accessing it.
61  // Note that logv itself is not mutex protected to allow maximum concurrency,
62  // as thread safety should have been handled by the underlying logger.
63  logger->Logv(format, ap);
64 }

66 bool AutoRollLogger::LogExpired() {
67  if (cached_now_access_count >= call_NowMicros_every_N_records_) {
68  cached_now = static_cast<uint64_t>(env_->NowMicros() * 1e-6);
69  cached_now_access_count = 0;
70  }
71 
72  ++cached_now_access_count;
73  return cached_now >= ctime_ + kLogFileTimeToRoll;
74 }
