template<typename Value, class SharedMutexType = SharedMutexAdapter <RecursiveMutex>>
class beast::SharedData< Value, SharedMutexType >::UnlockedAccess

Provides access to the contents of a SharedData. This acquires a shared lock on the underlying mutex.
