118  static string HexToString(const string& str) {
119  string parsed;
120  if (str[0] != '0' || str[1] != 'x') {
121  fprintf(stderr, "Invalid hex input %s. Must start with 0x\n",
122  str.c_str());
123  throw "Invalid hex input";
124  }
125 
126  for (unsigned int i = 2; i < str.length();) {
127  int c;
128  sscanf(str.c_str() + i, "%2X", &c);
129  parsed.push_back(c);
130  i += 2;
131  }
132  return parsed;
133  }

135  static string StringToHex(const string& str) {
136  string result = "0x";

211  void OpenDB() {
212  Options opt = PrepareOptionsForOpenDB();
213  if (!exec_state_.IsNotStarted()) {
214  return;
215  }
216  // Open the DB.
217  Status st;
218  if (is_db_ttl_) {
219  if (is_read_only_) {
220  st = UtilityDB::OpenTtlDB(opt, db_path_, &sdb_, 0, true);
221  } else {
222  st = UtilityDB::OpenTtlDB(opt, db_path_, &sdb_);
223  }
224  db_ = sdb_;
225  } else if (is_read_only_) {
226  st = DB::OpenForReadOnly(opt, db_path_, &db_);
227  } else {
228  st = DB::Open(opt, db_path_, &db_);
229  }
230  if (!st.ok()) {
231  string msg = st.ToString();
232  exec_state_ = LDBCommandExecuteResult::FAILED(msg);
233  }
234 
235  options_ = opt;
236  }
