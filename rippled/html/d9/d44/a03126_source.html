37  // Scale using load as well as base rate
38  std::uint64_t scaleFeeLoad (std::uint64_t fee, std::uint64_t baseFee, std::uint32_t referenceFeeUnits, bool bAdmin)
39  {
40  static std::uint64_t midrange (0x00000000FFFFFFFF);
41 
42  bool big = (fee > midrange);
43 
44  if (big) // big fee, divide first to avoid overflow
45  fee /= baseFee;
46  else // normal fee, multiply first for accuracy
47  fee *= referenceFeeUnits;
48 
49  std::uint32_t feeFactor = std::max (mLocalTxnLoadFee, mRemoteTxnLoadFee);
50 
51  // Let admins pay the normal fee until the local load exceeds four times the remote
52  std::uint32_t uRemFee = std::max(mRemoteTxnLoadFee, mClusterTxnLoadFee);
53  if (bAdmin && (feeFactor > uRemFee) && (feeFactor < (4 * uRemFee)))
54  feeFactor = uRemFee;
55 
56  {
57  ScopedLockType sl (mLock);
58  fee = mulDiv (fee, feeFactor, lftNormalFee);
59  }
60 
61  if (big) // Fee was big to start, must now multiply
62  fee *= referenceFeeUnits;
63  else // Fee was small to start, mst now divide
64  fee /= baseFee;
65 
66  return fee;
67  }

140  bool raiseLocalFee ()
141  {
142  ScopedLockType sl (mLock);
143 
144  if (++raiseCount < 2)
145  return false;
146 
147  std::uint32_t origFee = mLocalTxnLoadFee;
148 
149  if (mLocalTxnLoadFee < mRemoteTxnLoadFee) // make sure this fee takes effect
150  mLocalTxnLoadFee = mRemoteTxnLoadFee;
151 
152  mLocalTxnLoadFee += (mLocalTxnLoadFee / lftFeeIncFraction); // increment by 1/16th
153 
154  if (mLocalTxnLoadFee > lftFeeMax)
155  mLocalTxnLoadFee = lftFeeMax;
156 
157  if (origFee == mLocalTxnLoadFee)
158  return false;
159 
160  m_journal.debug << "Local load fee raised from " << origFee << " to " << mLocalTxnLoadFee;
161  return true;
162  }

164  bool lowerLocalFee ()
165  {
166  ScopedLockType sl (mLock);
167  std::uint32_t origFee = mLocalTxnLoadFee;
168  raiseCount = 0;
169 
170  mLocalTxnLoadFee -= (mLocalTxnLoadFee / lftFeeDecFraction ); // reduce by 1/4
171 
172  if (mLocalTxnLoadFee < lftNormalFee)
173  mLocalTxnLoadFee = lftNormalFee;
174 
175  if (origFee == mLocalTxnLoadFee)
176  return false;
177 
178  m_journal.debug << "Local load fee lowered from " << origFee << " to " << mLocalTxnLoadFee;
179  return true;
