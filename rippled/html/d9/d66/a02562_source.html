496 // Helper for quickly generating random data.
497 class RandomGenerator {
498  private:
499  std::string data_;
500  unsigned int pos_;
501 
502  public:
503  RandomGenerator() {
504  // We use a limited amount of data over and over again and ensure
505  // that it is larger than the compression window (32KB), and also
506  // large enough to serve all typical value sizes we want to write.
507  Random rnd(301);
508  std::string piece;
509  while (data_.size() < (unsigned)std::max(1048576, FLAGS_value_size)) {
510  // Add a short fragment that is as compressible as specified
511  // by FLAGS_compression_ratio.
512  test::CompressibleString(&rnd, FLAGS_compression_ratio, 100, &piece);
513  data_.append(piece);
514  }
515  pos_ = 0;
516  }
517 
518  Slice Generate(unsigned int len) {
519  if (pos_ + len > data_.size()) {
520  pos_ = 0;
521  assert(len < data_.size());
522  }
523  pos_ += len;
524  return Slice(data_.data() + pos_ - len, len);
525  }
526 };

1507  const int test_duration = write_mode == RANDOM ? FLAGS_duration : 0;
1508  const int num_ops = writes_ == 0 ? num_ : writes_ ;
1509  Duration duration(test_duration, num_ops);

1511

1514  }
1515

1553  }

1557  found = true;

1559  break;
1560  }
1561  }
1562  }
1563  }

1565  break;
1566  }

1574  if (!s.ok()) {
1575  fprintf(stderr, "put error: %s\n", s.ToString().c_str());
1576  exit(1);
1577  }
1578  i += entries_per_batch_;
1579  }
1580  thread->stats.AddBytes(bytes);
