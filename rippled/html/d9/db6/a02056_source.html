118  ScopedPointer& operator= (ScopedPointer& objectToTransferFrom)
119  {
120  if (this != objectToTransferFrom.getAddress())
121  {
122  // Two ScopedPointers should never be able to refer to the same object - if
123  // this happens, you must have done something dodgy!
124  bassert (object == nullptr || object != objectToTransferFrom.object);
125 
126  ObjectType* const oldObject = object;
127  object = objectToTransferFrom.object;

130  }
