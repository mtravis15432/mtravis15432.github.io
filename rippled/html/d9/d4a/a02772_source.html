145 void VectorRep::Iterator::DoSort() const {
146  // vrep is non-null means that we are working on an immutable memtable
147  if (!sorted_ && vrep_ != nullptr) {
148  WriteLock l(&vrep_->rwlock_);
149  if (!vrep_->sorted_) {
150  std::sort(bucket_->begin(), bucket_->end(), Compare(compare_));
151  cit_ = bucket_->begin();
152  vrep_->sorted_ = true;
153  }
154  sorted_ = true;
155  }
156  if (!sorted_) {
157  std::sort(bucket_->begin(), bucket_->end(), Compare(compare_));
158  cit_ = bucket_->begin();
159  sorted_ = true;
160  }
161  assert(sorted_);
162  assert(vrep_ == nullptr || vrep_->sorted_);
163 }

188 // Advances to the previous position.
189 // REQUIRES: Valid()
190 void VectorRep::Iterator::Prev() {
191  assert(Valid());
192  if (cit_ == bucket_->begin()) {
193  // If you try to go back from the first element, the iterator should be
194  // invalidated. So we set it to past-the-end. This means that you can
195  // treat the container circularly.
196  cit_ = bucket_->end();
197  } else {
198  --cit_;
199  }
200 }

202 // Advance to the first entry with a key >= target
203 void VectorRep::Iterator::Seek(const char* target) {
204  DoSort();
205  // Do binary search to find first value not less than the target
206  cit_ = std::equal_range(bucket_->begin(),
207  bucket_->end(),
208  target,
209  [this] (const char* a, const char* b) {
210  return compare_(a, b) < 0;
211  }).first;
212 }

231 std::shared_ptr<MemTableRep::Iterator> VectorRep::GetIterator() {
232  ReadLock l(&rwlock_);
233  // Do not sort here. The sorting would be done the first time
234  // a Seek is performed on the iterator.
235  if (immutable_) {
236  return std::make_shared<Iterator>(this, bucket_, compare_);
237  } else {
238  std::shared_ptr<Bucket> tmp;
239  tmp.reset(new Bucket(*bucket_)); // make a copy
240  return std::make_shared<Iterator>(nullptr, tmp, compare_);
241  }
