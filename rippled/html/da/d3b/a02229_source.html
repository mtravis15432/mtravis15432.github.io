31 bool SemanticVersion::parse (String input)
32 {
33  // May not have leading or trailing whitespace
34  if (input.trim () != input)
35  return false;
36 
37  // Must have major version number
38  if (! chopUInt (&majorVersion, std::numeric_limits <int>::max (), input))
39  return false;
40 
41  if (! chop (".", input))
42  return false;
43 
44  // Must have minor version number
45  if (! chopUInt (&minorVersion, std::numeric_limits <int>::max (), input))
46  return false;
47 
48  if (! chop (".", input))
49  return false;
50 
51  // Must have patch version number
52  if (! chopUInt (&patchVersion, std::numeric_limits <int>::max (), input))
53  return false;
54 
55  // May have pre-release identifier list
56  if (chop ("-", input))
57  {
58  chopIdentifiers (&preReleaseIdentifiers, false, input);
59 
60  // Must not be empty
61  if (preReleaseIdentifiers.size () <= 0)
62  return false;
63  }
64 
65  // May have metadata identifier list
66  if (chop ("+", input))
67  {
68  chopIdentifiers (&metaData, true, input);
69 
70  // Must not be empty
71  if (metaData.size () <= 0)
72  return false;
73  }
74 
75  // May not have anything left
76  if (input.length () > 0)
77  return false;
78 
79  return true;
80 }

97 int SemanticVersion::compare (SemanticVersion const& rhs) const noexcept
98 {
99  if (majorVersion > rhs.majorVersion)
100  return 1;
101  else if (majorVersion < rhs.majorVersion)
102  return -1;
103 
104  if (minorVersion > rhs.minorVersion)
105  return 1;
106  else if (minorVersion < rhs.minorVersion)
107  return -1;
108 
109  if (patchVersion > rhs.patchVersion)
110  return 1;
111  else if (patchVersion < rhs.patchVersion)
112  return -1;
113 
114  if (isPreRelease () || rhs.isPreRelease ())
115  {
116  // Pre-releases have a lower precedence
117  if (isRelease () && rhs.isPreRelease ())
118  return 1;
119  else if (isPreRelease () && rhs.isRelease ())
120  return -1;
121 
122  // Compare pre-release identifiers
123  for (int i = 0; i < bmax (preReleaseIdentifiers.size (), rhs.preReleaseIdentifiers.size ()); ++i)
124  {
125  // A larger list of identifiers has a higher precedence
126  if (i >= rhs.preReleaseIdentifiers.size ())
127  return 1;
128  else if (i >= preReleaseIdentifiers.size ())
129  return -1;
130 
131  String const& left (preReleaseIdentifiers [i]);
132  String const& right (rhs.preReleaseIdentifiers [i]);
133 
134  // Numeric identifiers have lower precedence
135  if (! isNumeric (left) && isNumeric (right))
136  return 1;
137  else if (isNumeric (left) && ! isNumeric (right))
138  return -1;
139 
140  if (isNumeric (left))
141  {
142  bassert (isNumeric (right));
143 
144  int const iLeft (left.getIntValue ());
145  int const iRight (right.getIntValue ());
146 
147  if (iLeft > iRight)
148  return 1;
149  else if (iLeft < iRight)
150  return -1;
151  }
152  else
153  {
154  bassert (! isNumeric (right));
155 
156  int result = left.compareLexicographically (right);
157 
158  if (result != 0)
159  return result;
160  }
161  }
162  }
163 
164  // metadata is ignored
165 
166  return 0;
167 }

201 bool SemanticVersion::chopUInt (int* value, int limit, String& input)
202 {
203  // Must not be empty
204  if (input.length () <= 0)
205  return false;
206 
207  int firstNonDigit = 0;
208  for (; firstNonDigit < input.length (); ++firstNonDigit)
209  {
210  if (! CharacterFunctions::isDigit (input [firstNonDigit]))
211  break;
212  }
213 
214  String const s = input.substring (0, firstNonDigit);
215 
216  // Must not be empty
217  if (s.length () <= 0)
218  return false;
219 
220  int const n = s.getIntValue ();
221 
222  // Must not have leading zeroes
223  if (String (n) != s)
224  return false;
225 
226  // Must not be out of range
227  if (n < 0 || n > limit)
228  return false;
229 
230  input = input.substring (s.length ());
231 
232  *value = n;
233 
234  return true;
235 }

237 bool SemanticVersion::chopIdentifier (String* value, bool allowLeadingZeroes, String& input)
238 {
239  // Must not be empty
240  if (input.length () <= 0)
241  return false;
242 
243  // Must not have a leading 0
244  if (! allowLeadingZeroes && input [0] == '0')
245  return false;
246 
247  // Find the first character that cannot be part of an identifier
248  int i;
249  for (i = 0; i < input.length (); ++i)
250  {
251  static char const* validSet =
252  "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-";
253 
254  if (! String (validSet).contains (String::charToString (input [i])))
255  break;
256  }
257 
258  // Must not be empty
259  if (i <= 0)
260  return false;
261 
262  *value = input.substring (0, i);
263 
264  input = input.substring (i);
265 
266  return true;
267 }

269 bool SemanticVersion::chopIdentifiers (StringArray* value, bool allowLeadingZeroes, String& input)
270 {
271  if (input.length () <= 0)
272  return false;
273 
274  for (;;)
275  {
276  String s;
277 
278  if (! chopIdentifier (&s, allowLeadingZeroes, input))
279  return false;
280 
281  value->add (s);
282 
283  if (! chop (".", input))
284  break;
285  }
286 
287  return true;
