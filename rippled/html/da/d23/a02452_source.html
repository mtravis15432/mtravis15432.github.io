445 Value::Value ( const Value& other )
446  : type_ ( other.type_ )
447  , comments_ ( 0 )
448 # ifdef JSON_VALUE_USE_INTERNAL_MAP
449  , itemIsUsed_ ( 0 )
450 #endif
451 {
452  switch ( type_ )
453  {
454  case nullValue:
455  case intValue:
456  case uintValue:
457  case realValue:
458  case booleanValue:
459  value_ = other.value_;
460  break;
461 
462  case stringValue:
463  if ( other.value_.string_ )
464  {
465  value_.string_ = valueAllocator ()->duplicateStringValue ( other.value_.string_ );
466  allocated_ = true;
467  }
468  else
469  value_.string_ = 0;
470 
471  break;
472 #ifndef JSON_VALUE_USE_INTERNAL_MAP
473 
474  case arrayValue:
475  case objectValue:
476  value_.map_ = new ObjectValues ( *other.value_.map_ );
477  break;
478 #else
479 
480  case arrayValue:
481  value_.array_ = arrayAllocator ()->newArrayCopy ( *other.value_.array_ );
482  break;
483 
484  case objectValue:
485  value_.map_ = mapAllocator ()->newMapCopy ( *other.value_.map_ );
486  break;
487 #endif
488 
489  default:
490  JSON_ASSERT_UNREACHABLE;
491  }
492 
493  if ( other.comments_ )
494  {
495  comments_ = new CommentInfo[numberOfCommentPlacement];
496 
497  for ( int comment = 0; comment < numberOfCommentPlacement; ++comment )
498  {
499  const CommentInfo& otherComment = other.comments_[comment];

501  if ( otherComment.comment_ )
