28  {
29  // e.g.
30  //

32  //
33  static boost::regex re (
34  "\\G" // end of last match (or start)
35  "(?:[\\v\\h]*)" // white (optional)
36  "([^\\h\\v]+)" // [1] non-white run
37  "(?:\\h*)" // horiz-white (optional)
38  "([^\\v]*?)" // [2] non vert-white text (optional)
39  "(?:\\h*)" // white run (optional)
40  "(?:\\v*)" // vert-white (optional)
41 
42  //"(?:\\')" // buffer boundary
43 
44  , boost::regex::perl
45  //| boost::regex_constants::match_flags::match_not_dot_newline
46  );
47 
48  return re;
49  }

52  //

77 bool Utilities::parseInfoLine (
78  Source::Item& item,
79  std::string const& line,
80  beast::Journal journal)
81 {
82  bool success (false);
83 
84  boost::smatch match;
85 
86  if (boost::regex_match (line, match, Helpers::reInfo ()))
87  {
88  std::string const encodedKey (match [1]);
89  std::string const commentText (match [2]);
90 
91  std::pair <RipplePublicKey, bool> results (
92  RipplePublicKey::from_string (encodedKey));
93 
94  if (results.second)
95  {
96  // We got a public key.
97  item.label = commentText;
98  item.publicKey = results.first;
99  success = true;
100  }
101  else
102  {
103  // Some other junk.
104  journal.error << "Invalid RipplePublicKey: '" << encodedKey << "'";
105  }
106  }
107  else if (boost::regex_match (line, match, Helpers::reComment ()))
108  {
109  // it's a comment
110  }
111  else
112  {
113  // Log a warning about a parsing error
114  journal.error << "Invalid Validators source line:" << std::endl << line;
115  }
116 
117  return success;
118 }

160 int Utilities::stoi (beast::String& s, int fieldSize, int minValue, int maxValue,
161  beast::beast_wchar delimiter)
162 {
163  int const needed (fieldSize + ((delimiter != 0) ? 1 : 0));
164  beast::String const v (s.substring (0, needed));
165  s = s.substring (v.length ());
166  if (s.length() == needed)
167  {
168  int const v (s.getIntValue());
169  if (s.startsWith (itos (v, fieldSize)) &&
170  v >= minValue && v <= maxValue &&
171  (delimiter == 0 || s.endsWithChar (delimiter)))
172  {
173  return v;
174  }
175  }
176  return -1; // fail
177 }
