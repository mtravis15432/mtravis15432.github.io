78 void PeerSet::invokeOnTimer ()
79 {
80  ScopedLockType sl (mLock);
81 
82  if (isDone ())

84 
85  if (!isProgress())
86  {

88  WriteLog (lsWARNING, InboundLedger) << "Timeout(" << mTimeouts << ") pc=" << mPeers.size () << " acquiring " << mHash;

101 void PeerSet::TimerEntry (boost::weak_ptr<PeerSet> wptr, const boost::system::error_code& result)
102 {
103  if (result == boost::asio::error::operation_aborted)
104  return;
105 
106  boost::shared_ptr<PeerSet> ptr = wptr.lock ();
107 
108  if (ptr)
109  {
110  // VFALCO NOTE So this function is really two different functions depending on
111  // the value of mTxnData, which is directly tied to whether we are
112  // a base class of IncomingLedger or TransactionAcquire
113  //
114  if (ptr->mTxnData)
115  {
116  getApp().getJobQueue ().addJob (jtTXN_DATA, "timerEntryTxn",
117  BIND_TYPE (&PeerSet::TimerJobEntry, P_1, ptr));
118  }
119  else
120  {
121  int jc = getApp().getJobQueue ().getJobCountTotal (jtLEDGER_DATA);
122 
123  if (jc > 4)
124  {
125  WriteLog (lsDEBUG, InboundLedger) << "Deferring PeerSet timer due to load";
126  ptr->setTimer ();
127  }
128  else
129  getApp().getJobQueue ().addJob (jtLEDGER_DATA, "timerEntryLgr",
130  BIND_TYPE (&PeerSet::TimerJobEntry, P_1, ptr));
