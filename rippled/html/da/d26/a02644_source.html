55 // `TablePropertiesCollector` provides the mechanism for users to collect
56 // their own interested properties. This class is essentially a collection
57 // of callback functions that will be invoked during table building.
58 class TablePropertiesCollector {
59  public:
60  virtual ~TablePropertiesCollector() { }
61

65  virtual Status Add(const Slice& key, const Slice& value) = 0;
66
