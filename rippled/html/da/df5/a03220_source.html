166  bool doQuery ()
167  {
168  ScopedLockType sl (mLock);
169  std::vector< std::pair<std::string, time_t> >::iterator best = mServers.end ();
170 
171  for (std::vector< std::pair<std::string, time_t> >::iterator it = mServers.begin (), end = best;
172  it != end; ++it)
173  if ((best == end) || (it->second == (time_t) - 1) || (it->second < best->second))
174  best = it;
175 
176  if (best == mServers.end ())
177  {
178  WriteLog (lsTRACE, SNTPClient) << "SNTP: No server to query";
179  return false;
180  }
181 
182  time_t now = time (nullptr);
183 
184  if ((best->second != (time_t) - 1) && ((best->second + NTP_MIN_QUERY) >= now))
185  {
186  WriteLog (lsTRACE, SNTPClient) << "SNTP: All servers recently queried";
187  return false;
188  }
189 
190  best->second = now;
191 
192  boost::asio::ip::udp::resolver::query query (boost::asio::ip::udp::v4 (), best->first, "ntp");
193  mResolver.async_resolve (query,
194  boost::bind (&SNTPClientImp::resolveComplete, this,
195  boost::asio::placeholders::error, boost::asio::placeholders::iterator));
196  #ifdef SNTP_DEBUG
197  WriteLog (lsTRACE, SNTPClient) << "SNTP: Resolve pending for " << best->first;
198  #endif
199  return true;
200  }

202  void resolveComplete (const boost::system::error_code& error, boost::asio::ip::udp::resolver::iterator it)
203  {
204  if (!error)
205  {
206  boost::asio::ip::udp::resolver::iterator sel = it;
207  int i = 1;
208 
209  while (++it != boost::asio::ip::udp::resolver::iterator ())
210  if ((rand () % ++i) == 0)
211  sel = it;
212 
213  if (sel != boost::asio::ip::udp::resolver::iterator ())
214  {
215  ScopedLockType sl (mLock);
216  SNTPQuery& query = mQueries[*sel];
217  time_t now = time (nullptr);
218 
219  if ((query.mLocalTimeSent == now) || ((query.mLocalTimeSent + 1) == now))
220  {
221  // This can happen if the same IP address is reached through multiple names
222  WriteLog (lsTRACE, SNTPClient) << "SNTP: Redundant query suppressed";
223  return;
224  }
225 
226  query.mReceivedReply = false;
227  query.mLocalTimeSent = now;
228  RandomNumbers::getInstance ().fill (&query.mQueryNonce);
229  reinterpret_cast<std::uint32_t*> (SNTPQueryData)[NTP_OFF_XMITTS_INT] = static_cast<std::uint32_t> (time (nullptr)) + NTP_UNIX_OFFSET;
230  reinterpret_cast<std::uint32_t*> (SNTPQueryData)[NTP_OFF_XMITTS_FRAC] = query.mQueryNonce;
231  mSocket.async_send_to (boost::asio::buffer (SNTPQueryData, 48), *sel,
232  boost::bind (&SNTPClientImp::sendComplete, this,
233  boost::asio::placeholders::error, boost::asio::placeholders::bytes_transferred));
234  }
235  }

238  void receivePacket (const boost::system::error_code& error, std::size_t bytes_xferd)
239  {
240  if (!error)
241  {
242  ScopedLockType sl (mLock);
243  #ifdef SNTP_DEBUG
244  WriteLog (lsTRACE, SNTPClient) << "SNTP: Packet from " << mReceiveEndpoint;
245  #endif
246  std::map<boost::asio::ip::udp::endpoint, SNTPQuery>::iterator query = mQueries.find (mReceiveEndpoint);
247 
248  if (query == mQueries.end ())
249  WriteLog (lsDEBUG, SNTPClient) << "SNTP: Reply from " << mReceiveEndpoint << " found without matching query";
250  else if (query->second.mReceivedReply)
251  WriteLog (lsDEBUG, SNTPClient) << "SNTP: Duplicate response from " << mReceiveEndpoint;
252  else
253  {
254  query->second.mReceivedReply = true;
255 
256  if (time (nullptr) > (query->second.mLocalTimeSent + 1))
257  WriteLog (lsWARNING, SNTPClient) << "SNTP: Late response from " << mReceiveEndpoint;
258  else if (bytes_xferd < 48)
259  WriteLog (lsWARNING, SNTPClient) << "SNTP: Short reply from " << mReceiveEndpoint
260  << " (" << bytes_xferd << ") " << mReceiveBuffer.size ();
261  else if (reinterpret_cast<std::uint32_t*> (&mReceiveBuffer[0])[NTP_OFF_ORGTS_FRAC] != query->second.mQueryNonce)
262  WriteLog (lsWARNING, SNTPClient) << "SNTP: Reply from " << mReceiveEndpoint << "had wrong nonce";
263  else
264  processReply ();
265  }
266  }
267 
268  mSocket.async_receive_from (boost::asio::buffer (mReceiveBuffer, 256), mReceiveEndpoint,
269  boost::bind (&SNTPClientImp::receivePacket, this, boost::asio::placeholders::error,
270  boost::asio::placeholders::bytes_transferred));
271  }

278  void processReply ()
279  {
280  assert (mReceiveBuffer.size () >= 48);
281  std::uint32_t* recvBuffer = reinterpret_cast<std::uint32_t*> (&mReceiveBuffer.front ());
282 
283  unsigned info = ntohl (recvBuffer[NTP_OFF_INFO]);
284  int64_t timev = ntohl (recvBuffer[NTP_OFF_RECVTS_INT]);
285  unsigned stratum = (info >> 16) & 0xff;
286 
287  if ((info >> 30) == 3)
288  {
289  WriteLog (lsINFO, SNTPClient) << "SNTP: Alarm condition " << mReceiveEndpoint;
290  return;
291  }
292 
293  if ((stratum == 0) || (stratum > 14))
294  {
295  WriteLog (lsINFO, SNTPClient) << "SNTP: Unreasonable stratum (" << stratum << ") from " << mReceiveEndpoint;
296  return;
297  }
298 
299  std::int64_t now = static_cast<int> (time (nullptr));
300  timev -= now;
301  timev -= NTP_UNIX_OFFSET;
302 
303  // add offset to list, replacing oldest one if appropriate
304  mOffsetList.push_back (timev);
305 
306  if (mOffsetList.size () >= NTP_SAMPLE_WINDOW)
307  mOffsetList.pop_front ();
308 
309  mLastOffsetUpdate = now;
310 
311  // select median time
312  std::list<int> offsetList = mOffsetList;
313  offsetList.sort ();
314  int j = offsetList.size ();
315  std::list<int>::iterator it = offsetList.begin ();
316 
317  for (int i = 0; i < (j / 2); ++i)
318  ++it;
319 
320  mOffset = *it;
321 
322  if ((j % 2) == 0)
323  mOffset = (mOffset + (*--it)) / 2;
324 
325  if ((mOffset == -1) || (mOffset == 1)) // small corrections likely do more harm than good
326  mOffset = 0;
327 
328  CondLog (timev || mOffset, lsTRACE, SNTPClient) << "SNTP: Offset is " << timev << ", new system offset is " << mOffset;
329  }
