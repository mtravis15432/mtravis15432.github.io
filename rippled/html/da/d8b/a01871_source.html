172 template <class Derived>
173 class enable_wait_for_async
174 {
175 private:
176  BEAST_DEFINE_IS_CALL_POSSIBLE(
177  has_on_wait_for_async,on_wait_for_async);
178 
179  void increment()
180  {
181  std::lock_guard <decltype(m_mutex)> lock (m_mutex);
182  ++m_count;
183  }
184 
185  void notify (std::true_type)
186  {
187  static_cast <Derived*> (this)->on_wait_for_async();
188  }
189 
190  void notify (std::false_type)
191  {
192  }
193 
194  void decrement()
195  {
196  std::lock_guard <decltype(m_mutex)> lock (m_mutex);
197  --m_count;
198  if (m_count == 0)
199  {
200  m_cond.notify_all();
201  notify (std::integral_constant <bool,
202  has_on_wait_for_async<Derived, void(void)>::value>());
203  }
204  }
