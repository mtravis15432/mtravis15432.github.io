159  hdfsFS connectToPath(const std::string& uri) {
160  if (uri.empty()) {
161  return NULL;
162  }
163  if (uri.find(kProto) != 0) {
164  // uri doesn't start with hdfs:// -> use default:0, which is special
165  // to libhdfs.
166  return hdfsConnectNewInstance("default", 0);
167  }
168  const std::string hostport = uri.substr(kProto.length());
169 
170  std::vector <std::string> parts;
171  split(hostport, ':', parts);
172  if (parts.size() != 2) {
173  throw HdfsFatalException("Bad uri for hdfs " + uri);
174  }
175  // parts[0] = hosts, parts[1] = port/xxx/yyy
176  std::string host(parts[0]);
177  std::string remaining(parts[1]);
178 
179  int rem = remaining.find(pathsep);
180  std::string portStr = (rem == 0 ? remaining :
181  remaining.substr(0, rem));
182 
183  tPort port;
184  port = atoi(portStr.c_str());
185  if (port == 0) {
186  throw HdfsFatalException("Bad host-port for hdfs " + uri);
187  }
188  hdfsFS fs = hdfsConnectNewInstance(host.c_str(), port);
189  return fs;
190  }

192  void split(const std::string &s, char delim,
193  std::vector<std::string> &elems) {
194  elems.clear();
195  size_t prev = 0;
196  size_t pos = s.find(delim);
197  while (pos != std::string::npos) {
198  elems.push_back(s.substr(prev, pos));
199  prev = pos + 1;
200  pos = s.find(delim, prev);
201  }
202  elems.push_back(s.substr(prev, s.size()));
203  }
