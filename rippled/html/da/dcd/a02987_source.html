183  void doBind (SqliteStatement& statement, NodeObject::ref object)
184  {
185  char const* type;
186  switch (object->getType())
187  {
188  case hotLEDGER: type = "L"; break;
189  case hotTRANSACTION: type = "T"; break;
190  case hotACCOUNT_NODE: type = "A"; break;
191  case hotTRANSACTION_NODE: type = "N"; break;
192  default: type = "U";
193  }
194 
195  statement.bind(1, object->getHash().GetHex());
196  statement.bind(2, type);
197  statement.bind(3, object->getIndex());
198  statement.bindStatic(4, object->getData());
199  }
