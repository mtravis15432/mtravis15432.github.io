30 /*
31 Config file format:
32 
33  Syntactically a series of lines, where line has this format:
34  [ <vertical whitespace> ] <anything> ( <vertical-whitespace> OR <end-of-file> )
35 
36  Semantically a series of of zero or more sections, where each section
37  has a name and optional data. Specifically, the format:
38  ( <start-of-file> OR <header> ) <data>
39 
40  Data appearing before the first header goes into the section whose
41  name is the empty string "".
42 
43  All lines are valid, errors are not possible. Each line matches one of
44  the Comment, Header, or Data format:
45 
46  Comment:
47  [ <horizontal whitespace> ] [ '#' <anything> ]
48 
49  Comment lines are ignored; The file is treated as if
50  the comment lines do not exist.
51 
52  Header:
53  [ <horizontal whitespace> ] '[' <anything> ']' [ <anything> ]
54 
55  Data:
56  Anything not matching a comment or header.
57 
58  Lines in a data block are added to the section with the last name parsed,
59  or the empty string if no header line has been seen yet.
60 */

157  void processResponse (beast::URL const& url, beast::HTTPResponse const& response)
158  {
159  SharedState::Access state (m_state);
160 
161  std::pair <SiteFiles::iterator, bool> result (
162  state->files.emplace (url, 0));
163 
164  if (! result.second)
165  {
166  m_journal.error <<
167  "Duplicate URL '" << url << "' ignored";
168  return;
169  }
170 
171  SiteFile& siteFile (result.first->second);
172  parse (siteFile, response);
173 
174  for (Listeners::iterator iter (state->listeners.begin());
175  iter != state->listeners.end(); ++iter)
176  {
177  Listener* const listener (*iter);
178  listener->onSiteFileFetch (url.to_string(), siteFile);
179  }

261  template <typename BidirectionalIterator>
262  void parse (
263  SiteFile& siteFile,
264  BidirectionalIterator start,
265  BidirectionalIterator end)
266  {
267  Section* section (&siteFile.insert (""));
268 
269  boost::match_results <BidirectionalIterator> m;
270  for (;start != end;)
271  {
272  if (boost::regex_search (start, end, m, reHeader(),
273  boost::regex_constants::match_continuous))
274  {
275  std::string const& s (m[1]);
276  section = &siteFile.insert (s);
277  }
278  else
279  {
280  boost::regex_search (start, end, m, reData(),
281  boost::regex_constants::match_continuous);
282 
283  std::string const& s (m[1]);
284  section->push_back (s);
285  }
286 
287  start = m[0].second;
288  }
289  }
