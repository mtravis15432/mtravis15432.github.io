78 inline static std::string sqlEscape (Blob const& vecSrc)
79 {
80  size_t size = vecSrc.size ();
81 
82  if (size == 0)
83  return "X''";
84 
85  std::string j (size * 2 + 3, 0);
86 
87  unsigned char* oPtr = reinterpret_cast<unsigned char*> (&*j.begin ());
88  const unsigned char* iPtr = &vecSrc[0];
89 
90  *oPtr++ = 'X';
91  *oPtr++ = '\'';
92 
93  for (int i = size; i != 0; --i)
94  {
95  unsigned char c = *iPtr++;
96  *oPtr++ = charHex (c >> 4);
97  *oPtr++ = charHex (c & 15);
98  }
99 
100  *oPtr++ = '\'';
101  return j;
102 }
