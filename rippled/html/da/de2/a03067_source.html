28 void TransactionEngine::txnWrite ()
29 {
30  // Write back the account states
31  typedef std::map<uint256, LedgerEntrySetEntry>::value_type u256_LES_pair;
32  BOOST_FOREACH (u256_LES_pair & it, mNodes)
33  {
34  SLE::ref sleEntry = it.second.mEntry;
35 
36  switch (it.second.mAction)
37  {
38  case taaNONE:
39  assert (false);
40  break;
41 
42  case taaCACHED:
43  break;
44 
45  case taaCREATE:
46  {
47  WriteLog (lsINFO, TransactionEngine) << "applyTransaction: taaCREATE: " << sleEntry->getText ();
48 
49  if (mLedger->writeBack (lepCREATE, sleEntry) & lepERROR)
50  assert (false);
51  }
52  break;
53 
54  case taaMODIFY:
55  {
56  WriteLog (lsINFO, TransactionEngine) << "applyTransaction: taaMODIFY: " << sleEntry->getText ();
57 
58  if (mLedger->writeBack (lepNONE, sleEntry) & lepERROR)
59  assert (false);
60  }
61  break;
62 
63  case taaDELETE:
64  {
65  WriteLog (lsINFO, TransactionEngine) << "applyTransaction: taaDELETE: " << sleEntry->getText ();
66 
67  if (!mLedger->peekAccountStateMap ()->delItem (it.first))
68  assert (false);
69  }
70  break;
71  }
72  }
73 }

75 TER TransactionEngine::applyTransaction (const SerializedTransaction& txn, TransactionEngineParams params,
76  bool& didApply)
77 {
78  WriteLog (lsTRACE, TransactionEngine) << "applyTransaction>";
79  didApply = false;
80  assert (mLedger);
81  mNodes.init (mLedger, txn.getTransactionID (), mLedger->getLedgerSeq (), params);
82 
83 #ifdef BEAST_DEBUG
84 
85  if (1)
86  {
87  Serializer ser;
88  txn.add (ser);
89  SerializerIterator sit (ser);
90  SerializedTransaction s2 (sit);
91 
92  if (!s2.isEquivalent (txn))
93  {
94  WriteLog (lsFATAL, TransactionEngine) << "Transaction serdes mismatch";
95  Json::StyledStreamWriter ssw;
96  WriteLog (lsINFO, TransactionEngine) << txn.getJson (0);
97  WriteLog (lsFATAL, TransactionEngine) << s2.getJson (0);
98  assert (false);
99  }
100  }
101 
102 #endif
103 
104  uint256 txID = txn.getTransactionID ();
105 
106  if (!txID)
107  {
108  WriteLog (lsWARNING, TransactionEngine) << "applyTransaction: invalid transaction id";
109  return temINVALID;
110  }
111 
112  std::unique_ptr<Transactor> transactor = Transactor::makeTransactor (txn, params, this);
113 
114  if (transactor.get () == nullptr)
115  {
116  WriteLog (lsWARNING, TransactionEngine) << "applyTransaction: Invalid transaction: unknown transaction type";
117  return temUNKNOWN;
118  }
119 
120  TER terResult = transactor->apply ();
121  std::string strToken;
122  std::string strHuman;
123 
124  transResultInfo (terResult, strToken, strHuman);
125 
126  WriteLog (lsINFO, TransactionEngine) << "applyTransaction: terResult=" << strToken << " : " << terResult << " : " << strHuman;
127 
128  if (isTesSuccess (terResult))
129  didApply = true;
130  else if (isTecClaim (terResult) && !is_bit_set (params, tapRETRY))
131  {
132  // only claim the transaction fee
133  WriteLog (lsDEBUG, TransactionEngine) << "Reprocessing to only claim fee";
134  mNodes.clear ();
135 
136  SLE::pointer txnAcct = entryCache (ltACCOUNT_ROOT, Ledger::getAccountRootIndex (txn.getSourceAccount ()));
137 
138  if (!txnAcct)
139  terResult = terNO_ACCOUNT;
140  else
141  {
142  std::uint32_t t_seq = txn.getSequence ();
143  std::uint32_t a_seq = txnAcct->getFieldU32 (sfSequence);
144 
145  if (a_seq < t_seq)
146  terResult = terPRE_SEQ;
147  else if (a_seq > t_seq)
148  terResult = tefPAST_SEQ;
149  else
150  {
151  STAmount fee = txn.getTransactionFee ();
152  STAmount balance = txnAcct->getFieldAmount (sfBalance);
153 
154  if (balance < fee)
155  terResult = terINSUF_FEE_B;
156  else
157  {
158  txnAcct->setFieldAmount (sfBalance, balance - fee);
159  txnAcct->setFieldU32 (sfSequence, t_seq + 1);
160  entryModify (txnAcct);
161  didApply = true;
162  }
163  }
164  }
165  }
166  else
167  WriteLog (lsDEBUG, TransactionEngine) << "Not applying transaction " << txID;
168 
169  if (didApply)
170  {
171  if (!checkInvariants (terResult, txn, params))
172  {
173  WriteLog (lsFATAL, TransactionEngine) << "Transaction violates invariants";
174  WriteLog (lsFATAL, TransactionEngine) << txn.getJson (0);
175  WriteLog (lsFATAL, TransactionEngine) << transToken (terResult) << ": " << transHuman (terResult);
176  WriteLog (lsFATAL, TransactionEngine) << mNodes.getJson (0);
177  didApply = false;
178  terResult = tefINTERNAL;
179  }
180  else
181  {
182  // Transaction succeeded fully or (retries are not allowed and the transaction could claim a fee)
183  Serializer m;
184  mNodes.calcRawMeta (m, terResult, mTxnSeq++);
185 
186  txnWrite ();
187 
188  Serializer s;
189  txn.add (s);
190 
191  if (is_bit_set (params, tapOPEN_LEDGER))
192  {
193  if (!mLedger->addTransaction (txID, s))
194  {
195  WriteLog (lsFATAL, TransactionEngine) << "Tried to add transaction to open ledger that already had it";
196  assert (false);
197  throw std::runtime_error ("Duplicate transaction applied");
198  }
199  }
200  else
201  {
202  if (!mLedger->addTransaction (txID, s, m))
203  {
204  WriteLog (lsFATAL, TransactionEngine) << "Tried to add transaction to ledger that already had it";
205  assert (false);
206  throw std::runtime_error ("Duplicate transaction applied to closed ledger");
207  }
208 
209  // Charge whatever fee they specified.
210  STAmount saPaid = txn.getTransactionFee ();
211  mLedger->destroyCoins (saPaid.getNValue ());
212  }
213  }
214  }
215 
216  mTxnAccount.reset ();
217  mNodes.clear ();
218 
219  if (!is_bit_set (params, tapOPEN_LEDGER) && isTemMalformed (terResult))
220  {
221  // XXX Malformed or failed transaction in closed ledger must bow out.
222  }
223 
224  return terResult;
225 }
