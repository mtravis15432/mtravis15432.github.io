36 // SimpleTable is a simple table format for UNIT TEST ONLY. It is not built
37 // as production quality.
38 // SimpleTable requires the input key size to be fixed 16 bytes, value cannot
39 // be longer than 150000 bytes and stored data on disk in this format:
40 // +--------------------------------------------+ <= key1 offset
41 // | key1 | value_size (4 bytes) | |
42 // +----------------------------------------+ |
43 // | value1 |
44 // | |
45 // +----------------------------------------+---+ <= key2 offset
46 // | key2 | value_size (4 bytes) | |
47 // +----------------------------------------+ |
48 // | value2 |
49 // | |
50 // | ...... |
51 // +-----------------+--------------------------+ <= index_block_offset
52 // | key1 | key1 offset (8 bytes) |
53 // +-----------------+--------------------------+
54 // | key2 | key2 offset (8 bytes) |
55 // +-----------------+--------------------------+
56 // | key3 | key3 offset (8 bytes) |
57 // +-----------------+--------------------------+
58 // | ...... |
59 // +-----------------+------------+-------------+
60 // | index_block_offset (8 bytes) |

189 Status SimpleTableReader::Open(const Options& options,
190  const EnvOptions& soptions,
191  unique_ptr<RandomAccessFile> && file,
192  uint64_t size,
193  unique_ptr<TableReader>* table_reader) {
194  char footer_space[Rep::offset_length];
195  Slice footer_input;
196  Status s = file->Read(size - Rep::offset_length, Rep::offset_length,
197  &footer_input, footer_space);
198  if (s.ok()) {
199  uint64_t index_start_offset = DecodeFixed64(footer_space);
200 
201  int num_entries = (size - Rep::offset_length - index_start_offset)
202  / (Rep::GetInternalKeyLength() + Rep::offset_length);
203  SimpleTableReader::Rep* rep = new SimpleTableReader::Rep(soptions,
204  index_start_offset,
205  num_entries);
206 
207  rep->file = std::move(file);
208  rep->options = options;
209  table_reader->reset(new SimpleTableReader(rep));
210  }
211  return s;
212 }

229 Status SimpleTableReader::GetOffset(const Slice& target, uint64_t* offset) {
230  uint32_t left = 0;
231  uint32_t right = rep_->num_entries - 1;
232  char key_chars[Rep::GetInternalKeyLength()];
233  Slice tmp_slice;
234 
235  uint32_t target_offset = 0;
236  while (left <= right) {
237  uint32_t mid = (left + right + 1) / 2;
238 
239  uint64_t offset_to_read = rep_->index_start_offset
240  + (Rep::GetInternalKeyLength() + Rep::offset_length) * mid;
241  Status s = rep_->file->Read(offset_to_read, Rep::GetInternalKeyLength(),
242  &tmp_slice, key_chars);
243  if (!s.ok()) {
244  return s;
245  }
246 
247  int compare_result = rep_->options.comparator->Compare(tmp_slice, target);
248 
249  if (compare_result < 0) {
250  if (left == right) {
251  target_offset = right + 1;
252  break;
253  }
254  left = mid;
255  } else {
256  if (left == right) {
257  target_offset = left;
258  break;
259  }
260  right = mid - 1;
261  }
262  }
263 
264  if (target_offset >= (uint32_t) rep_->num_entries) {
265  *offset = rep_->index_start_offset;
266  return Status::OK();
267  }
268 
269  char value_offset_chars[Rep::offset_length];
270 
271  int64_t offset_for_value_offset = rep_->index_start_offset
272  + (Rep::GetInternalKeyLength() + Rep::offset_length) * target_offset
273  + Rep::GetInternalKeyLength();
274  Status s = rep_->file->Read(offset_for_value_offset, Rep::offset_length,
275  &tmp_slice, value_offset_chars);
276  if (s.ok()) {
277  *offset = DecodeFixed64(value_offset_chars);
278  }
279  return s;
280 }

342 void SimpleTableIterator::Next() {
343  offset_ = next_offset_;
344  if (offset_ >= table_->rep_->index_start_offset) {
345  return;
346  }
347  Slice result;
348  int internal_key_size = SimpleTableReader::Rep::GetInternalKeyLength();
349 
350  Status s = table_->rep_->file->Read(next_offset_, internal_key_size, &result,
351  key_str_);
352  next_offset_ += internal_key_size;
353  key_ = result;
354 
355  Slice value_size_slice;
356  s = table_->rep_->file->Read(next_offset_, 4, &value_size_slice, tmp_str_);
357  next_offset_ += 4;
358  uint32_t value_size = DecodeFixed32(tmp_str_);
359 
360  Slice value_slice;
361  if ((int) value_size > value_str_len_) {
362  if (value_str_len_ >= 0) {
363  delete[] value_str_;
364  }
365  value_str_ = new char[value_size];
366  value_str_len_ = value_size;
367  }
368  s = table_->rep_->file->Read(next_offset_, value_size, &value_slice,
369  value_str_);
370  next_offset_ += value_size;
371  value_ = value_slice;
372 }

477 void SimpleTableBuilder::Add(const Slice& key, const Slice& value) {
478  assert((int ) key.size() == Rep::GetInternalKeyLength());
479 
480  // Update index
481  rep_->index.append(key.data(), key.size());
482  PutFixed64(&(rep_->index), rep_->offset);
483 
484  // Write key-value pair
485  rep_->file->Append(key);
486  rep_->offset += Rep::GetInternalKeyLength();
487 
488  std::string size;
489  int value_size = value.size();
490  PutFixed32(&size, value_size);
491  Slice sizeSlice(size);
492  rep_->file->Append(sizeSlice);
493  rep_->file->Append(value);
494  rep_->offset += value_size + 4;
495 
496  rep_->num_entries++;
497 }
