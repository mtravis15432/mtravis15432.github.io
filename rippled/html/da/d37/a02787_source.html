177 class TtlCompactionFilter : public CompactionFilter {
178 
179  public:
180  TtlCompactionFilter(
181  int32_t ttl,
182  const CompactionFilter* user_comp_filter,
183  std::unique_ptr<const CompactionFilter>
184  user_comp_filter_from_factory = nullptr)
185  : ttl_(ttl),
186  user_comp_filter_(user_comp_filter),
187  user_comp_filter_from_factory_(std::move(user_comp_filter_from_factory)) {
188  // Unlike the merge operator, compaction filter is necessary for TTL, hence
189  // this would be called even if user doesn't specify any compaction-filter
190  if (!user_comp_filter_) {
191  user_comp_filter_ = user_comp_filter_from_factory_.get();
192  }
193  }

195  virtual bool Filter(int level,
196  const Slice& key,
197  const Slice& old_val,
198  std::string* new_val,
199  bool* value_changed) const override {
200  if (DBWithTTL::IsStale(old_val, ttl_)) {
201  return true;
202  }
203  if (user_comp_filter_ == nullptr) {
204  return false;
205  }
206  assert(old_val.size() >= DBWithTTL::kTSLength);
207  Slice old_val_without_ts(old_val.data(),
208  old_val.size() - DBWithTTL::kTSLength);
209  if (user_comp_filter_->Filter(level, key, old_val_without_ts, new_val,
210  value_changed)) {
211  return true;
212  }
213  if (*value_changed) {
214  new_val->append(old_val.data() + old_val.size() - DBWithTTL::kTSLength,
215  DBWithTTL::kTSLength);
216  }
217  return false;
218  }
219 
220  virtual const char* Name() const override {
221  return "Delete By TTL";
222  }
223 
224  private:
225  int32_t ttl_;
226  const CompactionFilter* user_comp_filter_;
227  std::unique_ptr<const CompactionFilter> user_comp_filter_from_factory_;
228 };

258 class TtlMergeOperator : public MergeOperator {
259 
260  public:
261  explicit TtlMergeOperator(const std::shared_ptr<MergeOperator> merge_op)
262  : user_merge_op_(merge_op) {

266  virtual bool FullMerge(const Slice& key,
267  const Slice* existing_value,
268  const std::deque<std::string>& operands,
269  std::string* new_value,
270  Logger* logger) const override {
271  const uint32_t ts_len = DBWithTTL::kTSLength;
272  if (existing_value && existing_value->size() < ts_len) {
273  Log(logger, "Error: Could not remove timestamp from existing value.");
274  return false;
275  }
276 
277  // Extract time-stamp from each operand to be passed to user_merge_op_
278  std::deque<std::string> operands_without_ts;
279  for (const auto &operand : operands) {
280  if (operand.size() < ts_len) {
281  Log(logger, "Error: Could not remove timestamp from operand value.");
282  return false;
283  }
284  operands_without_ts.push_back(operand.substr(0, operand.size() - ts_len));
285  }
286 
287  // Apply the user merge operator (store result in *new_value)
288  bool good = true;
289  if (existing_value) {
290  Slice existing_value_without_ts(existing_value->data(),
291  existing_value->size() - ts_len);
292  good = user_merge_op_->FullMerge(key, &existing_value_without_ts,
293  operands_without_ts, new_value, logger);
294  } else {
295  good = user_merge_op_->FullMerge(key, nullptr, operands_without_ts,
296  new_value, logger);
297  }
298 
299  // Return false if the user merge operator returned false
300  if (!good) {
301  return false;
302  }
303 
304  // Augment the *new_value with the ttl time-stamp
305  int64_t curtime;
306  if (!DBWithTTL::GetCurrentTime(curtime).ok()) {
307  Log(logger, "Error: Could not get current time to be attached internally "
308  "to the new value.");
309  return false;
310  } else {
311  char ts_string[ts_len];
312  EncodeFixed32(ts_string, (int32_t)curtime);
313  new_value->append(ts_string, ts_len);
314  return true;
315  }
316  }
317 
318  virtual bool PartialMerge(const Slice& key,
319  const Slice& left_operand,
320  const Slice& right_operand,
321  std::string* new_value,
322  Logger* logger) const override {
323  const uint32_t ts_len = DBWithTTL::kTSLength;
324 
325  if (left_operand.size() < ts_len || right_operand.size() < ts_len) {
326  Log(logger, "Error: Could not remove timestamp from value.");
327  return false;
328  }
329 
330  // Apply the user partial-merge operator (store result in *new_value)
331  assert(new_value);
332  Slice left_without_ts(left_operand.data(), left_operand.size() - ts_len);
333  Slice right_without_ts(right_operand.data(), right_operand.size() - ts_len);
334  if (!user_merge_op_->PartialMerge(key, left_without_ts, right_without_ts,
335  new_value, logger)) {
336  return false;
337  }
338 
339  // Augment the *new_value with the ttl time-stamp
340  int64_t curtime;
341  if (!DBWithTTL::GetCurrentTime(curtime).ok()) {
342  Log(logger, "Error: Could not get current time to be attached internally "
343  "to the new value.");
344  return false;
345  } else {
346  char ts_string[ts_len];
347  EncodeFixed32(ts_string, (int32_t)curtime);
348  new_value->append(ts_string, ts_len);
349  return true;
350  }
351 
352  }
