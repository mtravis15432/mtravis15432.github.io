48 Status TableCache::FindTable(const EnvOptions& toptions,
49  uint64_t file_number, uint64_t file_size,
50  Cache::Handle** handle, bool* table_io,
51  const bool no_io) {
52  Status s;
53  char buf[sizeof(file_number)];
54  EncodeFixed64(buf, file_number);
55  Slice key(buf, sizeof(buf));
56  *handle = cache_->Lookup(key);
57  if (*handle == nullptr) {
58  if (no_io) { // Dont do IO and return a not-found status
59  return Status::Incomplete("Table not found in table_cache, no_io is set");
60  }
61  if (table_io != nullptr) {
62  *table_io = true; // we had to do IO from storage
63  }
64  std::string fname = TableFileName(dbname_, file_number);
65  unique_ptr<RandomAccessFile> file;
66  unique_ptr<TableReader> table_reader;
67  s = env_->NewRandomAccessFile(fname, &file, toptions);
68  RecordTick(options_->statistics, NO_FILE_OPENS);
69  if (s.ok()) {
70  if (options_->advise_random_on_open) {
71  file->Hint(RandomAccessFile::RANDOM);
72  }
73  StopWatch sw(env_, options_->statistics, TABLE_OPEN_IO_MICROS);
74  s = options_->table_factory->GetTableReader(*options_, toptions,
75  std::move(file), file_size,
76  &table_reader);
77  }
78 
79  if (!s.ok()) {
80  assert(table_reader == nullptr);
81  RecordTick(options_->statistics, NO_FILE_ERRORS);
82  // We do not cache error results so that if the error is transient,
83  // or somebody repairs the file, we recover automatically.
84  } else {
85  assert(file.get() == nullptr);
86  *handle = cache_->Insert(key, table_reader.release(), 1, &DeleteEntry);
87  }
88  }
89  return s;
90 }

92 Iterator* TableCache::NewIterator(const ReadOptions& options,
93  const EnvOptions& toptions,
94  uint64_t file_number,
95  uint64_t file_size,
96  TableReader** table_reader_ptr,
97  bool for_compaction) {
98  if (table_reader_ptr != nullptr) {
99  *table_reader_ptr = nullptr;
100  }
101 
102  Cache::Handle* handle = nullptr;
103  Status s = FindTable(toptions, file_number, file_size, &handle,
104  nullptr, options.read_tier == kBlockCacheTier);
105  if (!s.ok()) {
106  return NewErrorIterator(s);
107  }
108 
109  TableReader* table_reader =
110  reinterpret_cast<TableReader*>(cache_->Value(handle));
111  Iterator* result = table_reader->NewIterator(options);
112  result->RegisterCleanup(&UnrefEntry, cache_.get(), handle);
113  if (table_reader_ptr != nullptr) {
114  *table_reader_ptr = table_reader;
115  }
116 
117  if (for_compaction) {
118  table_reader->SetupForCompaction();
119  }
120 
121  return result;
122 }

124 Status TableCache::Get(const ReadOptions& options,
125  uint64_t file_number,
126  uint64_t file_size,
127  const Slice& k,
128  void* arg,
129  bool (*saver)(void*, const Slice&, const Slice&, bool),
130  bool* table_io,
131  void (*mark_key_may_exist)(void*)) {
132  Cache::Handle* handle = nullptr;
133  Status s = FindTable(storage_options_, file_number, file_size,
134  &handle, table_io,
135  options.read_tier == kBlockCacheTier);
136  if (s.ok()) {
137  TableReader* t =
138  reinterpret_cast<TableReader*>(cache_->Value(handle));
139  s = t->Get(options, k, arg, saver, mark_key_may_exist);
140  cache_->Release(handle);
141  } else if (options.read_tier && s.IsIncomplete()) {
142  // Couldnt find Table in cache but treat as kFound if no_io set
143  (*mark_key_may_exist)(arg);
144  return Status::OK();
145  }
146  return s;
147 }

150  uint64_t file_number,
