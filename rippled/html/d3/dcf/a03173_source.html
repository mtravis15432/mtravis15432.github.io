22 // ECIES uses elliptic curve keys to send an encrypted message.
23 
24 // A shared secret is generated from one public key and one private key.
25 // The same key results regardless of which key is public and which private.
26 
27 // Anonymous messages can be sent by generating an ephemeral public/private
28 // key pair, using that private key with the recipient's public key to
29 // encrypt and publishing the ephemeral public key. Non-anonymous messages
30 // can be sent by using your own private key with the recipient's public key.
31 
32 // A random IV is used to encrypt the message and an HMAC is used to ensure
33 // message integrity. If you need timestamps or need to tell the recipient
34 // which key to use (his, yours, or ephemeral) you must add that data.
35 // (Obviously, key information can't go in the encrypted portion anyway.)
36 
37 // Our ciphertext is all encrypted except the IV. The encrypted data decodes as follows:
38 // 1) IV (unencrypted)
39 // 2) Encrypted: HMAC of original plaintext
40 // 3) Encrypted: Original plaintext
41 // 4) Encrypted: Rest of block/padding
42 
43 // ECIES operations throw on any error such as a corrupt message or incorrect
44 // key. They *must* be called in try/catch blocks.
45 
46 // Algorithmic choices:
47 #define ECIES_KEY_HASH SHA512 // Hash used to expand shared secret
48 #define ECIES_KEY_LENGTH (512/8) // Size of expanded shared secret
49 #define ECIES_MIN_SEC (128/8) // The minimum equivalent security
50 #define ECIES_ENC_ALGO EVP_aes_256_cbc() // Encryption algorithm
51 #define ECIES_ENC_KEY_TYPE uint256 // Type used to hold shared secret
52 #define ECIES_ENC_KEY_SIZE (256/8) // Encryption key size
53 #define ECIES_ENC_BLK_SIZE (128/8) // Encryption block size
54 #define ECIES_ENC_IV_TYPE uint128 // Type used to hold IV
55 #define ECIES_HMAC_ALGO EVP_sha256() // HMAC algorithm
56 #define ECIES_HMAC_KEY_TYPE uint256 // Type used to hold HMAC key
57 #define ECIES_HMAC_KEY_SIZE (256/8) // Size of HMAC key
58 #define ECIES_HMAC_TYPE uint256 // Type used to hold HMAC value
59 #define ECIES_HMAC_SIZE (256/8) // Size of HMAC value

61 void CKey::getECIESSecret (CKey& otherKey, ECIES_ENC_KEY_TYPE& enc_key, ECIES_HMAC_KEY_TYPE& hmac_key)
62 {
63  // Retrieve a secret generated from an EC key pair. At least one private key must be known.
64  if (!pkey || !otherKey.pkey)
65  throw std::runtime_error ("missing key");
66 
67  EC_KEY* pubkey, *privkey;
68 
69  if (EC_KEY_get0_private_key (pkey))
70  {
71  privkey = pkey;
72  pubkey = otherKey.pkey;
73  }
74  else if (EC_KEY_get0_private_key (otherKey.pkey))
75  {
76  privkey = otherKey.pkey;
77  pubkey = pkey;
78  }
79  else throw std::runtime_error ("no private key");
80 
81  unsigned char rawbuf[512];
82  int buflen = ECDH_compute_key (rawbuf, 512, EC_KEY_get0_public_key (pubkey), privkey, nullptr);
83 
84  if (buflen < ECIES_MIN_SEC)
85  throw std::runtime_error ("ecdh key failed");
86 
87  unsigned char hbuf[ECIES_KEY_LENGTH];
88  ECIES_KEY_HASH (rawbuf, buflen, hbuf);
89  memset (rawbuf, 0, ECIES_HMAC_KEY_SIZE);
90 
91  assert ((ECIES_ENC_KEY_SIZE + ECIES_HMAC_KEY_SIZE) >= ECIES_KEY_LENGTH);
92  memcpy (enc_key.begin (), hbuf, ECIES_ENC_KEY_SIZE);
93  memcpy (hmac_key.begin (), hbuf + ECIES_ENC_KEY_SIZE, ECIES_HMAC_KEY_SIZE);
94  memset (hbuf, 0, ECIES_KEY_LENGTH);
95 }

97 static ECIES_HMAC_TYPE makeHMAC (const ECIES_HMAC_KEY_TYPE& secret, Blob const& data)
98 {
99  HMAC_CTX ctx;
100  HMAC_CTX_init (&ctx);
101 
102  if (HMAC_Init_ex (&ctx, secret.begin (), ECIES_HMAC_KEY_SIZE, ECIES_HMAC_ALGO, nullptr) != 1)
103  {
104  HMAC_CTX_cleanup (&ctx);
105  throw std::runtime_error ("init hmac");
106  }
107 
108  if (HMAC_Update (&ctx, & (data.front ()), data.size ()) != 1)
109  {
110  HMAC_CTX_cleanup (&ctx);
111  throw std::runtime_error ("update hmac");
112  }
113 
114  ECIES_HMAC_TYPE ret;
115  unsigned int ml = ECIES_HMAC_SIZE;
116 
117  if (HMAC_Final (&ctx, ret.begin (), &ml) != 1)
118  {
119  HMAC_CTX_cleanup (&ctx);
120  throw std::runtime_error ("finalize hmac");
121  }
122 
123  assert (ml == ECIES_HMAC_SIZE);
124  HMAC_CTX_cleanup (&ctx);
125 
126  return ret;
127 }

129 Blob CKey::encryptECIES (CKey& otherKey, Blob const& plaintext)
130 {
131 
132  ECIES_ENC_IV_TYPE iv;
133  RandomNumbers::getInstance ().fillBytes (iv.begin (), ECIES_ENC_BLK_SIZE);
134 
135  ECIES_ENC_KEY_TYPE secret;
136  ECIES_HMAC_KEY_TYPE hmacKey;
137 
138  getECIESSecret (otherKey, secret, hmacKey);
139  ECIES_HMAC_TYPE hmac = makeHMAC (hmacKey, plaintext);
140  hmacKey.zero ();
141 
142  EVP_CIPHER_CTX ctx;
143  EVP_CIPHER_CTX_init (&ctx);
144 
145  if (EVP_EncryptInit_ex (&ctx, ECIES_ENC_ALGO, nullptr, secret.begin (), iv.begin ()) != 1)
146  {
147  EVP_CIPHER_CTX_cleanup (&ctx);
148  secret.zero ();
149  throw std::runtime_error ("init cipher ctx");
150  }
151 
152  secret.zero ();
153 
154  Blob out (plaintext.size () + ECIES_HMAC_SIZE + ECIES_ENC_KEY_SIZE + ECIES_ENC_BLK_SIZE, 0);
155  int len = 0, bytesWritten;
156 
157  // output IV
158  memcpy (& (out.front ()), iv.begin (), ECIES_ENC_BLK_SIZE);
159  len = ECIES_ENC_BLK_SIZE;
160 
161  // Encrypt/output HMAC
162  bytesWritten = out.capacity () - len;
163  assert (bytesWritten > 0);
164 
165  if (EVP_EncryptUpdate (&ctx, & (out.front ()) + len, &bytesWritten, hmac.begin (), ECIES_HMAC_SIZE) < 0)
166  {
167  EVP_CIPHER_CTX_cleanup (&ctx);
168  throw std::runtime_error ("");
169  }
170 
171  len += bytesWritten;
172 
173  // encrypt/output plaintext
174  bytesWritten = out.capacity () - len;
175  assert (bytesWritten > 0);
176 
177  if (EVP_EncryptUpdate (&ctx, & (out.front ()) + len, &bytesWritten, & (plaintext.front ()), plaintext.size ()) < 0)
178  {
179  EVP_CIPHER_CTX_cleanup (&ctx);
180  throw std::runtime_error ("");
181  }
182 
183  len += bytesWritten;
184 
185  // finalize
186  bytesWritten = out.capacity () - len;
187 
188  if (EVP_EncryptFinal_ex (&ctx, & (out.front ()) + len, &bytesWritten) < 0)
189  {
190  EVP_CIPHER_CTX_cleanup (&ctx);
191  throw std::runtime_error ("encryption error");
192  }
193 
194  len += bytesWritten;
195 
196  // Output contains: IV, encrypted HMAC, encrypted data, encrypted padding
197  assert (len <= (plaintext.size () + ECIES_HMAC_SIZE + (2 * ECIES_ENC_BLK_SIZE)));
198  assert (len >= (plaintext.size () + ECIES_HMAC_SIZE + ECIES_ENC_BLK_SIZE)); // IV, HMAC, data
199  out.resize (len);
200  EVP_CIPHER_CTX_cleanup (&ctx);
201  return out;
202 }

204 Blob CKey::decryptECIES (CKey& otherKey, Blob const& ciphertext)
205 {
206  // minimum ciphertext = IV + HMAC + 1 block
207  if (ciphertext.size () < ((2 * ECIES_ENC_BLK_SIZE) + ECIES_HMAC_SIZE) )
208  throw std::runtime_error ("ciphertext too short");
209 
210  // extract IV
211  ECIES_ENC_IV_TYPE iv;
212  memcpy (iv.begin (), & (ciphertext.front ()), ECIES_ENC_BLK_SIZE);
213 
214  // begin decrypting
215  EVP_CIPHER_CTX ctx;
216  EVP_CIPHER_CTX_init (&ctx);
217 
218  ECIES_ENC_KEY_TYPE secret;
219  ECIES_HMAC_KEY_TYPE hmacKey;
220  getECIESSecret (otherKey, secret, hmacKey);
221 
222  if (EVP_DecryptInit_ex (&ctx, ECIES_ENC_ALGO, nullptr, secret.begin (), iv.begin ()) != 1)
223  {
224  secret.zero ();
225  hmacKey.zero ();
226  EVP_CIPHER_CTX_cleanup (&ctx);
227  throw std::runtime_error ("unable to init cipher");
228  }
229 
230  // decrypt mac
231  ECIES_HMAC_TYPE hmac;
232  int outlen = ECIES_HMAC_SIZE;
233 
234  if ( (EVP_DecryptUpdate (&ctx, hmac.begin (), &outlen,
235  & (ciphertext.front ()) + ECIES_ENC_BLK_SIZE, ECIES_HMAC_SIZE + 1) != 1) || (outlen != ECIES_HMAC_SIZE) )
236  {
237  secret.zero ();
238  hmacKey.zero ();
239  EVP_CIPHER_CTX_cleanup (&ctx);
240  throw std::runtime_error ("unable to extract hmac");
241  }
242 
243  // decrypt plaintext (after IV and encrypted mac)
244  Blob plaintext (ciphertext.size () - ECIES_HMAC_SIZE - ECIES_ENC_BLK_SIZE);
245  outlen = plaintext.size ();
246 
247  if (EVP_DecryptUpdate (&ctx, & (plaintext.front ()), &outlen,
248  & (ciphertext.front ()) + ECIES_ENC_BLK_SIZE + ECIES_HMAC_SIZE + 1,
249  ciphertext.size () - ECIES_ENC_BLK_SIZE - ECIES_HMAC_SIZE - 1) != 1)
250  {
251  secret.zero ();
252  hmacKey.zero ();
253  EVP_CIPHER_CTX_cleanup (&ctx);
254  throw std::runtime_error ("unable to extract plaintext");
255  }
256 
257  // decrypt padding
258  int flen = 0;
259 
260  if (EVP_DecryptFinal (&ctx, & (plaintext.front ()) + outlen, &flen) != 1)
261  {
262  secret.zero ();
263  hmacKey.zero ();
264  EVP_CIPHER_CTX_cleanup (&ctx);
265  throw std::runtime_error ("plaintext had bad padding");
266  }
267 
268  plaintext.resize (flen + outlen);
269 
270  // verify integrity
271  if (hmac != makeHMAC (hmacKey, plaintext))
272  {
273  secret.zero ();
274  hmacKey.zero ();
275  EVP_CIPHER_CTX_cleanup (&ctx);
276  throw std::runtime_error ("plaintext had bad hmac");
277  }
278 
279  secret.zero ();
280  hmacKey.zero ();
281 
282  EVP_CIPHER_CTX_cleanup (&ctx);
283  return plaintext;
284 }

286 bool checkECIES (void)
287 {
288  CKey senderPriv, recipientPriv, senderPub, recipientPub;
289 
290  for (int i = 0; i < 30000; ++i)
291  {
292  if ((i % 100) == 0)
293  {
294  // generate new keys every 100 times
295  // Log::out() << "new keys";
296  senderPriv.MakeNewKey ();
297  recipientPriv.MakeNewKey ();
298 
299  if (!senderPub.SetPubKey (senderPriv.GetPubKey ()))
300  throw std::runtime_error ("key error");
301 
302  if (!recipientPub.SetPubKey (recipientPriv.GetPubKey ()))
303  throw std::runtime_error ("key error");
304  }
305 
306  // generate message
307  Blob message (4096);
308  int msglen = i % 3000;
309 
310  RandomNumbers::getInstance ().fillBytes (&message.front (), msglen);
311  message.resize (msglen);
312 
313  // encrypt message with sender's private key and recipient's public key
314  Blob ciphertext = senderPriv.encryptECIES (recipientPub, message);
315 
316  // decrypt message with recipient's private key and sender's public key
317  Blob decrypt = recipientPriv.decryptECIES (senderPub, ciphertext);
318 
319  if (decrypt != message)
320  {
321  assert (false);
322  return false;
323  }
324 
325  //Log::out() << "Msg(" << msglen << ") ok " << ciphertext.size();
326  }
327 
328  return true;
329 }
