30 // A Mutex represents an exclusive lock.
31 class Mutex {
32  public:
33  Mutex();
34  ~Mutex();
35 
36  // Lock the mutex. Waits until other lockers have exited.
37  // Will deadlock if the mutex is already locked by this thread.
38  void Lock();
39 
40  // Unlock the mutex.
41  // REQUIRES: This mutex was locked by this thread.
42  void Unlock();
43 
44  // Optionally crash if this thread does not hold this mutex.
45  // The implementation must be fast, especially if NDEBUG is
46  // defined. The implementation is allowed to skip all checks.
47  void AssertHeld();
48 };

79 // (i.e., without word-tearing.)
80 class AtomicPointer {
81  private:
82  intptr_t rep_;
83  public:
84  // Initialize to arbitrary value

87  // Initialize to hold v

90  // Read and return the stored pointer with the guarantee that no
91  // later memory access (read or write) by this thread can be
92  // reordered ahead of this read.
93  void* Acquire_Load() const;
94
