46 // VFALCO NOTE if this function is called quickly to reduce then
47 // increase the number of threads, it could result in
48 // more paused threads being created than expected.
49 //
50 void Workers::setNumberOfThreads (int numberOfThreads)
51 {
52  if (m_numberOfThreads != numberOfThreads)
53  {
54  if (numberOfThreads > m_numberOfThreads)
55  {
56  // Increasing the number of working threads
57 
58  int const amount = numberOfThreads - m_numberOfThreads;
59 
60  for (int i = 0; i < amount; ++i)
61  {
62  // See if we can reuse a paused worker
63  Worker* worker = m_paused.pop_front ();
64 
65  if (worker != nullptr)
66  {
67  // If we got here then the worker thread is at [1]
68  // This will unblock their call to wait()
69  //
70  worker->notify ();
71  }
72  else
73  {
74  worker = new Worker (*this, m_threadNames);
75  }
76 
77  m_everyone.push_front (worker);
78  }
79  }
80  else if (numberOfThreads < m_numberOfThreads)
81  {
82  // Decreasing the number of working threads
83 
84  int const amount = m_numberOfThreads - numberOfThreads;
85 
86  for (int i = 0; i < amount; ++i)
87  {
88  ++m_pauseCount;
89 
90  // Pausing a thread counts as one "internal task"
91  m_semaphore.signal ();
92  }
93  }
94 
95  m_numberOfThreads = numberOfThreads;
96  }
97 }

150 void Workers::Worker::run ()
151 {
152  while (! threadShouldExit ())
153  {
154  // Increment the count of active workers, and if
155  // we are the first one then reset the "all paused" event
156  //
157  if (++m_workers.m_activeCount == 1)
158  m_workers.m_allPaused.reset ();
159 
160  for (;;)
161  {
162  // Acquire a task or "internal task."
163  //
164  m_workers.m_semaphore.wait ();
165 
166  // See if there's a pause request. This
167  // counts as an "internal task."
168  //
169  int pauseCount = m_workers.m_pauseCount.get ();
170 
171  if (pauseCount > 0)
172  {
173  // Try to decrement
174  pauseCount = --m_workers.m_pauseCount;
175 
176  if (pauseCount >= 0)
177  {
178  // We got paused
179  break;
180  }
181  else
182  {
183  // Undo our decrement
184  ++m_workers.m_pauseCount;
185  }
186  }
187 
188  // We couldn't pause so we must have gotten
189  // unblocked in order to process a task.
190  //
191  ++m_workers.m_runningTaskCount;
192  m_workers.m_callback.processTask ();
193  --m_workers.m_runningTaskCount;
194 
195  // Put the name back in case the callback changed it
196  Thread::setCurrentThreadName (Thread::getThreadName());
197  }
198 
199  // Any worker that goes into the paused list must
200  // guarantee that it will eventually block on its
201  // event object.
202  //
203  m_workers.m_paused.push_front (this);
204 
205  // Decrement the count of active workers, and if we
206  // are the last one then signal the "all paused" event.
207  //
208  if (--m_workers.m_activeCount == 0)
209  m_workers.m_allPaused.signal ();
210 
211  Thread::setCurrentThreadName ("(" + getThreadName() + ")");
212 
213  // [1] We will be here when the paused list is popped
214  //
215  // We block on our event object, a requirement of being
216  // put into the paused list.
217  //
218  // This will get signaled on either a reactivate or a stopThread()
219  //
220  wait ();
221  }
222 }
