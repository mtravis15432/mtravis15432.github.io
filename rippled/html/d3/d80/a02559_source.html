113  virtual void CreateFilter(const Slice* keys, int n, std::string* dst) const {
114  std::vector<const char*> key_pointers(n);
115  std::vector<size_t> key_sizes(n);
116  for (int i = 0; i < n; i++) {
117  key_pointers[i] = keys[i].data();
118  key_sizes[i] = keys[i].size();
119  }
120  size_t len;
121  char* filter = (*create_)(state_, &key_pointers[0], &key_sizes[0], n, &len);
122  dst->append(filter, len);
123  free(filter);
124  }

137 static bool SaveError(char** errptr, const Status& s) {
138  assert(errptr != NULL);
139  if (s.ok()) {
140  return false;
141  } else if (*errptr == NULL) {
142  *errptr = strdup(s.ToString().c_str());
143  } else {
144  // TODO(sanjay): Merge with existing error?
145  free(*errptr);
146  *errptr = strdup(s.ToString().c_str());
147  }
148  return true;
149 }

371  void (*deleted)(void*, const char* k, size_t klen)) {

383  };

387  handler.deleted_ = deleted;
