163  // VFALCO NOTE Where's the thread object? It's not a data member...
164  //
165  void run ()
166  {
167  // VFALCO TODO replace this with a beast Time object?
168  //
169  // Initialize the clock to the current time.
170  boost::posix_time::ptime t = boost::posix_time::microsec_clock::universal_time ();
171 
172  while (! threadShouldExit ())
173  {
174  {
175  // VFALCO NOTE What is this lock protecting?
176  ScopedLockType sl (mLock);
177 
178  // VFALCO NOTE I think this is to reduce calls to the operating system
179  // for retrieving the current time.
180  //
181  // TODO Instead of incrementing can't we just retrieve the current
182  // time again?
183  //
184  // Manually update the timer.
185  UptimeTimer::getInstance ().incrementElapsedTime ();
186 
187  // Measure the amount of time we have been deadlocked, in seconds.
188  //
189  // VFALCO NOTE mDeadLock is a canary for detecting the condition.
190  int const timeSpentDeadlocked = UptimeTimer::getInstance ().getElapsedSeconds () - mDeadLock;
191 
192  // VFALCO NOTE I think that "armed" refers to the deadlock detector
193  //
194  int const reportingIntervalSeconds = 10;
195  if (mArmed && (timeSpentDeadlocked >= reportingIntervalSeconds))
196  {
197  // Report the deadlocked condition every 10 seconds
198  if ((timeSpentDeadlocked % reportingIntervalSeconds) == 0)
199  {
200  logDeadlock (timeSpentDeadlocked);
201  }
202 
203  // If we go over 500 seconds spent deadlocked, it means that the
204  // deadlock resolution code has failed, which qualifies as undefined
205  // behavior.
206  //
207  assert (timeSpentDeadlocked < 500);
208  }
209  }
210 
211  bool change;
212 
213  // VFALCO TODO Eliminate the dependence on the Application object.
214  // Choices include constructing with the job queue / feetracker.
215  // Another option is using an observer pattern to invert the dependency.
216  if (getApp().getJobQueue ().isOverloaded ())
217  {
218  m_journal.info << getApp().getJobQueue ().getJson (0);
219  change = getApp().getFeeTrack ().raiseLocalFee ();
220  }
221  else
222  {
223  change = getApp().getFeeTrack ().lowerLocalFee ();
224  }
225 
226  if (change)
227  {
228  // VFALCO TODO replace this with a Listener / observer and subscribe in NetworkOPs or Application
229  getApp().getOPs ().reportFeeChange ();
230  }
231 
232  t += boost::posix_time::seconds (1);
233  boost::posix_time::time_duration when = t - boost::posix_time::microsec_clock::universal_time ();
234 
235  if ((when.is_negative ()) || (when.total_seconds () > 1))
236  {
237  m_journal.warning << "time jump";
238  t = boost::posix_time::microsec_clock::universal_time ();
239  }
240  else
241  {
242  boost::this_thread::sleep (when);
243  }
244  }
245 
246  stopped ();
247  }
