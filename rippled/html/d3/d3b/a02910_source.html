56  {
57 
58  if (reason == InboundLedger::fcCONSENSUS)

61  {
62  ripple::unordered_map<uint256, InboundLedger::pointer>::iterator it = mLedgers.find (mConsensusLedger);

65  oldLedger = it->second;
66  mLedgers.erase (it);
67  }
68  }
69  mConsensusLedger = hash;
70  }
71  else if (reason == InboundLedger::fcVALIDATION)
72  {
73  if (mValidationLedger.isNonZero() && (mValidationLedger != mConsensusLedger) && (hash != mValidationLedger))

76  if (it != mLedgers.end ())
77  {

79  mLedgers.erase (it);
80  }
81  }

83  }
84 
85  ripple::unordered_map<uint256, InboundLedger::pointer>::iterator it = mLedgers.find (hash);
86  if (it != mLedgers.end ())
87  {
88  ret = it->second;
89  // FIXME: Should set the sequence if it's not set

323  void sweep ()
324  {
325  mRecentFailures.sweep ();
326 
327  clock_type::time_point const now (m_clock.now());
328 
329  // Make a list of things to sweep, while holding the lock
330  std::vector <MapType::mapped_type> stuffToSweep;
331  std::size_t total;
332  {
333  ScopedLockType sl (mLock);
334  MapType::iterator it (mLedgers.begin ());
335  total = mLedgers.size ();
336  stuffToSweep.reserve (total);
337 
338  while (it != mLedgers.end ())
339  {
340  if (it->second->getLastAction () > now)
341  {
342  it->second->touch ();
343  ++it;
344  }
345  else if ((it->second->getLastAction () + std::chrono::minutes (1)) < now)
346  {
347  stuffToSweep.push_back (it->second);
348  // shouldn't cause the actual final delete
349  // since we are holding a reference in the vector.
350  it = mLedgers.erase (it);
351  }
352  else
353  {
354  ++it;
355  }
356  }
357  }
358 
359  WriteLog (lsDEBUG, InboundLedger) <<
360  "Sweeped " << stuffToSweep.size () <<
361  " out of " << total << " inbound ledgers.";
362  }
