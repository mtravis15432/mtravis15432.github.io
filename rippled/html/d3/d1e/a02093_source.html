192  // Loop until we satisfy the allocation from an
193  // active page, or we run out of active pages.
194  //
195  for (;;)
196  {
197  // Acquire ownership of an active page
198  // This prevents other threads from seeing it.
199  Page* page (state->active.pop_front());
200  if (page == nullptr)
201  break;
202 
203  void* p = page->allocate (n);
204  if (p != nullptr)
205  {
206  // Put the page back so other threads can use it
207  state->active.push_front (page);
208  return p;
209  }
210 
211  // Page is full, count it for diagnostics
212  ++state->full;
213  }
214 
215  // No active page, get a recycled page or create a new page.
216  //
217  Page* page (state->recycle.pop_front());
218  if (page == nullptr)
219  {
220  page = Page::create (std::max (m_maxBytes, n));
221  ++state->page_count;
222  }
223 
224  void* p = page->allocate (n);
225  bassert (p != nullptr);
226  // Throw page into the active list so other threads can use it
227  state->active.push_front (page);
228  return p;
229  }
