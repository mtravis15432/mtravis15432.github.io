22 TER TrustSetTransactor::doApply ()
23 {
24  TER terResult = tesSUCCESS;
25 
26  STAmount const saLimitAmount (mTxn.getFieldAmount (sfLimitAmount));
27  bool const bQualityIn (mTxn.isFieldPresent (sfQualityIn));
28  bool const bQualityOut (mTxn.isFieldPresent (sfQualityOut));
29 
30  uint160 const uCurrencyID (saLimitAmount.getCurrency ());
31  uint160 uDstAccountID (saLimitAmount.getIssuer ());
32 
33  // true, iff current is high account.
34  bool const bHigh = mTxnAccountID > uDstAccountID;
35 
36  std::uint32_t uQualityIn (bQualityIn ? mTxn.getFieldU32 (sfQualityIn) : 0);
37  std::uint32_t uQualityOut (bQualityOut ? mTxn.getFieldU32 (sfQualityOut) : 0);
38 
39  if (!saLimitAmount.isLegalNet ())
40  return temBAD_AMOUNT;
41 
42  if (bQualityIn && QUALITY_ONE == uQualityIn)
43  uQualityIn = 0;
44 
45  if (bQualityOut && QUALITY_ONE == uQualityOut)
46  uQualityOut = 0;
47 
48  std::uint32_t const uTxFlags = mTxn.getFlags ();
49 
50  if (uTxFlags & tfTrustSetMask)
51  {
52  m_journal.trace <<
53  "Malformed transaction: Invalid flags set.";
54  return temINVALID_FLAG;
55  }
56 
57  bool const bSetAuth = is_bit_set (uTxFlags, tfSetfAuth);
58  bool const bSetNoRipple = is_bit_set (uTxFlags, tfSetNoRipple);
59  bool const bClearNoRipple = is_bit_set (uTxFlags, tfClearNoRipple);
60 
61  if (bSetAuth && !is_bit_set (mTxnAccount->getFieldU32 (sfFlags), lsfRequireAuth))
62  {
63  m_journal.trace <<
64  "Retry: Auth not required.";
65  return tefNO_AUTH_REQUIRED;
66  }
67 
68  if (saLimitAmount.isNative ())
69  {
70  m_journal.trace <<
71  "Malformed transaction: Native credit limit: " <<
72  saLimitAmount.getFullText ();
73  return temBAD_LIMIT;
74  }
75 
76  if (saLimitAmount < zero)
77  {
78  m_journal.trace <<
79  "Malformed transaction: Negative credit limit.";
80  return temBAD_LIMIT;
81  }
82 
83  // Check if destination makes sense.
84  if (!uDstAccountID || uDstAccountID == ACCOUNT_ONE)
85  {
86  m_journal.trace <<
87  "Malformed transaction: Destination account not specified.";
88 
89  return temDST_NEEDED;
90  }
91 
92  if (mTxnAccountID == uDstAccountID)
93  {
94  SLE::pointer selDelete (
95  mEngine->entryCache (ltRIPPLE_STATE,
96  Ledger::getRippleStateIndex (
97  mTxnAccountID, uDstAccountID, uCurrencyID)));
98 
99  if (selDelete)
100  {
101  m_journal.warning <<
102  "Clearing redundant line.";
103 
104  return mEngine->view ().trustDelete (
105  selDelete, mTxnAccountID, uDstAccountID);
106  }
107  else
108  {
109  m_journal.trace <<
110  "Malformed transaction: Can not extend credit to self.";
111  return temDST_IS_SRC;
112  }
113  }
114 
115  SLE::pointer sleDst (mEngine->entryCache (
116  ltACCOUNT_ROOT, Ledger::getAccountRootIndex (uDstAccountID)));
117 
118  if (!sleDst)
119  {
120  m_journal.trace <<
121  "Delay transaction: Destination account does not exist.";
122  return tecNO_DST;
123  }
124 
125  std::uint32_t const uOwnerCount (mTxnAccount->getFieldU32 (sfOwnerCount));
126  // The reserve required to create the line.
127  std::uint64_t const uReserveCreate =
128  (uOwnerCount < 2)
129  ? 0
130  : mEngine->getLedger ()->getReserve (uOwnerCount + 1);
131 
132  STAmount saLimitAllow = saLimitAmount;
133  saLimitAllow.setIssuer (mTxnAccountID);
134 
135  SLE::pointer sleRippleState (mEngine->entryCache (ltRIPPLE_STATE,
136  Ledger::getRippleStateIndex (mTxnAccountID, uDstAccountID, uCurrencyID)));
137 
138  if (sleRippleState)
139  {
140  STAmount saLowBalance;
141  STAmount saLowLimit;
142  STAmount saHighBalance;
143  STAmount saHighLimit;
144  std::uint32_t uLowQualityIn;
145  std::uint32_t uLowQualityOut;
146  std::uint32_t uHighQualityIn;
147  std::uint32_t uHighQualityOut;
148  const uint160& uLowAccountID = !bHigh ? mTxnAccountID : uDstAccountID;
149  const uint160& uHighAccountID = bHigh ? mTxnAccountID : uDstAccountID;
150  SLE::ref sleLowAccount = !bHigh ? mTxnAccount : sleDst;
151  SLE::ref sleHighAccount = bHigh ? mTxnAccount : sleDst;
152 
153  //
154  // Balances
155  //
156 
157  saLowBalance = sleRippleState->getFieldAmount (sfBalance);
158  saHighBalance = -saLowBalance;
159 
160  //
161  // Limits
162  //
163 
164  sleRippleState->setFieldAmount (!bHigh ? sfLowLimit : sfHighLimit, saLimitAllow);
165 
166  saLowLimit = !bHigh ? saLimitAllow : sleRippleState->getFieldAmount (sfLowLimit);
167  saHighLimit = bHigh ? saLimitAllow : sleRippleState->getFieldAmount (sfHighLimit);
168 
169  //
170  // Quality in
171  //
172 
173  if (!bQualityIn)
174  {
175  // Not setting. Just get it.
176 
177  uLowQualityIn = sleRippleState->getFieldU32 (sfLowQualityIn);
178  uHighQualityIn = sleRippleState->getFieldU32 (sfHighQualityIn);
179  }
180  else if (uQualityIn)
181  {
182  // Setting.
183 
184  sleRippleState->setFieldU32 (!bHigh ? sfLowQualityIn : sfHighQualityIn, uQualityIn);
185 
186  uLowQualityIn = !bHigh ? uQualityIn : sleRippleState->getFieldU32 (sfLowQualityIn);
187  uHighQualityIn = bHigh ? uQualityIn : sleRippleState->getFieldU32 (sfHighQualityIn);
188  }
189  else
190  {
191  // Clearing.
192 
193  sleRippleState->makeFieldAbsent (!bHigh ? sfLowQualityIn : sfHighQualityIn);
194 
195  uLowQualityIn = !bHigh ? 0 : sleRippleState->getFieldU32 (sfLowQualityIn);
196  uHighQualityIn = bHigh ? 0 : sleRippleState->getFieldU32 (sfHighQualityIn);
197  }
198 
199  if (QUALITY_ONE == uLowQualityIn) uLowQualityIn = 0;
200 
201  if (QUALITY_ONE == uHighQualityIn) uHighQualityIn = 0;
202 
203  //
204  // Quality out
205  //
206 
207  if (!bQualityOut)
208  {
209  // Not setting. Just get it.
210 
211  uLowQualityOut = sleRippleState->getFieldU32 (sfLowQualityOut);
212  uHighQualityOut = sleRippleState->getFieldU32 (sfHighQualityOut);
213  }
214  else if (uQualityOut)
215  {
216  // Setting.
217 
218  sleRippleState->setFieldU32 (!bHigh ? sfLowQualityOut : sfHighQualityOut, uQualityOut);
219 
220  uLowQualityOut = !bHigh ? uQualityOut : sleRippleState->getFieldU32 (sfLowQualityOut);
221  uHighQualityOut = bHigh ? uQualityOut : sleRippleState->getFieldU32 (sfHighQualityOut);
222  }
223  else
224  {
225  // Clearing.
226 
227  sleRippleState->makeFieldAbsent (!bHigh ? sfLowQualityOut : sfHighQualityOut);
228 
229  uLowQualityOut = !bHigh ? 0 : sleRippleState->getFieldU32 (sfLowQualityOut);
230  uHighQualityOut = bHigh ? 0 : sleRippleState->getFieldU32 (sfHighQualityOut);
231  }
232 
233  std::uint32_t const uFlagsIn (sleRippleState->getFieldU32 (sfFlags));
234  std::uint32_t uFlagsOut (uFlagsIn);
235 
236  if (bSetNoRipple && !bClearNoRipple && (bHigh ? saHighBalance : saLowBalance) >= zero)
237  {
238  uFlagsOut |= (bHigh ? lsfHighNoRipple : lsfLowNoRipple);
239  }
240  else if (bClearNoRipple && !bSetNoRipple)
241  {
242  uFlagsOut &= ~(bHigh ? lsfHighNoRipple : lsfLowNoRipple);
243  }
244 
245  if (QUALITY_ONE == uLowQualityOut) uLowQualityOut = 0;
246 
247  if (QUALITY_ONE == uHighQualityOut) uHighQualityOut = 0;
248 
249 
250  bool const bLowReserveSet = uLowQualityIn || uLowQualityOut ||
251  is_bit_set (uFlagsOut, lsfLowNoRipple) ||
252  !!saLowLimit || saLowBalance > zero;
253  bool const bLowReserveClear = !bLowReserveSet;
254 
255  bool const bHighReserveSet = uHighQualityIn || uHighQualityOut ||
256  is_bit_set (uFlagsOut, lsfHighNoRipple) ||
257  !!saHighLimit || saHighBalance > zero;
258  bool const bHighReserveClear = !bHighReserveSet;
259 
260  bool const bDefault = bLowReserveClear && bHighReserveClear;
261 
262  bool const bLowReserved = is_bit_set (uFlagsIn, lsfLowReserve);
263  bool const bHighReserved = is_bit_set (uFlagsIn, lsfHighReserve);
264 
265  bool bReserveIncrease = false;
266 
267  if (bSetAuth)
268  {
269  uFlagsOut |= (bHigh ? lsfHighAuth : lsfLowAuth);
270  }
271 
272  if (bLowReserveSet && !bLowReserved)
273  {
274  // Set reserve for low account.
275 
276  mEngine->view ().ownerCountAdjust (uLowAccountID, 1, sleLowAccount);
277  uFlagsOut |= lsfLowReserve;
278 
279  if (!bHigh)
280  bReserveIncrease = true;
281  }
282 
283  if (bLowReserveClear && bLowReserved)
284  {
285  // Clear reserve for low account.
286 
287  mEngine->view ().ownerCountAdjust (uLowAccountID, -1, sleLowAccount);
288  uFlagsOut &= ~lsfLowReserve;
289  }
290 
291  if (bHighReserveSet && !bHighReserved)
292  {
293  // Set reserve for high account.
294 
295  mEngine->view ().ownerCountAdjust (uHighAccountID, 1, sleHighAccount);
296  uFlagsOut |= lsfHighReserve;
297 
298  if (bHigh)
299  bReserveIncrease = true;
300  }
301 
302  if (bHighReserveClear && bHighReserved)
303  {
304  // Clear reserve for high account.
305 
306  mEngine->view ().ownerCountAdjust (uHighAccountID, -1, sleHighAccount);
307  uFlagsOut &= ~lsfHighReserve;
308  }
309 
310  if (uFlagsIn != uFlagsOut)
311  sleRippleState->setFieldU32 (sfFlags, uFlagsOut);
312 
313  if (bDefault || CURRENCY_BAD == uCurrencyID)
314  {
315  // Delete.
316 
317  terResult = mEngine->view ().trustDelete (sleRippleState, uLowAccountID, uHighAccountID);
318  }
319  else if (bReserveIncrease
320  && mPriorBalance.getNValue () < uReserveCreate) // Reserve is not scaled by load.
321  {
322  m_journal.trace <<
323  "Delay transaction: Insufficent reserve to add trust line.";
324 
325  // Another transaction could provide XRP to the account and then
326  // this transaction would succeed.
327  terResult = tecINSUF_RESERVE_LINE;
328  }
329  else
330  {
331  mEngine->entryModify (sleRippleState);
332 
333  m_journal.trace <<
334  "Modify ripple line";
335  }
336  }
337  // Line does not exist.
338  else if (!saLimitAmount // Setting default limit.
339  && (!bQualityIn || !uQualityIn) // Not setting quality in or setting default quality in.
340  && (!bQualityOut || !uQualityOut)) // Not setting quality out or setting default quality out.
341  {
342  m_journal.trace <<
343  "Redundant: Setting non-existent ripple line to defaults.";
344  return tecNO_LINE_REDUNDANT;
345  }
346  else if (mPriorBalance.getNValue () < uReserveCreate) // Reserve is not scaled by load.
347  {
348  m_journal.trace <<
349  "Delay transaction: Line does not exist. Insufficent reserve to create line.";
350 
351  // Another transaction could create the account and then this transaction would succeed.
352  terResult = tecNO_LINE_INSUF_RESERVE;
353  }
354  else if (CURRENCY_BAD == uCurrencyID)
355  {
356  terResult = temBAD_CURRENCY;
357  }
358  else
359  {
360  // Zero balance in currency.
361  STAmount saBalance (STAmount (uCurrencyID, ACCOUNT_ONE));
362 
363 
364  m_journal.trace <<
365  "doTrustSet: Creating ripple line: " <<
366  Ledger::getRippleStateIndex (mTxnAccountID, uDstAccountID, uCurrencyID).ToString ();
367 
368  // Create a new ripple line.
369  terResult = mEngine->view ().trustCreate (
370  bHigh,
371  mTxnAccountID,
372  uDstAccountID,
373  Ledger::getRippleStateIndex (mTxnAccountID, uDstAccountID, uCurrencyID),
374  mTxnAccount,
375  bSetAuth,
376  bSetNoRipple && !bClearNoRipple,
377  saBalance,
378  saLimitAllow, // Limit for who is being charged.
379  uQualityIn,
380  uQualityOut);
381  }
382 
383  return terResult;
384 }
