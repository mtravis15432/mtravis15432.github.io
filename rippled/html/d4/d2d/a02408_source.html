62  static std::string getRawDHParams (int keySize)
63  {
64  std::string params;
65 
66  // Original code provided the 512-bit keySize parameters
67  // when 1024 bits were requested so we will do the same.
68  if (keySize == 1024)
69  keySize = 512;
70 
71  switch (keySize)
72  {
73  case 512:
74  {
75  // These are the DH parameters that OpenCoin has chosen for Ripple
76  //
77  std::uint8_t const raw [] = {
78  0x30, 0x46, 0x02, 0x41, 0x00, 0x98, 0x15, 0xd2, 0xd0, 0x08, 0x32, 0xda,
79  0xaa, 0xac, 0xc4, 0x71, 0xa3, 0x1b, 0x11, 0xf0, 0x6c, 0x62, 0xb2, 0x35,
80  0x8a, 0x10, 0x92, 0xc6, 0x0a, 0xa3, 0x84, 0x7e, 0xaf, 0x17, 0x29, 0x0b,
81  0x70, 0xef, 0x07, 0x4f, 0xfc, 0x9d, 0x6d, 0x87, 0x99, 0x19, 0x09, 0x5b,
82  0x6e, 0xdb, 0x57, 0x72, 0x4a, 0x7e, 0xcd, 0xaf, 0xbd, 0x3a, 0x97, 0x55,
83  0x51, 0x77, 0x5a, 0x34, 0x7c, 0xe8, 0xc5, 0x71, 0x63, 0x02, 0x01, 0x02

126  void initAuthenticated (
127  std::string key_file, std::string cert_file, std::string chain_file)
128  {
129  initCommon ();
130 
131  SSL_CTX* const ssl = m_context.native_handle ();
132 
133  bool cert_set = false;
134 
135  if (! cert_file.empty ())
136  {
137  boost::system::error_code ec;
138 
139  m_context.use_certificate_file (
140  cert_file, boost::asio::ssl::context::pem, ec);
141 
142  if (ec)
143  {
144  beast::FatalError (error_message (
145  "Problem with SSL certificate file.", ec).c_str(),
146  __FILE__, __LINE__);
147  }
148 
149  cert_set = true;
150  }
151 
152  if (! chain_file.empty ())
153  {
154  // VFALCO Replace fopen() with RAII
155  FILE* f = fopen (chain_file.c_str (), "r");
156 
157  if (!f)
158  {
159  beast::FatalError (error_message (
160  "Problem opening SSL chain file.", boost::system::error_code (errno,
161  boost::system::generic_category())).c_str(),
162  __FILE__, __LINE__);
163  }
164 
165  try
166  {
167  for (;;)
168  {
169  X509* const x = PEM_read_X509 (f, nullptr, nullptr, nullptr);
170 
171  if (x == nullptr)
172  break;
173 
174  if (! cert_set)
175  {
176  if (SSL_CTX_use_certificate (ssl, x) != 1)
177  beast::FatalError ("Problem retrieving SSL certificate from chain file.",
178  __FILE__, __LINE__);
179 
180  cert_set = true;
181  }
182  else if (SSL_CTX_add_extra_chain_cert (ssl, x) != 1)
183  {
184  X509_free (x);
185  beast::FatalError ("Problem adding SSL chain certificate.",
186  __FILE__, __LINE__);
187  }
188  }
189 
190  fclose (f);
191  }
192  catch (...)
193  {
194  fclose (f);
195  beast::FatalError ("Reading the SSL chain file generated an exception.",
196  __FILE__, __LINE__);
197  }
198  }
199 
200  if (! key_file.empty ())
201  {
202  boost::system::error_code ec;
203 
204  m_context.use_private_key_file (key_file,
205  boost::asio::ssl::context::pem, ec);
206 
207  if (ec)
208  {
209  beast::FatalError (error_message (
210  "Problem using the SSL private key file.", ec).c_str(),
211  __FILE__, __LINE__);
212  }
213  }
214 
215  if (SSL_CTX_check_private_key (ssl) != 1)
216  {
217  beast::FatalError ("Invalid key in SSL private key file.",
218  __FILE__, __LINE__);
219  }
220  }
