55 // Returns the memtables that need to be flushed.
56 void MemTableList::PickMemtablesToFlush(std::vector<MemTable*>* ret) {
57  for (auto it = memlist_.rbegin(); it != memlist_.rend(); it++) {
58  MemTable* m = *it;
59  if (!m->flush_in_progress_) {
60  assert(!m->flush_completed_);
61  num_flush_not_started_--;
62  if (num_flush_not_started_ == 0) {
63  imm_flush_needed.Release_Store(nullptr);
64  }
65  m->flush_in_progress_ = true; // flushing will start very soon
66  ret->push_back(m);
67  }
68  }
69  flush_requested_ = false; // start-flush request is complete
70 }

72 // Record a successful flush in the manifest file
73 Status MemTableList::InstallMemtableFlushResults(
74  const std::vector<MemTable*> &mems,
75  VersionSet* vset, Status flushStatus,
76  port::Mutex* mu, Logger* info_log,
77  uint64_t file_number,
78  std::set<uint64_t>& pending_outputs) {
79  mu->AssertHeld();
80 
81  // If the flush was not successful, then just reset state.
82  // Maybe a suceeding attempt to flush will be successful.
83  if (!flushStatus.ok()) {
84  for (MemTable* m : mems) {
85  assert(m->flush_in_progress_);
86  assert(m->file_number_ == 0);
87 
88  m->flush_in_progress_ = false;
89  m->flush_completed_ = false;
90  m->edit_.Clear();
91  num_flush_not_started_++;
92  imm_flush_needed.Release_Store((void *)1);
93  pending_outputs.erase(file_number);
94  }
95  return flushStatus;
96  }
97 
98  // flush was sucessful
99  for (size_t i = 0; i < mems.size(); ++i) {
100  // All the edits are associated with the first memtable of this batch.
101  assert(i == 0 || mems[i]->GetEdits()->NumEntries() == 0);
102 
103  mems[i]->flush_completed_ = true;
104  mems[i]->file_number_ = file_number;
105  }
106 
107  // if some other thread is already commiting, then return
108  Status s;
109  if (commit_in_progress_) {
110  return s;
111  }
112 
113  // Only a single thread can be executing this piece of code
114  commit_in_progress_ = true;
115 
116  // scan all memtables from the earliest, and commit those
117  // (in that order) that have finished flushing. Memetables
118  // are always committed in the order that they were created.
119  while (!memlist_.empty() && s.ok()) {
120  MemTable* m = memlist_.back(); // get the last element
121  if (!m->flush_completed_) {
122  break;
123  }
124 
125  Log(info_log,
126  "Level-0 commit table #%lu started",
127  (unsigned long)m->file_number_);
128 
129  // this can release and reacquire the mutex.
130  s = vset->LogAndApply(&m->edit_, mu);
131 
132  // All the later memtables that have the same filenum
133  // are part of the same batch. They can be committed now.
134  uint64_t mem_id = 1; // how many memtables has been flushed.
135  do {
136  if (s.ok()) { // commit new state
137  Log(info_log,
138  "Level-0 commit table #%lu: memtable #%lu done",
139  (unsigned long)m->file_number_,
140  (unsigned long)mem_id);
141  memlist_.remove(m);
142  assert(m->file_number_ > 0);
143 
144  // pending_outputs can be cleared only after the newly created file
145  // has been written to a committed version so that other concurrently
146  // executing compaction threads do not mistakenly assume that this
147  // file is not live.
148  pending_outputs.erase(m->file_number_);
149  m->Unref();
150  size_--;
151  } else {
152  //commit failed. setup state so that we can flush again.
153  Log(info_log,
154  "Level-0 commit table #%lu: memtable #%lu failed",
155  (unsigned long)m->file_number_,
156  (unsigned long)mem_id);
157  m->flush_completed_ = false;
158  m->flush_in_progress_ = false;
159  m->edit_.Clear();
160  num_flush_not_started_++;
161  pending_outputs.erase(m->file_number_);
162  m->file_number_ = 0;
163  imm_flush_needed.Release_Store((void *)1);
164  s = Status::IOError("Unable to commit flushed memtable");
165  }
166  ++mem_id;
167  } while (!memlist_.empty() && (m = memlist_.back()) &&
168  m->file_number_ == file_number);
169  }
170  commit_in_progress_ = false;
171  return s;
172 }

174 // New memtables are inserted at the front of the list.
175 void MemTableList::Add(MemTable* m) {
176  assert(size_ >= num_flush_not_started_);
177  size_++;
178  memlist_.push_front(m);
179  m->MarkImmutable();
180  num_flush_not_started_++;
181  if (num_flush_not_started_ == 1) {
182  imm_flush_needed.Release_Store((void *)1);
183  }
184 }
