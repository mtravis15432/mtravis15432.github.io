66 Running a new prepared statement:
67 
68 sqlite3_prepare() // create prepared statement
69 sqlite3_bind() // define input variables
70 sqlite3_step() // retrieve next row

72 sqlite3_finalize() // destroy prepared statement

75 
76 sqlite3_reset() // reset the statement (input bindings are not cleared)

151 void statement_imp::prepare(std::string const& query, bool bRepeatable)
152 {
153  m_query = query;
154  m_session.log_query(query);
155  m_last_insert_rowid = 0;
156 
157  release_resources();
158 
159  char const* tail = 0;
160  int result = sqlite3_prepare_v2(
161  m_session.get_connection(),
162  query.c_str(),
163  static_cast<int>(query.size()),
164  &m_stmt,
165  &tail);
166 
167  if (result == SQLITE_OK)
168  {
169  m_bReady = true;
170  }
171  else
172  {
173  throw detail::sqliteError(__FILE__, __LINE__, result);
174  }
175 }

177 Error statement_imp::execute()
178 {
179  Error error;
180 
181  assert (m_stmt != nullptr);
182 
183  // ???
184  m_bGotData = false;
185  m_session.set_got_data(m_bGotData);
186 
187  // binds
188 
189  int iCol = 0;
190 
191  for (intos_t::iterator iter = m_intos.begin(); iter != m_intos.end(); ++iter)
192  (*iter)->bind(*this, iCol);
193 
194  int iParam = 1;
195 
196  for (uses_t::iterator iter = m_uses.begin(); iter != m_uses.end(); ++iter)
197  (*iter)->bind(*this, iParam);
198 
199  // reset
200  error = detail::sqliteError(__FILE__, __LINE__, sqlite3_reset(m_stmt));
201 
202  if (!error)
203  {
204  // set input variables
205  do_uses();
206 
207  m_bReady = true;
208  m_bFirstTime = true;
209  }
210 
211  return error;
212 }

214 bool statement_imp::fetch(Error& error)
215 {
216  int result = sqlite3_step(m_stmt);
217 
218  if (result == SQLITE_ROW ||
219  result == SQLITE_DONE)
220  {
221  if (m_bFirstTime)
222  {
223  m_last_insert_rowid = m_session.last_insert_rowid();
224  m_bFirstTime = false;
225  }
226 
227  if (result == SQLITE_ROW)
228  {
229  m_bGotData = true;
230  m_session.set_got_data(m_bGotData);
231 
232  do_intos();
233  }
234  else
235  {
236  m_bGotData = false;
237  m_session.set_got_data(m_bGotData);
238 
239  if (result == SQLITE_DONE)
240  {
241  m_bReady = false;
242  }
243  }
244  }
245  else if (result != SQLITE_OK)
246  {
247  m_bGotData = false;
248  error = detail::sqliteError(__FILE__, __LINE__, result);
249  }
250  else
251  {
252  // should never get SQLITE_OK here
253  throw std::invalid_argument ("invalid result");
254  }
255 
256  return m_bGotData;
257 }
