464 void LedgerEntrySet::calcRawMeta (Serializer& s, TER result, std::uint32_t index)
465 {
466  // calculate the raw meta data and return it. This must be called before the set is committed
467 
468  // Entries modified only as a result of building the transaction metadata
469  ripple::unordered_map<uint256, SLE::pointer> newMod;
470 
471  typedef std::map<uint256, LedgerEntrySetEntry>::value_type u256_LES_pair;
472  BOOST_FOREACH (u256_LES_pair & it, mEntries)
473  {
474  SField::ptr type = &sfGeneric;
475 
476  switch (it.second.mAction)
477  {
478  case taaMODIFY:
479 #ifdef META_DEBUG
480  WriteLog (lsTRACE, LedgerEntrySet) << "Modified Node " << it.first;
481 #endif
482  type = &sfModifiedNode;
483  break;
484 
485  case taaDELETE:
486 #ifdef META_DEBUG
487  WriteLog (lsTRACE, LedgerEntrySet) << "Deleted Node " << it.first;
488 #endif
489  type = &sfDeletedNode;
490  break;
491 
492  case taaCREATE:
493 #ifdef META_DEBUG
494  WriteLog (lsTRACE, LedgerEntrySet) << "Created Node " << it.first;
495 #endif
496  type = &sfCreatedNode;
497  break;
498 
499  default: // ignore these
500  break;
501  }
502 
503  if (type == &sfGeneric)
504  continue;
505 
506  SLE::pointer origNode = mLedger->getSLEi (it.first);
507  SLE::pointer curNode = it.second.mEntry;
508 
509  if ((type == &sfModifiedNode) && (*curNode == *origNode))
510  continue;
511 
512  std::uint16_t nodeType = curNode ? curNode->getFieldU16 (sfLedgerEntryType)
513  : origNode->getFieldU16 (sfLedgerEntryType);
514 
515  mSet.setAffectedNode (it.first, *type, nodeType);
516 
517  if (type == &sfDeletedNode)
518  {
519  assert (origNode && curNode);
520  threadOwners (origNode, mLedger, newMod); // thread transaction to owners
521 
522  STObject prevs (sfPreviousFields);
523  BOOST_FOREACH (const SerializedType & obj, *origNode)
524  {
525  // go through the original node for modified fields saved on modification
526  if (obj.getFName ().shouldMeta (SField::sMD_ChangeOrig) && !curNode->hasMatchingEntry (obj))
527  prevs.addObject (obj);
528  }
529 
530  if (!prevs.empty ())
531  mSet.getAffectedNode (it.first).addObject (prevs);
532 
533  STObject finals (sfFinalFields);
534  BOOST_FOREACH (const SerializedType & obj, *curNode)
535  {
536  // go through the final node for final fields
537  if (obj.getFName ().shouldMeta (SField::sMD_Always | SField::sMD_DeleteFinal))
538  finals.addObject (obj);
539  }
540 
541  if (!finals.empty ())
542  mSet.getAffectedNode (it.first).addObject (finals);
543  }
544  else if (type == &sfModifiedNode)
545  {
546  assert (curNode && origNode);
547 
548  if (curNode->isThreadedType ()) // thread transaction to node it modified
549  threadTx (curNode, mLedger, newMod);
550 
551  STObject prevs (sfPreviousFields);
552  BOOST_FOREACH (const SerializedType & obj, *origNode)
553  {
554  // search the original node for values saved on modify
555  if (obj.getFName ().shouldMeta (SField::sMD_ChangeOrig) && !curNode->hasMatchingEntry (obj))
556  prevs.addObject (obj);
557  }
558 
559  if (!prevs.empty ())
560  mSet.getAffectedNode (it.first).addObject (prevs);
561 
562  STObject finals (sfFinalFields);
563  BOOST_FOREACH (const SerializedType & obj, *curNode)
564  {
565  // search the final node for values saved always
566  if (obj.getFName ().shouldMeta (SField::sMD_Always | SField::sMD_ChangeNew))
567  finals.addObject (obj);
568  }
569 
570  if (!finals.empty ())
571  mSet.getAffectedNode (it.first).addObject (finals);
572  }
573  else if (type == &sfCreatedNode) // if created, thread to owner(s)
574  {
575  assert (curNode && !origNode);
576  threadOwners (curNode, mLedger, newMod);
577 
578  if (curNode->isThreadedType ()) // always thread to self
579  threadTx (curNode, mLedger, newMod);
580 
581  STObject news (sfNewFields);
582  BOOST_FOREACH (const SerializedType & obj, *curNode)
583  {
584  // save non-default values
585  if (!obj.isDefault () && obj.getFName ().shouldMeta (SField::sMD_Create | SField::sMD_Always))
586  news.addObject (obj);
587  }
588 
589  if (!news.empty ())
590  mSet.getAffectedNode (it.first).addObject (news);
591  }
592  else assert (false);
593  }
594 
595  // add any new modified nodes to the modification set
596  typedef std::map<uint256, SLE::pointer>::value_type u256_sle_pair;
597  BOOST_FOREACH (u256_sle_pair & it, newMod)
598  entryModify (it.second);
599 
600  mSet.addRaw (s, result, index);
601  WriteLog (lsTRACE, LedgerEntrySet) << "Metadata:" << mSet.getJson (0);
602 }

746 TER LedgerEntrySet::dirDelete (

748  const std::uint64_t& uNodeDir, // --> Node containing entry.
749  uint256 const& uRootIndex, // --> The index of the base of the directory. Nodes are based off of this.
750  uint256 const& uLedgerIndex, // --> Value to remove from directory.
751  const bool bStable, // --> True, not to change relative order of entries.
752  const bool bSoft) // --> True, uNodeDir is not hard and fast (pass uNodeDir=0).
753 {
754  std::uint64_t uNodeCur = uNodeDir;
755  SLE::pointer sleNode = entryCache (ltDIR_NODE, Ledger::getDirNodeIndex (uRootIndex, uNodeCur));
756 
757  if (!sleNode)
758  {

760  " uRootIndex=" << uRootIndex.ToString () <<
761  " uNodeDir=" << strHex (uNodeDir) <<
762  " uLedgerIndex=" << uLedgerIndex.ToString ();

764  if (!bSoft)
765  {
766  assert (false);
767  return tefBAD_LEDGER;
768  }
769  else if (uNodeDir < 20)
770  {
771  // Go the extra mile. Even if node doesn't exist, try the next node.
772 
773  return dirDelete (bKeepRoot, uNodeDir + 1, uRootIndex, uLedgerIndex, bStable, true);
774  }
775  else
776  {
777  return tefBAD_LEDGER;
778  }
779  }
780 
781  STVector256 svIndexes = sleNode->getFieldV256 (sfIndexes);
782  std::vector<uint256>& vuiIndexes = svIndexes.peekValue ();
783  std::vector<uint256>::iterator it;
784 
785  it = std::find (vuiIndexes.begin (), vuiIndexes.end (), uLedgerIndex);
786 
787  if (vuiIndexes.end () == it)
788  {
789  if (!bSoft)
790  {
791  assert (false);
792 
793  WriteLog (lsWARNING, LedgerEntrySet) << "dirDelete: no such entry";
794 
795  return tefBAD_LEDGER;
796  }
797  else if (uNodeDir < 20)
798  {
799  // Go the extra mile. Even if entry not in node, try the next node.
800 
801  return dirDelete (bKeepRoot, uNodeDir + 1, uRootIndex, uLedgerIndex, bStable, true);
802  }
803  else
804  {
805  return tefBAD_LEDGER;
806  }
807  }
808 
809  // Remove the element.
810  if (vuiIndexes.size () > 1)
811  {
812  if (bStable)
813  {
814  vuiIndexes.erase (it);
815  }
816  else
817  {
818  *it = vuiIndexes[vuiIndexes.size () - 1];

820  }
821  }
822  else
823  {
824  vuiIndexes.clear ();
825  }

828  entryModify (sleNode);
829

834  std::uint64_t uNodeNext = sleNode->getFieldU64 (sfIndexNext);
835

839

843  entryDelete (sleNode);
844  }
845  // Root overflowed.
846  else if (bKeepRoot)
847  {
848  // If root overflowed and not allowed to delete overflowed root node.
849

851  }
852  else if (uNodePrevious != uNodeNext)
853  {
854  // Have more than 2 nodes. Can't delete root node.
855 
856  nothing ();
857  }
858  else

861  SLE::pointer sleLast = entryCache (ltDIR_NODE, Ledger::getDirNodeIndex (uRootIndex, uNodeNext));
862 
863  assert (sleLast);
864

866  {
867  // Both nodes are empty.
868 
869  entryDelete (sleNode); // Delete root.
870  entryDelete (sleLast); // Delete last.
871  }
872  else
873  {
874  // Have an entry, can't delete root node.
875 
876  nothing ();
877  }
878  }
879  }
880  // Just emptied a non-root node.
881  else if (uNodeNext)
882  {
883  // Not root and not last node. Can delete node.
884 
885  SLE::pointer slePrevious = entryCache (ltDIR_NODE, Ledger::getDirNodeIndex (uRootIndex, uNodePrevious));
886 
887  assert (slePrevious);
888 
889  SLE::pointer sleNext = entryCache (ltDIR_NODE, Ledger::getDirNodeIndex (uRootIndex, uNodeNext));
890 
891  assert (slePrevious);
892  assert (sleNext);

899  }

905  return tefBAD_LEDGER;
906  }

909  slePrevious->setFieldU64 (sfIndexNext, uNodeNext);
910  entryModify (slePrevious);
911 
912  // Fix next to point to its new previous.
913  sleNext->setFieldU64 (sfIndexPrevious, uNodePrevious);
914  entryModify (sleNext);
915 
916  entryDelete(sleNode);
917  }
918  // Last node.
919  else if (bKeepRoot || uNodePrevious)
920  {
921  // Not allowed to delete last node as root was overflowed.
922  // Or, have pervious entries preventing complete delete.
923 
924  nothing ();
925  }
926  else
927  {
928  // Last and only node besides the root.
929  SLE::pointer sleRoot = entryCache (ltDIR_NODE, uRootIndex);
930 
931  assert (sleRoot);
932 
933  if (sleRoot->getFieldV256 (sfIndexes).peekValue ().empty ())
934  {
935  // Both nodes are empty.
936 
937  entryDelete (sleRoot); // Delete root.
938  entryDelete (sleNode); // Delete last.
939  }
940  else
941  {
942  // Root has an entry, can't delete.
943 
944  nothing ();
945  }
946  }
947  }
948 
949  return tesSUCCESS;
