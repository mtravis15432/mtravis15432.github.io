22 bool ParameterNode::setValue (const std::string& name, const Json::Value& value, Json::Value& error)
23 {
24  if (name.empty ()) // this node
25  return setValue (value, error);
26 
27  size_t dot = name.find ('.');
28 
29  if (dot == std::string::npos) // a child of this node
30  {
31  std::map<std::string, Parameter::pointer>::iterator it = mChildren.find (name);
32 
33  if (it == mChildren.end ())
34  {
35  error = Json::objectValue;
36  error["error"] = "Name not found";
37  error["name"] = name;
38  return false;
39  }
40 
41  return it->second->setValue (value, error);
42  }
43 
44  std::map<std::string, Parameter::pointer>::iterator it = mChildren.find (name.substr (0, dot));
45 
46  if (it == mChildren.end ())
47  {
48  error = Json::objectValue;
49  error["error"] = "Name not found";
50  error["name"] = name;
51  return false;
52  }
53 
54  ParameterNode* n = dynamic_cast<ParameterNode*> (it->second.get ());
55 
56  if (!n)

58  error = Json::objectValue;

60  error["name"] = it->second->getName ();
61  return false;
62  }
63 
64  return n->setValue (name.substr (dot + 1), value, error);
65 }

67 bool ParameterNode::addNode (const std::string& name, Parameter::ref node)
68 {
69  if (name.empty ()) // this node
70  return false;
71 
72  size_t dot = name.find ('.');
73 
74  if (dot == std::string::npos) // a child of this node
75  {
76  std::map<std::string, Parameter::pointer>::iterator it = mChildren.find (name);
77 
78  if (it != mChildren.end ())
79  return false;
80 
81  mChildren[name] = node;
82  return true;
83  }
84 
85  std::map<std::string, Parameter::pointer>::iterator it = mChildren.find (name.substr (0, dot));
86  ParameterNode* n;
87 
88  if (it == mChildren.end ())
89  {
90  // create a new inner node
91  ParameterNode::pointer node = boost::make_shared<ParameterNode> (getShared (), name.substr (0, dot));
92  n = dynamic_cast<ParameterNode*> (node.get ());
93  assert (n);
94  mChildren[name] = node;
95  }
96  else
97  {
98  // existing node passed through must be inner
99  n = dynamic_cast<ParameterNode*> (it->second.get ());
100 
101  if (!n)
102  return false;
103  }
104 
105  return n->addNode (name.substr (dot + 1), node);
106 }
