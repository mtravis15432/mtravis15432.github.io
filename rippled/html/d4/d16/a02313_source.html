29  static struct tm millisToLocal (const std::int64_t millis) noexcept
30  {
31  struct tm result;
32  const std::int64_t seconds = millis / 1000;
33 
34  if (seconds < 86400LL || seconds >= 2145916800LL)
35  {
36  // use extended maths for dates beyond 1970 to 2037..
37  const int timeZoneAdjustment = 31536000 - (int) (Time (1971, 0, 1, 0, 0).toMilliseconds() / 1000);
38  const std::int64_t jdm = seconds + timeZoneAdjustment + 210866803200LL;
39 
40  const int days = (int) (jdm / 86400LL);
41  const int a = 32044 + days;
42  const int b = (4 * a + 3) / 146097;
43  const int c = a - (b * 146097) / 4;
44  const int d = (4 * c + 3) / 1461;
45  const int e = c - (d * 1461) / 4;
46  const int m = (5 * e + 2) / 153;
47 
48  result.tm_mday = e - (153 * m + 2) / 5 + 1;
49  result.tm_mon = m + 2 - 12 * (m / 10);
50  result.tm_year = b * 100 + d - 6700 + (m / 10);
51  result.tm_wday = (days + 1) % 7;
52  result.tm_yday = -1;
53 
54  int t = (int) (jdm % 86400LL);
55  result.tm_hour = t / 3600;
56  t %= 3600;
57  result.tm_min = t / 60;
58  result.tm_sec = t % 60;
59  result.tm_isdst = -1;
60  }
61  else
62  {
63  time_t now = static_cast <time_t> (seconds);
64 
65  #if BEAST_WINDOWS
66  #ifdef _INC_TIME_INL
67  if (now >= 0 && now <= 0x793406fff)
68  localtime_s (&result, &now);
69  else
70  zerostruct (result);
71  #else
72  result = *localtime (&now);
73  #endif
74  #else
75 
76  localtime_r (&now, &result); // more thread-safe
77  #endif
78  }
79 
80  return result;
81  }

136 Time::Time (const int year,
137  const int month,
138  const int day,
139  const int hours,
140  const int minutes,
141  const int seconds,
142  const int milliseconds,
143  const bool useLocalTime) noexcept
144 {
145  bassert (year > 100); // year must be a 4-digit version
146 
147  if (year < 1971 || year >= 2038 || ! useLocalTime)
148  {
149  // use extended maths for dates beyond 1970 to 2037..
150  const int timeZoneAdjustment = useLocalTime ? (31536000 - (int) (Time (1971, 0, 1, 0, 0).toMilliseconds() / 1000))
151  : 0;
152  const int a = (13 - month) / 12;
153  const int y = year + 4800 - a;
154  const int jd = day + (153 * (month + 12 * a - 2) + 2) / 5
155  + (y * 365) + (y / 4) - (y / 100) + (y / 400)
156  - 32045;
157 
158  const std::int64_t s = ((std::int64_t) jd) * 86400LL - 210866803200LL;
159 
160  millisSinceEpoch = 1000 * (s + (hours * 3600 + minutes * 60 + seconds - timeZoneAdjustment))
161  + milliseconds;
162  }
163  else
164  {
165  struct tm t;
166  t.tm_year = year - 1900;
167  t.tm_mon = month;
168  t.tm_mday = day;
169  t.tm_hour = hours;
170  t.tm_min = minutes;
171  t.tm_sec = seconds;
172  t.tm_isdst = -1;
173 
174  millisSinceEpoch = 1000 * (std::int64_t) mktime (&t);
175 
176  if (millisSinceEpoch < 0)
177  millisSinceEpoch = 0;
178  else
179  millisSinceEpoch += milliseconds;
180  }
181 }
