197  HostAndPort parseName(std::string const& str)
198  {
199  // Attempt to find the first and last non-whitespace
200  auto const find_whitespace = std::bind (
201  &std::isspace <std::string::value_type>,
202  std::placeholders::_1,
203  std::locale ());
204 
205  auto host_first = std::find_if_not (
206  str.begin (), str.end (), find_whitespace);
207 
208  auto port_last = std::find_if_not (
209  str.rbegin (), str.rend(), find_whitespace).base();
210 
211  // This should only happen for all-whitespace strings
212  if (host_first >= port_last)
213  return std::make_pair(std::string (), std::string ());
214 
215  // Attempt to find the first and last valid port separators
216  auto const find_port_separator = [](char const c) -> bool
217  {
218  if (std::isspace (c))
219  return true;
220 
221  if (c == ':')
222  return true;
223 
224  return false;
225  };
226 
227  auto host_last = std::find_if (
228  host_first, port_last, find_port_separator);
229 
230  auto port_first = std::find_if_not (
231  host_last, port_last, find_port_separator);
232 
233  return make_pair (
234  std::string (host_first, host_last),
235  std::string (port_first, port_last));
236  }

238  void do_work (CompletionCounter)
239  {
240  if (m_stop_called == true)
241  return;
242 
243  // We don't have any work to do at this time
244  if (m_work.empty ())
245  return;
246 
247  std::string const name (m_work.front ().names.back());
248  HandlerType handler (m_work.front ().handler);
249 
250  m_work.front ().names.pop_back ();
251 
252  if (m_work.front ().names.empty ())
253  m_work.pop_front();
254 
255  HostAndPort const hp (parseName (name));
256 
257  if (hp.first.empty ())
258  {
259  m_journal.error <<
260  "Unable to parse '" << name << "'";
261 
262  m_io_service.post (m_strand.wrap (boost::bind (
263  &ResolverAsioImpl::do_work, this,
264  CompletionCounter (this))));
265 
266  return;
267  }
268 
269  boost::asio::ip::tcp::resolver::query query (
270  hp.first, hp.second);
271 
272  m_resolver.async_resolve (query, boost::bind (
273  &ResolverAsioImpl::do_finish, this, name,
274  boost::asio::placeholders::error, handler,
275  boost::asio::placeholders::iterator,
276  CompletionCounter (this)));
277  }

279  void do_resolve (std::vector <std::string> const& names,
280  HandlerType const& handler, CompletionCounter)
281  {
282  assert (! names.empty());
283 
284  if (m_stop_called == false)
285  {
286  m_work.emplace_back (names, handler);
287 
288  m_journal.debug <<
289  "Queued new job with " << names.size() <<
290  " tasks. " << m_work.size() << " jobs outstanding.";
291 
292  if (m_work.size() == 1)
293  {
294  m_io_service.post (m_strand.wrap (boost::bind (
295  &ResolverAsioImpl::do_work, this,
296  CompletionCounter (this))));
297  }
298  }
