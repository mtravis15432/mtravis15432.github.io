373  void doConnectPeers (IPAddresses const& addresses)
374  {
375  for (IPAddresses::const_iterator iter (addresses.begin());
376  iter != addresses.end(); ++iter)
377  {
378  IP::Endpoint const& remote_endpoint (*iter);
379  Node* const remote_node (m_network.find (remote_endpoint));
380  // Acquire slot
381  Slot::ptr const local_slot (
382  m_logic->new_outbound_slot (remote_endpoint));
383  if (! local_slot)
384  continue;
385  // See if the address is connectible
386  if (remote_node == nullptr || ! remote_node->canAccept())
387  {
388  // Firewalled or no one listening
389  // Post notification
390  m_network.post (std::bind (&Logic::on_closed,
391  &logic(), local_slot));
392  continue;
393  }
394  IP::Endpoint const local_endpoint (
395  listening_endpoint().at_port (m_next_port++));
396  // Acquire slot
397  Slot::ptr const remote_slot (
398  remote_node->logic().new_inbound_slot (
399  remote_endpoint, local_endpoint));
400  if (! remote_slot)
401  continue;
402  // Connection established, create links
403  m_links.emplace_back (*this, local_slot, local_endpoint,
404  *remote_node, remote_endpoint, false);
405  remote_node->m_links.emplace_back (*remote_node, remote_slot,
406  remote_endpoint, *this, local_endpoint, true);
407  // Post notifications
408  m_network.post (std::bind (&Logic::on_connected,
409  &logic(), local_endpoint, remote_endpoint));
410  m_network.post (std::bind (&Node::doCheckAccept,
411  remote_node, boost::ref (*this), local_endpoint));
412  }
