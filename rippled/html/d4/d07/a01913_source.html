130  void run ()
131  {
132  unique_lock lock (m_mutex);;
133 
134  for (;;)
135  {
136  for (auto iter : m_workers)
137  iter->sample();
138 
139  clock_type::time_point const when (
140  floor <seconds> (
141  clock_type::now().time_since_epoch()) +

143
