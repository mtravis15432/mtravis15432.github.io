22 TER PaymentTransactor::doApply ()
23 {
24  // Ripple if source or destination is non-native or if there are paths.
25  std::uint32_t const uTxFlags = mTxn.getFlags ();
26  bool const bPartialPayment = is_bit_set (uTxFlags, tfPartialPayment);
27  bool const bLimitQuality = is_bit_set (uTxFlags, tfLimitQuality);
28  bool const bNoRippleDirect = is_bit_set (uTxFlags, tfNoRippleDirect);
29  bool const bPaths = mTxn.isFieldPresent (sfPaths);
30  bool const bMax = mTxn.isFieldPresent (sfSendMax);
31  uint160 const uDstAccountID = mTxn.getFieldAccount160 (sfDestination);
32  STAmount const saDstAmount = mTxn.getFieldAmount (sfAmount);
33  STAmount const saMaxAmount = bMax
34  ? mTxn.getFieldAmount (sfSendMax)
35  : saDstAmount.isNative ()
36  ? saDstAmount
37  : STAmount (saDstAmount.getCurrency (),
38  mTxnAccountID,
39  saDstAmount.getMantissa (),
40  saDstAmount.getExponent (),
41  saDstAmount < zero);
42  uint160 const uSrcCurrency = saMaxAmount.getCurrency ();
43  uint160 const uDstCurrency = saDstAmount.getCurrency ();
44  bool const bXRPDirect = uSrcCurrency.isZero () && uDstCurrency.isZero ();
45 
46  m_journal.trace <<
47  "saMaxAmount=" << saMaxAmount.getFullText () <<
48  " saDstAmount=" << saDstAmount.getFullText ();
49 
50  if (!saDstAmount.isLegalNet () || !saMaxAmount.isLegalNet ())
51  return temBAD_AMOUNT;
52 
53  if (uTxFlags & tfPaymentMask)
54  {
55  m_journal.trace <<
56  "Malformed transaction: Invalid flags set.";
57 
58  return temINVALID_FLAG;
59  }
60  else if (!uDstAccountID)
61  {
62  m_journal.trace <<
63  "Malformed transaction: Payment destination account not specified.";
64 
65  return temDST_NEEDED;
66  }
67  else if (bMax && saMaxAmount <= zero)
68  {
69  m_journal.trace <<
70  "Malformed transaction: bad max amount: " << saMaxAmount.getFullText ();
71 
72  return temBAD_AMOUNT;
73  }
74  else if (saDstAmount <= zero)
75  {
76  m_journal.trace <<
77  "Malformed transaction: bad dst amount: " << saDstAmount.getFullText ();
78 
79  return temBAD_AMOUNT;
80  }
81  else if (CURRENCY_BAD == uSrcCurrency || CURRENCY_BAD == uDstCurrency)
82  {
83  m_journal.trace <<
84  "Malformed transaction: Bad currency.";
85 
86  return temBAD_CURRENCY;
87  }
88  else if (mTxnAccountID == uDstAccountID && uSrcCurrency == uDstCurrency && !bPaths)
89  {
90  m_journal.trace <<
91  "Malformed transaction: Redundant transaction:" <<
92  " src=" << mTxnAccountID.ToString () <<
93  " dst=" << uDstAccountID.ToString () <<
94  " src_cur=" << uSrcCurrency.ToString () <<
95  " dst_cur=" << uDstCurrency.ToString ();
96 
97  return temREDUNDANT;
98  }
99  else if (bMax && saMaxAmount == saDstAmount && saMaxAmount.getCurrency () == saDstAmount.getCurrency ())
100  {
101  m_journal.trace <<
102  "Malformed transaction: Redundant SendMax.";
103 
104  return temREDUNDANT_SEND_MAX;
105  }
106  else if (bXRPDirect && bMax)
107  {
108  m_journal.trace <<
109  "Malformed transaction: SendMax specified for XRP to XRP.";
110 
111  return temBAD_SEND_XRP_MAX;
112  }
113  else if (bXRPDirect && bPaths)
114  {
115  m_journal.trace <<
116  "Malformed transaction: Paths specified for XRP to XRP.";
117 
118  return temBAD_SEND_XRP_PATHS;
119  }
120  else if (bXRPDirect && bPartialPayment)
121  {
122  m_journal.trace <<
123  "Malformed transaction: Partial payment specified for XRP to XRP.";
124 
125  return temBAD_SEND_XRP_PARTIAL;
126  }
127  else if (bXRPDirect && bLimitQuality)
128  {
129  m_journal.trace <<
130  "Malformed transaction: Limit quality specified for XRP to XRP.";
131 
132  return temBAD_SEND_XRP_LIMIT;
133  }
134  else if (bXRPDirect && bNoRippleDirect)
135  {
136  m_journal.trace <<
137  "Malformed transaction: No ripple direct specified for XRP to XRP.";
138 
139  return temBAD_SEND_XRP_NO_DIRECT;
140  }
141 
142  SLE::pointer sleDst (mEngine->entryCache (
143  ltACCOUNT_ROOT, Ledger::getAccountRootIndex (uDstAccountID)));
144 
145  if (!sleDst)
146  {
147  // Destination account does not exist.
148 
149  if (!saDstAmount.isNative ())
150  {
151  m_journal.trace <<
152  "Delay transaction: Destination account does not exist.";
153 
154  // Another transaction could create the account and then this transaction would succeed.
155  return tecNO_DST;
156  }
157  else if (is_bit_set (mParams, tapOPEN_LEDGER) && bPartialPayment)
158  {
159  // Make retry work smaller, by rejecting this.
160  m_journal.trace <<
161  "Delay transaction: Partial payment not allowed to create account.";
162 
163 
164  // Another transaction could create the account and then this
165  // transaction would succeed.
166  return telNO_DST_PARTIAL;
167  }
168  // Note: Reserve is not scaled by load.
169  else if (saDstAmount.getNValue () < mEngine->getLedger ()->getReserve (0))
170  {
171  m_journal.trace <<
172  "Delay transaction: Destination account does not exist. " <<
173  "Insufficent payment to create account.";
174 
175  // Another transaction could create the account and then this
176  // transaction would succeed.
177  return tecNO_DST_INSUF_XRP;
178  }
179 
180  // Create the account.
181  sleDst = mEngine->entryCreate (
182  ltACCOUNT_ROOT, Ledger::getAccountRootIndex (uDstAccountID));
183 
184  sleDst->setFieldAccount (sfAccount, uDstAccountID);
185  sleDst->setFieldU32 (sfSequence, 1);
186  }
187  else if ((sleDst->getFlags () & lsfRequireDestTag) && !mTxn.isFieldPresent (sfDestinationTag))
188  {
189  m_journal.trace <<
190  "Malformed transaction: DestinationTag required.";
191 
192  return tefDST_TAG_NEEDED;
193  }
194  else
195  {
196  mEngine->entryModify (sleDst);
197  }
198 
199  TER terResult;
200  // XXX Should bMax be sufficient to imply ripple?
201  bool const bRipple = bPaths || bMax || !saDstAmount.isNative ();
202 
203  if (bRipple)
204  {
205  // Ripple payment
206 
207  STPathSet spsPaths = mTxn.getFieldPathSet (sfPaths);
208  std::vector<PathState::pointer> vpsExpanded;
209  STAmount saMaxAmountAct;
210  STAmount saDstAmountAct;
211 
212  try
213  {
214  bool const openLedger = is_bit_set (mParams, tapOPEN_LEDGER);
215  bool const tooManyPaths = spsPaths.size () > MaxPathSize;
216 
217  terResult = openLedger && tooManyPaths
218  ? telBAD_PATH_COUNT // Too many paths for proposed ledger.
219  : RippleCalc::rippleCalc (
220  mEngine->view (),
221  saMaxAmountAct,
222  saDstAmountAct,
223  vpsExpanded,
224  saMaxAmount,
225  saDstAmount,
226  uDstAccountID,
227  mTxnAccountID,
228  spsPaths,
229  bPartialPayment,
230  bLimitQuality,
231  bNoRippleDirect, // Always compute for finalizing ledger.
232  false, // Not standalone, delete unfundeds.
233  is_bit_set (mParams, tapOPEN_LEDGER));
234 
235  if (isTerRetry(terResult))
236  terResult = tecPATH_DRY;
237 
238  if ((tesSUCCESS == terResult) && (saDstAmountAct != saDstAmount))
239  mEngine->view().setDeliveredAmount (saDstAmountAct);
240  }
241  catch (std::exception const& e)
242  {
243  m_journal.trace <<
244  "Caught throw: " << e.what ();
245 
246  terResult = tefEXCEPTION;
247  }
248  }
249  else
250  {
251  // Direct XRP payment.
252 
253  std::uint32_t const uOwnerCount (mTxnAccount->getFieldU32 (sfOwnerCount));
254  std::uint64_t const uReserve (mEngine->getLedger ()->getReserve (uOwnerCount));
255 
256  // Make sure have enough reserve to send. Allow final spend to use reserve for fee.
257  if (mPriorBalance < saDstAmount + std::max(uReserve, mTxn.getTransactionFee ().getNValue ()))
258  {
259  // Vote no. However, transaction might succeed, if applied in a different order.
260  m_journal.trace << "Delay transaction: Insufficient funds: " <<
261  " " << mPriorBalance.getText () <<
262  " / " << (saDstAmount + uReserve).getText () <<
263  " (" << uReserve << ")";
264 
265  terResult = tecUNFUNDED_PAYMENT;
266  }
267  else
268  {
269  mTxnAccount->setFieldAmount (sfBalance, mSourceBalance - saDstAmount);
270  sleDst->setFieldAmount (sfBalance, sleDst->getFieldAmount (sfBalance) + saDstAmount);
271 
272  // re-arm the password change fee if we can and need to
273  if ((sleDst->getFlags () & lsfPasswordSpent))
274  sleDst->clearFlag (lsfPasswordSpent);
275 
276  terResult = tesSUCCESS;
277  }
278  }
279 
280  std::string strToken;
281  std::string strHuman;
282 
283  if (transResultInfo (terResult, strToken, strHuman))
284  {
285  m_journal.trace <<
286  strToken << ": " << strHuman;
287  }
288  else
289  {
290  assert (false);
291  }
292 
293  return terResult;
294 }
