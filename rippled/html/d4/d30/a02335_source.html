67 // used to order into and uses that need conversions
68 template<typename T>
69 struct base_value_holder
70 {
71  typename type_conversion<T>::base_type m_value;
72 };

74 // create into_type from type_conversion
75 template <typename T>
76 class conversion_into_type
77  : private base_value_holder<T>
78  , public into_type <typename type_conversion<T>::base_type>
79  , public Uncopyable
80 {
81 public:
82  typedef typename type_conversion<T>::base_type BASE_TYPE;
83 
84  conversion_into_type(T& value)
85  : into_type<BASE_TYPE>(detail::base_value_holder<T>::m_value, m_own_ind)
86  , m_value(value), m_ind(m_own_ind) {}
87 
88  conversion_into_type(T& value, indicator& ind)

90  , m_value(value), m_ind(ind) {}
91 
92 private:
93  void convert_from_base()

105 // used to create a use_type from a type_conversion
106 template<typename T>
107 class conversion_use_type
108  : private base_value_holder<T>
109  , public use_type<typename type_conversion<T>::base_type>
110  , public Uncopyable
111 {
112 public:
113  typedef typename type_conversion<T>::base_type BASE_TYPE;
114 
115  conversion_use_type(T& value)
116  : use_type<BASE_TYPE>(detail::base_value_holder<T>::m_value, m_own_ind)
117  , m_value(value), m_ind(m_own_ind), m_bReadOnly(false) {}
118 
119  conversion_use_type(T const& value)
120  : use_type<BASE_TYPE>(detail::base_value_holder<T>::m_value, m_own_ind)
121  , m_value(const_cast<T&>(value)), m_ind(m_own_ind), m_bReadOnly(true) {}
122 
123  conversion_use_type(T& value, indicator& ind)
124  : use_type<BASE_TYPE>(detail::base_value_holder<T>::m_value, ind)
125  , m_value(value), m_ind(ind), m_bReadOnly(false) {}
126 
127  conversion_use_type(T const& value, indicator& ind)
128  : use_type<BASE_TYPE>(detail::base_value_holder<T>::m_value, ind)
129  , m_value(const_cast<T&>(value)), m_ind(ind), m_bReadOnly(true) {}
130 
131  void convert_from_base()
132  {
133  if (!m_bReadOnly)
134  {
135  type_conversion<T>::from_base(
136  detail::base_value_holder<T>::m_value, m_ind, m_value);
137  }
138  }
139 
140  void convert_to_base()
141  {
142  type_conversion<T>::to_base(m_value,
143  detail::base_value_holder<T>::m_value, m_ind);
144  }
145 
146 private:
147  T& m_value;
148  indicator m_own_ind;
149  indicator& m_ind;
150  bool m_bReadOnly;
151 };
