25 RippleLineCache::pointer PathRequests::getLineCache (Ledger::pointer& ledger, bool authoritative)
26 {
27  ScopedLockType sl (mLock);
28 
29  std::uint32_t lineSeq = mLineCache ? mLineCache->getLedger()->getLedgerSeq() : 0;
30  std::uint32_t lgrSeq = ledger->getLedgerSeq();
31 
32  if ( (lineSeq == 0) || // no ledger
33  (authoritative && (lgrSeq > lineSeq)) || // newer authoritative ledger
34  (authoritative && ((lgrSeq + 8) < lineSeq)) || // we jumped way back for some reason
35  (lgrSeq > (lineSeq + 8))) // we jumped way forward for some reason
36  {
37  ledger = boost::make_shared<Ledger>(*ledger, false); // Take a snapshot of the ledger
38  mLineCache = boost::make_shared<RippleLineCache> (ledger);
39  }
40  else
41  {
42  ledger = mLineCache->getLedger();
43  }
44  return mLineCache;
45 }

47 void PathRequests::updateAll (Ledger::ref inLedger, CancelCallback shouldCancel)
48 {
49  std::vector<PathRequest::wptr> requests;
50 
51  LoadEvent::autoptr event (getApp().getJobQueue().getLoadEventAP(jtPATH_FIND, "PathRequest::updateAll"));
52 
53  // Get the ledger and cache we should be using
54  Ledger::pointer ledger = inLedger;
55  RippleLineCache::pointer cache;
56  {
57  ScopedLockType sl (mLock);
58  requests = mRequests;
59  cache = getLineCache (ledger, true);
60  }
61 
62  bool newRequests = getApp().getLedgerMaster().isNewPathRequest();
63  bool mustBreak = false;
64 
65  mJournal.trace << "updateAll seq=" << ledger->getLedgerSeq() << ", " <<
66  requests.size() << " requests";
67  int processed = 0, removed = 0;
68 
69  do
70  {
71 
72  BOOST_FOREACH (PathRequest::wref wRequest, requests)
73  {
74  if (shouldCancel())
75  break;
76 
77  bool remove = true;
78  PathRequest::pointer pRequest = wRequest.lock ();
79 
80  if (pRequest)
81  {
82  if (!pRequest->needsUpdate (newRequests, ledger->getLedgerSeq ()))
83  remove = false;
84  else
85  {
86  InfoSub::pointer ipSub = pRequest->getSubscriber ();
87  if (ipSub)
88  {
89  ipSub->getConsumer ().charge (Resource::feePathFindUpdate);
90  if (!ipSub->getConsumer ().warn ())
91  {
92  Json::Value update = pRequest->doUpdate (cache, false);
93  update["type"] = "path_find";
94  ipSub->send (update, false);
95  remove = false;
96  ++processed;
97  }
98  }
99  }
100  }
101 
102  if (remove)
103  {
104  PathRequest::pointer pRequest = wRequest.lock ();
105 
106  ScopedLockType sl (mLock);
107 
108  // Remove any dangling weak pointers or weak pointers that refer to this path request.
109  std::vector<PathRequest::wptr>::iterator it = mRequests.begin();
110  while (it != mRequests.end())
111  {
112  PathRequest::pointer itRequest = it->lock ();
113  if (!itRequest || (itRequest == pRequest))
114  {
115  ++removed;
116  it = mRequests.erase (it);
117  }
118  else
119  ++it;
120  }
121  }
122 
123  mustBreak = !newRequests && getApp().getLedgerMaster().isNewPathRequest();
124  if (mustBreak) // We weren't handling new requests and then there was a new request
125  break;
126 
127  }
128 
129  if (mustBreak)
130  { // a new request came in while we were working
131  newRequests = true;
132  }
133  else if (newRequests)
134  { // we only did new requests, so we always need a last pass
135  newRequests = getApp().getLedgerMaster().isNewPathRequest();
136  }
137  else
138  { // check if there are any new requests, otherwise we are done
139  newRequests = getApp().getLedgerMaster().isNewPathRequest();
140  if (!newRequests) // We did a full pass and there are no new requests
141  return;
142  }
143 
144  {
145  // Get the latest requests, cache, and ledger for next pass
146  ScopedLockType sl (mLock);
147 
148  if (mRequests.empty())
149  break;
150  requests = mRequests;
151 
152  cache = getLineCache (ledger, false);
153  }
154 
155  }
156  while (!shouldCancel ());
157 
158  mJournal.debug << "updateAll complete " << processed << " process and " <<
159  removed << " removed";
160 }

162 Json::Value PathRequests::makePathRequest(
163  boost::shared_ptr <InfoSub> const& subscriber,
164  const boost::shared_ptr<Ledger>& inLedger,
165  const Json::Value& requestJson)
166 {
167  PathRequest::pointer req = boost::make_shared<PathRequest> (
168  subscriber, ++mLastIdentifier, *this, mJournal);
169 
170  Ledger::pointer ledger = inLedger;
171  RippleLineCache::pointer cache;
172 
173  {
174  ScopedLockType sl (mLock);
175  cache = getLineCache (ledger, false);
176  }
177 
178  bool valid = false;
179  Json::Value result = req->doCreate (ledger, cache, requestJson, valid);
180 
181  if (valid)
182  {
183  {
184  ScopedLockType sl (mLock);
185 
186  // Insert after any older unserviced requests but before any serviced requests
187  std::vector<PathRequest::wptr>::iterator it = mRequests.begin ();
188  while (it != mRequests.end ())
189  {
190  PathRequest::pointer req = it->lock ();
191  if (req && !req->isNew ())
192  break; // This request has been handled, we come before it
193 
194  // This is a newer request, we come after it
195  ++it;
196  }
197  mRequests.insert (it, PathRequest::wptr (req));
198 
199  }
200  subscriber->setPathRequest (req);
201  getApp().getLedgerMaster().newPathRequest();
202  }
203  return result;
204 }
