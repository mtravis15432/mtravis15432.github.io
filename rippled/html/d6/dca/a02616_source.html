69 Status WriteBatch::Iterate(Handler* handler) const {
70  Slice input(rep_);
71  if (input.size() < kHeader) {
72  return Status::Corruption("malformed WriteBatch (too small)");
73  }
74 
75  input.remove_prefix(kHeader);
76  Slice key, value, blob;
77  int found = 0;
78  while (!input.empty() && handler->Continue()) {
79  char tag = input[0];
80  input.remove_prefix(1);
81  switch (tag) {
82  case kTypeValue:
83  if (GetLengthPrefixedSlice(&input, &key) &&
84  GetLengthPrefixedSlice(&input, &value)) {
85  handler->Put(key, value);
86  found++;
87  } else {
88  return Status::Corruption("bad WriteBatch Put");
89  }
90  break;
91  case kTypeDeletion:
92  if (GetLengthPrefixedSlice(&input, &key)) {
93  handler->Delete(key);
94  found++;
95  } else {
96  return Status::Corruption("bad WriteBatch Delete");
97  }
98  break;
99  case kTypeMerge:
100  if (GetLengthPrefixedSlice(&input, &key) &&
101  GetLengthPrefixedSlice(&input, &value)) {
102  handler->Merge(key, value);
103  found++;
104  } else {
105  return Status::Corruption("bad WriteBatch Merge");
106  }
107  break;
108  case kTypeLogData:
109  if (GetLengthPrefixedSlice(&input, &blob)) {
110  handler->LogData(blob);
111  } else {
112  return Status::Corruption("bad WriteBatch Blob");
113  }
114  break;
115  default:
116  return Status::Corruption("unknown WriteBatch tag");
117  }
118  }
119  if (found != WriteBatchInternal::Count(this)) {
120  return Status::Corruption("WriteBatch has wrong count");
121  } else {
122  return Status::OK();
123  }
124 }

175 class MemTableInserter : public WriteBatch::Handler {
176  public:
177  SequenceNumber sequence_;
178  MemTable* mem_;
179  const Options* options_;
180  DBImpl* db_;
181  const bool filter_deletes_;
182 
183  MemTableInserter(SequenceNumber sequence, MemTable* mem, const Options* opts,
184  DB* db, const bool filter_deletes)
185  : sequence_(sequence),
186  mem_(mem),
187  options_(opts),
188  db_(reinterpret_cast<DBImpl*>(db)),
189  filter_deletes_(filter_deletes) {
190  assert(mem_);
191  if (filter_deletes_) {
192  assert(options_);
193  assert(db_);
194  }
195  }
196 
197  virtual void Put(const Slice& key, const Slice& value) {
198  if (options_->inplace_update_support
199  && mem_->Update(sequence_, kTypeValue, key, value)) {
200  RecordTick(options_->statistics, NUMBER_KEYS_UPDATED);
201  } else {
202  mem_->Add(sequence_, kTypeValue, key, value);
203  }
204  sequence_++;
205  }
206  virtual void Merge(const Slice& key, const Slice& value) {
207  mem_->Add(sequence_, kTypeMerge, key, value);
208  sequence_++;
209  }
210  virtual void Delete(const Slice& key) {
211  if (filter_deletes_) {
212  SnapshotImpl read_from_snapshot;
213  read_from_snapshot.number_ = sequence_;
214  ReadOptions ropts;
215  ropts.snapshot = &read_from_snapshot;
216  std::string value;
217  if (!db_->KeyMayExist(ropts, key, &value)) {
218  RecordTick(options_->statistics, NUMBER_FILTERED_DELETES);
219  return;
220  }
221  }
222  mem_->Add(sequence_, kTypeDeletion, key, Slice());
223  sequence_++;
224  }
225 };
226 } // namespace
