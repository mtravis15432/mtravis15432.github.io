166  void parseBootstrapIPs (std::string const& name, SiteFiles::Section const& section)
167  {
168  std::size_t n (0);
169  for (SiteFiles::Section::DataType::const_iterator iter (
170  section.data().begin()); iter != section.data().end(); ++iter)
171  {
172  std::string const& s (*iter);
173  beast::IP::Endpoint addr (beast::IP::Endpoint::from_string (s));
174  if (is_unspecified (addr))
175  addr = beast::IP::Endpoint::from_string_altform(s);
176  if (! is_unspecified (addr))
177  {
178  // add IP::Endpoint to bootstrap cache
179  ++n;
180  }
181  }
182 
183  m_journal.info <<
184  "Added " << n <<
185  " bootstrap IPs from " << name;
186  }
