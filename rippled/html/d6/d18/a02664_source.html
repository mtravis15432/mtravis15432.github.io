52 // Helper routine: decode the next block entry starting at "p",
53 // storing the number of shared key bytes, non_shared key bytes,
54 // and the length of the value in "*shared", "*non_shared", and
55 // "*value_length", respectively. Will not derefence past "limit".
56 //
57 // If any errors are detected, returns nullptr. Otherwise, returns a
58 // pointer to the key delta (just past the three decoded values).
59 static inline const char* DecodeEntry(const char* p, const char* limit,
60  uint32_t* shared,
61  uint32_t* non_shared,
62  uint32_t* value_length) {
63  if (limit - p < 3) return nullptr;
64  *shared = reinterpret_cast<const unsigned char*>(p)[0];
65  *non_shared = reinterpret_cast<const unsigned char*>(p)[1];
66  *value_length = reinterpret_cast<const unsigned char*>(p)[2];
67  if ((*shared | *non_shared | *value_length) < 128) {
68  // Fast path: all three values are encoded in one byte each
69  p += 3;
70  } else {
71  if ((p = GetVarint32Ptr(p, limit, shared)) == nullptr) return nullptr;
72  if ((p = GetVarint32Ptr(p, limit, non_shared)) == nullptr) return nullptr;
73  if ((p = GetVarint32Ptr(p, limit, value_length)) == nullptr) return nullptr;
74  }
75 
76  if (static_cast<uint32_t>(limit - p) < (*non_shared + *value_length)) {
77  return nullptr;
78  }
79  return p;
80 }

110  void SeekToRestartPoint(uint32_t index) {
111  key_.clear();
112  restart_index_ = index;
113  // current_ will be fixed by ParseNextKey();
114 
115  // ParseNextKey() starts at the end of value_, so set value_ accordingly
116  uint32_t offset = GetRestartPoint(index);
117  value_ = Slice(data_ + offset, 0);

150  virtual void Prev() {
151  assert(Valid());
152 
153  // Scan backwards to a restart point before current_
154  const uint32_t original = current_;
155  while (GetRestartPoint(restart_index_) >= original) {
156  if (restart_index_ == 0) {
157  // No more entries
158  current_ = restarts_;
159  restart_index_ = num_restarts_;
160  return;
161  }
162  restart_index_--;
163  }
164 
165  SeekToRestartPoint(restart_index_);
166  do {
167  // Loop until end of current entry hits the start of original entry
168  } while (ParseNextKey() && NextEntryOffset() < original);
169  }

171  virtual void Seek(const Slice& target) {
172  // Binary search in restart array to find the first restart point
173  // with a key >= target
174  uint32_t left = 0;
175  uint32_t right = num_restarts_ - 1;
176  while (left < right) {
177  uint32_t mid = (left + right + 1) / 2;
178  uint32_t region_offset = GetRestartPoint(mid);
179  uint32_t shared, non_shared, value_length;
180  const char* key_ptr = DecodeEntry(data_ + region_offset,
181  data_ + restarts_,
182  &shared, &non_shared, &value_length);
183  if (key_ptr == nullptr || (shared != 0)) {
184  CorruptionError();
185  return;
186  }
187  Slice mid_key(key_ptr, non_shared);
188  if (Compare(mid_key, target) < 0) {
189  // Key at "mid" is smaller than "target". Therefore all
190  // blocks before "mid" are uninteresting.
191  left = mid;
192  } else {
193  // Key at "mid" is >= "target". Therefore all blocks at or
194  // after "mid" are uninteresting.
195  right = mid - 1;
196  }
197  }
198 
199  // Linear search (within restart block) for first key >= target
200  SeekToRestartPoint(left);
201  while (true) {
202  if (!ParseNextKey()) {
203  return;
204  }
205  if (Compare(key_, target) >= 0) {
206  return;
207  }
208  }
209  }

224  void CorruptionError() {

228  key_.clear();

232  bool ParseNextKey() {
233  current_ = NextEntryOffset();
234  const char* p = data_ + current_;
235  const char* limit = data_ + restarts_; // Restarts come right after data
236  if (p >= limit) {
237  // No more entries to return. Mark as invalid.
238  current_ = restarts_;
239  restart_index_ = num_restarts_;
240  return false;
241  }
242 
243  // Decode next entry
244  uint32_t shared, non_shared, value_length;
245  p = DecodeEntry(p, limit, &shared, &non_shared, &value_length);
246  if (p == nullptr || key_.size() < shared) {
247  CorruptionError();
248  return false;
249  } else {
250  key_.resize(shared);
251  key_.append(p, non_shared);
252  value_ = Slice(p + non_shared, value_length);
253  while (restart_index_ + 1 < num_restarts_ &&
254  GetRestartPoint(restart_index_ + 1) < current_) {
255  ++restart_index_;
256  }
257  return true;
258  }
259  }

262 Iterator* Block::NewIterator(const Comparator* cmp) {
263  if (size_ < 2*sizeof(uint32_t)) {
264  return NewErrorIterator(Status::Corruption("bad block contents"));
265  }
266  const uint32_t num_restarts = NumRestarts();
267  if (num_restarts == 0) {
268  return NewEmptyIterator();
