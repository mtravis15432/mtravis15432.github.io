131 TER Interpreter::interpret (Contract* contract, const SerializedTransaction& txn, Blob& code)
132 {
133  mContract = contract;
134  mCode = &code;
135  mTotalFee = 0;
136  mInstructionPointer = 0;
137 
138  while (mInstructionPointer < code.size ())
139  {
140  unsigned int fun = (*mCode)[mInstructionPointer];
141  mInstructionPointer++;
142 
143  if (fun >= mFunctionTable.size ())
144  {
145  // TODO: log
146  return (temMALFORMED); // TODO: is this actually what we want to do?
147  }
148 
149  mTotalFee += mFunctionTable[ fun ]->getFee (); // FIXME: You can't use fees this way, there's no consensus
150 
151  if (mTotalFee > txn.getTransactionFee ().getNValue ())
152  {
153  // TODO: log
154  return (telINSUF_FEE_P);
155  }
156  else
157  {
158  if (!mFunctionTable[ fun ]->work (this))
159  {
160  // TODO: log
161  return (temMALFORMED); // TODO: is this actually what we want to do?
162  }
163  }
164  }
165 
166  return (tesSUCCESS);
167 }

193 bool Interpreter::jumpTo (int offset)
194 {
195  if (offset < 0)
196  {
197  if (-offset > mInstructionPointer)
198  return false;
199  }
200  else
201  {
202  if (offset > mCode->size () ||
203  mInstructionPointer > mCode->size () - offset)
204  return false;
205  }
206 
207  mInstructionPointer += offset;
208  return (true);
