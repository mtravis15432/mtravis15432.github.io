28 Json::Value RPCHandler::doLedgerEntry (Json::Value params, Resource::Charge& loadType, Application::ScopedLockType& masterLockHolder)

32  Ledger::pointer lpLedger;
33  Json::Value jvResult = RPC::lookupLedger (params, lpLedger, *mNetOps);
34 
35  if (!lpLedger)
36  return jvResult;
37 
38  uint256 uNodeIndex;
39  bool bNodeBinary = false;
40 
41  if (params.isMember ("index"))
42  {
43  // XXX Needs to provide proof.
44  uNodeIndex.SetHex (params["index"].asString ());
45  bNodeBinary = true;
46  }
47  else if (params.isMember ("account_root"))
48  {
49  RippleAddress naAccount;
50 
51  if (!naAccount.setAccountID (params["account_root"].asString ())
52  || !naAccount.getAccountID ())
53  {
54  jvResult["error"] = "malformedAddress";
55  }
56  else
57  {
58  uNodeIndex = Ledger::getAccountRootIndex (naAccount.getAccountID ());
59  }
60  }
61  else if (params.isMember ("directory"))
62  {
63  if (!params["directory"].isObject ())
64  {
65  uNodeIndex.SetHex (params["directory"].asString ());
66  }
67  else if (params["directory"].isMember ("sub_index")
68  && !params["directory"]["sub_index"].isIntegral ())
69  {
70  jvResult["error"] = "malformedRequest";
71  }
72  else
73  {
74  std::uint64_t uSubIndex = params["directory"].isMember ("sub_index")
75  ? params["directory"]["sub_index"].asUInt ()
76  : 0;
77 
78  if (params["directory"].isMember ("dir_root"))
79  {
80  uint256 uDirRoot;
81 
82  uDirRoot.SetHex (params["dir_root"].asString ());
83 
84  uNodeIndex = Ledger::getDirNodeIndex (uDirRoot, uSubIndex);
85  }
86  else if (params["directory"].isMember ("owner"))
87  {
88  RippleAddress naOwnerID;
89 
90  if (!naOwnerID.setAccountID (params["directory"]["owner"].asString ()))
91  {
92  jvResult["error"] = "malformedAddress";
93  }
94  else
95  {
96  uint256 uDirRoot = Ledger::getOwnerDirIndex (naOwnerID.getAccountID ());
97 
98  uNodeIndex = Ledger::getDirNodeIndex (uDirRoot, uSubIndex);
99  }
100  }
101  else
102  {
103  jvResult["error"] = "malformedRequest";
104  }
105  }
106  }
107  else if (params.isMember ("generator"))
108  {
109  RippleAddress naGeneratorID;
110 
111  if (!params["generator"].isObject ())
112  {
113  uNodeIndex.SetHex (params["generator"].asString ());
114  }
115  else if (!params["generator"].isMember ("regular_seed"))
116  {
117  jvResult["error"] = "malformedRequest";
118  }
119  else if (!naGeneratorID.setSeedGeneric (params["generator"]["regular_seed"].asString ()))
120  {
121  jvResult["error"] = "malformedAddress";
122  }
123  else
124  {
125  RippleAddress na0Public; // To find the generator's index.
126  RippleAddress naGenerator = RippleAddress::createGeneratorPublic (naGeneratorID);
127 
128  na0Public.setAccountPublic (naGenerator, 0);
129 
130  uNodeIndex = Ledger::getGeneratorIndex (na0Public.getAccountID ());
131  }
132  }
133  else if (params.isMember ("offer"))
134  {
135  RippleAddress naAccountID;
136 
137  if (!params["offer"].isObject ())
138  {
139  uNodeIndex.SetHex (params["offer"].asString ());
140  }
141  else if (!params["offer"].isMember ("account")
142  || !params["offer"].isMember ("seq")
143  || !params["offer"]["seq"].isIntegral ())
144  {
145  jvResult["error"] = "malformedRequest";
146  }
147  else if (!naAccountID.setAccountID (params["offer"]["account"].asString ()))
148  {
149  jvResult["error"] = "malformedAddress";
150  }
151  else
152  {
153  std::uint32_t uSequence = params["offer"]["seq"].asUInt ();
154

156  }
157  }
158  else if (params.isMember ("ripple_state"))
159  {
160  RippleAddress naA;
161  RippleAddress naB;
162  uint160 uCurrency;
163  Json::Value jvRippleState = params["ripple_state"];
164

167  || !jvRippleState.isMember ("accounts")
168  || !jvRippleState["accounts"].isArray ()
169  || 2 != jvRippleState["accounts"].size ()
170  || !jvRippleState["accounts"][0u].isString ()
171  || !jvRippleState["accounts"][1u].isString ()
172  || jvRippleState["accounts"][0u].asString () == jvRippleState["accounts"][1u].asString ()
173  )
174  {
175  jvResult["error"] = "malformedRequest";
176  }
177  else if (!naA.setAccountID (jvRippleState["accounts"][0u].asString ())
178  || !naB.setAccountID (jvRippleState["accounts"][1u].asString ()))
179  {
180  jvResult["error"] = "malformedAddress";
181  }
182  else if (!STAmount::currencyFromString (uCurrency, jvRippleState["currency"].asString ()))
183  {
184  jvResult["error"] = "malformedCurrency";
185  }
186  else
187  {
188  uNodeIndex = Ledger::getRippleStateIndex (naA, naB, uCurrency);
189  }
190  }
191  else
192  {

194  }
195 
196  if (uNodeIndex.isNonZero ())

202 
203  if (!sleNode)

213

227 }
