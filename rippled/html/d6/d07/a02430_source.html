137 // Updates our Door list based on settings.
138 //
139 void ServerImpl::handle_update ()
140 {
141  if (! stopping())
142  {
143  // Make a local copy to shorten the lock
144  //
145  Ports ports;
146  {
147  SharedState::ConstAccess state (m_state);
148  ports = state->ports;
149  }
150 
151  std::sort (ports.begin(), ports.end());
152 
153  // Walk the Door list and the Port list simultaneously and
154  // build a replacement Door vector which we will then swap in.
155  //
156  Doors doors;
157  Doors::iterator door (m_doors.begin());
158  for (Ports::const_iterator port (ports.begin());
159  port != ports.end(); ++port)
160  {
161  int comp;
162 
163  while (door != m_doors.end() &&
164  ((comp = compare (*port, (*door)->port())) > 0))
165  {
166  (*door)->cancel();
167  ++door;
168  }
169 
170  if (door != m_doors.end())
171  {
172  if (comp < 0)
173  {
174  doors.push_back (new Door (*this, *port));
175  }
176  else
177  {
178  // old Port and new Port are the same
179  doors.push_back (*door);
180  ++door;
181  }
182  }
183  else
184  {
185  doors.push_back (new Door (*this, *port));
186  }
187  }
188 
189  // Any remaining Door objects are not in the new set, so cancel them.
190  //
191  for (;door != m_doors.end();)
192  (*door)->cancel();
193 
194  m_doors.swap (doors);
195  }
196  else
197  {
198  // Cancel pending I/O on all doors.
199  //
200  for (Doors::iterator iter (m_doors.begin());
201  iter != m_doors.end(); ++iter)
202  {
203  (*iter)->cancel();
204  }
205 
206  // Remove our references to the old doors.
207  //
208  m_doors.resize (0);
209  }
210 }
