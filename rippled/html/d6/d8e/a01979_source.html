49 void printspeed(char *caption, unsigned long bytes, double time) {
50  if (bytes / 1073741824UL > 0) {
51  printf("%s %.4f sec (%.3f GBps)\n", caption, time, (double)bytes/1073741824UL/time);
52  } else if (bytes / 1048576 > 0) {
