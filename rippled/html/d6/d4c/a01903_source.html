274 template <typename Type>
275 inline Type Atomic<Type>::get() const noexcept
276 {
277  #if BEAST_ATOMICS_MAC
278  return sizeof (Type) == 4 ? castFrom32Bit ((std::int32_t) OSAtomicAdd32Barrier ((int32_t) 0, (BEAST_MAC_ATOMICS_VOLATILE int32_t*) &value))
279  : castFrom64Bit ((std::int64_t) OSAtomicAdd64Barrier ((int64_t) 0, (BEAST_MAC_ATOMICS_VOLATILE int64_t*) &value));
280  #elif BEAST_ATOMICS_WINDOWS
281  return sizeof (Type) == 4 ? castFrom32Bit ((std::int32_t) beast_InterlockedExchangeAdd ((volatile long*) &value, (long) 0))
282  : castFrom64Bit ((std::int64_t) beast_InterlockedExchangeAdd64 ((volatile __int64*) &value, (__int64) 0));
283  #elif BEAST_ATOMICS_GCC
284  return sizeof (Type) == 4 ? castFrom32Bit ((std::int32_t) __sync_add_and_fetch ((volatile std::int32_t*) &value, 0))
285  : castFrom64Bit ((std::int64_t) __sync_add_and_fetch ((volatile std::int64_t*) &value, 0));
286  #endif
287 }

289 template <typename Type>
290 inline Type Atomic<Type>::exchange (const Type newValue) noexcept
291 {
292  #if BEAST_ATOMICS_MAC || BEAST_ATOMICS_GCC
293  Type currentVal = value;
294  while (! compareAndSetBool (newValue, currentVal)) { currentVal = value; }
295  return currentVal;
296  #elif BEAST_ATOMICS_WINDOWS
297  return sizeof (Type) == 4 ? castFrom32Bit ((std::int32_t) beast_InterlockedExchange ((volatile long*) &value, (long) castTo32Bit (newValue)))

299  #endif

374 template <typename Type>
375 inline bool Atomic<Type>::compareAndSetBool (const Type newValue, const Type valueToCompare) noexcept
376 {
377  #if BEAST_ATOMICS_MAC
378  return sizeof (Type) == 4 ? OSAtomicCompareAndSwap32Barrier ((int32_t) castTo32Bit (valueToCompare), (int32_t) castTo32Bit (newValue), (BEAST_MAC_ATOMICS_VOLATILE int32_t*) &value)
379  : OSAtomicCompareAndSwap64Barrier ((int64_t) castTo64Bit (valueToCompare), (int64_t) castTo64Bit (newValue), (BEAST_MAC_ATOMICS_VOLATILE int64_t*) &value);
380  #elif BEAST_ATOMICS_WINDOWS
381  return compareAndSetValue (newValue, valueToCompare) == valueToCompare;
382  #elif BEAST_ATOMICS_GCC
383  return sizeof (Type) == 4 ? __sync_bool_compare_and_swap ((volatile std::int32_t*) &value, castTo32Bit (valueToCompare), castTo32Bit (newValue))
384  : __sync_bool_compare_and_swap ((volatile std::int64_t*) &value, castTo64Bit (valueToCompare), castTo64Bit (newValue));

388 template <typename Type>
389 inline Type Atomic<Type>::compareAndSetValue (const Type newValue, const Type valueToCompare) noexcept
390 {
391  #if BEAST_ATOMICS_MAC
392  for (;;) // Annoying workaround for only having a bool CAS operation..
393  {
394  if (compareAndSetBool (newValue, valueToCompare))
395  return valueToCompare;
396 
397  const Type result = value;
398  if (result != valueToCompare)
399  return result;
400  }
401 
402  #elif BEAST_ATOMICS_WINDOWS
403  return sizeof (Type) == 4 ? castFrom32Bit ((std::int32_t) beast_InterlockedCompareExchange ((volatile long*) &value, (long) castTo32Bit (newValue), (long) castTo32Bit (valueToCompare)))
404  : castFrom64Bit ((std::int64_t) beast_InterlockedCompareExchange64 ((volatile __int64*) &value, (__int64) castTo64Bit (newValue), (__int64) castTo64Bit (valueToCompare)));
405  #elif BEAST_ATOMICS_GCC
406  return sizeof (Type) == 4 ? castFrom32Bit ((std::int32_t) __sync_val_compare_and_swap ((volatile std::int32_t*) &value, castTo32Bit (valueToCompare), castTo32Bit (newValue)))
407  : castFrom64Bit ((std::int64_t) __sync_val_compare_and_swap ((volatile std::int64_t*) &value, castTo64Bit (valueToCompare), castTo64Bit (newValue)));
