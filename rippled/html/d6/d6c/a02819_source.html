484  template <typename RandomType, typename PeerContainer>
485  void make_connections (
486  PeerContainer& s,
487  typename PeerContainer::size_type outDegree,
488  RandomType r = RandomType())
489  {
490  // Turn the PeerContainer into a PeerSequence
491  typedef typename PeerContainer::pointer pointer;
492  typedef std::vector <pointer> PeerSequence;
493  PeerSequence v;
494  v.reserve (s.size());
495  for (typename PeerContainer::iterator iter (s.begin());
496  iter != s.end(); ++iter)
497  v.push_back (&(*iter));
498 
499  // Make random connections
500  typedef typename PeerSequence::size_type size_type;
501  size_type const n (v.size());
502  for (typename PeerSequence::iterator iter (v.begin());
503  iter != v.end(); ++iter)
504  {
505  for (size_type i (0); i < outDegree; ++i)
506  {
507  for(;;)
508  {
509  size_type const j (r.nextInt (n));
510 
511  // prohibit self connection
512  if (*iter == v[j])
513  continue;
514 
515  // prohibit duplicate connection
516  if (is_connected (**iter, *v[j]))
517  continue;
518 
519  (*iter)->connect (*v [j]);
520  break;
521  }
522  }
523  }
524  }
