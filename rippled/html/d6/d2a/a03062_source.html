280 bool Transaction::convertToTransactions (std::uint32_t firstLedgerSeq, std::uint32_t secondLedgerSeq,
281  bool checkFirstTransactions, bool checkSecondTransactions, const SHAMap::Delta& inMap,
282  std::map<uint256, std::pair<Transaction::pointer, Transaction::pointer> >& outMap)
283 {
284  // convert a straight SHAMap payload difference to a transaction difference table
285  // return value: true=ledgers are valid, false=a ledger is invalid
286  SHAMap::Delta::const_iterator it;
287 
288  for (it = inMap.begin (); it != inMap.end (); ++it)
289  {
290  uint256 const& id = it->first;
291  SHAMapItem::ref first = it->second.first;
292  SHAMapItem::ref second = it->second.second;
293 
294  Transaction::pointer firstTrans, secondTrans;
295 
296  if (!!first)
297  {
298  // transaction in our table
299  firstTrans = sharedTransaction (first->peekData (), checkFirstTransactions);
300 
301  if ((firstTrans->getStatus () == INVALID) || (firstTrans->getID () != id ))
302  {
303  firstTrans->setStatus (INVALID, firstLedgerSeq);
304  return false;
305  }
306  else firstTrans->setStatus (INCLUDED, firstLedgerSeq);
307  }
308 
309  if (!!second)
310  {
311  // transaction in other table
312  secondTrans = sharedTransaction (second->peekData (), checkSecondTransactions);
313 
314  if ((secondTrans->getStatus () == INVALID) || (secondTrans->getID () != id))
315  {
316  secondTrans->setStatus (INVALID, secondLedgerSeq);
317  return false;
318  }
319  else secondTrans->setStatus (INCLUDED, secondLedgerSeq);
320  }
321 
322  assert (firstTrans || secondTrans);
323 
324  if (firstTrans && secondTrans && (firstTrans->getStatus () != INVALID) && (secondTrans->getStatus () != INVALID))
325  return false; // one or the other SHAMap is structurally invalid or a miracle has happened
326 
327  outMap[id] = std::pair<Transaction::pointer, Transaction::pointer> (firstTrans, secondTrans);
328  }
329 
330  return true;
331 }
