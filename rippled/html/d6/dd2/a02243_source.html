105 class RandomAccessFile_test : public unit_test::suite
106 {
107 public:
108  enum
109  {
110  maxPayload = 8192
111  };
112 
113  /* For this test we will create a file which consists of a fixed
114  number of variable length records. Each record is numbered sequentially
115  starting at 0. To calculate the position of each record we first build
116  a table of size/offset pairs using a pseudorandom number generator.
117  */
118  struct Record
119  {
120  int index;
121  int bytes;
122  int offset;
123  };
124 
125  typedef HeapBlock <Record> Records;
126 
127  // Produce the pseudo-random set of records.
128  static void createRecords (HeapBlock <Record>& records,
129  int numRecords,
130  int maxBytes,
131  std::int64_t seedValue)
132  {
133  using namespace UnitTestUtilities;
134 
135  Random r (seedValue);
136 
137  records.malloc (numRecords);
138 
139  int offset = 0;
140 
141  for (int i = 0; i < numRecords; ++i)
142  {
143  int const bytes = r.nextInt (maxBytes) + 1;
144 
145  records [i].index = i;
146  records [i].bytes = bytes;
147  records [i].offset = offset;
148 
149  offset += bytes;
150  }
151 
152  repeatableShuffle (numRecords, records, seedValue);
153  }
154 
155  // Write all the records to the file.
156  // The payload is pseudo-randomly generated.
157  void writeRecords (RandomAccessFile& file,
158  int numRecords,
159  HeapBlock <Record> const& records,
160  std::int64_t seedValue)
161  {
162  using namespace UnitTestUtilities;
163 
164  for (int i = 0; i < numRecords; ++i)
165  {
166  Payload p (records [i].bytes);
167 
168  p.repeatableRandomFill (records [i].bytes,
169  records [i].bytes,
170  records [i].index + seedValue);
171 
172  file.setPosition (records [i].offset);
173 
174  Result result = file.write (p.data.getData (), p.bytes);
175 
176  expect (result.wasOk (), "Should be ok");
177  }
178  }
179 
180  // Read the records and verify the consistency.
181  void readRecords (RandomAccessFile& file,
182  int numRecords,
183  HeapBlock <Record> const& records,
184  std::int64_t seedValue)
185  {
186  using namespace UnitTestUtilities;
187 
188  for (int i = 0; i < numRecords; ++i)
189  {
190  Record const& record (records [i]);
191 
192  int const bytes = record.bytes;
193 
194  Payload p1 (bytes);
195  Payload p2 (bytes);
196 
197  p1.repeatableRandomFill (bytes, bytes, record.index + seedValue);
198 
199  file.setPosition (record.offset);
200 
201  Result result = file.read (p2.data.getData (), bytes);
202 
203  expect (result.wasOk (), "Should be ok");
204 
205  if (result.wasOk ())
206  {
207  p2.bytes = bytes;
208 
209  expect (p1 == p2, "Should be equal");
210  }
211  }
212  }

214  // Perform the test at the given buffer size.
215  void testFile (int const numRecords)
216  {
217  using namespace UnitTestUtilities;
218 
219  int const seedValue = 50;
220 
221  std::stringstream ss;
222  ss << numRecords << " records";
223  testcase (ss.str());
224 
225  // Calculate the path
226  File const path (File::createTempFile ("RandomAccessFile"));
227 
228  // Create a predictable set of records
229  HeapBlock <Record> records (numRecords);
230  createRecords (records, numRecords, maxPayload, seedValue);
231 
232  Result result (Result::ok ());
233 
234  {
235  // Create the file
236  RandomAccessFile file;
237  result = file.open (path, RandomAccessFile::readWrite);
238  expect (result.wasOk (), "Should be ok");
239 
240  if (result.wasOk ())
241  {
242  writeRecords (file, numRecords, records, seedValue);
243 
244  readRecords (file, numRecords, records, seedValue);
245 
246  repeatableShuffle (numRecords, records, seedValue);
247 
248  readRecords (file, numRecords, records, seedValue);
249  }
250  }
251 
252  if (result.wasOk ())
253  {
254  // Re-open the file in read only mode
255  RandomAccessFile file;
256  result = file.open (path, RandomAccessFile::readOnly);
257  expect (result.wasOk (), "Should be ok");
258 
259  if (result.wasOk ())
260  {
261  readRecords (file, numRecords, records, seedValue);
262  }
263  }
264  }
