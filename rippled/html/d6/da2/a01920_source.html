58 template <class _Ratio, class _CharT>
59 struct ratio_string
60 {
61  static basic_string<_CharT> symbol() {return prefix();}
62  static basic_string<_CharT> prefix();
63 };
64 
65 template <class _Ratio, class _CharT>
66 basic_string<_CharT>
67 ratio_string<_Ratio, _CharT>::prefix()
68 {
69  basic_ostringstream<_CharT> __os;
70  __os << _CharT('[') << _Ratio::num << _CharT('/')
71  << _Ratio::den << _CharT(']');
72  return __os.str();
73 }
