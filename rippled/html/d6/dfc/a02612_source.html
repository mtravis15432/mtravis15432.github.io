60 int FindFile(const InternalKeyComparator& icmp,
61  const std::vector<FileMetaData*>& files,
62  const Slice& key) {
63  uint32_t left = 0;
64  uint32_t right = files.size();
65  while (left < right) {
66  uint32_t mid = (left + right) / 2;
67  const FileMetaData* f = files[mid];
68  if (icmp.InternalKeyComparator::Compare(f->largest.Encode(), key) < 0) {
69  // Key at "mid.largest" is < "target". Therefore all
70  // files at or before "mid" are uninteresting.
71  left = mid + 1;
72  } else {
73  // Key at "mid.largest" is >= "target". Therefore all files
74  // after "mid" are uninteresting.
75  right = mid;
76  }
77  }
78  return right;
79 }

95 bool SomeFileOverlapsRange(
96  const InternalKeyComparator& icmp,
97  bool disjoint_sorted_files,
98  const std::vector<FileMetaData*>& files,
99  const Slice* smallest_user_key,
100  const Slice* largest_user_key) {
101  const Comparator* ucmp = icmp.user_comparator();
102  if (!disjoint_sorted_files) {
103  // Need to check against all files
104  for (size_t i = 0; i < files.size(); i++) {
105  const FileMetaData* f = files[i];
106  if (AfterFile(ucmp, smallest_user_key, f) ||
107  BeforeFile(ucmp, largest_user_key, f)) {
108  // No overlap
109  } else {
110  return true; // Overlap
111  }
112  }
113  return false;
114  }
115 
116  // Binary search over file list
117  uint32_t index = 0;
118  if (smallest_user_key != nullptr) {
119  // Find the earliest possible internal key for smallest_user_key
120  InternalKey small(*smallest_user_key, kMaxSequenceNumber,kValueTypeForSeek);
121  index = FindFile(icmp, files, small.Encode());
122  }
123 
124  if (index >= files.size()) {
125  // beginning of range is after all files, so no overlap.
126  return false;
127  }
128 
129  return !BeforeFile(ucmp, largest_user_key, files[index]);
130 }

132 // An internal iterator. For a given version/level pair, yields
133 // information about the files in the level. For a given entry, key()
134 // is the largest key that occurs in the file, and value() is an
135 // 16-byte value containing the file number and file size, both
136 // encoded using EncodeFixed64.
137 class Version::LevelFileNumIterator : public Iterator {
138  public:
139  LevelFileNumIterator(const InternalKeyComparator& icmp,
140  const std::vector<FileMetaData*>* flist)
141  : icmp_(icmp),
142  flist_(flist),
143  index_(flist->size()) { // Marks as invalid
144  }
145  virtual bool Valid() const {
146  return index_ < flist_->size();
147  }
148  virtual void Seek(const Slice& target) {
149  index_ = FindFile(icmp_, *flist_, target);
150  }
151  virtual void SeekToFirst() { index_ = 0; }
152  virtual void SeekToLast() {
153  index_ = flist_->empty() ? 0 : flist_->size() - 1;
154  }
155  virtual void Next() {
156  assert(Valid());
157  index_++;
158  }
159  virtual void Prev() {
160  assert(Valid());
161  if (index_ == 0) {
162  index_ = flist_->size(); // Marks as invalid
163  } else {
164  index_--;
165  }
166  }
167  Slice key() const {
168  assert(Valid());
169  return (*flist_)[index_]->largest.Encode();
170  }
171  Slice value() const {
172  assert(Valid());
173  EncodeFixed64(value_buf_, (*flist_)[index_]->number);
174  EncodeFixed64(value_buf_+8, (*flist_)[index_]->file_size);
175  return Slice(value_buf_, sizeof(value_buf_));
176  }
177  virtual Status status() const { return Status::OK(); }
178  private:
179  const InternalKeyComparator icmp_;
180  const std::vector<FileMetaData*>* const flist_;
181  uint32_t index_;
182 
183  // Backing store for value(). Holds the file number and size.
184  mutable char value_buf_[16];
185 };

310 static bool SaveValue(void* arg, const Slice& ikey, const Slice& v, bool didIO){
311  Saver* s = reinterpret_cast<Saver*>(arg);
312  std::deque<std::string>* const ops = s->merge_operands; // shorter alias
313  std::string merge_result; // temporary area for merge results later
314 
315  assert(s != nullptr && ops != nullptr);
316 
317  ParsedInternalKey parsed_key;
318  // TODO: didIO and Merge?
319  s->didIO = didIO;
320  if (!ParseInternalKey(ikey, &parsed_key)) {
321  // TODO: what about corrupt during Merge?
322  s->state = kCorrupt;
323  } else {
324  if (s->ucmp->Compare(parsed_key.user_key, s->user_key) == 0) {
325  // Key matches. Process it
326  switch (parsed_key.type) {
327  case kTypeValue:
328  if (kNotFound == s->state) {
329  s->state = kFound;
330  s->value->assign(v.data(), v.size());
331  } else if (kMerge == s->state) {
332  assert(s->merge_operator != nullptr);
333  s->state = kFound;
334  if (!s->merge_operator->FullMerge(s->user_key, &v, *ops,
335  s->value, s->logger)) {
336  RecordTick(s->statistics, NUMBER_MERGE_FAILURES);
337  s->state = kCorrupt;
338  }
339  } else {
340  assert(false);
341  }
342  return false;
343 
344  case kTypeDeletion:
345  if (kNotFound == s->state) {
346  s->state = kDeleted;
347  } else if (kMerge == s->state) {
348  s->state = kFound;
349  if (!s->merge_operator->FullMerge(s->user_key, nullptr, *ops,
350  s->value, s->logger)) {
351  RecordTick(s->statistics, NUMBER_MERGE_FAILURES);
352  s->state = kCorrupt;
353  }
354  } else {
355  assert(false);
356  }
357  return false;
358 
359  case kTypeMerge:
360  assert(s->state == kNotFound || s->state == kMerge);
361  s->state = kMerge;
362  ops->push_front(v.ToString());
363  while (ops->size() >= 2) {
364  // Attempt to merge operands together via user associateive merge
365  if (s->merge_operator->PartialMerge(s->user_key,
366  Slice((*ops)[0]),
367  Slice((*ops)[1]),
368  &merge_result,
369  s->logger)) {
370  ops->pop_front();
371  swap(ops->front(), merge_result);
372  } else {
373  // Associative merge returns false ==> stack the operands
374  break;
375  }
376  }
377  return true;
378 
379  case kTypeLogData:
380  assert(false);
381  break;
382  }
383  }
384  }
385 
386  // s->state could be Corrupt, merge or notfound
387 
388  return false;
389 }

416 void Version::Get(const ReadOptions& options,
417  const LookupKey& k,
418  std::string* value,
419  Status* status,
420  std::deque<std::string>* operands,
421  GetStats* stats,
422  const Options& db_options,
423  bool* value_found) {
424  Slice ikey = k.internal_key();
425  Slice user_key = k.user_key();
426  const Comparator* ucmp = vset_->icmp_.user_comparator();
427 
428  auto merge_operator = db_options.merge_operator.get();
429  auto logger = db_options.info_log;
430 
431  assert(status->ok() || status->IsMergeInProgress());
432  Saver saver;
433  saver.state = status->ok()? kNotFound : kMerge;
434  saver.ucmp = ucmp;
435  saver.user_key = user_key;
436  saver.value_found = value_found;
437  saver.value = value;
438  saver.merge_operator = merge_operator;
439  saver.merge_operands = operands;
440  saver.logger = logger.get();
441  saver.didIO = false;
442  saver.statistics = db_options.statistics;
443 
444  stats->seek_file = nullptr;
445  stats->seek_file_level = -1;
446  FileMetaData* last_file_read = nullptr;
447  int last_file_read_level = -1;
448 
449  // We can search level-by-level since entries never hop across
450  // levels. Therefore we are guaranteed that if we find data
451  // in an smaller level, later levels are irrelevant (unless we
452  // are MergeInProgress).
453  std::vector<FileMetaData*> important_files;
454  for (int level = 0; level < vset_->NumberLevels(); level++) {
455  size_t num_files = files_[level].size();
456  if (num_files == 0) continue;
457 
458  // Get the list of files to search in this level
459  FileMetaData* const* files = &files_[level][0];
460  important_files.clear();
461  important_files.reserve(num_files);
462 
463  // Some files may overlap each other. We find
464  // all files that overlap user_key and process them in order from
465  // newest to oldest. In the context of merge-operator,
466  // this can occur at any level. Otherwise, it only occurs
467  // at Level-0 (since Put/Deletes are always compacted into a single entry).
468  uint32_t start_index;
469  if (level == 0) {
470  // On Level-0, we read through all files to check for overlap.
471  start_index = 0;
472  } else {
473  // On Level-n (n>=1), files are sorted.
474  // Binary search to find earliest index whose largest key >= ikey.
475  // We will also stop when the file no longer overlaps ikey
476  start_index = FindFile(vset_->icmp_, files_[level], ikey);
477  }
478 
479  // Traverse the list, finding all overlapping files.
480  for (uint32_t i = start_index; i < num_files; i++) {
481  FileMetaData* f = files[i];
482  if (ucmp->Compare(user_key, f->smallest.user_key()) >= 0 &&
483  ucmp->Compare(user_key, f->largest.user_key()) <= 0) {
484  important_files.push_back(f);
485  } else if (level > 0) {
486  // If on Level-n (n>=1) then the files are sorted.
487  // So we can stop looking when we are past the ikey.
488  break;
489  }
490  }
491 
492  if (important_files.empty()) continue;
493 
494  if (level == 0) {
495  if (vset_->options_->compaction_style == kCompactionStyleUniversal) {
496  std::sort(important_files.begin(), important_files.end(), NewestFirstBySeqNo);
497  } else {
498  std::sort(important_files.begin(), important_files.end(), NewestFirst);
499  }
500  } else {
501  // Sanity check to make sure that the files are correctly sorted
502 #ifndef NDEBUG
503  num_files = important_files.size();
504  for (uint32_t i = 1; i < num_files; ++i) {
505  FileMetaData* a = important_files[i-1];
506  FileMetaData* b = important_files[i];
507  int comp_sign = vset_->icmp_.Compare(a->largest, b->smallest);
508  assert(comp_sign < 0);
509  }
510 #endif
511  }
512 
513  // Traverse each relevant file to find the desired key
514  num_files = important_files.size();
515  for (uint32_t i = 0; i < num_files; ++i) {
516  FileMetaData* f = important_files[i];
517  bool tableIO = false;
518  *status = vset_->table_cache_->Get(options, f->number, f->file_size,
519  ikey, &saver, SaveValue, &tableIO,
520  MarkKeyMayExist);
521  // TODO: examine the behavior for corrupted key
522  if (!status->ok()) {
523  return;
524  }
525 
526  if (last_file_read != nullptr && stats->seek_file == nullptr) {
527  // We have had more than one seek for this read. Charge the 1st file.
528  stats->seek_file = last_file_read;
529  stats->seek_file_level = last_file_read_level;
530  }
531 
532  // If we did any IO as part of the read, then we remember it because
533  // it is a possible candidate for seek-based compaction. saver.didIO
534  // is true if the block had to be read in from storage and was not
535  // pre-exisiting in the block cache. Also, if this file was not pre-
536  // existing in the table cache and had to be freshly opened that needed
537  // the index blocks to be read-in, then tableIO is true. One thing
538  // to note is that the index blocks are not part of the block cache.
539  if (saver.didIO || tableIO) {
540  last_file_read = f;
541  last_file_read_level = level;
542  }
543 
544  switch (saver.state) {
545  case kNotFound:
546  break; // Keep searching in other files
547  case kFound:
548  return;
549  case kDeleted:
550  *status = Status::NotFound(Slice()); // Use empty error message for speed
551  return;
552  case kCorrupt:
553  *status = Status::Corruption("corrupted key for ", user_key);
554  return;
555  case kMerge:
556  break;
557  }
558  }
559  }
560 
561 
562  if (kMerge == saver.state) {
563  // merge_operands are in saver and we hit the beginning of the key history
564  // do a final merge of nullptr and operands;
565  if (merge_operator->FullMerge(user_key, nullptr, *saver.merge_operands,
566  value, logger.get())) {
567  *status = Status::OK();
568  } else {
569  RecordTick(db_options.statistics, NUMBER_MERGE_FAILURES);
570  *status = Status::Corruption("could not perform end-of-key merge for ",
571  user_key);
572  }
573  } else {
574  *status = Status::NotFound(Slice()); // Use an empty error message for speed
575  }
576 }

611 int Version::PickLevelForMemTableOutput(
612  const Slice& smallest_user_key,
613  const Slice& largest_user_key) {
614  int level = 0;
615  if (!OverlapInLevel(0, &smallest_user_key, &largest_user_key)) {
616  // Push to next level if there is no overlap in next level,
617  // and the #bytes overlapping in the level after that are limited.
618  InternalKey start(smallest_user_key, kMaxSequenceNumber, kValueTypeForSeek);
619  InternalKey limit(largest_user_key, 0, static_cast<ValueType>(0));
620  std::vector<FileMetaData*> overlaps;
621  int max_mem_compact_level = vset_->options_->max_mem_compaction_level;
622  while (max_mem_compact_level > 0 && level < max_mem_compact_level) {
623  if (OverlapInLevel(level + 1, &smallest_user_key, &largest_user_key)) {
624  break;
625  }
626  if (level + 2 >= vset_->NumberLevels()) {
627  level++;
628  break;
629  }
630  GetOverlappingInputs(level + 2, &start, &limit, &overlaps);
631  const uint64_t sum = TotalFileSize(overlaps);
632  if (sum > vset_->MaxGrandParentOverlapBytes(level)) {
633  break;
634  }
635  level++;
636  }
637  }
638 
639  return level;

701 // Employ binary search to find at least one file that overlaps the
702 // specified range. From that file, iterate backwards and
703 // forwards to find all overlapping files.
704 void Version::GetOverlappingInputsBinarySearch(
705  int level,
706  const Slice& user_begin,
707  const Slice& user_end,
708  std::vector<FileMetaData*>* inputs,
709  int hint_index,
710  int* file_index) {
711  assert(level > 0);
712  int min = 0;
713  int mid = 0;
714  int max = files_[level].size() -1;
715  bool foundOverlap = false;
716  const Comparator* user_cmp = vset_->icmp_.user_comparator();
717 
718  // if the caller already knows the index of a file that has overlap,
719  // then we can skip the binary search.
720  if (hint_index != -1) {
721  mid = hint_index;
722  foundOverlap = true;
723  }
724

726  mid = (min + max)/2;
727  FileMetaData* f = files_[level][mid];
728  const Slice file_start = f->smallest.user_key();
729  const Slice file_limit = f->largest.user_key();
730  if (user_cmp->Compare(file_limit, user_begin) < 0) {
731  min = mid + 1;
732  } else if (user_cmp->Compare(user_end, file_start) < 0) {
733  max = mid - 1;
734  } else {
735  foundOverlap = true;
736  break;
737  }
738  }
739

741  if (!foundOverlap) {
742  return;
743  }

745  if (file_index) {
746  *file_index = mid;
747  }

751 // Store in "*inputs" all files in "level" that overlap [begin,end]
752 // The midIndex specifies the index of at least one file that
753 // overlaps the specified range. From that file, iterate backward
754 // and forward to find all overlapping files.
755 void Version::ExtendOverlappingInputs(
756  int level,
757  const Slice& user_begin,
758  const Slice& user_end,
759  std::vector<FileMetaData*>* inputs,
760  unsigned int midIndex) {
761 
762  const Comparator* user_cmp = vset_->icmp_.user_comparator();
763 #ifndef NDEBUG
764  {
765  // assert that the file at midIndex overlaps with the range
766  assert(midIndex < files_[level].size());
767  FileMetaData* f = files_[level][midIndex];
768  const Slice fstart = f->smallest.user_key();
769  const Slice flimit = f->largest.user_key();
770  if (user_cmp->Compare(fstart, user_begin) >= 0) {
771  assert(user_cmp->Compare(fstart, user_end) <= 0);
772  } else {
773  assert(user_cmp->Compare(flimit, user_begin) >= 0);
774  }
775  }
776 #endif
777  int startIndex = midIndex + 1;
778  int endIndex = midIndex;
779  int count __attribute__((unused)) = 0;
780 
781  // check backwards from 'mid' to lower indices
782  for (int i = midIndex; i >= 0 ; i--) {
783  FileMetaData* f = files_[level][i];
784  const Slice file_limit = f->largest.user_key();
785  if (user_cmp->Compare(file_limit, user_begin) >= 0) {
786  startIndex = i;
787  assert((count++, true));
788  } else {
789  break;
790  }
791  }
792  // check forward from 'mid+1' to higher indices
793  for (unsigned int i = midIndex+1; i < files_[level].size(); i++) {
794  FileMetaData* f = files_[level][i];
795  const Slice file_start = f->smallest.user_key();
796  if (user_cmp->Compare(file_start, user_end) <= 0) {
797  assert((count++, true));
798  endIndex = i;
799  } else {
800  break;
801  }
802  }
803  assert(count == endIndex - startIndex + 1);
804 
805  // insert overlapping files into vector
806  for (int i = startIndex; i <= endIndex; i++) {
807  FileMetaData* f = files_[level][i];
808  inputs->push_back(f);
809  }
810 }

816 bool Version::HasOverlappingUserKey(

818  int level) {

820  // If inputs empty, there is no overlap.
821  // If level == 0, it is assumed that all needed files were already included.
822  if (inputs->empty() || level == 0){
823  return false;
824  }

831  size_t last_file = FindFile(vset_->icmp_, files,
832  inputs->back()->largest.Encode());
833  assert(0 <= last_file && last_file < kNumFiles); // File should exist!
834  if (last_file < kNumFiles-1) { // If not the last file
835  const Slice last_key_in_input = files[last_file]->largest.user_key();
836  const Slice first_key_after = files[last_file+1]->smallest.user_key();
837  if (user_cmp->Compare(last_key_in_input, first_key_after) == 0) {
838  // The last user key in input overlaps with the next file's first key
839  return true;
840  }
841  }
842 
843  // Check the first file in inputs against the file just before it
844  size_t first_file = FindFile(vset_->icmp_, files,
845  inputs->front()->smallest.Encode());
846  assert(0 <= first_file && first_file <= last_file); // File should exist!
847  if (first_file > 0) { // If not first file
848  const Slice& first_key_in_input = files[first_file]->smallest.user_key();
849  const Slice& last_key_before = files[first_file-1]->largest.user_key();
850  if (user_cmp->Compare(first_key_in_input, last_key_before) == 0) {
851  // The first user key in input overlaps with the previous file's last key
852  return true;
853  }
854  }
855 
856  return false;
857 }

1032  // Apply all of the edits in *edit to the current state.
1033  void Apply(VersionEdit* edit) {
1034  CheckConsistency(base_);
1035 
1036  // Update compaction pointers
1037  for (size_t i = 0; i < edit->compact_pointers_.size(); i++) {
1038  const int level = edit->compact_pointers_[i].first;
1039  vset_->compact_pointer_[level] =
1040  edit->compact_pointers_[i].second.Encode().ToString();
1041  }
1042 
1043  // Delete files
1044  const VersionEdit::DeletedFileSet& del = edit->deleted_files_;
1045  for (VersionEdit::DeletedFileSet::const_iterator iter = del.begin();
1046  iter != del.end();
1047  ++iter) {
1048  const int level = iter->first;
1049  const uint64_t number = iter->second;
1050  levels_[level].deleted_files.insert(number);
1051  CheckConsistencyForDeletes(edit, number, level);
1052  }
1053 
1054  // Add new files
1055  for (size_t i = 0; i < edit->new_files_.size(); i++) {
1056  const int level = edit->new_files_[i].first;
1057  FileMetaData* f = new FileMetaData(edit->new_files_[i].second);
1058  f->refs = 1;
1059 
1060  // We arrange to automatically compact this file after
1061  // a certain number of seeks. Let's assume:
1062  // (1) One seek costs 10ms
1063  // (2) Writing or reading 1MB costs 10ms (100MB/s)
1064  // (3) A compaction of 1MB does 25MB of IO:
1065  // 1MB read from this level
1066  // 10-12MB read from next level (boundaries may be misaligned)
1067  // 10-12MB written to next level
1068  // This implies that 25 seeks cost the same as the compaction
1069  // of 1MB of data. I.e., one seek costs approximately the
1070  // same as the compaction of 40KB of data. We are a little
1071  // conservative and allow approximately one seek for every 16KB
1072  // of data before triggering a compaction.
1073  f->allowed_seeks = (f->file_size / 16384);
1074  if (f->allowed_seeks < 100) f->allowed_seeks = 100;
1075 
1076  levels_[level].deleted_files.erase(f->number);
1077  levels_[level].added_files->insert(f);
1078  }
1079  }

1081  // Save the current state in *v.
1082  void SaveTo(Version* v) {
1083  CheckConsistency(base_);
1084  CheckConsistency(v);
1085  BySmallestKey cmp;
1086  cmp.internal_comparator = &vset_->icmp_;
1087  for (int level = 0; level < vset_->NumberLevels(); level++) {
1088  // Merge the set of added files with the set of pre-existing files.
1089  // Drop any deleted files. Store the result in *v.
1090  const std::vector<FileMetaData*>& base_files = base_->files_[level];
1091  std::vector<FileMetaData*>::const_iterator base_iter = base_files.begin();
1092  std::vector<FileMetaData*>::const_iterator base_end = base_files.end();
1093  const FileSet* added = levels_[level].added_files;
1094  v->files_[level].reserve(base_files.size() + added->size());
1095  for (FileSet::const_iterator added_iter = added->begin();
1096  added_iter != added->end();
1097  ++added_iter) {
1098  // Add all smaller files listed in base_
1099  for (std::vector<FileMetaData*>::const_iterator bpos
1100  = std::upper_bound(base_iter, base_end, *added_iter, cmp);
1101  base_iter != bpos;
1102  ++base_iter) {
1103  MaybeAddFile(v, level, *base_iter);
1104  }
1105 
1106  MaybeAddFile(v, level, *added_iter);
1107  }
1108 
1109  // Add remaining base files
1110  for (; base_iter != base_end; ++base_iter) {
1111  MaybeAddFile(v, level, *base_iter);
1112  }
1113  }

1674 void VersionSet::Finalize(Version* v,
1675  std::vector<uint64_t>& size_being_compacted) {
1676 
1677  double max_score = 0;
1678  int max_score_level = 0;
1679 
1680  int num_levels_to_check =
1681  (options_->compaction_style != kCompactionStyleUniversal) ?
1682  NumberLevels() - 1 : 1;
1683 
1684  for (int level = 0; level < num_levels_to_check; level++) {
1685 
1686  double score;
1687  if (level == 0) {
1688  // We treat level-0 specially by bounding the number of files
1689  // instead of number of bytes for two reasons:
1690  //
1691  // (1) With larger write-buffer sizes, it is nice not to do too
1692  // many level-0 compactions.
1693  //
1694  // (2) The files in level-0 are merged on every read and
1695  // therefore we wish to avoid too many files when the individual
1696  // file size is small (perhaps because of a small write-buffer
1697  // setting, or very high compression ratios, or lots of
1698  // overwrites/deletions).
1699  int numfiles = 0;
1700  for (unsigned int i = 0; i < v->files_[level].size(); i++) {
1701  if (!v->files_[level][i]->being_compacted) {
1702  numfiles++;
1703  }
1704  }
1705 
1706  // If we are slowing down writes, then we better compact that first
1707  if (numfiles >= options_->level0_stop_writes_trigger) {
1708  score = 1000000;
1709  // Log(options_->info_log, "XXX score l0 = 1000000000 max");
1710  } else if (numfiles >= options_->level0_slowdown_writes_trigger) {
1711  score = 10000;
1712  // Log(options_->info_log, "XXX score l0 = 1000000 medium");
1713  } else {
1714  score = numfiles /
1715  static_cast<double>(options_->level0_file_num_compaction_trigger);
1716  if (score >= 1) {
1717  // Log(options_->info_log, "XXX score l0 = %d least", (int)score);
1718  }
1719  }
1720  } else {
1721  // Compute the ratio of current size to size limit.
1722  const uint64_t level_bytes = TotalFileSize(v->files_[level]) -
1723  size_being_compacted[level];
1724  score = static_cast<double>(level_bytes) / MaxBytesForLevel(level);
1725  if (score > 1) {
1726  // Log(options_->info_log, "XXX score l%d = %d ", level, (int)score);
1727  }
1728  if (max_score < score) {
1729  max_score = score;
1730  max_score_level = level;
1731  }
1732  }
1733  v->compaction_level_[level] = level;
1734  v->compaction_score_[level] = score;
1735  }
1736 
1737  // update the max compaction score in levels 1 to n-1
1738  v->max_compaction_score_ = max_score;
1739  v->max_compaction_score_level_ = max_score_level;
1740 
1741  // sort all the levels based on their score. Higher scores get listed
1742  // first. Use bubble sort because the number of entries are small.
1743  for (int i = 0; i < NumberLevels()-2; i++) {
1744  for (int j = i+1; j < NumberLevels()-1; j++) {
1745  if (v->compaction_score_[i] < v->compaction_score_[j]) {
1746  double score = v->compaction_score_[i];
1747  int level = v->compaction_level_[i];
1748  v->compaction_score_[i] = v->compaction_score_[j];
1749  v->compaction_level_[i] = v->compaction_level_[j];
1750  v->compaction_score_[j] = score;
1751  v->compaction_level_[j] = level;
1752  }
1753  }
1754  }
1755 }
