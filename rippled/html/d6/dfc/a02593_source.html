25 std::shared_ptr<DB> OpenDb(const string& dbname, const bool ttl = false) {
26  DB* db;

31  Status s;

34  cout << "Opening database with TTL\n";

41  cerr << s.ToString() << endl;

43  }
44  return std::shared_ptr<DB>(db);
45 }

47 // Imagine we are maintaining a set of uint64 counters.
48 // Each counter has a distinct name. And we would like
49 // to support four high level operations:
50 // set, add, get and remove
51 // This is a quick implementation without a Merge operation.
52 class Counters {
53 
54  protected:
55  std::shared_ptr<DB> db_;
56 
57  WriteOptions put_option_;
58  ReadOptions get_option_;
59  WriteOptions delete_option_;
60 
61  uint64_t default_;
62 
63  public:
64  explicit Counters(std::shared_ptr<DB> db, uint64_t defaultCount = 0)
65  : db_(db),
66  put_option_(),
67  get_option_(),
68  delete_option_(),
69  default_(defaultCount) {
70  assert(db_);
71  }
72 
73  virtual ~Counters() {}
74 
75  // public interface of Counters.
76  // All four functions return false
77  // if the underlying level db operation failed.
78 
79  // mapped to a levedb Put
80  bool set(const string& key, uint64_t value) {
81  // just treat the internal rep of int64 as the string
82  Slice slice((char *)&value, sizeof(value));
83  auto s = db_->Put(put_option_, key, slice);
84 
85  if (s.ok()) {
86  return true;
87  } else {
88  cerr << s.ToString() << endl;
89  return false;
90  }
91  }
92 
93  // mapped to a rocksdb Delete
94  bool remove(const string& key) {
95  auto s = db_->Delete(delete_option_, key);
96 
97  if (s.ok()) {
98  return true;
99  } else {
100  cerr << s.ToString() << std::endl;
101  return false;
102  }
103  }
104 
105  // mapped to a rocksdb Get
106  bool get(const string& key, uint64_t *value) {
107  string str;
108  auto s = db_->Get(get_option_, key, &str);
109 
110  if (s.IsNotFound()) {
111  // return default value if not found;
112  *value = default_;
113  return true;
114  } else if (s.ok()) {
115  // deserialization
116  if (str.size() != sizeof(uint64_t)) {
117  cerr << "value corruption\n";
118  return false;
119  }
120  *value = DecodeFixed64(&str[0]);
121  return true;
122  } else {
123  cerr << s.ToString() << std::endl;
124  return false;
125  }
126  }
127 
128  // 'add' is implemented as get -> modify -> set
129  // An alternative is a single merge operation, see MergeBasedCounters
130  virtual bool add(const string& key, uint64_t value) {
131  uint64_t base = default_;
132  return get(key, &base) && set(key, base + value);
133  }
134 
135 
136  // convenience functions for testing
137  void assert_set(const string& key, uint64_t value) {
138  assert(set(key, value));
139  }
140 
141  void assert_remove(const string& key) {
142  assert(remove(key));
143  }
144 
145  uint64_t assert_get(const string& key) {
146  uint64_t value = default_;
147  assert(get(key, &value));
148  return value;
149  }
150 
151  void assert_add(const string& key, uint64_t value) {
152  assert(add(key, value));
153  }
154 };

167  // mapped to a rocksdb Merge operation
168  virtual bool add(const string& key, uint64_t value) override {
169  char encoded[sizeof(uint64_t)];
170  EncodeFixed64(encoded, value);
171  Slice slice(encoded, sizeof(uint64_t));
172  auto s = db_->Merge(merge_option_, key, slice);
173 
174  if (s.ok()) {
175  return true;

192 void testCounters(Counters& counters, DB* db, bool test_compaction) {
193 
194  FlushOptions o;

197  counters.assert_set("a", 1);
198 
199  if (test_compaction) db->Flush(o);
200 
201  assert(counters.assert_get("a") == 1);
202 
203  counters.assert_remove("b");
204 
205  // defaut value is 0 if non-existent
206  assert(counters.assert_get("b") == 0);

208  counters.assert_add("a", 2);

212  // 1+2 = 3
