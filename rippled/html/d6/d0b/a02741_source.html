69 LDBCommand* LDBCommand::InitFromCmdLineArgs(
70  const vector<string>& args,
71  const Options& options
72 ) {
73  // --x=y command line arguments are added as x->y map entries.
74  map<string, string> option_map;
75 
76  // Command-line arguments of the form --hex end up in this array as hex
77  vector<string> flags;
78 
79  // Everything other than option_map and flags. Represents commands
80  // and their parameters. For eg: put key1 value1 go into this vector.
81  vector<string> cmdTokens;
82 
83  const string OPTION_PREFIX = "--";
84 
85  for (const auto& arg : args) {
86  if (arg[0] == '-' && arg[1] == '-'){
87  vector<string> splits = stringSplit(arg, '=');
88  if (splits.size() == 2) {
89  string optionKey = splits[0].substr(OPTION_PREFIX.size());
90  option_map[optionKey] = splits[1];
91  } else {
92  string optionKey = splits[0].substr(OPTION_PREFIX.size());
93  flags.push_back(optionKey);
94  }
95  } else {
96  cmdTokens.push_back(arg);
97  }
98  }
99 
100  if (cmdTokens.size() < 1) {
101  fprintf(stderr, "Command not specified!");
102  return nullptr;
103  }
104 
105  string cmd = cmdTokens[0];
106  vector<string> cmdParams(cmdTokens.begin()+1, cmdTokens.end());
107  LDBCommand* command = LDBCommand::SelectCommand(
108  cmd,
109  cmdParams,
110  option_map,
111  flags
112  );
113 
114  if (command) {
115  command->SetOptions(options);
116  }
117  return command;
118 }

491 void ManifestDumpCommand::DoCommand() {
492 
493  std::string manifestfile;
494 
495  if (!path_.empty()) {
496  manifestfile = path_;
497  } else {
498  bool found = false;
499  // We need to find the manifest file by searching the directory
500  // containing the db for files of the form MANIFEST_[0-9]+
501  DIR* d = opendir(db_path_.c_str());
502  if (d == nullptr) {
503  exec_state_ = LDBCommandExecuteResult::FAILED(
504  db_path_ + " is not a directory");
505  return;
506  }
507  struct dirent* entry;
508  while ((entry = readdir(d)) != nullptr) {
509  unsigned int match;
510  unsigned long long num;
511  if (sscanf(entry->d_name,
512  "MANIFEST-%ln%ln",
513  (unsigned long*)&num,
514  (unsigned long*)&match)
515  && match == strlen(entry->d_name)) {
516  if (!found) {
517  manifestfile = db_path_ + "/" + std::string(entry->d_name);
518  found = true;
519  } else {
520  exec_state_ = LDBCommandExecuteResult::FAILED(
521  "Multiple MANIFEST files found; use --path to select one");
522  return;
523  }
524  }
525  }
526  closedir(d);
527  }
528 
529  if (verbose_) {
530  printf("Processing Manifest file %s\n", manifestfile.c_str());
531  }
532 
533  Options options;
534  EnvOptions sopt;
535  std::string file(manifestfile);
536  std::string dbname("dummy");
537  TableCache* tc = new TableCache(dbname, &options, sopt, 10);
538  const InternalKeyComparator* cmp =
539  new InternalKeyComparator(options.comparator);
