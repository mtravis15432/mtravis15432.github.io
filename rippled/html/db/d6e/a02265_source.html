56  String getCpuInfo (const char* const key)
57  {
58  StringArray lines;
59  File ("/proc/cpuinfo").readLines (lines);
60 
61  for (int i = lines.size(); --i >= 0;) // (NB - it's important that this runs in reverse order)
62  if (lines[i].startsWithIgnoreCase (key))
63  return lines[i].fromFirstOccurrenceOf (":", false, false).trim();
64 
65  return String::empty;
66  }
67 }
