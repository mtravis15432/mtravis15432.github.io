271 std::pair <PropertyStream::Source*, bool> PropertyStream::Source::find (std::string path)
272 {
273  bool const deep (peel_trailing_slashstar (&path));
274  bool const rooted (peel_leading_slash (&path));
275  Source* source (this);
276  if (! path.empty())
277  {
278  if (! rooted)
279  {
280  std::string const name (peel_name (&path));
281  source = find_one_deep (name);
282  if (source == nullptr)
283  return std::make_pair (nullptr, deep);
284  }
285  source = source->find_path (path);
286  }
287  return std::make_pair (source, deep);

300 bool PropertyStream::Source::peel_trailing_slashstar (std::string* path)

315 std::string PropertyStream::Source::peel_name (std::string* path)
316 {
317  if (path->empty())
318  return "";
319 
320  std::string::const_iterator first = (*path).begin();
321  std::string::const_iterator last = (*path).end();
322  std::string::const_iterator pos (std::find (first, last, '/'));
323  std::string s (first, pos);
324 
325  if (pos != last)
326  *path = std::string (pos+1, last);
327  else
328  *path = std::string ();
329 
330  return s;
331 }

333 // Recursive search through the whole tree until name is found
334 PropertyStream::Source* PropertyStream::Source::find_one_deep (std::string const& name)
335 {
336  Source* found = find_one (name);
337  if (found != nullptr)
338  return found;
339  SharedState::Access state (this->m_state);
340  for (auto& s : state->children)
341  {
342  found = s.source().find_one_deep (name);
343  if (found != nullptr)
344  return found;
345  }
346  return nullptr;
347 }

349 PropertyStream::Source* PropertyStream::Source::find_path (std::string path)
350 {
351  if (path.empty())
352  return this;
353  Source* source (this);
354  do
355  {
356  std::string const name (peel_name (&path));
357  if(name.empty ())
358  break;
359  source = source->find_one(name);
360  }
361  while (source != nullptr);
362  return source;
363 }
