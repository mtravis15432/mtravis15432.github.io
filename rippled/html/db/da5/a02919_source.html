25 enum TransactionEngineParams
26 {
27  tapNONE = 0x00,
28 
29  tapNO_CHECK_SIGN = 0x01, // Signature already checked
30 
31  tapOPEN_LEDGER = 0x10, // Transaction is running against an open ledger
32  // true = failures are not forwarded, check transaction fee
33  // false = debit ledger for consumed funds
34 
35  tapRETRY = 0x20, // This is not the transaction's last pass
36  // Transaction can be retried, soft failures allowed
37 
38  tapADMIN = 0x400, // Transaction came from a privileged source
39 };
