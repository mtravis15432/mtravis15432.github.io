254  template <class CharPointer>
255  void appendCharPointer (const CharPointer textToAppend, size_t maxCharsToTake)
256  {
257  if (textToAppend.getAddress() != nullptr)
258  {
259  size_t extraBytesNeeded = 0;
260  size_t numChars = 0;
261 
262  for (CharPointer t (textToAppend); numChars < maxCharsToTake && ! t.isEmpty();)
263  {
264  extraBytesNeeded += CharPointerType::getBytesRequiredFor (t.getAndAdvance());
265  ++numChars;
266  }
267 
268  if (numChars > 0)
269  {
270  const size_t byteOffsetOfNull = getByteOffsetOfEnd();
271 
272  preallocateBytes (byteOffsetOfNull + extraBytesNeeded);
273  CharPointerType (addBytesToPointer (text.getAddress(), (int) byteOffsetOfNull)).writeWithCharLimit (textToAppend, (int) (numChars + 1));
274  }
275  }
276  }
