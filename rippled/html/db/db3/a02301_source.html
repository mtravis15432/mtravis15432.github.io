45  // Okay to call on an active timer.
46  // However, an extra notification may still happen due to concurrency.
47  //
48  void activate (DeadlineTimer& timer,
49  double secondsRecurring, RelativeTime const& when)
50  {
51  bassert (secondsRecurring >= 0);
52 
53  std::lock_guard <LockType> lock (m_mutex);
54 
55  if (timer.m_isActive)
56  {
57  m_items.erase (m_items.iterator_to (timer));
58 
59  timer.m_isActive = false;
60  }
61 
62  timer.m_secondsRecurring = secondsRecurring;
63  timer.m_notificationTime = when;
64 
65  insertSorted (timer);
66  timer.m_isActive = true;
67 
68  notify ();
69  }

88  void run ()
89  {
90  while (! threadShouldExit ())
91  {
92  RelativeTime const currentTime (
93  RelativeTime::fromStartup ());
94 
95  double seconds (0);
96  DeadlineTimer* timer (nullptr);
97 
98  {
99  std::lock_guard <LockType> lock (m_mutex);
100 
101  // See if a timer expired
102  if (! m_items.empty ())
103  {
104  timer = &m_items.front ();
105 
106  // Has this timer expired?
107  if (timer->m_notificationTime <= currentTime)
108  {
109  // Expired, remove it from the list.
110  bassert (timer->m_isActive);
111  m_items.pop_front ();
112 
113  // Is the timer recurring?
114  if (timer->m_secondsRecurring > 0)
115  {
116  // Yes so set the timer again.
117  timer->m_notificationTime =
118  currentTime + timer->m_secondsRecurring;
119 
120  // Put it back into the list as active
121  insertSorted (*timer);
122  }
123  else
124  {
125  // Not a recurring timer, deactivate it.
126  timer->m_isActive = false;
127  }
128 
129  timer->m_listener->onDeadlineTimer (*timer);
130 
131  // re-loop
132  seconds = -1;
133  }
134  else
135  {
136  seconds = (
137  timer->m_notificationTime - currentTime).inSeconds ();
138 
139  // Can't be zero and come into the else clause.
140  bassert (seconds != 0);
141 
142  // Don't call the listener
143  timer = nullptr;
144  }
145  }
146  }
147 
148  // Note that we have released the lock here.
149 
150  if (seconds > 0)
151  {
152  // Wait until interrupt or next timer.
153  //
154  int const milliSeconds (std::max (
155  static_cast <int> (seconds * 1000 + 0.5), 1));
156  bassert (milliSeconds > 0);
157  wait (milliSeconds);
158  }
159  else if (seconds == 0)
160  {
161  // Wait until interrupt
162  //
163  wait ();
164  }
165  else
166  {
167  // Do not wait. This can happen if the recurring timer duration
168  // is extremely short, or if a listener wastes too much time in
169  // their callback.
170  }
171  }
172  }

174  // Caller is responsible for locking
175  void insertSorted (DeadlineTimer& timer)
176  {
177  if (! m_items.empty ())
178  {
179  Items::iterator before = m_items.begin ();
180 
181  for (;;)
182  {
183  if (before->m_notificationTime >= timer.m_notificationTime)
184  {
185  m_items.insert (before, timer);
186  break;
187  }
188 
189  ++before;
190 
191  if (before == m_items.end ())
192  {
193  m_items.push_back (timer);
194  break;
195  }
196  }
197  }
198  else
199  {
200  m_items.push_back (timer);
201  }
202  }
