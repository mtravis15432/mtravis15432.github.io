23 DecodedBlob::DecodedBlob (void const* key, void const* value, int valueBytes)
24 {
25  /* Data format:
26 
27  Bytes
28 
29  0...3 LedgerIndex 32-bit big endian integer
30  4...7 Unused? An unused copy of the LedgerIndex
31  8 char One of NodeObjectType
32  9...end The body of the object data
33  */
34 
35  m_success = false;
36  m_key = key;
37  // VFALCO NOTE Ledger indexes should have started at 1
38  m_ledgerIndex = LedgerIndex (-1);
39  m_objectType = hotUNKNOWN;
40  m_objectData = nullptr;
41  m_dataBytes = beast::bmax (0, valueBytes - 9);
42 
43  if (valueBytes > 4)
44  {
45  LedgerIndex const* index = static_cast <LedgerIndex const*> (value);
46  m_ledgerIndex = beast::ByteOrder::swapIfLittleEndian (*index);
47  }
48 
49  // VFALCO NOTE What about bytes 4 through 7 inclusive?
50 
51  if (valueBytes > 8)
52  {
53  unsigned char const* byte = static_cast <unsigned char const*> (value);
54  m_objectType = static_cast <NodeObjectType> (byte [8]);
55  }
56 
57  if (valueBytes > 9)
58  {
59  m_objectData = static_cast <unsigned char const*> (value) + 9;
60 
61  switch (m_objectType)
62  {
63  case hotUNKNOWN:
64  default:
65  break;
66 
67  case hotLEDGER:
68  case hotTRANSACTION:
69  case hotACCOUNT_NODE:
70  case hotTRANSACTION_NODE:
71  m_success = true;
72  break;
73  }
74  }
75 }
