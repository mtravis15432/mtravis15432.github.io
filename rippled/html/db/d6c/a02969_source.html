544 void NetworkOPsImp::processHeartbeatTimer ()
545 {
546  {
547  Application::ScopedLockType lock (getApp().getMasterLock ());
548 
549  // VFALCO NOTE This is for diagnosing a crash on exit
550  Application& app (getApp ());
551  LoadManager& mgr (app.getLoadManager ());
552  mgr.resetDeadlockDetector ();
553 
554  std::size_t const numPeers = getApp().overlay ().size ();
555 
556  // do we have sufficient peers? If not, we are disconnected.
557  if (numPeers < getConfig ().NETWORK_QUORUM)
558  {
559  if (mMode != omDISCONNECTED)
560  {
561  setMode (omDISCONNECTED);
562  m_journal.warning
563  << "Node count (" << numPeers << ") "
564  << "has fallen below quorum (" << getConfig ().NETWORK_QUORUM << ").";
565  }
566 
567  setHeartbeatTimer ();
568 
569  return;
570  }
571 
572  if (mMode == omDISCONNECTED)
573  {
574  setMode (omCONNECTED);
575  m_journal.info << "Node count (" << numPeers << ") is sufficient.";
576  }
577 
578  // Check if the last validated ledger forces a change between these states
579  if (mMode == omSYNCING)
580  {
581  setMode (omSYNCING);
582  }
583  else if (mMode == omCONNECTED)
584  {
585  setMode (omCONNECTED);
586  }
587 
588  if (!mConsensus)
589  tryStartConsensus ();
590 
591  if (mConsensus)
592  mConsensus->timerEntry ();
593  }
594 
595  setHeartbeatTimer ();
596 }

598 void NetworkOPsImp::processClusterTimer ()
599 {
600  bool synced = (m_ledgerMaster.getValidatedLedgerAge() <= 240);
601  ClusterNodeStatus us("", synced ? getApp().getFeeTrack().getLocalFee() : 0, getNetworkTimeNC());
602  if (!getApp().getUNL().nodeUpdate(getApp().getLocalCredentials().getNodePublic(), us))
603  {
604  m_journal.debug << "To soon to send cluster update";
605  return;
606  }
607 
608  std::map<RippleAddress, ClusterNodeStatus> nodes = getApp().getUNL().getClusterStatus();
609 
610  protocol::TMCluster cluster;
611  for (std::map<RippleAddress, ClusterNodeStatus>::iterator it = nodes.begin(),
612  end = nodes.end(); it != end; ++it)
613  {
614  protocol::TMClusterNode& node = *cluster.add_clusternodes();
615  node.set_publickey(it->first.humanNodePublic());
616  node.set_reporttime(it->second.getReportTime());
617  node.set_nodeload(it->second.getLoadFee());
618  if (!it->second.getName().empty())
619  node.set_nodename(it->second.getName());
620  }
621 
622  Resource::Gossip gossip = getApp().getResourceManager().exportConsumers();
623  BOOST_FOREACH (Resource::Gossip::Item const& item, gossip.items)
624  {
625  protocol::TMLoadSource& node = *cluster.add_loadsources();
626  node.set_name (to_string (item.address));
627  node.set_cost (item.balance);
628  }
629  getApp ().overlay ().foreach (send_if (
630  boost::make_shared<Message>(cluster, protocol::mtCLUSTER),
631  peer_in_cluster ()));
632  setClusterTimer ();
