822  getApp().getJobQueue ().getLoadEventAP (jtPEER, "Peer::read"));

826  // and it must be received *exactly* once during a connection; any

828

837  {

844

1250  {
1251  SSL* ssl = m_socket->ssl_handle ();
1252 
1253  if (!ssl)
1254  {
1255  m_journal.error << "Cookie generation: No underlying connection";
1256  return false;
1257  }
1258 
1259  unsigned char sha1[64];
1260  unsigned char sha2[64];
1261 
1262  if (!hashLatestFinishedMessage(ssl, sha1, SSL_get_finished))
1263  {
1264  m_journal.error << "Cookie generation: local setup not complete";
1265  return false;
1266  }
1267

1269  {
1270  m_journal.error << "Cookie generation: peer setup not complete";
1271  return false;
1272  }
1273 
1274  // If both messages hash to the same value (i.e. match) something is
1275  // wrong. This would cause the resulting cookie to be 0.
1276  if (memcmp (sha1, sha2, sizeof (sha1)) == 0)
1277  {
1278  m_journal.error << "Cookie generation: identical finished messages";
1279  return false;
1280  }
1281 
1282  for (size_t i = 0; i < sizeof (sha1); ++i)
1283  sha1[i] ^= sha2[i];
1284 
1285  // Finally, derive the actual cookie for the values that we have
1286  // calculated.
1287  m_secureCookie = Serializer::getSHA512Half (sha1, sizeof(sha1));
1288 
1289  return true;
