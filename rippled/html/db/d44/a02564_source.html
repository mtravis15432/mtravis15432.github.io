120 Options SanitizeOptions(const std::string& dbname,
121  const InternalKeyComparator* icmp,
122  const InternalFilterPolicy* ipolicy,
123  const Options& src) {
124  Options result = src;
125  result.comparator = icmp;
126  result.filter_policy = (src.filter_policy != nullptr) ? ipolicy : nullptr;
127  ClipToRange(&result.max_open_files, 20, 1000000);
128  ClipToRange(&result.write_buffer_size, ((size_t)64)<<10,
129  ((size_t)64)<<30);
130  ClipToRange(&result.block_size, 1<<10, 4<<20);
131 
132  // if user sets arena_block_size, we trust user to use this value. Otherwise,
133  // calculate a proper value from writer_buffer_size;
134  if (result.arena_block_size <= 0) {
135  result.arena_block_size = result.write_buffer_size / 10;
136  }
137 
138  result.min_write_buffer_number_to_merge = std::min(
139  result.min_write_buffer_number_to_merge, result.max_write_buffer_number-1);
140  if (result.info_log == nullptr) {
141  Status s = CreateLoggerFromOptions(dbname, result.db_log_dir, src.env,
142  result, &result.info_log);
143  if (!s.ok()) {
144  // No place suitable for logging
145  result.info_log = nullptr;
146  }
147  }
148  if (result.block_cache == nullptr && !result.no_block_cache) {
149  result.block_cache = NewLRUCache(8 << 20);
150  }
151  result.compression_per_level = src.compression_per_level;
152  if (result.block_size_deviation < 0 || result.block_size_deviation > 100) {
153  result.block_size_deviation = 0;
154  }
155  if (result.max_mem_compaction_level >= result.num_levels) {
156  result.max_mem_compaction_level = result.num_levels - 1;
157  }
158  if (result.soft_rate_limit > result.hard_rate_limit) {
159  result.soft_rate_limit = result.hard_rate_limit;
160  }
161  if (result.compaction_filter) {
162  Log(result.info_log, "Compaction filter specified, ignore factory");
163  }
164  if (result.prefix_extractor) {
165  // If a prefix extractor has been supplied and a PrefixHashRepFactory is
166  // being used, make sure that the latter uses the former as its transform
167  // function.
168  auto factory = dynamic_cast<PrefixHashRepFactory*>(
169  result.memtable_factory.get());
170  if (factory &&
171  factory->GetTransform() != result.prefix_extractor) {
172  Log(result.info_log, "A prefix hash representation factory was supplied "
173  "whose prefix extractor does not match options.prefix_extractor. "
174  "Falling back to skip list representation factory");
175  result.memtable_factory = std::make_shared<SkipListFactory>();
176  } else if (factory) {
177  Log(result.info_log, "Prefix hash memtable rep is in use.");
178  }
179  }
180 
181  if (result.wal_dir.empty()) {
182  // Use dbname as default
183  result.wal_dir = dbname;
184  }
185 
186  // -- Sanitize the table properties collector
187  // All user defined properties collectors will be wrapped by
188  // UserKeyTablePropertiesCollector since for them they only have the
189  // knowledge of the user keys; internal keys are invisible to them.
190  auto& collectors = result.table_properties_collectors;
191  for (size_t i = 0; i < result.table_properties_collectors.size(); ++i) {
192  assert(collectors[i]);
193  collectors[i] =
194  std::make_shared<UserKeyTablePropertiesCollector>(collectors[i]);
195  }
196 
197  // Add collector to collect internal key statistics
198  collectors.push_back(
199  std::make_shared<InternalKeyPropertiesCollector>()
200  );
201 
202  return result;
203 }

205 CompressionType GetCompressionType(const Options& options, int level,
206  const bool enable_compression) {
207  if (!enable_compression) {
208  // disable compression
209  return kNoCompression;
210  }
211  // If the use has specified a different compression level for each level,
212  // then pick the compresison for that level.
213  if (!options.compression_per_level.empty()) {
214  const int n = options.compression_per_level.size() - 1;
215  // It is possible for level_ to be -1; in that case, we use level
216  // 0's compression. This occurs mostly in backwards compatibility
217  // situations when the builder doesn't know what level the file
218  // belongs to. Likewise, if level_ is beyond the end of the
219  // specified compression levels, use the last value.
220  return options.compression_per_level[std::max(0, std::min(level, n))];
221  } else {
222  return options.compression;
223  }
224 }

439 // of all files in the filesystem in 'all_files'.

444 void DBImpl::FindObsoleteFiles(DeletionState& deletion_state,
445  bool force,
446  bool no_full_scan) {
447  mutex_.AssertHeld();
448 
449  // if deletion is disabled, do nothing
450  if (disable_delete_obsolete_files_) {
451  return;
452  }
453 
454  bool doing_the_full_scan = false;
455 
456  // logic for figurint out if we're doing the full scan
457  if (no_full_scan) {
458  doing_the_full_scan = false;
459  } else if (force || options_.delete_obsolete_files_period_micros == 0) {
460  doing_the_full_scan = true;
461  } else {
462  const uint64_t now_micros = env_->NowMicros();
463  if (delete_obsolete_files_last_run_ +
464  options_.delete_obsolete_files_period_micros < now_micros) {
465  doing_the_full_scan = true;
466  delete_obsolete_files_last_run_ = now_micros;
467  }
468  }
469 
470  // get obsolete files
471  versions_->GetObsoleteFiles(&deletion_state.sst_delete_files);
472 
473  // store the current filenum, lognum, etc
474  deletion_state.manifest_file_number = versions_->ManifestFileNumber();
475  deletion_state.log_number = versions_->LogNumber();
476  deletion_state.prev_log_number = versions_->PrevLogNumber();
477 
478  if (!doing_the_full_scan && !deletion_state.HaveSomethingToDelete()) {
479  // avoid filling up sst_live if we're sure that we
480  // are not going to do the full scan and that we don't have
481  // anything to delete at the moment
482  return;
483  }
484 
485  // don't delete live files
486  deletion_state.sst_live.assign(pending_outputs_.begin(),
487  pending_outputs_.end());
488  versions_->AddLiveFiles(&deletion_state.sst_live);
489 
490  if (doing_the_full_scan) {
491  // set of all files in the directory
492  env_->GetChildren(dbname_, &deletion_state.all_files); // Ignore errors
493 
494  //Add log files in wal_dir
495  if (options_.wal_dir != dbname_) {
496  std::vector<std::string> log_files;
497  env_->GetChildren(options_.wal_dir, &log_files); // Ignore errors
498  deletion_state.all_files.insert(
499  deletion_state.all_files.end(),
500  log_files.begin(),
501  log_files.end()
502  );
503  }
504  }
505 }

507 // Diffs the files listed in filenames and those that do not
508 // belong to live files are posibly removed. Also, removes all the
509 // files in sst_delete_files and log_delete_files.
510 // It is not necessary to hold the mutex when invoking this method.
511 void DBImpl::PurgeObsoleteFiles(DeletionState& state) {
512  // this checks if FindObsoleteFiles() was run before. If not, don't do
513  // PurgeObsoleteFiles(). If FindObsoleteFiles() was run, we need to also
514  // run PurgeObsoleteFiles(), even if disable_delete_obsolete_files_ is true
515  if (state.manifest_file_number == 0) {
516  return;
517  }
518 
519  uint64_t number;
520  FileType type;
521  std::vector<std::string> old_log_files;
522 
523  // Now, convert live list to an unordered set, WITHOUT mutex held;
524  // set is slow.
525  std::unordered_set<uint64_t> live_set(state.sst_live.begin(),
526  state.sst_live.end());
527 
528  state.all_files.reserve(state.all_files.size() +
529  state.sst_delete_files.size());
530  for (auto file : state.sst_delete_files) {
531  state.all_files.push_back(TableFileName("", file->number).substr(1));
532  delete file;
533  }
534 
535  state.all_files.reserve(state.all_files.size() +
536  state.log_delete_files.size());
537  for (auto filenum : state.log_delete_files) {
538  if (filenum > 0) {
539  state.all_files.push_back(LogFileName("", filenum).substr(1));
540  }
541  }
542 
543  // dedup state.all_files so we don't try to delete the same
544  // file twice
545  sort(state.all_files.begin(), state.all_files.end());
546  auto unique_end = unique(state.all_files.begin(), state.all_files.end());
547 
548  for (size_t i = 0; state.all_files.begin() + i < unique_end; i++) {
549  if (ParseFileName(state.all_files[i], &number, &type)) {
550  bool keep = true;
551  switch (type) {
552  case kLogFile:
553  keep = ((number >= state.log_number) ||
554  (number == state.prev_log_number));
555  break;
556  case kDescriptorFile:
557  // Keep my manifest file, and any newer incarnations'
558  // (in case there is a race that allows other incarnations)
559  keep = (number >= state.manifest_file_number);
560  break;
561  case kTableFile:
562  keep = (live_set.find(number) != live_set.end());
563  break;
564  case kTempFile:
565  // Any temp files that are currently being written to must
566  // be recorded in pending_outputs_, which is inserted into "live"
567  keep = (live_set.find(number) != live_set.end());
568  break;
569  case kInfoLogFile:
570  keep = true;
571  if (number != 0) {
572  old_log_files.push_back(state.all_files[i]);
573  }
574  break;
575  case kCurrentFile:
576  case kDBLockFile:
577  case kIdentityFile:
578  case kMetaDatabase:
579  keep = true;
580  break;
581  }
582 
583  if (!keep) {
584  if (type == kTableFile) {
585  // evict from cache
586  table_cache_->Evict(number);
587  }
588  std::string fname = ((type == kLogFile) ? options_.wal_dir : dbname_) +
589  "/" + state.all_files[i];
590  Log(options_.info_log,
591  "Delete type=%d #%lu",
592  int(type),
593  (unsigned long)number);
594 
595  Status st;
596  if (type == kLogFile && (options_.WAL_ttl_seconds > 0 ||
597  options_.WAL_size_limit_MB > 0)) {
598  st = env_->RenameFile(fname,
599  ArchivedLogFileName(options_.wal_dir, number));
600  if (!st.ok()) {
601  Log(options_.info_log,
602  "RenameFile logfile #%lu FAILED -- %s\n",
603  (unsigned long)number, st.ToString().c_str());
604  }
605  } else {
606  st = env_->DeleteFile(fname);
607  if (!st.ok()) {
608  Log(options_.info_log, "Delete type=%d #%lu FAILED -- %s\n",
609  int(type), (unsigned long)number, st.ToString().c_str());
610  }
611  }
612  }
613  }
614  }
615 
616  // Delete old info log files.
617  size_t old_log_file_count = old_log_files.size();
618  // NOTE: Currently we only support log purge when options_.db_log_dir is
619  // located in `dbname` directory.
620  if (old_log_file_count >= options_.keep_log_file_num &&
621  options_.db_log_dir.empty()) {
622  std::sort(old_log_files.begin(), old_log_files.end());
623  size_t end = old_log_file_count - options_.keep_log_file_num;
624  for (unsigned int i = 0; i <= end; i++) {
625  std::string& to_delete = old_log_files.at(i);
626  // Log(options_.info_log, "Delete type=%d %s\n",
627  // int(kInfoLogFile), to_delete.c_str());
628  env_->DeleteFile(dbname_ + "/" + to_delete);
629  }

769  bool error_if_log_file_exist) {

771 
772  assert(db_lock_ == nullptr);
773  if (!external_table) {
774  // We call CreateDirIfMissing() as the directory may already exist (if we
775  // are reopening a DB), when this happens we don't want creating the
776  // directory to cause an error. However, we need to check if creating the
777  // directory fails or else we may get an obscure message about the lock
778  // file not existing. One real-world example of this occurring is if
779  // env->CreateDirIfMissing() doesn't create intermediate directories, e.g.
780  // when dbname_ is "dir/db" but when "dir" doesn't exist.
781  Status s = env_->CreateDirIfMissing(dbname_);
782  if (!s.ok()) {

784  }
785 
786  s = env_->LockFile(LockFileName(dbname_), &db_lock_);
787  if (!s.ok()) {
788  return s;
789  }
790 
791  if (!env_->FileExists(CurrentFileName(dbname_))) {
792  if (options_.create_if_missing) {
793  // TODO: add merge_operator name check
794  s = NewDB();
795  if (!s.ok()) {
796  return s;
797  }
798  } else {
799  return Status::InvalidArgument(
800  dbname_, "does not exist (create_if_missing is false)");
801  }
802  } else {
803  if (options_.error_if_exists) {
804  return Status::InvalidArgument(
805  dbname_, "exists (error_if_exists is true)");
806  }
807  }
808  // Check for the IDENTITY file and create it if not there
809  if (!env_->FileExists(IdentityFileName(dbname_))) {
810  s = SetIdentityFile(env_, dbname_);
811  if (!s.ok()) {
812  return s;
813  }
814  }
815  }
816 
817  Status s = versions_->Recover();

819  SequenceNumber max_sequence(0);
820 
821  // Recover from all newer log files than the ones named in the
822  // descriptor (new log files may have been added by the previous
823  // incarnation without registering them in the descriptor).
824  //
825  // Note that PrevLogNumber() is no longer used, but we pay
826  // attention to it in case we are recovering a database
827  // produced by an older version of rocksdb.
828  const uint64_t min_log = versions_->LogNumber();
829  const uint64_t prev_log = versions_->PrevLogNumber();

831  s = env_->GetChildren(options_.wal_dir, &filenames);
832  if (!s.ok()) {
833  return s;
834  }
835  uint64_t number;
836  FileType type;

838  for (size_t i = 0; i < filenames.size(); i++) {
839  if (ParseFileName(filenames[i], &number, &type)
840  && type == kLogFile
841  && ((number >= min_log) || (number == prev_log))) {
842  logs.push_back(number);
843  }
844  }
845

847  return Status::Corruption(""
848  "The db was opened in readonly mode with error_if_log_file_exist"
849  "flag but a log file already exists");
850  }
851 
852  // Recover in the order in which the logs were generated
853  std::sort(logs.begin(), logs.end());
854  for (size_t i = 0; i < logs.size(); i++) {
855  s = RecoverLogFile(logs[i], edit, &max_sequence, external_table);
856  // The previous incarnation may not have written any MANIFEST
857  // records after allocating this log number. So we manually
858  // update the file number allocation counter in VersionSet.
859  versions_->MarkFileNumberUsed(logs[i]);
860  }
861 
862  if (s.ok()) {
863  if (versions_->LastSequence() < max_sequence) {
864  versions_->SetLastSequence(max_sequence);
865  }
866  SetTickerCount(options_.statistics, SEQUENCE_NUMBER,
867  versions_->LastSequence());
868  }
869  }
870 
871  return s;

874 Status DBImpl::RecoverLogFile(uint64_t log_number,
875  VersionEdit* edit,
876  SequenceNumber* max_sequence,
877  MemTable* external_table) {
878  struct LogReporter : public log::Reader::Reporter {
879  Env* env;
880  Logger* info_log;
881  const char* fname;
882  Status* status; // nullptr if options_.paranoid_checks==false or
883  // options_.skip_log_error_on_recovery==true
884  virtual void Corruption(size_t bytes, const Status& s) {
885  Log(info_log, "%s%s: dropping %d bytes; %s",
886  (this->status == nullptr ? "(ignoring error) " : ""),
887  fname, static_cast<int>(bytes), s.ToString().c_str());
888  if (this->status != nullptr && this->status->ok()) *this->status = s;
889  }
890  };
891 
892  mutex_.AssertHeld();
893 
894  // Open the log file
895  std::string fname = LogFileName(options_.wal_dir, log_number);
896  unique_ptr<SequentialFile> file;
897  Status status = env_->NewSequentialFile(fname, &file, storage_options_);
898  if (!status.ok()) {
899  MaybeIgnoreError(&status);
900  return status;
901  }
902 
903  // Create the log reader.
904  LogReporter reporter;
905  reporter.env = env_;
906  reporter.info_log = options_.info_log.get();
907  reporter.fname = fname.c_str();
908  reporter.status = (options_.paranoid_checks &&
909  !options_.skip_log_error_on_recovery ? &status : nullptr);
910  // We intentially make log::Reader do checksumming even if
911  // paranoid_checks==false so that corruptions cause entire commits
912  // to be skipped instead of propagating bad information (like overly
913  // large sequence numbers).
914  log::Reader reader(std::move(file), &reporter, true/*checksum*/,
915  0/*initial_offset*/);
916  Log(options_.info_log, "Recovering log #%lu",
917  (unsigned long) log_number);
918 
919  // Read all the records and add to a memtable
920  std::string scratch;
921  Slice record;
922  WriteBatch batch;
923  MemTable* mem = nullptr;
924  if (external_table) {
925  mem = external_table;
926  }
927  while (reader.ReadRecord(&record, &scratch) && status.ok()) {
928  if (record.size() < 12) {
929  reporter.Corruption(
930  record.size(), Status::Corruption("log record too small"));
931  continue;
932  }
933  WriteBatchInternal::SetContents(&batch, record);
934 
935  if (mem == nullptr) {
936  mem = new MemTable(internal_comparator_, mem_rep_factory_,
937  NumberLevels(), options_);
938  mem->Ref();
939  }
940  status = WriteBatchInternal::InsertInto(&batch, mem, &options_);
941  MaybeIgnoreError(&status);
942  if (!status.ok()) {
943  break;
944  }
945  const SequenceNumber last_seq =
946  WriteBatchInternal::Sequence(&batch) +
947  WriteBatchInternal::Count(&batch) - 1;
948  if (last_seq > *max_sequence) {
949  *max_sequence = last_seq;
950  }
951 
952  if (!external_table &&
953  mem->ApproximateMemoryUsage() > options_.write_buffer_size) {
954  status = WriteLevel0TableForRecovery(mem, edit);
955  if (!status.ok()) {
956  // Reflect errors immediately so that conditions like full
957  // file-systems cause the DB::Open() to fail.
958  break;
959  }
960  mem->Unref();
961  mem = nullptr;
962  }
963  }
964 
965  if (status.ok() && mem != nullptr && !external_table) {
966  status = WriteLevel0TableForRecovery(mem, edit);
967  // Reflect errors immediately so that conditions like full
968  // file-systems cause the DB::Open() to fail.
969  }
970 
971  if (mem != nullptr && !external_table) mem->Unref();
972  return status;
973 }

1025 Status DBImpl::WriteLevel0Table(std::vector<MemTable*> &mems, VersionEdit* edit,
1026  uint64_t* filenumber) {
1027  mutex_.AssertHeld();

1029  FileMetaData meta;
1030  meta.number = versions_->NewFileNumber();
1031  *filenumber = meta.number;
1032  pending_outputs_.insert(meta.number);
1033 
1034  std::vector<Iterator*> list;
1035  for (MemTable* m : mems) {
1036  Log(options_.info_log,
1037  "Flushing memtable with log file: %lu\n",
1038  (unsigned long)m->GetLogNumber());
1039  list.push_back(m->NewIterator());
1040  }
1041  Iterator* iter = NewMergingIterator(&internal_comparator_, &list[0],
1042  list.size());
1043  const SequenceNumber newest_snapshot = snapshots_.GetNewest();
1044  const SequenceNumber earliest_seqno_in_memtable =
1045  mems[0]->GetFirstSequenceNumber();
1046  Log(options_.info_log,
1047  "Level-0 flush table #%lu: started",
1048  (unsigned long)meta.number);
1049 
1050  Version* base = versions_->current();
1051  base->Ref(); // it is likely that we do not need this reference
1052  Status s;
1053  {
1054  mutex_.Unlock();
1055  // We skip compression if universal compression is used and the size
1056  // threshold is set for compression.
1057  bool enable_compression = (options_.compaction_style
1058  != kCompactionStyleUniversal ||
1059  options_.compaction_options_universal.compression_size_percent < 0);
1060  s = BuildTable(dbname_, env_, options_, storage_options_,
1061  table_cache_.get(), iter, &meta,
1062  user_comparator(), newest_snapshot,
1063  earliest_seqno_in_memtable, enable_compression);
1064  LogFlush(options_.info_log);
1065  mutex_.Lock();
1066  }
1067  base->Unref();
1068 
1069  Log(options_.info_log, "Level-0 flush table #%lu: %lu bytes %s",
1070  (unsigned long) meta.number,
1071  (unsigned long) meta.file_size,
1072  s.ToString().c_str());

1074 
1075  // re-acquire the most current version
1076  base = versions_->current();
1077

1079  // The pending_outputs cannot be cleared here, otherwise this newly
1080  // created file might not be considered as a live-file by another
1081  // compaction thread that is concurrently deleting obselete files.
1082  // The pending_outputs can be cleared only after the new version is
1083  // committed so that other threads can recognize this file as a
1084  // valid one.
1085  // pending_outputs_.erase(meta.number);
1086 
1087  // Note that if file_size is zero, the file has been deleted and
1088  // should not be added to the manifest.
1089  int level = 0;
1090  if (s.ok() && meta.file_size > 0) {
1091  const Slice min_user_key = meta.smallest.user_key();
1092  const Slice max_user_key = meta.largest.user_key();
1093  // if we have more than 1 background thread, then we cannot
1094  // insert files directly into higher levels because some other
1095  // threads could be concurrently producing compacted files for
1096  // that key range.

1098  options_.compaction_style == kCompactionStyleLevel) {

1102  meta.smallest, meta.largest,
1103  meta.smallest_seqno, meta.largest_seqno);
1104  }
1105 
1106  CompactionStats stats;
1107  stats.micros = env_->NowMicros() - start_micros;
1108  stats.bytes_written = meta.file_size;
1109  stats_[level].Add(stats);
1110  return s;

1227 
1228  // only allow one thread refitting
1229  if (refitting_level_) {
1230  Log(options_.info_log, "ReFitLevel: another thread is refitting");

1233  refitting_level_ = true;
1234

1236  bg_work_gate_closed_ = true;

1238  Log(options_.info_log,

1330 Status DBImpl::RetainProbableWalFiles(VectorLogPtr& all_logs,
1331  const SequenceNumber target) {
1332  long start = 0; // signed to avoid overflow when target is < first file.

1343  } else {

1345  }
1346  }

1485 void DBImpl::TEST_CompactRange(int level, const Slice* begin,const Slice* end) {
1486  assert(level >= 0);
1487 
1488  InternalKey begin_storage, end_storage;
1489 
1490  ManualCompaction manual;
1491  manual.level = level;
1492  manual.done = false;
1493  manual.in_progress = false;
1494  // For universal compaction, we enforce every manual compaction to compact
1495  // all files.
1496  if (begin == nullptr ||
1497  options_.compaction_style == kCompactionStyleUniversal) {
1498  manual.begin = nullptr;
1499  } else {
1500  begin_storage = InternalKey(*begin, kMaxSequenceNumber, kValueTypeForSeek);
1501  manual.begin = &begin_storage;
1502  }
1503  if (end == nullptr ||
1504  options_.compaction_style == kCompactionStyleUniversal) {
1505  manual.end = nullptr;
1506  } else {
1507  end_storage = InternalKey(*end, 0, static_cast<ValueType>(0));
1508  manual.end = &end_storage;
1509  }
1510 
1511  MutexLock l(&mutex_);
1512 
1513  // When a manual compaction arrives, temporarily throttle down
1514  // the number of background compaction threads to 1. This is
1515  // needed to ensure that this manual compaction can compact
1516  // any range of keys/files. We artificialy increase
1517  // bg_compaction_scheduled_ by a large number, this causes
1518  // the system to have a single background thread. Now,
1519  // this manual compaction can progress without stomping
1520  // on any other concurrent compactions.
1521  const int LargeNumber = 10000000;
1522  const int newvalue = options_.max_background_compactions-1;
1523  bg_compaction_scheduled_ += LargeNumber;
1524  while (bg_compaction_scheduled_ > LargeNumber) {
1525  Log(options_.info_log, "Manual compaction request waiting for background threads to fall below 1");
1526  bg_cv_.Wait();
1527  }
1528  Log(options_.info_log, "Manual compaction starting");
1529 
1530  while (!manual.done) {
1531  while (manual_compaction_ != nullptr) {
1532  bg_cv_.Wait();
1533  }
1534  manual_compaction_ = &manual;
1535  if (bg_compaction_scheduled_ == LargeNumber) {
1536  bg_compaction_scheduled_ = newvalue;
1537  }
1538  MaybeScheduleFlushOrCompaction();
1539  while (manual_compaction_ == &manual) {
1540  bg_cv_.Wait();
1541  }
1542  }
1543  assert(!manual.in_progress);
1544 
1545  // wait till there are no background threads scheduled
1546  bg_compaction_scheduled_ += LargeNumber;
1547  while (bg_compaction_scheduled_ > LargeNumber + newvalue) {
1548  Log(options_.info_log, "Manual compaction resetting background threads");
1549  bg_cv_.Wait();
1550  }
1551  bg_compaction_scheduled_ = 0;
1552 }

1748  DeletionState& deletion_state) {
1749  *madeProgress = false;
1750  mutex_.AssertHeld();
1751

1753  while (imm_.IsFlushPending(options_.min_write_buffer_number_to_merge)) {

1758  Status stat = FlushMemTableToOutputFile(madeProgress, deletion_state);
1759  if (!stat.ok()) {
1760  return stat;
1761  }
1762  }
1763 
1764  unique_ptr<Compaction> c;
1765  bool is_manual = (manual_compaction_ != nullptr) &&
1766  (manual_compaction_->in_progress == false);
1767  InternalKey manual_end;
1768  if (is_manual) {
1769  ManualCompaction* m = manual_compaction_;
1770  assert(!m->in_progress);
1771  m->in_progress = true; // another thread cannot pick up the same work
1772  c.reset(versions_->CompactRange(m->level, m->begin, m->end));
1773  if (c) {
1774  manual_end = c->input(0, c->num_input_files(0) - 1)->largest;
1775  } else {
1776  m->done = true;
1777  }
1778  Log(options_.info_log,
1779  "Manual compaction at level-%d from %s .. %s; will stop at %s\n",
1780  m->level,
1781  (m->begin ? m->begin->DebugString().c_str() : "(begin)"),
1782  (m->end ? m->end->DebugString().c_str() : "(end)"),
1783  (m->done ? "(end)" : manual_end.DebugString().c_str()));
1784  } else if (!options_.disable_auto_compactions) {
1785  c.reset(versions_->PickCompaction());
1786  }
1787 
1788  Status status;
1789  if (!c) {
1790  // Nothing to do
1791  Log(options_.info_log, "Compaction nothing to do");
1792  } else if (!is_manual && c->IsTrivialMove()) {
1793  // Move file to next level
1794  assert(c->num_input_files(0) == 1);
1795  FileMetaData* f = c->input(0, 0);
1796  c->edit()->DeleteFile(c->level(), f->number);
1797  c->edit()->AddFile(c->level() + 1, f->number, f->file_size,
1798  f->smallest, f->largest,
1799  f->smallest_seqno, f->largest_seqno);
1800  status = versions_->LogAndApply(c->edit(), &mutex_);
1801  VersionSet::LevelSummaryStorage tmp;
1802  Log(options_.info_log, "Moved #%lld to level-%d %lld bytes %s: %s\n",
1803  static_cast<unsigned long long>(f->number),
1804  c->level() + 1,
1805  static_cast<unsigned long long>(f->file_size),
1806  status.ToString().c_str(),
1807  versions_->LevelSummary(&tmp));
1808  versions_->ReleaseCompactionFiles(c.get(), status);
1809  *madeProgress = true;
1810  } else {
1811  MaybeScheduleFlushOrCompaction(); // do more compaction work in parallel.
1812  CompactionState* compact = new CompactionState(c.get());
1813  status = DoCompactionWork(compact, deletion_state);
1814  CleanupCompaction(compact, status);
1815  versions_->ReleaseCompactionFiles(c.get(), status);
1816  c->ReleaseInputs();
1817  *madeProgress = true;
1818  }
1819  c.reset();
1820 
1821  if (status.ok()) {
1822  // Done
1823  } else if (shutting_down_.Acquire_Load()) {
1824  // Ignore compaction errors found during shutting down
1825  } else {
1826  Log(options_.info_log,
1827  "Compaction error: %s", status.ToString().c_str());
1828  if (options_.paranoid_checks && bg_error_.ok()) {
1829  bg_error_ = status;
1830  }
1831  }
1832 
1833  if (is_manual) {
1834  ManualCompaction* m = manual_compaction_;
1835  if (!status.ok()) {
1836  m->done = true;
1837  }
1838  // For universal compaction:
1839  // Because universal compaction always happens at level 0, so one
1840  // compaction will pick up all overlapped files. No files will be
1841  // filtered out due to size limit and left for a successive compaction.
1842  // So we can safely conclude the current compaction.
1843  //
1844  // Also note that, if we don't stop here, then the current compaction
1845  // writes a new file back to level 0, which will be used in successive
1846  // compaction. Hence the manual compaction will never finish.
1847  if (options_.compaction_style == kCompactionStyleUniversal) {
1848  m->done = true;
1849  }
1850  if (!m->done) {
1851  // We only compacted part of the requested range. Update *m
1852  // to the range that is left to be compacted.
1853  m->tmp_storage = manual_end;
1854  m->begin = &m->tmp_storage;
1855  }
1856  m->in_progress = false; // not being processed anymore
1857  manual_compaction_ = nullptr;
1858  }
1859  return status;
1860 }

1863  mutex_.AssertHeld();
1864  if (compact->builder != nullptr) {
1865  // May happen if we get a shutdown call in the middle of compaction
1866  compact->builder->Abandon();
1867  compact->builder.reset();
1868  } else {
1869  assert(compact->outfile == nullptr);
1870  }
1871  for (size_t i = 0; i < compact->outputs.size(); i++) {
1872  const CompactionState::Output& out = compact->outputs[i];
1873  pending_outputs_.erase(out.number);

2864 // REQUIRES: Writer list must be non-empty
2865 // REQUIRES: First writer must have a non-nullptr batch
2866 WriteBatch* DBImpl::BuildBatchGroup(Writer** last_writer) {
2867  assert(!writers_.empty());
2868  Writer* first = writers_.front();
2869  WriteBatch* result = first->batch;
2870  assert(result != nullptr);
2871 
2872  size_t size = WriteBatchInternal::ByteSize(first->batch);
2873 
2874  // Allow the group to grow up to a maximum size, but if the
2875  // original write is small, limit the growth so we do not slow
2876  // down the small write too much.
2877  size_t max_size = 1 << 20;
2878  if (size <= (128<<10)) {
2879  max_size = size + (128<<10);
2880  }
2881 
2882  *last_writer = first;
2883  std::deque<Writer*>::iterator iter = writers_.begin();
2884  ++iter; // Advance past "first"
2885  for (; iter != writers_.end(); ++iter) {
2886  Writer* w = *iter;
2887  if (w->sync && !first->sync) {
2888  // Do not include a sync write into a batch handled by a non-sync write.
2889  break;
2890  }
2891 
2892  if (!w->disableWAL && first->disableWAL) {
2893  // Do not include a write that needs WAL into a batch that has
2894  // WAL disabled.
2895  break;
2896  }
2897 
2898  if (w->batch != nullptr) {
2899  size += WriteBatchInternal::ByteSize(w->batch);
2900  if (size > max_size) {
2901  // Do not make batch too big
2902  break;
2903  }
2904 
2905  // Append to *reuslt
2906  if (result == first->batch) {
2907  // Switch to temporary batch instead of disturbing caller's batch
2908  result = &tmp_batch_;

2910  WriteBatchInternal::Append(result, first->batch);

2919 // This function computes the amount of time in microseconds by which a write
2920 // should be delayed based on the number of level-0 files according to the
2921 // following formula:
2922 // if n < bottom, return 0;
2923 // if n >= top, return 1000;
2924 // otherwise, let r = (n - bottom) /
2925 // (top - bottom)
2926 // and return r^2 * 1000.
2927 // The goal of this formula is to gradually increase the rate at which writes
2928 // are slowed. We also tried linear delay (r * 1000), but it seemed to do
2929 // slightly worse. There is no other particular reason for choosing quadratic.
2930 uint64_t DBImpl::SlowdownAmount(int n, int top, int bottom) {
2931  uint64_t delay;
2932  if (n >= top) {
2933  delay = 1000;
2934  }
2935  else if (n < bottom) {
2936  delay = 0;
2937  }
2938  else {
2939  // If we are here, we know that:
2940  // level0_start_slowdown <= n < level0_slowdown
2941  // since the previous two conditions are false.
2942  float how_much =
2943  (float) (n - bottom) /
2944  (top - bottom);
2945  delay = how_much * how_much * 1000;
2946  }
2947  assert(delay <= 1000);
2948  return delay;
2949 }
