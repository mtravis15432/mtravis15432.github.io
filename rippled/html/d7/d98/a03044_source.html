92 TER ChangeTransactor::applyFeature ()
93 {
94  uint256 feature (mTxn.getFieldH256 (sfFeature));
95 
96  SLE::pointer featureObject (mEngine->entryCache (
97  ltFEATURES, Ledger::getLedgerFeatureIndex ()));
98 
99  if (!featureObject)
100  featureObject = mEngine->entryCreate (
101  ltFEATURES, Ledger::getLedgerFeatureIndex ());
102 
103  STVector256 features (featureObject->getFieldV256 (sfFeatures));
104 
105  if (features.hasValue (feature))
106  return tefALREADY;
107 
108  features.addValue (feature);
109  featureObject->setFieldV256 (sfFeatures, features);
110  mEngine->entryModify (featureObject);
111 
112  getApp().getFeatureTable ().enable (feature);
113 
114  if (!getApp().getFeatureTable ().isSupported (feature))
115  getApp().getOPs ().setFeatureBlocked ();
116 
117  return tesSUCCESS;
118 }
