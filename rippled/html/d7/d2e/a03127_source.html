56 // VFALCO NOTE WHY do we need "the mutex?" This dependence on
57 // a hidden global, especially a synchronization primitive,
58 // is a flawed design.
59 // It's not clear exactly which data needs to be protected.
60 //
61 // call with the mutex
62 void LoadMonitor::update ()
63 {
64  int now = UptimeTimer::getInstance ().getElapsedSeconds ();
65 
66  // VFALCO TODO stop returning from the middle of functions.
67 
68  if (now == mLastUpdate) // current
69  return;
70 
71  // VFALCO TODO Why 8?
72  if ((now < mLastUpdate) || (now > (mLastUpdate + 8)))
73  {
74  // way out of date
75  mCounts = 0;
76  mLatencyEvents = 0;
77  mLatencyMSAvg = 0;
78  mLatencyMSPeak = 0;
79  mLastUpdate = now;
80  // VFALCO TODO don't return from the middle...
81  return;
82  }
83 
84  // do exponential decay
85  /*
86  David:
87 
88  "Imagine if you add 10 to something every second. And you
89  also reduce it by 1/4 every second. It will "idle" at 40,
90  correponding to 10 counts per second."
91  */
92  do
93  {
94  ++mLastUpdate;
95  mCounts -= ((mCounts + 3) / 4);
96  mLatencyEvents -= ((mLatencyEvents + 3) / 4);
97  mLatencyMSAvg -= (mLatencyMSAvg / 4);
98  mLatencyMSPeak -= (mLatencyMSPeak / 4);
99  }
100  while (mLastUpdate < now);
101 }

111 void LoadMonitor::addLatency (int latency)
112 {
113  // VFALCO NOTE Why does 1 become 0?
114  if (latency == 1)
115  latency = 0;
116 
117  ScopedLockType sl (mLock);
118 
119  update ();
120 
121  ++mLatencyEvents;
122  mLatencyMSAvg += latency;
123  mLatencyMSPeak += latency;
124 
125  // VFALCO NOTE Why are we multiplying by 4?
126  int const latencyPeak = mLatencyEvents * latency * 4;
127 
128  if (mLatencyMSPeak < latencyPeak)
129  mLatencyMSPeak = latencyPeak;
130 }

195 LoadMonitor::Stats LoadMonitor::getStats ()
196 {
197  Stats stats;
198 
199  ScopedLockType sl (mLock);
200 
201  update ();
202 
203  stats.count = mCounts / 4;
204 
205  if (mLatencyEvents == 0)
206  {
207  stats.latencyAvg = 0;
208  stats.latencyPeak = 0;
209  }
210  else
211  {
212  stats.latencyAvg = mLatencyMSAvg / (mLatencyEvents * 4);
213  stats.latencyPeak = mLatencyMSPeak / (mLatencyEvents * 4);

215
