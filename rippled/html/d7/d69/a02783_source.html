118 std::vector<std::string> RedisLists::Range(const std::string& key,
119  int32_t first, int32_t last) {

123

125  int listLen = Length(key);

127  first = listLen - (-first); // Replace (-x) with (N-x)

449 // Remove the last "num" occurrences of value in (list: key).
450 // TODO: I traverse the list 2x. Make faster. Might require MergeOperator.
451 // : throws RedisListException
452 int RedisLists::RemoveLast(const std::string& key, int32_t num,
453  const std::string& value) {
454  // Ensure that the number is positive
455  assert(num >= 0);
456 
457  // Extract the original list data
458  std::string data;
459  db_->Get(get_option_, key, &data);
460 
461  // Temporary variable to hold the "current element" in the blocks below
462  Slice elem;
463 
464  // Count the total number of occurrences of value
465  int totalOccs = 0;
466  for (RedisListIterator it(data); !it.Done(); it.Skip()) {
467  it.GetCurrent(&elem);
468  if (elem == value) {
469  ++totalOccs;
470  }
471  }
472 
473  // Construct an iterator to the data. Reserve enough space for the result.
474  RedisListIterator it(data);
475  int bytesRemoved = std::min(num,totalOccs)*it.SizeOf(value);
476  it.Reserve(it.Size() - bytesRemoved);
477 
478  // Traverse the list, appending all but the desired occurrences of value.
479  // Note: "Drop the last k occurrences" is equivalent to
480  // "keep only the first n-k occurrences", where n is total occurrences.
481  int numKept = 0; // Keep track of the number of times value is kept
482  while(!it.Done()) {
483  it.GetCurrent(&elem);
484 
485  // If we are within the deletion range and equal to value, drop it.
486  // Otherwise, append/keep/push it.
487  if (elem == value) {
488  if (numKept < totalOccs - num) {
489  it.Push();
490  ++numKept;
491  } else {
492  it.Skip();
493  }
494  } else {
495  // Always append the others
496  it.Push();
497  }
498  }
499 
500  // Put the result back to the database
501  db_->Put(put_option_, key, it.WriteResult());
502 
503  // Return the number of elements removed
504  return totalOccs - numKept;
505 }
