84  class SignaturePart
85  {
86  private:
87  size_t m_skip;
88  BigNum m_bn;
89 
90  public:
91  SignaturePart (unsigned char const* sig, size_t size)
92  : m_skip (0)
93  {
94  // The format is: <02> <length of signature> <signature>
95  if ((sig[0] != 0x02) || (size < 3))
96  return;
97 
98  size_t const len (sig[1]);
99 
100  // Claimed length can't be longer than amount of data available
101  if (len > (size - 2))
102  return;
103 
104  // Signature must be between 1 and 33 bytes.
105  if ((len < 1) || (len > 33))
106  return;
107 
108  // The signature can't be negative
109  if ((sig[2] & 0x80) != 0)
110  return;
111 
112  // It can't be zero
113  if ((sig[2] == 0) && (len == 1))
114  return;
115 
116  // And it can't be padded
117  if ((sig[2] == 0) && ((sig[3] & 0x80) == 0))
118  return;
119 
120  // Load the signature (ignore the marker prefix and length) and if
121  // successful, count the number of bytes we consumed.
122  if (m_bn.set (sig + 2, len))
123  m_skip = len + 2;
124  }
125 
126  bool valid () const
127  {
128  return m_skip != 0;
129  }
130 
131  // The signature as a BIGNUM
132  BigNum getBigNum () const
133  {
134  return m_bn;
135  }
136 
137  // Returns the number of bytes to skip for this signature part
138  size_t skip () const
139  {
140  return m_skip;
141  }
142  };
143 
144  // The SECp256k1 modulus
145  static BigNum const modulus (
146  "FFFFFFFFFFFFFFFFFFFFFFFFFFFFFFFEBAAEDCE6AF48A03BBFD25E8CD0364141");
147 }

161 bool isCanonicalECDSASig (void const* vSig, size_t sigLen, ECDSA strict_param)
162 {
163  // The format of a signature should be:
164  // <30> <len> [ <02> <lenR> <R> ] [ <02> <lenS> <S> ]
165 
166  unsigned char const* sig = reinterpret_cast<unsigned char const*> (vSig);
167 
168  if ((sigLen < 8) || (sigLen > 72))
169  return false;
170 
171  if ((sig[0] != 0x30) || (sig[1] != (sigLen - 2)))
172  return false;
173 
174  // The first two bytes are verified. Eat them.
175  sig += 2;
176  sigLen -= 2;
177 
178  // Verify the R signature
179  detail::SignaturePart sigR (sig, sigLen);
180 
181  if (!sigR.valid ())
182  return false;
183 
184  // Eat the number of bytes we consumed
185  sig += sigR.skip ();
186  sigLen -= sigR.skip ();
187 
188  // Verify the S signature
189  detail::SignaturePart sigS (sig, sigLen);
190 
191  if (!sigS.valid ())
192  return false;
193 
194  // Eat the number of bytes we consumed
195  sig += sigS.skip ();
196  sigLen -= sigS.skip ();
197 
198  // Nothing should remain at this point.
199  if (sigLen != 0)
200  return false;
201 
202  // Check whether R or S are greater than the modulus.
203  auto bnR (sigR.getBigNum ());
204  auto bnS (sigS.getBigNum ());
205 
206  if (BN_cmp (bnR, detail::modulus) != -1)
207  return false;
208 
209  if (BN_cmp (bnS, detail::modulus) != -1)
210  return false;
211 
212  // For a given signature, (R,S), the signature (R, N-S) is also valid. For
213  // a signature to be fully-canonical, the smaller of these two values must
214  // be specified. If operating in strict mode, check that as well.
215  if (strict_param == ECDSA::strict)
216  {
217  detail::BigNum mS;
218 
219  if (BN_sub (mS, detail::modulus, bnS) == 0)
220  return false;
221 
222  if (BN_cmp (bnS, mS) == 1)
223  return false;
224  }
225 
226  return true;
227 }

236 bool makeCanonicalECDSASig (void* vSig, size_t& sigLen)
237 {
238  unsigned char * sig = reinterpret_cast<unsigned char *> (vSig);
239  bool ret = false;
240 
241  // Find internals
242  int rLen = sig[3];
243  int sPos = rLen + 6, sLen = sig[rLen + 5];
244 
245  detail::BigNum origS, newS;
246  BN_bin2bn (&sig[sPos], sLen, origS);
247  BN_sub (newS, detail::modulus, origS);
248 
249  if (BN_cmp (origS, newS) == 1)
250  { // original signature is not fully canonical
251  unsigned char newSbuf [64];
252  int newSlen = BN_bn2bin (newS, newSbuf);
253 
254  if ((newSbuf[0] & 0x80) == 0)
255  { // no extra padding byte is needed
256  sig[1] = sig[1] - sLen + newSlen;
257  sig[sPos - 1] = newSlen;
258  memcpy (&sig[sPos], newSbuf, newSlen);
259  }
260  else
261  { // an extra padding byte is needed
262  sig[1] = sig[1] - sLen + newSlen + 1;
263  sig[sPos - 1] = newSlen + 1;
264  sig[sPos] = 0;
265  memcpy (&sig[sPos + 1], newSbuf, newSlen);
266  }
267  sigLen = sig[1] + 2;
268  }
269  else
270  ret = true;
271 
272  return ret;
273 }

275 template <class FwdIter, class Container>
276 void hex_to_binary (FwdIter first, FwdIter last, Container& out)
277 {
278  struct Table
279  {
280  int val[256];
281  Table ()
282  {
283  std::fill (val, val+256, 0);
284  for (int i = 0; i < 10; ++i)
285  val ['0'+i] = i;
286  for (int i = 0; i < 6; ++i)
287  {
288  val ['A'+i] = 10 + i;
289  val ['a'+i] = 10 + i;
290  }
291  }
292  int operator[] (int i)
293  {
294  return val[i];
295  }
296  };
297 
298  static Table lut;
299  out.reserve (std::distance (first, last) / 2);
300  while (first != last)
301  {
302  auto const hi (lut[(*first++)]);
303  auto const lo (lut[(*first++)]);
304  out.push_back ((hi*16)+lo);
305  }
306 }
