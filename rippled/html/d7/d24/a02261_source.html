72 bool File::isOnHardDisk() const
73 {
74  struct statfs buf;
75 
76  if (statfs (getFullPathName().toUTF8(), &buf) == 0)
77  {
78  switch (buf.f_type)
79  {
80  case U_ISOFS_SUPER_MAGIC: // CD-ROM
81  case U_MSDOS_SUPER_MAGIC: // Probably floppy (but could be mounted FAT filesystem)
82  case U_NFS_SUPER_MAGIC: // Network NFS
83  case U_SMB_SUPER_MAGIC: // Network Samba
84  return false;
85 
86  default:
87  // Assume anything else is a hard-disk (but note it could
88  // be a RAM disk. There isn't a good way of determining
89  // this for sure)
90  return true;
91  }
92  }
93 
94  // Assume so if this fails for some reason
95  return true;
96 }

133 static File resolveXDGFolder (const char* const type, const char* const fallbackFolder)
134 {
135  StringArray confLines;
136  File ("~/.config/user-dirs.dirs").readLines (confLines);
137 
138  for (int i = 0; i < confLines.size(); ++i)
139  {
140  const String line (confLines[i].trimStart());
141 
142  if (line.startsWith (type))
143  {
144  // eg. resolve XDG_MUSIC_DIR="$HOME/Music" to /home/user/Music
145  const File f (line.replace ("$HOME", File ("~").getFullPathName())
146  .fromFirstOccurrenceOf ("=", false, false)
147  .trim().unquoted());
148 
149  if (f.isDirectory())
150  return f;
151  }
152  }
153 
154  return File (fallbackFolder);
155 }

160 File File::getSpecialLocation (const SpecialLocationType type)
161 {
162  switch (type)
163  {
164  case userHomeDirectory:
165  {
166  const char* homeDir = getenv ("HOME");
167 
168  if (homeDir == nullptr)
169  {
170  struct passwd* const pw = getpwuid (getuid());
171  if (pw != nullptr)
172  homeDir = pw->pw_dir;
173  }
174 
175  return File (CharPointer_UTF8 (homeDir));
176  }
177 
178  case userDocumentsDirectory: return resolveXDGFolder ("XDG_DOCUMENTS_DIR", "~");
179  case userMusicDirectory: return resolveXDGFolder ("XDG_MUSIC_DIR", "~");
180  case userMoviesDirectory: return resolveXDGFolder ("XDG_VIDEOS_DIR", "~");
181  case userPicturesDirectory: return resolveXDGFolder ("XDG_PICTURES_DIR", "~");
182  case userDesktopDirectory: return resolveXDGFolder ("XDG_DESKTOP_DIR", "~/Desktop");
183  case userApplicationDataDirectory: return File ("~");
184  case commonApplicationDataDirectory: return File ("/var");
185  case globalApplicationsDirectory: return File ("/usr");
186 
187  case tempDirectory:
188  {
189  File tmp ("/var/tmp");
190 
191  if (! tmp.isDirectory())
192  {
193  tmp = "/tmp";
194 
195  if (! tmp.isDirectory())
196  tmp = File::getCurrentWorkingDirectory();
197  }
198 
199  return tmp;
200  }
201 
202  case invokedExecutableFile:
203  if (beast_argv != nullptr && beast_argc > 0)
204  return File (CharPointer_UTF8 (beast_argv[0]));
205  // deliberate fall-through...
206 
207  case currentExecutableFile:
208  case currentApplicationFile:
209  return beast_getExecutableFile();
210 
211  case hostApplicationPath:
212  return beast_readlink ("/proc/self/exe", beast_getExecutableFile());
213 
214  default:
215  bassertfalse; // unknown type?
216  break;
217  }
218 
219  return File::nonexistent ();
220 }

263  bool next (String& filenameFound,
264  bool* const isDir, bool* const isHidden, std::int64_t* const fileSize,
265  Time* const modTime, Time* const creationTime, bool* const isReadOnly)
266  {
267  if (dir != nullptr)
268  {
269  const char* wildcardUTF8 = nullptr;
270 
271  for (;;)
272  {
273  struct dirent* const de = readdir (dir);
274 
275  if (de == nullptr)
276  break;
277 
278  if (wildcardUTF8 == nullptr)
279  wildcardUTF8 = wildCard.toUTF8();
280 
281  if (fnmatch (wildcardUTF8, de->d_name, FNM_CASEFOLD) == 0)
282  {
283  filenameFound = CharPointer_UTF8 (de->d_name);
284 
285  updateStatInfoForFile (parentDir + filenameFound, isDir, fileSize, modTime, creationTime, isReadOnly);
286 
287  if (isHidden != nullptr)
288  *isHidden = filenameFound.startsWithChar ('.');
289 
290  return true;
291  }
292  }
293  }
294 
295  return false;
296  }

330 bool Process::openDocument (const String& fileName, const String& parameters)
331 {
332  String cmdString (fileName.replace (" ", "\\ ",false));
333  cmdString << " " << parameters;
334 
335  if ( cmdString.startsWithIgnoreCase ("file:")
336  || File::createFileWithoutCheckingPath (fileName).isDirectory()
337  || ! isFileExecutable (fileName))
338  {
339  // create a command that tries to launch a bunch of likely browsers
340  const char* const browserNames[] = { "xdg-open", "firefox", "seamonkey",
341  "chrome", "opera", "konqueror" };
342  StringArray cmdLines;
343 
344  for (int i = 0; i < numElementsInArray (browserNames); ++i)
345  cmdLines.add (String (browserNames[i]) + " " + cmdString.trim().quoted());
346 
347  cmdString = cmdLines.joinIntoString (" || ");
348  }
349 
350  const char* const argv[4] = { "/bin/sh", "-c", cmdString.toUTF8(), 0 };
351 
352  const int cpid = fork();
353 
354  if (cpid == 0)
355  {
356  setsid();
357 
358  // Child process
359  execve (argv[0], (char**) argv, environ);
360  exit (0);
361  }
362 
363  return cpid >= 0;
364 }
