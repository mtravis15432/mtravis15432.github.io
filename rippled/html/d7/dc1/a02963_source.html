102 FeatureState*
103 FeaturesImpl::getCreate (uint256 const& featureHash, bool create)
104 {
105  // call with the mutex held
106  auto iter (mFeatureMap.find (featureHash));
107 
108  if (iter == mFeatureMap.end ())
109  {
110  if (!create)
111  return nullptr;
112 
113  FeatureState* feature = & (mFeatureMap[featureHash]);
114 
115  {
116  std::string query = "SELECT FirstMajority,LastMajority FROM Features WHERE hash='";
117  query.append (featureHash.GetHex ());
118  query.append ("';");
119 
120  DeprecatedScopedLock sl (getApp().getWalletDB ()->getDBLock ());
121  Database* db = getApp().getWalletDB ()->getDB ();
122 
123  if (db->executeSQL (query) && db->startIterRows ())
124  {
125  feature->m_firstMajority = db->getBigInt("FirstMajority");
126  feature->m_lastMajority = db->getBigInt("LastMajority");
127  db->endIterRows ();
128  }
129  }
130 
131  return feature;
132  }
133 
134  return & (iter->second);
135 }

268 FeaturesImpl::shouldEnable (std::uint32_t closeTime, const FeatureState& fs)
269 {
270  if (fs.mVetoed || fs.mEnabled || !fs.mSupported || (fs.m_lastMajority != m_lastReport))
271  return false;
272 
273  if (fs.m_firstMajority == m_firstReport)
274  {
275  // had a majority when we first started the server, relaxed check
276  // WRITEME
277  }

279  // didn't have a majority when we first started the server, normal check
280  return (fs.m_lastMajority - fs.m_firstMajority) > m_majorityTime.count();

316 void
317 FeaturesImpl::reportValidations (const FeatureSet& set)
318 {
319  if (set.mTrustedValidations == 0)
320  return;
321 
322  int threshold = (set.mTrustedValidations * mMajorityFraction) / 256;
323 
324  typedef std::map<uint256, int>::value_type u256_int_pair;
325 
326  ScopedLockType sl (mLock);
327 
328  if (m_firstReport == 0)
329  m_firstReport = set.mCloseTime;
330 
331  std::vector<uint256> changedFeatures;
332  changedFeatures.resize (set.mVotes.size ());
333 
334  for (auto const& e : set.mVotes)
335  {
336  FeatureState& state = mFeatureMap[e.first];
337  if (m_journal.debug) m_journal.debug <<
338  "Feature " << e.first.GetHex () <<
339  " has " << e.second <<
340  " votes, needs " << threshold;
341 
342  if (e.second >= threshold)
343  {
344  // we have a majority
345  state.m_lastMajority = set.mCloseTime;
346 
347  if (state.m_firstMajority == 0)
348  {
349  if (m_journal.warning) m_journal.warning <<
350  "Feature " << e.first <<
351  " attains a majority vote";
352 
353  state.m_firstMajority = set.mCloseTime;
354  changedFeatures.push_back (e.first);
355  }
356  }
357  else // we have no majority
358  {
359  if (state.m_firstMajority != 0)
360  {
361  if (m_journal.warning) m_journal.warning <<
362  "Feature " << e.first <<
363  " loses majority vote";
364 
365  state.m_firstMajority = 0;
366  state.m_lastMajority = 0;
367  changedFeatures.push_back (e.first);
368  }
369  }
370  }
371  m_lastReport = set.mCloseTime;
372 
373  if (!changedFeatures.empty ())
374  {
375  DeprecatedScopedLock sl (getApp().getWalletDB ()->getDBLock ());
376  Database* db = getApp().getWalletDB ()->getDB ();
377 
378  db->executeSQL ("BEGIN TRANSACTION;");
379  for (auto const& hash : changedFeatures)
380  {
381  FeatureState& fState = mFeatureMap[hash];
382  db->executeSQL (boost::str (boost::format (
383  "UPDATE Features SET FirstMajority = %d WHERE Hash = '%s';"
384  ) % fState.m_firstMajority % hash.GetHex ()));
385  db->executeSQL (boost::str (boost::format (
386  "UPDATE Features SET LastMajority = %d WHERE Hash = '%s';"
387  ) % fState.m_lastMajority % hash.GetHex()));
388  }
389  db->executeSQL ("END TRANSACTION;");
390  changedFeatures.clear ();
391  }

439 void
440 FeaturesImpl::doVoting (Ledger::ref lastClosedLedger, SHAMap::ref initialPosition)
441 {
442 
443  // LCL must be flag ledger
444  assert((lastClosedLedger->getLedgerSeq () % 256) == 0);
445 
446  FeatureSet featureSet (lastClosedLedger->getParentCloseTimeNC ());
447 
448  // get validations for ledger before flag ledger
449  ValidationSet valSet = getApp().getValidations ().getValidations (lastClosedLedger->getParentHash ());
450  for (auto const& entry : valSet)
451  {
452  auto const& val = *entry.second;
453 
454  if (val.isTrusted ())
455  {
456  featureSet.addVoter ();
457  if (val.isFieldPresent (sfFeatures))
458  {
459  for (auto const& feature : val.getFieldV256 (sfFeatures))
460  {
461  featureSet.addVote (feature);
462  }
463  }
464 
465  }
466  }
467  reportValidations (featureSet);
468 
469  featureList_t lFeatures = getToEnable (lastClosedLedger->getCloseTimeNC ());
470  for (auto const& uFeature : lFeatures)
471  {
472  if (m_journal.warning) m_journal.warning <<
473  "Voting for feature: " << uFeature;
474 
475  // Create the transaction to enable the feature
476  SerializedTransaction trans (ttFEATURE);
477  trans.setFieldAccount (sfAccount, uint160 ());
478  trans.setFieldH256 (sfFeature, uFeature);
479  uint256 txID = trans.getTransactionID ();
480 
481  if (m_journal.warning) m_journal.warning <<
482  "Vote ID: " << txID;
483 
484  // Inject the transaction into our initial proposal
485  Serializer s;
486  trans.add (s, true);
487 #if RIPPLE_PROPOSE_FEATURES
488  SHAMapItem::pointer tItem = boost::make_shared<SHAMapItem> (txID, s.peekData ());
489  if (!initialPosition->addGiveItem (tItem, true, false))
490  {
491  if (m_journal.warning) m_journal.warning <<
492  "Ledger already had feature transaction";
493  }
494 #endif
495  }
496 }
