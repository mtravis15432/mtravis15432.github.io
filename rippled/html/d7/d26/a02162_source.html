452  void handle_read (error_code ec,
453  std::size_t bytes_transferred)
454  {
455  if (ec == boost::asio::error::operation_aborted)
456  return;
457 
458  if (ec != 0)
459  {
460  failed (ec);
461  return;
462  }
463 
464  m_bytesReceived += bytes_transferred;
465  if (m_bytesReceived > m_messageLimitBytes)
466  {
467  failed (error_code (
468  boost::system::errc::invalid_argument,
469  boost::system::system_category ()));
470  return;
471  }
472 
473  std::size_t const bytes_parsed (m_parser.process (
474  m_buffer.getData (), bytes_transferred));
475 
476  if (m_parser.error ())
477  {
478  failed (error_code (
479  boost::system::errc::invalid_argument,
480  boost::system::system_category ()));
481  return;
482  }
483 
484  if (bytes_parsed != bytes_transferred)
485  {
486  failed (error_code (
487  boost::system::errc::invalid_argument,
488  boost::system::system_category ()));
489  return;
490  }
491 
492  if (ec == boost::asio::error::eof)
493  m_parser.process_eof ();
494 
495  if (m_parser.finished ())
496  {
497  if (m_stream->needs_handshake ())
498  {
499  m_stream->async_shutdown (m_strand.wrap (asio::wrap_handler (
500  std::bind (&Session::handle_shutdown,
501  Ptr(this), asio::placeholders::error), m_handler)));
502  }
503  else
504  {
505  handle_shutdown (error_code ());
506  }
507  return;
508  }
509 
510  async_read_some ();
511  }
