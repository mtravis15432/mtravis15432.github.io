20 /*
21  This code prevents scenarios like the following:
22 1) A client submits a transaction.
23 2) The transaction gets into the ledger this server
24  believes will be the consensus ledger.
25 3) The server builds a succeeding open ledger without the
26  transaction (because it's in the prior ledger).
27 4) The local consensus ledger is not the majority ledger
28  (due to network conditions, Byzantine fault, etcetera)
29  the majority ledger does not include the transaction.
30 5) The server builds a new open ledger that does not include
31  the transaction or have it in a prior ledger.
32 6) The client submits another transaction and gets a terPRE_SEQ
33  preliminary result.
34 7) The server does not relay that second transaction, at least
35  not yet.
36 
37 With this code, when step 5 happens, the first transaction will
38 be applied to that open ledger so the second transaction will
39 succeed normally at step 6. Transactions remain tracked and
40 test-applied to all new open ledgers until seen in a fully-
41 validated ledger
42 */

47 // This class wraps a pointer to a transaction along with
48 // its expiration ledger. It also caches the issuing account.
49 class LocalTx
50 {
51 public:
52 
53  // The number of ledgers to hold a transaction is essentially
54  // arbitrary. It should be sufficient to allow the transaction to
55  // get into a fully-validated ledger.
56  static int const holdLedgers = 5;
57 
58  LocalTx (LedgerIndex index, SerializedTransaction::ref txn)
59  : m_txn (txn)
60  , m_expire (index + holdLedgers)
61  , m_id (txn->getTransactionID ())
62  , m_account (txn->getSourceAccount ())
63  , m_seq (txn->getSequence())
64  {
65  if (txn->isFieldPresent (sfLastLedgerSequence))
66  {
67  LedgerIndex m_txnexpire = txn->getFieldU32 (sfLastLedgerSequence) + 1;
68  m_expire = std::min (m_expire, m_txnexpire);
69  }
70  }
71 
72  uint256 const& getID ()
73  {
74  return m_id;
75  }
76 
77  std::uint32_t getSeq ()
78  {
79  return m_seq;
80  }
81 
82  bool isExpired (LedgerIndex i)
83  {
84  return i > m_expire;
85  }
86 
87  SerializedTransaction::ref getTX ()
88  {
89  return m_txn;
90  }
91 
92  RippleAddress const& getAccount ()
93  {
94  return m_account;
95  }

99  SerializedTransaction::pointer m_txn;
100  LedgerIndex m_expire;

113  // Add a new transaction to the set of local transactions
114  void push_back (LedgerIndex index, SerializedTransaction::ref txn) override
115  {
116  std::lock_guard <std::mutex> lock (m_lock);
117 
118  m_txns.emplace_back (index, txn);
119  }
120 
121  bool can_remove (LocalTx& txn, Ledger::ref ledger)
122  {
123 
124  if (txn.isExpired (ledger->getLedgerSeq ()))
125  return true;
126 
127  if (ledger->hasTransaction (txn.getID ()))
128  return true;
129 
130  SLE::pointer sle = ledger->getAccountRoot (txn.getAccount ());
131  if (!sle)
132  return false;
133 
134  if (sle->getFieldU32 (sfSequence) > txn.getSeq ())
135  return true;
136 
137 
138  return false;
139  }

141  void apply (TransactionEngine& engine) override
142  {
143 
144  CanonicalTXSet tset (uint256 {});
145 
146  // Get the set of local transactions as a canonical
147  // set (so they apply in a valid order)
148  {
149  std::lock_guard <std::mutex> lock (m_lock);
150 
151  for (auto& it : m_txns)
152  tset.push_back (it.getTX());
153  }
154 
155  for (auto it : tset)
156  {
157  try
158  {
159  TransactionEngineParams parms = tapOPEN_LEDGER;
160  bool didApply;
161  engine.applyTransaction (*it.second, parms, didApply);
162  }
163  catch (...)
164  {
165  // Nothing special we need to do.
166  // It's possible a cleverly malformed transaction or
167  // corrupt back end database could cause an exception
168  // during transaction processing.
169  }
170  }
171  }
172 
173  // Remove transactions that have either been accepted into a fully-validated
174  // ledger, are (now) impossible, or have expired
175  void sweep (Ledger::ref validLedger) override
176  {
177  std::lock_guard <std::mutex> lock (m_lock);
178 
179  for (auto it = m_txns.begin (); it != m_txns.end (); )
180  {
181  if (can_remove (*it, validLedger))
182  it = m_txns.erase (it);
183  else
184  ++it;
185  }
186  }
187 
188  std::size_t size () override
189  {
190  std::lock_guard <std::mutex> lock (m_lock);
191 
192  return m_txns.size ();
193  }
