70 // default conversion (copy in to out)
71 template<typename T>
72 struct type_conversion
73 {
74  typedef T base_type;
75 
76  static void from_base(base_type const& in, indicator ind, T& out)
77  {
78  // null not allowed
79  assert (ind != i_null);
80  out = in;
81  }
82 
83  static void to_base(T const& in, base_type& out, indicator& ind)
84  {
85  out = in;
86  ind = i_ok;
87  }
88 };
89 
90 // Time
91 template <>
92 struct type_conversion <Time>
93 {
94  //typedef sqlite3_int64 base_type;
95  typedef std::int64_t base_type;
96 
97  static void from_base(const base_type v, indicator ind, Time& result)
98  {
99  if (ind == i_null)
100  {
101  // bassertfalse
102  result = Time(0);
103  }
104  else
105  {
106  result = Time(v);
107  }
108  }
109 
110  static void to_base(const Time& v, base_type& result, indicator& ind)
111  {
112  result = v.toMilliseconds();
113  ind = i_ok;
114  }
115 };
