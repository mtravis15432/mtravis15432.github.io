92 String File::parseAbsolutePath (const String& p)
93 {
94  if (p.isEmpty())
95  return String::empty;
96 
97 #if BEAST_WINDOWS
98  // Windows..
99  String path (p.replaceCharacter ('/', '\\'));
100 
101  if (path.startsWithChar (separator))
102  {
103  if (path[1] != separator)
104  {
105  /* When you supply a raw string to the File object constructor, it must be an absolute path.
106  If you're trying to parse a string that may be either a relative path or an absolute path,
107  you MUST provide a context against which the partial path can be evaluated - you can do
108  this by simply using File::getChildFile() instead of the File constructor. E.g. saying
109  "File::getCurrentWorkingDirectory().getChildFile (myUnknownPath)" would return an absolute
110  path if that's what was supplied, or would evaluate a partial path relative to the CWD.
111  */
112  bassertfalse;
113 
114  path = File::getCurrentWorkingDirectory().getFullPathName().substring (0, 2) + path;
115  }
116  }
117  else if (! path.containsChar (':'))
118  {
119  /* When you supply a raw string to the File object constructor, it must be an absolute path.
120  If you're trying to parse a string that may be either a relative path or an absolute path,
121  you MUST provide a context against which the partial path can be evaluated - you can do
122  this by simply using File::getChildFile() instead of the File constructor. E.g. saying
123  "File::getCurrentWorkingDirectory().getChildFile (myUnknownPath)" would return an absolute
124  path if that's what was supplied, or would evaluate a partial path relative to the CWD.
125  */
126  bassertfalse;
127 
128  return File::getCurrentWorkingDirectory().getChildFile (path).getFullPathName();
129  }
130 #else
131  // Mac or Linux..
132 
133  // Yes, I know it's legal for a unix pathname to contain a backslash, but this assertion is here
134  // to catch anyone who's trying to run code that was written on Windows with hard-coded path names.
135  // If that's why you've ended up here, use File::getChildFile() to build your paths instead.
136  bassert ((! p.containsChar ('\\')) || (p.indexOfChar ('/') >= 0 && p.indexOfChar ('/') < p.indexOfChar ('\\')));
137 
138  String path (p);
139 
140  if (path.startsWithChar ('~'))
141  {
142  if (path[1] == separator || path[1] == 0)
143  {
144  // expand a name of the form "~/abc"
145  path = File::getSpecialLocation (File::userHomeDirectory).getFullPathName()
146  + path.substring (1);
147  }
148  else
149  {
150  // expand a name of type "~dave/abc"
151  const String userName (path.substring (1).upToFirstOccurrenceOf ("/", false, false));
152 
153  if (struct passwd* const pw = getpwnam (userName.toUTF8()))
154  path = addTrailingSeparator (pw->pw_dir) + path.fromFirstOccurrenceOf ("/", false, false);
155  }
156  }
157  else if (! path.startsWithChar (separator))
158  {
159  #if BEAST_DEBUG || BEAST_LOG_ASSERTIONS
160  if (! (path.startsWith ("./") || path.startsWith ("../")))
161  {
162  /* When you supply a raw string to the File object constructor, it must be an absolute path.
163  If you're trying to parse a string that may be either a relative path or an absolute path,
164  you MUST provide a context against which the partial path can be evaluated - you can do
165  this by simply using File::getChildFile() instead of the File constructor. E.g. saying
166  "File::getCurrentWorkingDirectory().getChildFile (myUnknownPath)" would return an absolute
167  path if that's what was supplied, or would evaluate a partial path relative to the CWD.
168  */
169  bassertfalse;
170 
171  #if BEAST_LOG_ASSERTIONS
172  Logger::writeToLog ("Illegal absolute path: " + path);
173  #endif
174  }
175  #endif
176 
177  return File::getCurrentWorkingDirectory().getChildFile (path).getFullPathName();
178  }
179 #endif
180 
181  while (path.endsWithChar (separator) && path != separatorString) // careful not to turn a single "/" into an empty string.
182  path = path.dropLastCharacters (1);
183 
184  return path;
185 }

370 File File::getChildFile (String relativePath) const
371 {
372  if (isAbsolutePath (relativePath))
373  return File (relativePath);
374 
375  String path (fullPath);
376 
377  // It's relative, so remove any ../ or ./ bits at the start..
378  if (relativePath[0] == '.')
379  {
380  #if BEAST_WINDOWS
381  relativePath = relativePath.replaceCharacter ('/', '\\');
382  #endif
383 
384  while (relativePath[0] == '.')
385  {
386  const beast_wchar secondChar = relativePath[1];
387 
388  if (secondChar == '.')
389  {
390  const beast_wchar thirdChar = relativePath[2];
391 
392  if (thirdChar == 0 || thirdChar == separator)
393  {
394  const int lastSlash = path.lastIndexOfChar (separator);
395  if (lastSlash >= 0)
396  path = path.substring (0, lastSlash);
397 
398  relativePath = relativePath.substring (3);
399  }
400  else
401  {
402  break;
403  }
404  }
405  else if (secondChar == separator)
406  {
407  relativePath = relativePath.substring (2);
408  }
409  else
410  {
411  break;
412  }
413  }
414  }
415 
416  return File (addTrailingSeparator (path) + relativePath);
417 }

552 File File::getNonexistentChildFile (const String& suggestedPrefix,
553  const String& suffix,
554  bool putNumbersInBrackets) const
555 {
556  File f (getChildFile (suggestedPrefix + suffix));
557 
558  if (f.exists())
559  {
560  int number = 1;
561  String prefix (suggestedPrefix);
562 
563  // remove any bracketed numbers that may already be on the end..
564  if (prefix.trim().endsWithChar (')'))
565  {
566  putNumbersInBrackets = true;
567 
568  const int openBracks = prefix.lastIndexOfChar ('(');
569  const int closeBracks = prefix.lastIndexOfChar (')');
570 
571  if (openBracks > 0
572  && closeBracks > openBracks
573  && prefix.substring (openBracks + 1, closeBracks).containsOnly ("0123456789"))
574  {
575  number = prefix.substring (openBracks + 1, closeBracks).getIntValue();
576  prefix = prefix.substring (0, openBracks);
577  }
578  }

580  // also use brackets if it ends in a digit.

586  String newName (prefix);
587 
588  if (putNumbersInBrackets)

593  f = getChildFile (newName + suffix);

622 bool File::hasFileExtension (const String& possibleSuffix) const
623 {
624  if (possibleSuffix.isEmpty())
625  return fullPath.lastIndexOfChar ('.') <= fullPath.lastIndexOfChar (separator);
626 
627  const int semicolon = possibleSuffix.indexOfChar (0, ';');
628 
629  if (semicolon >= 0)
630  {
631  return hasFileExtension (possibleSuffix.substring (0, semicolon).trimEnd())
632  || hasFileExtension (possibleSuffix.substring (semicolon + 1).trimStart());
633  }
634  else
635  {
636  if (fullPath.endsWithIgnoreCase (possibleSuffix))
637  {
638  if (possibleSuffix.startsWithChar ('.'))
639  return true;
640 
641  const int dotPos = fullPath.length() - possibleSuffix.length() - 1;
642 
643  if (dotPos >= 0)
644  return fullPath [dotPos] == '.';
645  }
646  }
647 
648  return false;

833 String File::getRelativePathFrom (const File& dir) const
834 {
835  String thisPath (fullPath);
836 
837  while (thisPath.endsWithChar (separator))
838  thisPath = thisPath.dropLastCharacters (1);
839 
840  String dirPath (addTrailingSeparator (dir.existsAsFile() ? dir.getParentDirectory().getFullPathName()
841  : dir.fullPath));
842 
843  int commonBitLength = 0;
844  String::CharPointerType thisPathAfterCommon (thisPath.getCharPointer());
845  String::CharPointerType dirPathAfterCommon (dirPath.getCharPointer());
846 
847  {
848  String::CharPointerType thisPathIter (thisPath.getCharPointer());
849  String::CharPointerType dirPathIter (dirPath.getCharPointer());
850 
851  for (int i = 0;;)
852  {
853  const beast_wchar c1 = thisPathIter.getAndAdvance();
854  const beast_wchar c2 = dirPathIter.getAndAdvance();
855 
856  #if NAMES_ARE_CASE_SENSITIVE
857  if (c1 != c2
858  #else
859  if ((c1 != c2 && CharacterFunctions::toLowerCase (c1) != CharacterFunctions::toLowerCase (c2))
860  #endif
861  || c1 == 0)
862  break;
863 
864  ++i;
865 
866  if (c1 == separator)
867  {
868  thisPathAfterCommon = thisPathIter;
869  dirPathAfterCommon = dirPathIter;
870  commonBitLength = i;
871  }
872  }
873  }
874 
875  // if the only common bit is the root, then just return the full path..
876  if (commonBitLength == 0 || (commonBitLength == 1 && thisPath[1] == separator))
877  return fullPath;
878 
879  const int numUpDirectoriesNeeded = countNumberOfSeparators (dirPathAfterCommon);
880 
881  if (numUpDirectoriesNeeded == 0)
882  return thisPathAfterCommon;
883 
884  #if BEAST_WINDOWS
885  String s (String::repeatedString ("..\\", numUpDirectoriesNeeded));
886  #else
887  String s (String::repeatedString ("../", numUpDirectoriesNeeded));
888  #endif
889  s.appendCharPointer (thisPathAfterCommon);
890  return s;
891 }
