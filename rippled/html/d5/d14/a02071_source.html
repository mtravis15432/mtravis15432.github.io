445  static bool isValidString (const CharType* dataToTest, int maxBytesToRead)
446  {
447  maxBytesToRead /= sizeof (CharType);
448 
449  while (--maxBytesToRead >= 0 && *dataToTest != 0)
450  {
451  const std::uint32_t n = (std::uint32_t) (std::uint16_t) *dataToTest++;
452 
453  if (n >= 0xd800)
454  {
455  if (n > 0x10ffff)
456  return false;
457 
458  if (n <= 0xdfff)
459  {
460  if (n > 0xdc00)
461  return false;
462 
463  const std::uint32_t nextChar = (std::uint32_t) (std::uint16_t) *dataToTest++;
464 
465  if (nextChar < 0xdc00 || nextChar > 0xdfff)
466  return false;
467  }
468  }
469  }
