109  static bool launchExecutable (const String& pathAndArguments)
110  {
111  const char* const argv[4] = { "/bin/sh", "-c", pathAndArguments.toUTF8(), 0 };
112 
113  const int cpid = fork();
114 
115  if (cpid == 0)
116  {
117  // Child process
118  if (execve (argv[0], (char**) argv, 0) < 0)
119  exit (0);
120  }
121  else
122  {
123  if (cpid < 0)
124  return false;
125  }
126 
127  return true;
128  }
129 }

176 File File::getSpecialLocation (const SpecialLocationType type)
177 {
178  BEAST_AUTORELEASEPOOL
179  {
180  String resultPath;
181 
182  switch (type)
183  {
184  case userHomeDirectory: resultPath = nsStringToBeast (NSHomeDirectory()); break;
185 
186  #if BEAST_IOS
187  case userDocumentsDirectory: resultPath = FileHelpers::getIOSSystemLocation (NSDocumentDirectory); break;
188  case userDesktopDirectory: resultPath = FileHelpers::getIOSSystemLocation (NSDesktopDirectory); break;
189 
190  case tempDirectory:
191  {
192  File tmp (FileHelpers::getIOSSystemLocation (NSCachesDirectory));
193  tmp = tmp.getChildFile (beast_getExecutableFile().getFileNameWithoutExtension());
194  tmp.createDirectory();
195  return tmp.getFullPathName();
196  }
197 
198  #else
199  case userDocumentsDirectory: resultPath = "~/Documents"; break;
200  case userDesktopDirectory: resultPath = "~/Desktop"; break;
201 
202  case tempDirectory:
203  {
204  File tmp ("~/Library/Caches/" + beast_getExecutableFile().getFileNameWithoutExtension());
205  tmp.createDirectory();
206  return tmp.getFullPathName();
207  }
208  #endif
209  case userMusicDirectory: resultPath = "~/Music"; break;
210  case userMoviesDirectory: resultPath = "~/Movies"; break;
211  case userPicturesDirectory: resultPath = "~/Pictures"; break;
212  case userApplicationDataDirectory: resultPath = "~/Library"; break;
213  case commonApplicationDataDirectory: resultPath = "/Library"; break;
214  case commonDocumentsDirectory: resultPath = "/Users/Shared"; break;
215  case globalApplicationsDirectory: resultPath = "/Applications"; break;
216 
217  case invokedExecutableFile:
218  if (beast_argv != nullptr && beast_argc > 0)
219  return File (CharPointer_UTF8 (beast_argv[0]));
220  // deliberate fall-through...
221 
222  case currentExecutableFile:
223  return beast_getExecutableFile();
224 
225  case currentApplicationFile:
226  {
227  const File exe (beast_getExecutableFile());
228  const File parent (exe.getParentDirectory());
229 
230  #if BEAST_IOS
231  return parent;
232  #else
233  return parent.getFullPathName().endsWithIgnoreCase ("Contents/MacOS")
234  ? parent.getParentDirectory().getParentDirectory()
235  : exe;
236  #endif
237  }
238 
239  case hostApplicationPath:
240  {
241  unsigned int size = 8192;
242  HeapBlock<char> buffer;
243  buffer.calloc (size + 8);
244 
245  _NSGetExecutablePath (buffer.getData(), &size);
246  return String::fromUTF8 (buffer, (int) size);
247  }
248 
249  default:
250  bassertfalse; // unknown type?
251  break;
252  }
253 
254  if (resultPath.isNotEmpty())
255  return File (resultPath.convertToPrecomposedUnicode());
256  }
257 
258  return File::nonexistent ();
259 }

334  bool next (String& filenameFound,
335  bool* const isDir, bool* const isHidden, std::int64_t* const fileSize,
336  Time* const modTime, Time* const creationTime, bool* const isReadOnly)
337  {
338  BEAST_AUTORELEASEPOOL
339  {
340  const char* wildcardUTF8 = nullptr;
341 
342  for (;;)
343  {
344  NSString* file;
345  if (enumerator == nil || (file = [enumerator nextObject]) == nil)
346  return false;
347 
348  [enumerator skipDescendents];
349  filenameFound = nsStringToBeast (file);
350 
351  if (wildcardUTF8 == nullptr)
352  wildcardUTF8 = wildCard.toUTF8();
353 
354  if (fnmatch (wildcardUTF8, filenameFound.toUTF8(), FNM_CASEFOLD) != 0)
355  continue;
356 
357  const String fullPath (parentDir + filenameFound);
358  updateStatInfoForFile (fullPath, isDir, fileSize, modTime, creationTime, isReadOnly);
359 
360  if (isHidden != nullptr)
361  *isHidden = FileHelpers::isHiddenFile (fullPath);
362 
363  return true;
364  }
365  }
366  }
367 
368 private:
369  String parentDir, wildCard;
370  NSDirectoryEnumerator* enumerator;
371 };

391 bool Process::openDocument (const String& fileName, const String& parameters)
392 {
393  #if BEAST_IOS
394  return [[UIApplication sharedApplication] openURL: [NSURL URLWithString: beastStringToNS (fileName)]];
395  #else
396  BEAST_AUTORELEASEPOOL
397  {
398  if (parameters.isEmpty())
399  {
400  return [[NSWorkspace sharedWorkspace] openFile: beastStringToNS (fileName)]
401  || [[NSWorkspace sharedWorkspace] openURL: [NSURL URLWithString: beastStringToNS (fileName)]];
402  }
403 
404  bool ok = false;
405  const File file (fileName);
406 
407  if (file.isBundle())
408  {
409  NSMutableArray* urls = [NSMutableArray array];
410 
411  StringArray docs;
412  docs.addTokens (parameters, true);
413  for (int i = 0; i < docs.size(); ++i)
414  [urls addObject: beastStringToNS (docs[i])];
415 
416  ok = [[NSWorkspace sharedWorkspace] openURLs: urls
417  withAppBundleIdentifier: [[NSBundle bundleWithPath: beastStringToNS (fileName)] bundleIdentifier]
418  options: 0
419  additionalEventParamDescriptor: nil
420  launchIdentifiers: nil];
421  }
422  else if (file.exists())
423  {
424  ok = FileHelpers::launchExecutable ("\"" + fileName + "\" " + parameters);
425  }
426 
427  return ok;
428  }
429  #endif
430 }
