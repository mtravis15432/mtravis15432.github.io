62 {
63  HeapBlock<char> heapBuffer;
64 
65  char localBuffer [1024];
66  char* cwd = getcwd (localBuffer, sizeof (localBuffer) - 1);
67  size_t bufferSize = 4096;
68 
69  while (cwd == nullptr && errno == ERANGE)
70  {
71  heapBuffer.malloc (bufferSize);
72  cwd = getcwd (heapBuffer, bufferSize - 1);
73  bufferSize += 1024;
74  }
75 
76  return File (CharPointer_UTF8 (cwd));

416 Result RandomAccessFile::nativeOpen (File const& path, Mode mode)
417 {
418  bassert (! isOpen ());
419 
420  Result result (Result::ok ());
421 
422  if (path.exists())
423  {
424  int oflag;
425  switch (mode)
426  {
427  case readOnly:
428  oflag = O_RDONLY;
429  break;
430 
431  default:
432  case readWrite:
433  oflag = O_RDWR;
434  break;
435  };
436 
437  const int f = ::open (path.getFullPathName().toUTF8(), oflag, 00644);
438 
439  if (f != -1)
440  {
441  currentPosition = lseek (f, 0, SEEK_SET);
442 
443  if (currentPosition >= 0)
444  {
445  file = path;
446  fileHandle = fdToVoidPointer (f);
447  }
448  else
449  {
450  result = getResultForErrno();
451  ::close (f);
452  }
453  }
454  else
455  {
456  result = getResultForErrno();
457  }
458  }
459  else if (mode == readWrite)
460  {
461  const int f = ::open (path.getFullPathName().toUTF8(), O_RDWR + O_CREAT, 00644);
462 
463  if (f != -1)
464  {
465  file = path;
466  fileHandle = fdToVoidPointer (f);
467  }
468  else
469  {
470  result = getResultForErrno();
471  }
472  }
473  else
474  {
475  // file doesn't exist and we're opening read-only
476  Result::fail (String (strerror (ENOENT)));
477  }
478 
479  return result;
