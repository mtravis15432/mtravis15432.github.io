70 template <typename Type>
71 const Type findMinimum (const Type* data, int numValues)
72 {
73  if (numValues <= 0)
74  return Type();
75 
76  Type result (*data++);
77 
78  while (--numValues > 0) // (> 0 rather than >= 0 because we've already taken the first sample)
79  {
80  const Type& v = *data++;
81  if (v < result) result = v;
82  }
83

106 template <typename Type>
107 void findMinAndMax (const Type* values, int numValues, Type& lowest, Type& highest)
108 {
109  if (numValues <= 0)
110  {
111  lowest = Type();
112  highest = Type();
113  }
114  else
115  {
116  Type mn (*values++);
117  Type mx (mn);
118 
119  while (--numValues > 0) // (> 0 rather than >= 0 because we've already taken the first sample)
120  {
121  const Type& v = *values++;
122 
123  if (mx < v) mx = v;
124  if (v < mn) mn = v;
125  }
126 
127  lowest = mn;
128  highest = mx;
129  }
130 }
