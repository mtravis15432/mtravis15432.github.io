190  // Add each entry in the list to the map, incrementing the
191  // reference count if it already exists, and updating fields.
192  //
193  std::size_t merge (std::vector <Source::Item> const& list, Source* source)
194  {
195  std::size_t numAdded (0);
196  for (std::size_t i = 0; i < list.size (); ++i)
197  {
198  Source::Item const& item (list [i]);
199  std::pair <ValidatorTable::iterator, bool> results (
200  m_validators.emplace (item.publicKey, Validator ()));
201  Validator& validatorInfo (results.first->second);
202  validatorInfo.addRef();
203  if (results.second)
204  {
205  // This is a new one
206  ++numAdded;
207  dirtyChosen ();
208  }
209  }
210 
211  return numAdded;
212  }
213 
214  // Decrement the reference count of each item in the list
215  // in the map.
216  //
217  std::size_t remove (std::vector <Source::Item> const& list,
218  Source* source)
219  {
220  std::size_t numRemoved (0);
221  for (std::size_t i = 0; i < list.size (); ++i)
222  {
223  Source::Item const& item (list [i]);
224  ValidatorTable::iterator iter (m_validators.find (item.publicKey));
225  bassert (iter != m_validators.end ());
226  Validator& validatorInfo (iter->second);
227  if (validatorInfo.release())
228  {
229  // Last reference removed
230  ++numRemoved;
231  m_validators.erase (iter);
232  dirtyChosen ();
233  }
234  }
235 
236  return numRemoved;
237  }

296  void fetch (SourceDesc& desc)
297  {
298  beast::SharedPtr <Source> const& source (desc.source);
299  Source::Results results;
300 
301  {
302  {
303  SharedState::Access state (m_state);
304  if (state->stopping)
305  return;
306  state->fetchSource = source;
307  }
308 
309  source->fetch (results, m_journal);
310 
311  {
312  SharedState::Access state (m_state);
313  if (state->stopping)
314  return;
315  state->fetchSource = nullptr;
316  }
317  }
318 
319  // Reset fetch timer for the source->
320  desc.whenToFetch = beast::Time::getCurrentTime () +
321  beast::RelativeTime (secondsBetweenFetches);
322 
323  if (results.success)
324  {
325  // Count the number fetched
326  std::size_t const numFetched (
327  results.list.size());
328 
329  // Add the new source item to the map
330  std::size_t const numAdded (
331  merge (results.list, source));
332 
333  // Swap lists
334  std::swap (desc.results, results);
335 
336  // Remove the old source item from the map
337  std::size_t const numRemoved (remove (results.list, source));
338 
339  // Report
340  if (numAdded > numRemoved)
341  {
342  m_journal.info <<
343  "Fetched " << numFetched <<
344  "(" << (numAdded - numRemoved) << " new) " <<
345  " trusted validators from " << *source;
346  }
347  else if (numRemoved > numAdded)
348  {
349  m_journal.info <<
350  "Fetched " << numFetched <<
351  "(" << numRemoved - numAdded << " removed) " <<
352  " trusted validators from " << *source;
353  }
354  else
355  {
356  m_journal.debug <<
357  "Fetched " << numFetched <<
358  " trusted validators from " << *source;
359  }
360 
361  // See if we need to rebuild
362  checkChosen ();
363 
364  // Reset failure status
365  desc.numberOfFailures = 0;
366  desc.status = SourceDesc::statusFetched;
367 
368  // Update the source's list in the store
369  m_store.update (desc, true);
370  }
371  else
372  {
373  m_journal.error <<
374  "Failed to fetch " << *source;
375 
376  ++desc.numberOfFailures;
377  desc.status = SourceDesc::statusFailed;
378  // Record the failure in the Store
379  m_store.update (desc);
380  }
381  }

399 
400  for (SourceTable::iterator iter = m_sources.begin ();

405  // See if we should fetch
406  //

434  {

436  ValidatorTable::iterator iter (m_validators.find (rv.publicKey));
