71  static void RunCommand(int argc, char** argv, Options options) {
72  if (argc <= 2) {
73  PrintHelp(argv[0]);
74  exit(1);
75  }
76 
77  LDBCommand* cmdObj = LDBCommand::InitFromCmdLineArgs(argc, argv, options);
78  if (cmdObj == nullptr) {
79  fprintf(stderr, "Unknown command\n");
80  PrintHelp(argv[0]);

82  }
83 
84  if (!cmdObj->ValidateCmdLineOptions()) {

88  cmdObj->Run();

90  fprintf(stderr, "%s\n", ret.ToString().c_str());
91  delete cmdObj;

93  exit(ret.IsFailed());
