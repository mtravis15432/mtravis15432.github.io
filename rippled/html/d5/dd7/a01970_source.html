45 template <class Block, class Derived>
46 class block_stream
47 {
48 private:
49  Block m_block;
50  std::size_t m_size;
51 
52  std::size_t
53  needed() const noexcept
54  {
55  return sizeof(Block) - m_size;
56  }
57 
58  void*
59  tail() noexcept
60  {
61  return ((char *)&m_block) + m_size;
62  }
63 
64 protected:
65  void
66  finish()
67  {
68  if (m_size > 0)
69  {
70  // zero-pad
71  memset (tail(), 0, needed());
72  static_cast <Derived*> (this)->process_block (m_block);
73  }
74  }
75 
76 public:
77  block_stream ()
78  : m_size(0)
79  {
80  }
81 
82  void
83  operator() (void const* data, std::size_t bytes) noexcept
84  {
85  // handle leftovers
86  if (m_size > 0)
87  {
88  std::size_t const n (std::min (needed(), bytes));
89  std::memcpy (tail(), data, n);
90  data = ((char const*)data) + n;
91  bytes -= n;
92  m_size += n;
93 
94  if (m_size < sizeof(Block))
95  return;
96 
97  static_cast <Derived*> (this)->process_block (m_block);
98  }
99 
100  // loop over complete blocks
101  while (bytes >= sizeof(Block))
102  {
103  m_block = *((Block const*)data);
104  static_cast <Derived*> (this)->process_block (m_block);
105  data = ((char const*)data) + sizeof(m_block);
106  bytes -= sizeof(m_block);
107  }
108 
109  // save leftovers
110  if (bytes > 0)
111  {
112  memcpy (tail(), data, bytes);
113  m_size += bytes;
114  }
115  }
116 };
