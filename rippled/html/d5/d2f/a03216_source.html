99  void handle_read_line (const boost::system::error_code& e)
100  {
101  if (! e)
102  {
103  HTTPRequest::Action action = mHTTPRequest.consume (mLineBuffer);
104 
105  if (action == HTTPRequest::haDO_REQUEST)
106  {
107  // request with no body
108  WriteLog (lsWARNING, RPCServer) << "RPC HTTP request with no body";
109 
110  mSocket.async_shutdown (mStrand.wrap (boost::bind (
111  &RPCServerImp::handle_shutdown,
112  boost::static_pointer_cast <RPCServerImp> (shared_from_this ()),
113  boost::asio::placeholders::error)));
114  }
115  else if (action == HTTPRequest::haREAD_LINE)
116  {
117  boost::asio::async_read_until (
118  mSocket,
119  mLineBuffer,
120  "\r\n",
121  mStrand.wrap (boost::bind (
122  &RPCServerImp::handle_read_line,
123  boost::static_pointer_cast <RPCServerImp> (shared_from_this ()),
124  boost::asio::placeholders::error)));
125  }
126  else if (action == HTTPRequest::haREAD_RAW)
127  {
128  int rLen = mHTTPRequest.getDataSize ();
129 
130  if ((rLen < 0) || (rLen > maxQueryBytes))
131  {
132  WriteLog (lsWARNING, RPCServer) << "Illegal RPC request length " << rLen;
133 
134  mSocket.async_shutdown (mStrand.wrap (boost::bind (
135  &RPCServerImp::handle_shutdown,
136  boost::static_pointer_cast <RPCServerImp> (shared_from_this ()),
137  boost::asio::placeholders::error)));
138  }
139  else
140  {
141  int alreadyHave = mLineBuffer.size ();
142 
143  if (alreadyHave < rLen)
144  {
145  mQueryVec.resize (rLen - alreadyHave);
146 
147  boost::asio::async_read (
148  mSocket,
149  boost::asio::buffer (mQueryVec),
150  mStrand.wrap (boost::bind (
151  &RPCServerImp::handle_read_req,
152  boost::static_pointer_cast <RPCServerImp> (shared_from_this ()),
153  boost::asio::placeholders::error)));
154 
155  WriteLog (lsTRACE, RPCServer) << "Waiting for completed request: " << rLen;
156  }
157  else
158  {
159  // we have the whole thing
160  mQueryVec.resize (0);
161 
162  handle_read_req (e);
163  }
164  }
165  }
166  else
167  {
168  mSocket.async_shutdown (mStrand.wrap (boost::bind (
169  &RPCServerImp::handle_shutdown,
170  boost::static_pointer_cast <RPCServerImp> (shared_from_this ()),
171  boost::asio::placeholders::error)));
172  }
173  }
174  }

178  void handle_read_req (const boost::system::error_code& ec)
179  {
180  std::string req;
181 
182  if (mLineBuffer.size ())
183  {
184  req.assign (boost::asio::buffer_cast <const char*> (mLineBuffer.data ()), mLineBuffer.size ());
185 
186  mLineBuffer.consume (mLineBuffer.size ());
187  }
188 
189  req += strCopy (mQueryVec);
190 
191  if (! m_handler.isAuthorized (mHTTPRequest.peekHeaders ()))
192  {
193  mReplyStr = m_handler.createResponse (403, "Forbidden");
194  }
195  else
196  {
197  mReplyStr = handleRequest (req);
198  }
199 
200  boost::asio::async_write (
201  mSocket,
202  boost::asio::buffer (mReplyStr),
203  mStrand.wrap (boost::bind (
204  &RPCServerImp::handle_write,
205  boost::static_pointer_cast <RPCServerImp> (shared_from_this ()),
206  boost::asio::placeholders::error)));
207  }
