101 // Get entropy from the Windows crypto provider
102 bool RandomNumbers::platformAddEntropy (beast::Journal::Stream stream)
103 {
104  char name[512], rand[128];
105  DWORD count = 500;
106  HCRYPTPROV cryptoHandle;
107 
108  if (!CryptGetDefaultProviderA (PROV_RSA_FULL, nullptr, CRYPT_MACHINE_DEFAULT, name, &count))
109  {
110  stream << "Unable to get default crypto provider";
111  return false;
112  }
113 
114  if (!CryptAcquireContextA (&cryptoHandle, nullptr, name, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT | CRYPT_SILENT))
115  {
116  stream << "Unable to acquire crypto provider";
117  return false;
118  }
119 
120  if (!CryptGenRandom (cryptoHandle, 128, reinterpret_cast<BYTE*> (rand)))
121  {
122  stream << "Unable to get entropy from crypto provider";
123  CryptReleaseContext (cryptoHandle, 0);
124  return false;
125  }
126 
127  CryptReleaseContext (cryptoHandle, 0);
128  RAND_seed (rand, 128);
129 
130  return true;
131 }

135 bool RandomNumbers::platformAddEntropy (beast::Journal::Stream stream)
136 {
137  char rand[128];
138  std::ifstream reader;
139 
140  reader.open ("/dev/urandom", std::ios::in | std::ios::binary);
141 
142  if (!reader.is_open ())
143  {
144 #ifdef BEAST_DEBUG
145  stream << "Unable to open random source";
146 #endif
147  return false;
148  }
149 
150  reader.read (rand, 128);
151 
152  int bytesRead = reader.gcount ();
153 
154  if (bytesRead == 0)
155  {
156 #ifdef BEAST_DEBUG
157  stream << "Unable to read from random source";
158 #endif
159  return false;
160  }
161 
162  RAND_seed (rand, bytesRead);
163  return bytesRead >= 64;
164 }
