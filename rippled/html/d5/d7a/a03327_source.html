134  void async_handshake (handshake_type type, callback cbFunc)
135  {
136  if ((type == ssl_socket::client) || (mSecure))
137  {
138  // must be ssl
139  mSecure = true;
140  mSocket->async_handshake (type, cbFunc);
141  }
142  else if (mBuffer.empty ())
143  {
144  // must be plain
145  mSecure = false;
146  mSocket->get_io_service ().post (
147  beast::asio::bind_handler (cbFunc, error_code()));
148  }
149  else
150  {
151  // autodetect
152  mSocket->next_layer ().async_receive (boost::asio::buffer (mBuffer), boost::asio::socket_base::message_peek,
153  boost::bind (&AutoSocket::handle_autodetect, this, cbFunc,
154  boost::asio::placeholders::error, boost::asio::placeholders::bytes_transferred));
155  }
156  }

269 protected:
270  void handle_autodetect (callback cbFunc, const error_code& ec, size_t bytesTransferred)
271  {
272  using namespace ripple;
273 
274  if (ec)
275  {
276  Log (lsWARNING, AutoSocketPartition) << "Handle autodetect error: " << ec;
277  cbFunc (ec);
278  }
279  else if ((mBuffer[0] < 127) && (mBuffer[0] > 31) &&
280  ((bytesTransferred < 2) || ((mBuffer[1] < 127) && (mBuffer[1] > 31))) &&
281  ((bytesTransferred < 3) || ((mBuffer[2] < 127) && (mBuffer[2] > 31))) &&
282  ((bytesTransferred < 4) || ((mBuffer[3] < 127) && (mBuffer[3] > 31))))
283  {
284  // not ssl
285  if (AutoSocketPartition.doLog (lsTRACE))
286  Log (lsTRACE, AutoSocketPartition) << "non-SSL";
287 
288  mSecure = false;
289  cbFunc (ec);
290  }
291  else
292  {
293  // ssl
294  if (AutoSocketPartition.doLog (lsTRACE))
295  Log (lsTRACE, AutoSocketPartition) << "SSL";
296 
297  mSecure = true;
298  mSocket->async_handshake (ssl_socket::server, cbFunc);
299  }
300  }
