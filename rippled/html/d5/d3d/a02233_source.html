36 static bool fileMatches (const StringArray& wildCards, const String& filename)
37 {
38  for (int i = 0; i < wildCards.size(); ++i)
39  if (filename.matchesWildcard (wildCards[i], ! File::areFileNamesCaseSensitive()))
40  return true;

42  return false;

71 bool DirectoryIterator::next (bool* const isDirResult, bool* const isHiddenResult, std::int64_t* const fileSize,
72  Time* const modTime, Time* const creationTime, bool* const isReadOnly)
73 {
74  hasBeenAdvanced = true;
75 
76  if (subIterator != nullptr)
77  {
78  if (subIterator->next (isDirResult, isHiddenResult, fileSize, modTime, creationTime, isReadOnly))
79  return true;
80 
81  subIterator = nullptr;
82  }
83 
84  String filename;
85  bool isDirectory, isHidden = false;
86 
87  while (fileFinder.next (filename, &isDirectory,
88  (isHiddenResult != nullptr || (whatToLookFor & File::ignoreHiddenFiles) != 0) ? &isHidden : nullptr,
89  fileSize, modTime, creationTime, isReadOnly))
90  {
91  ++index;
92 
93  if (! filename.containsOnly ("."))
94  {
95  bool matches = false;
96 
97  if (isDirectory)
98  {
99  if (isRecursive && ((whatToLookFor & File::ignoreHiddenFiles) == 0 || ! isHidden))
100  subIterator = new DirectoryIterator (File::createFileWithoutCheckingPath (path + filename),
101  true, wildCard, whatToLookFor);
102 
103  matches = (whatToLookFor & File::findDirectories) != 0;
104  }
105  else
106  {
107  matches = (whatToLookFor & File::findFiles) != 0;
108  }
109 
110  // if we're not relying on the OS iterator to do the wildcard match, do it now..
111  if (matches && (isRecursive || wildCards.size() > 1))
112  matches = fileMatches (wildCards, filename);
113 
114  if (matches && (whatToLookFor & File::ignoreHiddenFiles) != 0)
115  matches = ! isHidden;
116 
117  if (matches)
118  {
119  currentFile = File::createFileWithoutCheckingPath (path + filename);
120  if (isHiddenResult != nullptr) *isHiddenResult = isHidden;
121  if (isDirResult != nullptr) *isDirResult = isDirectory;
122 
123  return true;
124  }
125 
126  if (subIterator != nullptr)
127  return next (isDirResult, isHiddenResult, fileSize, modTime, creationTime, isReadOnly);
128  }
129  }
130 
131  return false;
132 }
