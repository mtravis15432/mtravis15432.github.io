36 // Return reverse of "key".
37 // Used to test non-lexicographic comparators.
38 static std::string Reverse(const Slice& key) {
39  std::string str(key.ToString());
40  std::string rev("");
41  for (std::string::reverse_iterator rit = str.rbegin();
42  rit != str.rend(); ++rit) {
43  rev.push_back(*rit);
44  }
45  return rev;

76 static void Increment(const Comparator* cmp, std::string* key) {
77  if (cmp == BytewiseComparator()) {
78  key->push_back('\0');
79  } else {
80  assert(cmp == &reverse_key_comparator);
81  std::string rev = Reverse(*key);

83  *key = Reverse(rev);

521  for (int j =0; j < reverse_compare_len; j++)
522  for (int k =0; k < restart_interval_len; k++)
523  for (unsigned int n =0; n < compression_types.size(); n++) {

616  void TestRandomAccess(Random* rnd,
617  const std::vector<std::string>& keys,
618  const KVMap& data) {
619  static const bool kVerbose = false;
620  Iterator* iter = constructor_->NewIterator();
621  ASSERT_TRUE(!iter->Valid());
622  KVMap::const_iterator model_iter = data.begin();
623  if (kVerbose) fprintf(stderr, "---\n");
624  for (int i = 0; i < 200; i++) {
625  const int toss = rnd->Uniform(5);
626  switch (toss) {
627  case 0: {
628  if (iter->Valid()) {
629  if (kVerbose) fprintf(stderr, "Next\n");
630  iter->Next();
631  ++model_iter;
632  ASSERT_EQ(ToString(data, model_iter), ToString(iter));
633  }
634  break;
635  }
636 
637  case 1: {
638  if (kVerbose) fprintf(stderr, "SeekToFirst\n");
639  iter->SeekToFirst();
640  model_iter = data.begin();
641  ASSERT_EQ(ToString(data, model_iter), ToString(iter));
642  break;
643  }
644 
645  case 2: {
646  std::string key = PickRandomKey(rnd, keys);
647  model_iter = data.lower_bound(key);
648  if (kVerbose) fprintf(stderr, "Seek '%s'\n",
649  EscapeString(key).c_str());
650  iter->Seek(Slice(key));
651  ASSERT_EQ(ToString(data, model_iter), ToString(iter));
652  break;
653  }
654 
655  case 3: {
656  if (iter->Valid()) {
657  if (kVerbose) fprintf(stderr, "Prev\n");
658  iter->Prev();
659  if (model_iter == data.begin()) {
660  model_iter = data.end(); // Wrap around to invalid value
661  } else {
662  --model_iter;
663  }
664  ASSERT_EQ(ToString(data, model_iter), ToString(iter));
665  }
666  break;
667  }
668 
669  case 4: {
670  if (kVerbose) fprintf(stderr, "SeekToLast\n");
671  iter->SeekToLast();
672  if (keys.empty()) {
673  model_iter = data.end();
674  } else {
675  std::string last = data.rbegin()->first;
676  model_iter = data.lower_bound(last);
677  }
678  ASSERT_EQ(ToString(data, model_iter), ToString(iter));
679  break;
680  }
681  }
682  }
683  delete iter;

711  std::string PickRandomKey(Random* rnd, const std::vector<std::string>& keys) {
712  if (keys.empty()) {
713  return "foo";
714  } else {
715  const int index = rnd->Uniform(keys.size());
716  std::string result = keys[index];
717  switch (rnd->Uniform(3)) {
718  case 0:
719  // Return an existing key
720  break;
721  case 1: {
722  // Attempt to return something smaller than an existing key
723  if (result.size() > 0 && result[result.size()-1] > '\0') {
724  result[result.size()-1]--;
725  }
726  break;
727  }
728  case 2: {
729  // Return something larger than an existing key
730  Increment(options_.comparator, &result);
731  break;
732  }
733  }
734  return result;
735  }
736  }

978  // -- Table construction
979  Options options;
980  options.create_if_missing = true;
981  options.statistics = CreateDBStatistics();

991  // Since block_cache is disabled, no cache activities will be involved.
992  unique_ptr<Iterator> iter;
993 
994  // At first, no block will be accessed.
995  {

997  // index will be added to block cache.
998  props.AssertEqual(
999  1, // index block miss
1000  0,
1001  0,
1002  0
1003  );
1004  }
1005 
1006  // Only index block will be accessed
1007  {
1008  iter.reset(c.NewIterator());
1009  BlockCacheProperties props(options.statistics);
1010  // NOTE: to help better highlight the "detla" of each ticker, I use
1011  // <last_value> + <added_value> to indicate the increment of changed
1012  // value; other numbers remain the same.
1013  props.AssertEqual(
1014  1,
1015  0 + 1, // index block hit
1016  0,
1017  0
1018  );
1019  }
1020 
1021  // Only data block will be accessed
1022  {
1023  iter->SeekToFirst();
1024  BlockCacheProperties props(options.statistics);
1025  props.AssertEqual(
1026  1,
1027  1,
1028  0 + 1, // data block miss
1029  0
1030  );
1031  }
1032

1034  {
1035  iter.reset(c.NewIterator());
1036  iter->SeekToFirst();

1038  props.AssertEqual(
1039  1,
1040  1 + 1, // index block hit
1041  1,
1042  0 + 1 // data block hit
1043  );
1044  }
1045  // release the iterator so that the block cache can reset correctly.
1046  iter.reset();
1047 
1048  // -- PART 2: Open without block cache
1049  options.block_cache.reset();
1050  options.statistics = CreateDBStatistics(); // reset the props
1051  c.Reopen(options);
1052 
1053  {
1054  iter.reset(c.NewIterator());
1055  iter->SeekToFirst();
1056  ASSERT_EQ("key", iter->key().ToString());
1057  BlockCacheProperties props(options.statistics);
1058  // Nothing is affected at all
1059  props.AssertEqual(0, 0, 0, 0);
1060  }
1061 
1062  // -- PART 3: Open with very small block cache
1063  // In this test, no block will ever get hit since the block cache is
1064  // too small to fit even one entry.
1065  options.block_cache = NewLRUCache(1);
1066  c.Reopen(options);
1067  {
1068  BlockCacheProperties props(options.statistics);

1070  1, // index block miss
1071  0,
1072  0,
1073  0
1074  );
1075  }
1076 
1077 
1078  {

1080  // It first cache index block then data block. But since the cache size

1082  iter.reset(c.NewIterator());

1085  1 + 1, // index block miss
1086  0,
1087  0, // data block miss
1088  0
1089  );
