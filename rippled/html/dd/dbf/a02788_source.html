75  // Populates and returns a kv-map
76  void MakeKVMap(int64_t num_entries) {
77  kvmap_.clear();
78  int digits = 1;
79  for (int dummy = num_entries; dummy /= 10 ; ++digits);
80  int digits_in_i = 1;
81  for (int64_t i = 0; i < num_entries; i++) {
82  std::string key = "key";
83  std::string value = "value";
84  if (i % 10 == 0) {
85  digits_in_i++;
86  }
87  for(int j = digits_in_i; j < digits; j++) {
88  key.append("0");
89  value.append("0");
90  }
91  AppendNumberTo(&key, i);
92  AppendNumberTo(&value, i);
93  kvmap_[key] = value;
94  }
95  ASSERT_EQ((int)kvmap_.size(), num_entries);//check all insertions done
96  }

98  // Makes a write-batch with key-vals from kvmap_ and 'Write''s it
99  void MakePutWriteBatch(const BatchOperation* batch_ops, int num_ops) {
100  assert(num_ops <= (int)kvmap_.size());
101  static WriteOptions wopts;
102  static FlushOptions flush_opts;
103  WriteBatch batch;
104  kv_it_ = kvmap_.begin();
105  for (int i = 0; i < num_ops && kv_it_ != kvmap_.end(); i++, kv_it_++) {
106  switch (batch_ops[i]) {
107  case PUT:
108  batch.Put(kv_it_->first, kv_it_->second);
109  break;
110  case DELETE:
111  batch.Delete(kv_it_->first);
112  break;
113  default:
114  assert(false);
115  }
116  }
117  db_ttl_->Write(wopts, &batch);
118  db_ttl_->Flush(flush_opts);
119  }

164  // Sleeps for slp_tim then runs a manual compaction
165  // Checks span starting from st_pos from kvmap_ in the db and
166  // Gets should return true if check is true and false otherwise
167  // Also checks that value that we got is the same as inserted; and =kNewValue
168  // if test_compaction_change is true
169  void SleepCompactCheck(int slp_tim, int st_pos, int span, bool check = true,
170  bool test_compaction_change = false) {
171  assert(db_ttl_);
172  sleep(slp_tim);
173  ManualCompact();
174  static ReadOptions ropts;
175  kv_it_ = kvmap_.begin();
176  advance(kv_it_, st_pos);
177  std::string v;
178  for (int i = 0; kv_it_ != kvmap_.end() && i < span; i++, kv_it_++) {
179  Status s = db_ttl_->Get(ropts, kv_it_->first, &v);
180  if (s.ok() != check) {
181  fprintf(stderr, "key=%s ", kv_it_->first.c_str());
182  if (!s.ok()) {
183  fprintf(stderr, "is absent from db but was expected to be present\n");
184  } else {
185  fprintf(stderr, "is present in db but was expected to be absent\n");
186  }
187  assert(false);
188  } else if (s.ok()) {
189  if (test_compaction_change && v.compare(kNewValue_) != 0) {
190  fprintf(stderr, " value for key=%s present in database is %s but "
191  " should be %s\n", kv_it_->first.c_str(), v.c_str(),
192  kNewValue_.c_str());
193  assert(false);
194  } else if (!test_compaction_change && v.compare(kv_it_->second) !=0) {
195  fprintf(stderr, " value for key=%s present in database is %s but "
196  " should be %s\n", kv_it_->first.c_str(), v.c_str(),
197  kv_it_->second.c_str());
198  assert(false);
199  }
200  }
201  }

204  // Similar as SleepCompactCheck but uses TtlIterator to read from db
205  void SleepCompactCheckIter(int slp, int st_pos, int span, bool check=true) {
206  assert(db_ttl_);
207  sleep(slp);
208  ManualCompact();
209  static ReadOptions ropts;
210  Iterator *dbiter = db_ttl_->NewIterator(ropts);
211  kv_it_ = kvmap_.begin();
212  advance(kv_it_, st_pos);
213 
214  dbiter->Seek(kv_it_->first);
215  if (!check) {
216  if (dbiter->Valid()) {
217  ASSERT_NE(dbiter->value().compare(kv_it_->second), 0);
218  }
219  } else { // dbiter should have found out kvmap_[st_pos]
220  for (int i = st_pos;
221  kv_it_ != kvmap_.end() && i < st_pos + span;
222  i++, kv_it_++) {
223  ASSERT_TRUE(dbiter->Valid());
224  ASSERT_EQ(dbiter->value().compare(kv_it_->second), 0);
225  dbiter->Next();
226  }
227  }
228  delete dbiter;
229  }

238  // Works on keys of the form "key<number>"
239  // Drops key if number at the end of key is in [0, kSampleSize_/3),
240  // Keeps key if it is in [kSampleSize_/3, 2*kSampleSize_/3),
241  // Change value if it is in [2*kSampleSize_/3, kSampleSize_)
242  // Eg. kSampleSize_=6. Drop:key0-1...Keep:key2-3...Change:key4-5...
243  virtual bool Filter(int level, const Slice& key,
244  const Slice& value, std::string* new_value,
245  bool* value_changed) const override {
246  assert(new_value != nullptr);
247 
248  std::string search_str = "0123456789";
249  std::string key_string = key.ToString();
250  size_t pos = key_string.find_first_of(search_str);
251  int num_key_end;
252  if (pos != std::string::npos) {
253  num_key_end = stoi(key_string.substr(pos, key.size() - pos));
254  } else {
255  return false; // Keep keys not matching the format "key<NUMBER>"
256  }
257 
258  int partition = kSampleSize_ / 3;
259  if (num_key_end < partition) {
260  return true;
261  } else if (num_key_end < partition * 2) {
262  return false;
263  } else {
264  *new_value = kNewValue_;
265  *value_changed = true;
266  return false;
267  }
268  }
269 
270  virtual const char* Name() const override {
271  return "TestFilter";
272  }
273 
274  private:
275  const int64_t kSampleSize_;
276  const std::string kNewValue_;
277  };
