78 char* MemoryOutputStream::prepareToWrite (size_t numBytes)
79 {
80  bassert ((std::ptrdiff_t) numBytes >= 0);
81  size_t storageNeeded = position + numBytes;
82 
83  char* data;
84 
85  if (blockToUse != nullptr)
86  {
87  if (storageNeeded >= blockToUse->getSize())
88  blockToUse->ensureSize ((storageNeeded + bmin (storageNeeded / 2, (size_t) (1024 * 1024)) + 32) & ~31u);
89 
90  data = static_cast <char*> (blockToUse->getData());
91  }
92  else
93  {
94  if (storageNeeded > availableSize)
95  return nullptr;
96 
97  data = static_cast <char*> (externalData);
98  }
99 
100  char* const writePointer = data + position;
101  position += numBytes;
102  size = bmax (size, position);
103  return writePointer;
104 }

176 int MemoryOutputStream::writeFromInputStream (InputStream& source, std::int64_t maxNumBytesToWrite)
177 {
178  // before writing from an input, see if we can preallocate to make it more efficient..
