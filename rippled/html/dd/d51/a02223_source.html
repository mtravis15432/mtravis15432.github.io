119 template <class ElementType, class ElementComparator>
120 static int findInsertIndexInSortedArray (ElementComparator& comparator,
121  ElementType* const array,
122  const ElementType newElement,
123  int firstElement,
124  int lastElement)
125 {
126  bassert (firstElement <= lastElement);
127 
128  (void) comparator; // if you pass in an object with a static compareElements() method, this
129  // avoids getting warning messages about the parameter being unused
130 
131  while (firstElement < lastElement)
132  {
133  if (comparator.compareElements (newElement, array [firstElement]) == 0)
134  {
135  ++firstElement;
136  break;
137  }
138  else
139  {
140  const int halfway = (firstElement + lastElement) >> 1;
141 
142  if (halfway == firstElement)
143  {
144  if (comparator.compareElements (newElement, array [halfway]) >= 0)
145  ++firstElement;
146 
147  break;
148  }
149  else if (comparator.compareElements (newElement, array [halfway]) >= 0)
150  {
151  firstElement = halfway;
152  }
153  else
154  {
155  lastElement = halfway;
156  }
157  }
158  }
159 
160  return firstElement;
161 }
