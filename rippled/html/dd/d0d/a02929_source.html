37 void OrderBookDB::setup (Ledger::ref ledger)
38 {
39  {
40  ScopedLockType sl (mLock);
41 
42  // Do a full update every 256 ledgers
43  if (mSeq != 0)
44  {
45  if (ledger->getLedgerSeq () == mSeq)
46  return;
47  if ((ledger->getLedgerSeq () > mSeq) && ((ledger->getLedgerSeq () - mSeq) < 256))
48  return;
49  if ((ledger->getLedgerSeq () < mSeq) && ((mSeq - ledger->getLedgerSeq ()) < 16))
50  return;
51  }
52 
53  WriteLog (lsDEBUG, OrderBookDB) << "Advancing from " << mSeq << " to " << ledger->getLedgerSeq();
54 
55  mSeq = ledger->getLedgerSeq ();
56  }
57 
58  if (getConfig().RUN_STANDALONE)
59  update(ledger);
60  else
61  getApp().getJobQueue().addJob(jtUPDATE_PF, "OrderBookDB::update",
62  BIND_TYPE(&OrderBookDB::update, this, ledger));
63 }

65 static void updateHelper (SLE::ref entry,
66  boost::unordered_set< uint256 >& seen,
67  ripple::unordered_map< RippleAsset, std::vector<OrderBook::pointer> >& destMap,
68  ripple::unordered_map< RippleAsset, std::vector<OrderBook::pointer> >& sourceMap,
69  boost::unordered_set< RippleAsset >& XRPBooks,
70  int& books)
71 {
72  if ((entry->getType () == ltDIR_NODE) && (entry->isFieldPresent (sfExchangeRate)) &&
73  (entry->getFieldH256 (sfRootIndex) == entry->getIndex()))
74  {
75  const uint160& ci = entry->getFieldH160 (sfTakerPaysCurrency);
76  const uint160& co = entry->getFieldH160 (sfTakerGetsCurrency);
77  const uint160& ii = entry->getFieldH160 (sfTakerPaysIssuer);
78  const uint160& io = entry->getFieldH160 (sfTakerGetsIssuer);
79 
80  uint256 index = Ledger::getBookBase (ci, ii, co, io);
81 
82  if (seen.insert (index).second)
83  {
84  // VFALCO TODO Reduce the clunkiness of these parameter wrappers
85  OrderBook::pointer book = boost::make_shared<OrderBook> (boost::cref (index),
86  boost::cref (ci), boost::cref (co), boost::cref (ii), boost::cref (io));
87 
88  sourceMap[RippleAssetRef (ci, ii)].push_back (book);
89  destMap[RippleAssetRef (co, io)].push_back (book);
90  if (co.isZero())
91  XRPBooks.insert(RippleAssetRef (ci, ii));
92  ++books;
93  }
94  }
95 }

97 void OrderBookDB::update (Ledger::pointer ledger)
98 {
99  boost::unordered_set< uint256 > seen;
100  ripple::unordered_map< RippleAsset, std::vector<OrderBook::pointer> > destMap;
101  ripple::unordered_map< RippleAsset, std::vector<OrderBook::pointer> > sourceMap;
102  boost::unordered_set< RippleAsset > XRPBooks;
103 
104  WriteLog (lsDEBUG, OrderBookDB) << "OrderBookDB::update>";
105 
106  // walk through the entire ledger looking for orderbook entries
107  int books = 0;
108 
109  try
110  {
111  ledger->visitStateItems(BIND_TYPE(&updateHelper, P_1, boost::ref(seen), boost::ref(destMap),
112  boost::ref(sourceMap), boost::ref(XRPBooks), boost::ref(books)));
113  }
114  catch (const SHAMapMissingNode&)
115  {
116  WriteLog (lsINFO, OrderBookDB) << "OrderBookDB::update encountered a missing node";
117  ScopedLockType sl (mLock);
118  mSeq = 0;
119  return;
120  }
121 
122  WriteLog (lsDEBUG, OrderBookDB) << "OrderBookDB::update< " << books << " books found";
123  {
124  ScopedLockType sl (mLock);
125 
126  mXRPBooks.swap(XRPBooks);
127  mSourceMap.swap(sourceMap);
128  mDestMap.swap(destMap);
129  }
130  getApp().getLedgerMaster().newOrderBookDB();
131 }

133 void OrderBookDB::addOrderBook(const uint160& ci, const uint160& co,
134  const uint160& ii, const uint160& io)
135 {
136  bool toXRP = co.isZero();
137  ScopedLockType sl (mLock);
138 
139  if (toXRP)
140  { // We don't want to search through all the to-XRP or from-XRP order books!
141  BOOST_FOREACH(OrderBook::ref ob, mSourceMap[RippleAssetRef(ci, ii)])
142  {
143  if (ob->getCurrencyOut().isZero ()) // also to XRP
144  return;
145  }
146  }
147  else
148  {
149  BOOST_FOREACH(OrderBook::ref ob, mDestMap[RippleAssetRef(co, io)])
150  {
151  if ((ob->getCurrencyIn() == ci) && (ob->getIssuerIn() == ii))
152  return;
153  }
154  }
155 
156  uint256 index = Ledger::getBookBase(ci, ii, co, io);
157  OrderBook::pointer book = boost::make_shared<OrderBook> (boost::cref (index),
158  boost::cref (ci), boost::cref (co), boost::cref (ii), boost::cref (io));
159 
160  mSourceMap[RippleAssetRef (ci, ii)].push_back (book);
161  mDestMap[RippleAssetRef (co, io)].push_back (book);
162  if (toXRP)
163  mXRPBooks.insert(RippleAssetRef (ci, ii));
164 }

238 void OrderBookDB::processTxn (Ledger::ref ledger, const AcceptedLedgerTx& alTx, Json::Value const& jvObj)
239 {
240  ScopedLockType sl (mLock);
241 
242  if (alTx.getResult () == tesSUCCESS)
243  {
244  // check if this is an offer or an offer cancel or a payment that consumes an offer
245  //check to see what the meta looks like
246  BOOST_FOREACH (STObject & node, alTx.getMeta ()->getNodes ())
247  {
248  try
249  {
250  if (node.getFieldU16 (sfLedgerEntryType) == ltOFFER)
251  {
252  SField* field = nullptr;
253 
254  // We need a field that contains the TakerGets and TakerPays parameters
255  if (node.getFName () == sfModifiedNode)
256  {
257  field = &sfPreviousFields;
258  }
259  else if (node.getFName () == sfCreatedNode)
260  {
261  field = &sfNewFields;
262  }
263  else if (node.getFName () == sfDeletedNode)
264  {
265  field = &sfFinalFields;
266  }
267 
268  if (field)
269  {
270  const STObject* data = dynamic_cast<const STObject*> (node.peekAtPField (*field));
271 
272  if (data)
273  {
274  const STAmount& takerGets = data->getFieldAmount (sfTakerGets);
275  RippleCurrency const& currencyGets = takerGets.getCurrency ();
276  RippleIssuer const& issuerGets = takerGets.getIssuer ();
277 
278  const STAmount& takerPays = data->getFieldAmount (sfTakerPays);
279  RippleCurrency const& currencyPays = takerPays.getCurrency ();
280  RippleIssuer const& issuerPays = takerPays.getIssuer ();
281 
282  // determine the OrderBook
283  BookListeners::pointer book =
284  getBookListeners (currencyPays, currencyGets, issuerPays, issuerGets);
285 
286  if (book)
287  book->publish (jvObj);
288  }
289  }
290  }
291  }
292  catch (...)
293  {
294  WriteLog (lsINFO, OrderBookDB) << "Fields not found in OrderBookDB::processTxn";
295  }
296  }
297  }

318 void BookListeners::publish (Json::Value const& jvObj)
319 {
320  Json::FastWriter jfwWriter;
321  std::string sObj = jfwWriter.write (jvObj);
322 
323  ScopedLockType sl (mLock);
324  NetworkOPs::SubMapType::const_iterator it = mListeners.begin ();
325 
326  while (it != mListeners.end ())
327  {
328  InfoSub::pointer p = it->second.lock ();
329 
330  if (p)
