120 HashRouter::Entry& HashRouter::findCreateEntry (uint256 const& index, bool& created)
121 {
122  ripple::unordered_map<uint256, Entry>::iterator fit = mSuppressionMap.find (index);
123 
124  if (fit != mSuppressionMap.end ())
125  {
126  created = false;
127  return fit->second;
128  }
129 
130  created = true;
131 
132  int now = UptimeTimer::getInstance ().getElapsedSeconds ();
133  int expireTime = now - mHoldTime;
134 
135  // See if any supressions need to be expired
136  std::map< int, std::list<uint256> >::iterator it = mSuppressionTimes.begin ();
137 
138  if ((it != mSuppressionTimes.end ()) && (it->first <= expireTime))
139  {
140  BOOST_FOREACH (uint256 const & lit, it->second)
141  mSuppressionMap.erase (lit);
142  mSuppressionTimes.erase (it);
143  }
144 
145  mSuppressionTimes[now].push_back (index);
146  return mSuppressionMap.emplace (index, Entry ()).first->second;
147 }
