109 std::vector<RippleAddress> SerializedTransaction::getMentionedAccounts () const
110 {
111  std::vector<RippleAddress> accounts;
112 
113  BOOST_FOREACH (const SerializedType & it, peekData ())
114  {
115  const STAccount* sa = dynamic_cast<const STAccount*> (&it);
116 
117  if (sa != nullptr)
118  {
119  bool found = false;
120  RippleAddress na = sa->getValueNCA ();
121  BOOST_FOREACH (const RippleAddress & it, accounts)
122  {
123  if (it == na)
124  {
125  found = true;
126  break;
127  }
128  }
129 
130  if (!found)
131  accounts.push_back (na);
132  }
133 
134  const STAmount* sam = dynamic_cast<const STAmount*> (&it);
135 
136  if (sam)
137  {
138  uint160 issuer = sam->getIssuer ();
139 
140  if (issuer.isNonZero ())
141  {
142  RippleAddress na;
143  na.setAccountID (issuer);
144  bool found = false;
145  BOOST_FOREACH (const RippleAddress & it, accounts)
146  {
147  if (it == na)
148  {
149  found = true;
150  break;
151  }
152  }
153 
154  if (!found)
155  accounts.push_back (na);
156  }
157  }
158  }
159  return accounts;

192 bool SerializedTransaction::checkSign () const
193 {
194  if (mSigGood)
195  return true;
196 
197  if (mSigBad)
198  return false;
199 
200  try
201  {
202  RippleAddress n;
203  n.setAccountPublic (getFieldVL (sfSigningPubKey));
204 
205  if (checkSign (n))
206  {
207  mSigGood = true;
208  return true;
209  }
210  }
211  catch (...)
212  {
213  ;
214  }
215 
216  mSigBad = true;
217  return false;

317 bool isMemoOkay (STObject const& st)
318 {
319  if (!st.isFieldPresent (sfMemos))
320  return true;
321  // The number 2048 is a preallocation hint, not a hard limit
322  // to avoid allocate/copy/free's
323  Serializer s (2048);
324  st.getFieldArray (sfMemos).add (s);
325  // FIXME move the memo limit into a config tunable
326  if (s.getDataLength () > 1024)
327  return false;
328  return true;
329 }
