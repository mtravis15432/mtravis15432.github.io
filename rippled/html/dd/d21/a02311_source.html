39  void insert (Item& item)

58 private:
59  // Called at program exit when destructors for objects
60  // with static storage duration are invoked.
61  //
62  void doStaticDetruction ()
63  {
64  // In theory this shouldn't be needed (?)
65  ScopedLockType lock (m_mutex);
66 
67  bassert (! m_didStaticDestruction);
68 
69  for (List <Item>::iterator iter (m_list.begin()); iter != m_list.end();)
70  {
71  Item& item (*iter++);
72  AtExitHook* const hook (item.hook ());
73  hook->onExit ();
74  }
75 
76  // Now do the leak checking
77  //
78  LeakCheckedBase::checkForLeaks ();
79  }
