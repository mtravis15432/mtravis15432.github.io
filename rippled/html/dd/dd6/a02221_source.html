685  template <typename ElementComparator, typename TargetValueType>
686  int indexOfSorted (ElementComparator& comparator, TargetValueType elementToLookFor) const
687  {
688  (void) comparator; // if you pass in an object with a static compareElements() method, this
689  // avoids getting warning messages about the parameter being unused
690 
691  const ScopedLockType lock (getLock());
692 
693  for (int s = 0, e = numUsed;;)
694  {
695  if (s >= e)
696  return -1;
697 
698  if (comparator.compareElements (elementToLookFor, data.elements [s]) == 0)
699  return s;
700 
701  const int halfway = (s + e) / 2;
702  if (halfway == s)
703  return -1;
