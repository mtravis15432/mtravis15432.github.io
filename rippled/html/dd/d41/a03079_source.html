26 //
27 // This is a light weight, untrusted interface for web clients.
28 // For now we don't provide proof. Later we will.
29 //
30 // Might need to support this header for browsers: Access-Control-Allow-Origin: *
31 // - https://developer.mozilla.org/en-US/docs/HTTP_access_control
32 //
33 
34 //
35 // Strategy:
36 // - We only talk to NetworkOPs (so we will work even in thin mode)
37 // - NetworkOPs is smart enough to subscribe and or pass back messages
38 //
39 // VFALCO NOTE NetworkOPs isn't used here...
40 //

69 private:
70  void run ()
71  {
72  WriteLog (lsINFO, WSDoor) << boost::str (
73  boost::format ("Websocket: %s: Listening: %s %d ") %
74  (mPublic ? "Public" : "Private") % mIp % mPort);
75 
76  websocketpp::server_multitls::handler::ptr handler (
77  new WSServerHandler <websocketpp::server_multitls> (
78  m_resourceManager, m_source, m_ssl_context, mPublic, mProxy));
79 
80  {
81  ScopedLockType lock (m_endpointLock);
82 
83  m_endpoint = boost::make_shared<websocketpp::server_multitls> (handler);
84  }
85 
86  // Call the main-event-loop of the websocket server.
87  try
88  {
89  m_endpoint->listen (
90  boost::asio::ip::tcp::endpoint (
91  boost::asio::ip::address ().from_string (mIp), mPort));
92  }
93  catch (websocketpp::exception& e)
94  {
95  WriteLog (lsWARNING, WSDoor) << "websocketpp exception: " << e.what ();
96 
97  // temporary workaround for websocketpp throwing exceptions on access/close races
98  for (;;)
99  {
100  // https://github.com/zaphoyd/websocketpp/issues/98
101  try
102  {
103  m_endpoint->get_io_service ().run ();
104  break;
105  }
106  catch (websocketpp::exception& e)
107  {
108  WriteLog (lsWARNING, WSDoor) << "websocketpp exception: " << e.what ();
109  }
110  }
111  }
112 
113  {
114  ScopedLockType lock (m_endpointLock);
115 
116  m_endpoint.reset();
117  }
118 
119  stopped ();
120  }
121 
122  void onStop ()
123  {
124  boost::shared_ptr<websocketpp::server_multitls> endpoint;
125 
126  {
127  ScopedLockType lock (m_endpointLock);
128 
129  endpoint = m_endpoint;
130  }
131 
132  // VFALCO NOTE we probably dont want to block here
133  // but websocketpp is deficient and broken.
134  //
135  if (endpoint)
136  endpoint->stop ();
137 
138  signalThreadShouldExit ();
139  }
