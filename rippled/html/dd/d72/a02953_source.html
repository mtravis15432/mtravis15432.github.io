178  // Stolen directly from RPCServerHandler
179  std::string processRequest (std::string const& request, beast::IP::Endpoint const& remoteIPAddress)
180  {
181  Json::Value jvRequest;
182  {
183  Json::Reader reader;
184 
185  if ((request.size () > 1000000) ||
186  ! reader.parse (request, jvRequest) ||
187  jvRequest.isNull () ||
188  ! jvRequest.isObject ())
189  {
190  return createResponse (400, "Unable to parse request");
191  }
192  }
193 
194  Config::Role const role (getConfig ().getAdminRole (jvRequest, remoteIPAddress));
195 
196  Resource::Consumer usage;
197 
198  if (role == Config::ADMIN)
199  usage = m_resourceManager.newAdminEndpoint (remoteIPAddress.to_string());
200  else
201  usage = m_resourceManager.newInboundEndpoint(remoteIPAddress);
202 
203  if (usage.disconnect ())
204  return createResponse (503, "Server is overloaded");
205 
206  // Parse id now so errors from here on will have the id
207  //
208  // VFALCO NOTE Except that "id" isn't included in the following errors...
209  //
210  Json::Value const id = jvRequest ["id"];
211 
212  Json::Value const method = jvRequest ["method"];
213 
214  if (method.isNull ())
215  {
216  return createResponse (400, "Null method");
217  }
218  else if (! method.isString ())
219  {
220  return createResponse (400, "method is not string");
221  }
222 
223  std::string strMethod = method.asString ();
224 
225  // Parse params
226  Json::Value params = jvRequest ["params"];
227 
228  if (params.isNull ())
229  {
230  params = Json::Value (Json::arrayValue);
231  }
232  else if (!params.isArray ())
233  {
234  return HTTPReply (400, "params unparseable");
235  }
236 
237  // VFALCO TODO Shouldn't we handle this earlier?
238  //
239  if (role == Config::FORBID)
240  {
241  // VFALCO TODO Needs implementing
242  // FIXME Needs implementing
243  // XXX This needs rate limiting to prevent brute forcing password.
244  return HTTPReply (403, "Forbidden");
245  }
246 
247  // This code does all the work on the io_service thread and
248  // has no rate-limiting based on source IP or anything.
249  // This is a temporary safety
250  if ((role != Config::ADMIN) && (getApp().getFeeTrack().isLoadedLocal()))
251  {
252  return HTTPReply (503, "Unable to service at this time");
253  }
254 
255  std::string response;
256 
257  m_journal.debug << "Query: " << strMethod << params;
258 
259  RPCHandler rpcHandler (&m_networkOPs);
260 
261  Resource::Charge loadType = Resource::feeReferenceRPC;
262 
263  Json::Value const result (rpcHandler.doRpcCommand (
264  strMethod, params, role, loadType));
265 
266  usage.charge (loadType);
267 
268  m_journal.debug << "Reply: " << result;
269 
270  response = JSONRPCReply (result, Json::Value (), id);
271 
272  return createResponse (200, response);
273  }
