56 enum SHAMapState
57 {
58  smsModifying = 0, // Objects can be added and removed (like an open ledger)
59  smsImmutable = 1, // Map cannot be changed (like a closed ledger)
60  smsSynching = 2, // Map's hash is locked in, valid nodes can be added (like a peer's closing ledger)
61  smsFloating = 3, // Map is free to change hash (like a synching open ledger)
62  smsInvalid = 4, // Map is known not to be valid (usually synching a corrupt ledger)
63 };
