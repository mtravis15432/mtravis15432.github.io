76 String SystemStats::getStackBacktrace()
77 {
78  String result;
79 
80  #if BEAST_ANDROID || BEAST_MINGW || BEAST_BSD
81  bassertfalse; // sorry, not implemented yet!
82 
83  #elif BEAST_WINDOWS
84  HANDLE process = GetCurrentProcess();
85  SymInitialize (process, nullptr, TRUE);
86 
87  void* stack[128];
88  int frames = (int) CaptureStackBackTrace (0, numElementsInArray (stack), stack, nullptr);
89 
90  HeapBlock<SYMBOL_INFO> symbol;
91  symbol.calloc (sizeof (SYMBOL_INFO) + 256, 1);
92  symbol->MaxNameLen = 255;
93  symbol->SizeOfStruct = sizeof (SYMBOL_INFO);
94 
95  for (int i = 0; i < frames; ++i)
96  {
97  DWORD64 displacement = 0;
98 
99  if (SymFromAddr (process, (DWORD64) stack[i], &displacement, symbol))
100  {
101  result << i << ": ";
102 
103  IMAGEHLP_MODULE64 moduleInfo;
104  zerostruct (moduleInfo);
105  moduleInfo.SizeOfStruct = sizeof (moduleInfo);
106 
107  if (::SymGetModuleInfo64 (process, symbol->ModBase, &moduleInfo))
108  result << moduleInfo.ModuleName << ": ";
109 
110  result << symbol->Name << " + 0x" << String::toHexString ((std::int64_t) displacement) << newLine;
111  }
112  }
113 
114  #else
115  void* stack[128];
116  int frames = backtrace (stack, numElementsInArray (stack));
117  char** frameStrings = backtrace_symbols (stack, frames);
118 
119  for (int i = 0; i < frames; ++i)
120  result << frameStrings[i] << newLine;
121 
122  ::free (frameStrings);
123  #endif
124 
125  return result;
126 }

147 void SystemStats::setApplicationCrashHandler (CrashHandlerFunction handler)
148 {
149  bassert (handler != nullptr); // This must be a valid function.
150  globalCrashHandler = handler;
151 
152  #if BEAST_WINDOWS
153  SetUnhandledExceptionFilter (handleCrash);
154  #else
155  const int signals[] = { SIGFPE, SIGILL, SIGSEGV, SIGBUS, SIGABRT, SIGSYS };
156 
157  for (int i = 0; i < numElementsInArray (signals); ++i)
