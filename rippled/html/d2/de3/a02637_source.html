629 //
630 // An application can issue a read request (via Get/Iterators) and specify
631 // if that read should process data that ALREADY resides on a specified cache
632 // level. For example, if an application specifies kBlockCacheTier then the
633 // Get call will process data that is already processed in the memtable or
634 // the block cache. It will not page in data from the OS cache or data that
635 // resides in storage.

641 // Options that control read operations
642 struct ReadOptions {
643  // If true, all data read from underlying storage will be
644  // verified against corresponding checksums.
645  // Default: false
646  bool verify_checksums;
647 
648  // Should the "data block"/"index block"/"filter block" read for this
649  // iteration be cached in memory?
650  // Callers may wish to set this field to false for bulk scans.
651  // Default: true
652  bool fill_cache;
653 
654  // If this option is set and memtable implementation allows, Seek
655  // might only return keys with the same prefix as the seek-key
656  bool prefix_seek;
657 
658  // If "snapshot" is non-nullptr, read as of the supplied snapshot
659  // (which must belong to the DB that is being read and which must
660  // not have been released). If "snapshot" is nullptr, use an impliicit
661  // snapshot of the state at the beginning of this read operation.
662  // Default: nullptr
663  const Snapshot* snapshot;
664 
665  // If "prefix" is non-nullptr, and ReadOptions is being passed to
666  // db.NewIterator, only return results when the key begins with this
667  // prefix. This field is ignored by other calls (e.g., Get).
668  // Options.prefix_extractor must also be set, and
669  // prefix_extractor.InRange(prefix) must be true. The iterator
670  // returned by NewIterator when this option is set will behave just
671  // as if the underlying store did not contain any non-matching keys,

674  // option will sometimes reduce the number of read IOPs.
675  // Default: nullptr

677 
678  // Specify if this read request should process data that ALREADY
679  // resides on a particular cache. If the required data is not
680  // found at the specified cache, then Status::Incomplete is returned.

699 // Options that control write operations
700 struct WriteOptions {
701  // If true, the write will be flushed from the operating system
702  // buffer cache (by calling WritableFile::Sync()) before the write
703  // is considered complete. If this flag is true, writes will be
704  // slower.
705  //
706  // If this flag is false, and the machine crashes, some recent
707  // writes may be lost. Note that if it is just the process that
708  // crashes (i.e., the machine does not reboot), no writes will be
709  // lost even if sync==false.
710  //
711  // In other words, a DB write with sync==false has similar
712  // crash semantics as the "write()" system call. A DB write
713  // with sync==true has similar crash semantics to a "write()"
714  // system call followed by "fdatasync()".
715  //
716  // Default: false
717  bool sync;
718 
719  // If true, writes will not first go to the write ahead log,
720  // and the write may got lost after a crash.
721  bool disableWAL;
722 
723  WriteOptions()
724  : sync(false),
725  disableWAL(false) {
726  }
727 };
