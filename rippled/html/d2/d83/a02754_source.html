50  virtual void Logv(const char* format, va_list ap) {
51  const uint64_t thread_id = (*gettid_)();
52 
53  // We try twice: the first time with a fixed-size stack allocated buffer,
54  // and the second time with a much larger dynamically allocated buffer.
55  char buffer[500];
56  for (int iter = 0; iter < 2; iter++) {
57  char* base;
58  int bufsize;
59  if (iter == 0) {
60  bufsize = sizeof(buffer);
61  base = buffer;
62  } else {
63  bufsize = 30000;
64  base = new char[bufsize];
65  }
66  char* p = base;
67  char* limit = base + bufsize;
68 
69  struct timeval now_tv;
70  gettimeofday(&now_tv, nullptr);
71  const time_t seconds = now_tv.tv_sec;
72  struct tm t;
73  localtime_r(&seconds, &t);
74  p += snprintf(p, limit - p,
75  "%04d/%02d/%02d-%02d:%02d:%02d.%06d %llx ",
76  t.tm_year + 1900,
77  t.tm_mon + 1,
78  t.tm_mday,
79  t.tm_hour,
80  t.tm_min,
81  t.tm_sec,
82  static_cast<int>(now_tv.tv_usec),
83  static_cast<long long unsigned int>(thread_id));
84 
85  // Print the message
86  if (p < limit) {
87  va_list backup_ap;
88  va_copy(backup_ap, ap);
89  p += vsnprintf(p, limit - p, format, backup_ap);
90  va_end(backup_ap);
91  }
92 
93  // Truncate to available space if necessary
94  if (p >= limit) {
95  if (iter == 0) {
96  continue; // Try again with larger buffer
97  } else {
98  p = limit - 1;
99  }
100  }
101 
102  // Add newline if necessary
103  if (p == base || p[-1] != '\n') {
104  *p++ = '\n';
105  }
106 
107  assert(p <= limit);
108  const size_t write_size = p - base;
109 
110 #ifdef OS_LINUX
111  // If this write would cross a boundary of kDebugLogChunkSize
112  // space, pre-allocate more space to avoid overly large
113  // allocations from filesystem allocsize options.
114  const size_t log_size = log_size_;
115  const int last_allocation_chunk =
116  ((kDebugLogChunkSize - 1 + log_size) / kDebugLogChunkSize);
117  const int desired_allocation_chunk =
118  ((kDebugLogChunkSize - 1 + log_size + write_size) /
119  kDebugLogChunkSize);
120  if (last_allocation_chunk != desired_allocation_chunk) {
121  fallocate(fd_, FALLOC_FL_KEEP_SIZE, 0,
122  desired_allocation_chunk * kDebugLogChunkSize);
123  }
124 #endif
125 
126  size_t sz = fwrite(base, 1, write_size, file_);
127  assert(sz == write_size);
128  if (sz > 0) {
129  log_size_ += write_size;
130  }
131  uint64_t now_micros = static_cast<uint64_t>(now_tv.tv_sec) * 1000000 +
132  now_tv.tv_usec;
133  if (now_micros - last_flush_micros_ >= flush_every_seconds_ * 1000000) {
134  fflush(file_);
135  last_flush_micros_ = now_micros;
136  }
137  if (base != buffer) {
138  delete[] base;
139  }
140  break;
141  }
142  }
143  size_t GetLogFileSize() const {
144  return log_size_;
145  }
146 };
