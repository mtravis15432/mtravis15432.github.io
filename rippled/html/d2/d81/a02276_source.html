119 static bool isWindowsVersionOrLater (SystemStats::OperatingSystemType target)
120 {
121  OSVERSIONINFOEX info;
122  zerostruct (info);
123  info.dwOSVersionInfoSize = sizeof (OSVERSIONINFOEX);
124 
125  if (target >= SystemStats::WinVista)
126  {
127  info.dwMajorVersion = 6;
128 
129  switch (target)
130  {
131  case SystemStats::WinVista: info.dwMinorVersion = 0; break;
132  case SystemStats::Windows7: info.dwMinorVersion = 1; break;
133  case SystemStats::Windows8: info.dwMinorVersion = 2; break;
134  default: bassertfalse; break;
135  }
136  }
137  else
138  {
139  info.dwMajorVersion = 5;
140  info.dwMinorVersion = target >= SystemStats::WinXP ? 1 : 0;
141  }
142 
143  DWORDLONG mask = 0;
144 
145  VER_SET_CONDITION (mask, VER_MAJORVERSION, VER_GREATER_EQUAL);
146  VER_SET_CONDITION (mask, VER_MINORVERSION, VER_GREATER_EQUAL);
147  VER_SET_CONDITION (mask, VER_SERVICEPACKMAJOR, VER_GREATER_EQUAL);
148  VER_SET_CONDITION (mask, VER_SERVICEPACKMINOR, VER_GREATER_EQUAL);
149 
150  return VerifyVersionInfo (&info,
151  VER_MAJORVERSION | VER_MINORVERSION
152  | VER_SERVICEPACKMAJOR | VER_SERVICEPACKMINOR,
153  mask) != FALSE;
154 }

234 class HiResCounterHandler
235 {
236 public:
237  HiResCounterHandler()
238  : hiResTicksOffset (0)
239  {
240  const MMRESULT res = timeBeginPeriod (1);
241  (void) res;
242  bassert (res == TIMERR_NOERROR);
243 
244  LARGE_INTEGER f;
245  QueryPerformanceFrequency (&f);
246  hiResTicksPerSecond = f.QuadPart;
247  hiResTicksScaleFactor = 1000.0 / hiResTicksPerSecond;
248  }

250  inline std::int64_t getHighResolutionTicks() noexcept
251  {
252  LARGE_INTEGER ticks;
253  QueryPerformanceCounter (&ticks);
254 
255  const std::int64_t mainCounterAsHiResTicks = (beast_millisecondsSinceStartup() * hiResTicksPerSecond) / 1000;
256  const std::int64_t newOffset = mainCounterAsHiResTicks - ticks.QuadPart;
257 
258  // fix for a very obscure PCI hardware bug that can make the counter
259  // sometimes jump forwards by a few seconds..
260  const std::int64_t offsetDrift = abs64 (newOffset - hiResTicksOffset);
261 
262  if (offsetDrift > (hiResTicksPerSecond >> 1))
263  hiResTicksOffset = newOffset;
264 
265  return ticks.QuadPart + hiResTicksOffset;
266  }

268  inline double getMillisecondCounterHiRes() noexcept
269  {
270  return getHighResolutionTicks() * hiResTicksScaleFactor;
271  }
272 
273  std::int64_t hiResTicksPerSecond, hiResTicksOffset;
274  double hiResTicksScaleFactor;
275 };

284 static std::int64_t beast_getClockCycleCounter() noexcept
285 {
286  #if BEAST_USE_INTRINSICS
287  // MS intrinsics version...
288  return (std::int64_t) __rdtsc();
289 
290  #elif BEAST_GCC
291  // GNU inline asm version...
292  unsigned int hi = 0, lo = 0;
293 
294  __asm__ __volatile__ (
295  "xor %%eax, %%eax \n\
296  xor %%edx, %%edx \n\
297  rdtsc \n\
298  movl %%eax, %[lo] \n\
299  movl %%edx, %[hi]"
300  :
301  : [hi] "m" (hi),
302  [lo] "m" (lo)
303  : "cc", "eax", "ebx", "ecx", "edx", "memory");
304 
305  return (std::int64_t) ((((std::uint64_t) hi) << 32) | lo);
306  #else
307  // MSVC inline asm version...
308  unsigned int hi = 0, lo = 0;
309 
310  __asm
311  {
312  xor eax, eax
313  xor edx, edx
314  rdtsc
315  mov lo, eax
316  mov hi, edx
317  }
318 
319  return (std::int64_t) ((((std::uint64_t) hi) << 32) | lo);
320  #endif
321 }

323 int SystemStats::getCpuSpeedInMegaherz()
324 {
325  const std::int64_t cycles = beast_getClockCycleCounter();
326  const std::uint32_t millis = Time::getMillisecondCounter();
327  int lastResult = 0;
328 
329  for (;;)
330  {
331  int n = 1000000;
332  while (--n > 0) {}
333 
334  const std::uint32_t millisElapsed = Time::getMillisecondCounter() - millis;
335  const std::int64_t cyclesNow = beast_getClockCycleCounter();
336 
337  if (millisElapsed > 80)
338  {
339  const int newResult = (int) (((cyclesNow - cycles) / millisElapsed) / 1000);
340 
341  if (millisElapsed > 500 || (lastResult == newResult && newResult > 100))
342  return newResult;
343 
344  lastResult = newResult;
345  }
346  }
347 }
