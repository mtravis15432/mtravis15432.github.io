113  void addJob (JobType type, std::string const& name,
114  boost::function <void (Job&)> const& jobFunc)
115  {
116  assert (type != jtINVALID);
117 
118  JobDataMap::iterator iter (m_jobData.find (type));
119  assert (iter != m_jobData.end ());
120 
121  if (iter == m_jobData.end ())
122  return;
123 
124  JobTypeData& data (iter->second);
125 
126  // FIXME: Workaround incorrect client shutdown ordering
127  // do not add jobs to a queue with no threads
128  assert (type == jtCLIENT || m_workers.getNumberOfThreads () > 0);
129 
130  {
131  // If this goes off it means that a child didn't follow
132  // the Stoppable API rules. A job may only be added if:
133  //
134  // - The JobQueue has NOT stopped
135  // AND
136  // * We are currently processing jobs
137  // OR
138  // * We have have pending jobs
139  // OR
140  // * Not all children are stopped
141  //
142  ScopedLock lock (m_mutex);
143  assert (! isStopped() && (
144  m_processCount>0 ||
145  ! m_jobSet.empty () ||
146  ! areChildrenStopped()));
147  }
148 
149  // Don't even add it to the queue if we're stopping
150  // and the job type is marked for skipOnStop.
151  //
152  if (isStopping() && skipOnStop (type))
153  {
154  m_journal.debug <<
155  "Skipping addJob ('" << name << "')";
156  return;
157  }
158 
159  {
160  ScopedLock lock (m_mutex);
161 
162  std::pair <std::set <Job>::iterator, bool> result (
163  m_jobSet.insert (Job (type, name, ++m_lastJob,
164  data.load (), jobFunc, m_cancelCallback)));
165  queueJob (*result.first, lock);
166  }

356  // Signals the service stopped if the stopped condition is met.
357  //
358  void checkStopped (ScopedLock const& lock)
359  {
360  // We are stopped when all of the following are true:
361  //
362  // 1. A stop notification was received
363  // 2. All Stoppable children have stopped
364  // 3. There are no executing calls to processTask
365  // 4. There are no remaining Jobs in the job set
366  //
367  if (isStopping() &&
368  areChildrenStopped() &&
369  (m_processCount == 0) &&
370  m_jobSet.empty())
371  {
372  stopped();
373  }

376  //--------------------------------------------------------------------------
377  //
378  // Signals an added Job for processing.
379  //
380  // Pre-conditions:
381  // The JobType must be valid.
382  // The Job must exist in mJobSet.
383  // The Job must not have previously been queued.
384  //
385  // Post-conditions:
386  // Count of waiting jobs of that type will be incremented.
387  // If JobQueue exists, and has at least one thread, Job will eventually run.
388  //
389  // Invariants:
390  // The calling thread owns the JobLock
391  //
392  void queueJob (Job const& job, ScopedLock const& lock)
393  {
394  JobType const type (job.getType ());
395  assert (type != jtINVALID);
396  assert (m_jobSet.find (job) != m_jobSet.end ());
397 
398  JobTypeData& data (getJobTypeData (type));
399 
400  if (data.waiting + data.running < getJobLimit (type))
401  {
402  m_workers.addTask ();
403  }
404  else
405  {
406  // defer the task until we go below the limit
407  //
408  ++data.deferred;
409  }
410  ++data.waiting;
411  }
412 
413  //------------------------------------------------------------------------------
414  //
415  // Returns the next Job we should run now.
416  //
417  // RunnableJob:
418  // A Job in the JobSet whose slots count for its type is greater than zero.
419  //
420  // Pre-conditions:
421  // mJobSet must not be empty.
422  // mJobSet holds at least one RunnableJob
423  //
424  // Post-conditions:
425  // job is a valid Job object.
426  // job is removed from mJobQueue.
427  // Waiting job count of it's type is decremented
428  // Running job count of it's type is incremented
429  //
430  // Invariants:
431  // The calling thread owns the JobLock
432  //
433  void getNextJob (Job& job, ScopedLock const& lock)
434  {
435  assert (! m_jobSet.empty ());
436 
437  JobSet::const_iterator iter;
438  for (iter = m_jobSet.begin (); iter != m_jobSet.end (); ++iter)
439  {
440  JobTypeData& data (getJobTypeData (iter->getType ()));
441 
442  assert (data.running <= getJobLimit (data.type ()));
443 
444  // Run this job if we're running below the limit.
445  if (data.running < getJobLimit (data.type ()))
446  {
447  assert (data.waiting > 0);
448  break;
449  }
450  }
451 
452  assert (iter != m_jobSet.end ());
453 
454  JobType const type = iter->getType ();
455  JobTypeData& data (getJobTypeData (type));
456 
457  assert (type != jtINVALID);
458 
459  job = *iter;
460  m_jobSet.erase (iter);
461 
462  --data.waiting;
463  ++data.running;
464  }

466  //------------------------------------------------------------------------------
467  //
468  // Indicates that a running Job has completed its task.
469  //
470  // Pre-conditions:
471  // Job must not exist in mJobSet.
472  // The JobType must not be invalid.
473  //
474  // Post-conditions:
475  // The running count of that JobType is decremented
476  // A new task is signaled if there are more waiting Jobs than the limit, if any.
477  //
478  // Invariants:
479  // <none>
480  //
481  void finishJob (Job const& job, ScopedLock const& lock)
482  {
483  JobType const type = job.getType ();
484 
485  assert (m_jobSet.find (job) == m_jobSet.end ());
486  assert (type != jtINVALID);
487 
488  JobTypeData& data (getJobTypeData (type));
489 
490  // Queue a deferred task if possible
491  if (data.deferred > 0)
492  {
493  assert (data.running + data.waiting >= getJobLimit (type));
494 
495  --data.deferred;
496  m_workers.addTask ();
497  }
498 
499  --data.running;
500  }

523  //--------------------------------------------------------------------------
524  //
525  // Runs the next appropriate waiting Job.
526  //
527  // Pre-conditions:
528  // A RunnableJob must exist in the JobSet
529  //
530  // Post-conditions:
531  // The chosen RunnableJob will have Job::doJob() called.
532  //
533  // Invariants:
534  // <none>
535  //
536  void processTask ()
537  {
538  Job job;
539 
540  {
541  ScopedLock lock (m_mutex);
542  getNextJob (job, lock);
543  ++m_processCount;
544  }
545 
546  JobTypeData& data (getJobTypeData (job.getType ()));
547 
548  // Skip the job if we are stopping and the
549  // skipOnStop flag is set for the job type
550  //
551  if (!isStopping() || !data.info.skip ())
552  {
553  beast::Thread::setCurrentThreadName (data.name ());
554  m_journal.trace << "Doing " << data.name () << " job";
555 
556  Job::clock_type::time_point const start_time (
557  Job::clock_type::now());
558 
559  on_dequeue (job.getType (), start_time - job.queue_time ());
560  job.doJob ();
561  on_execute (job.getType (), Job::clock_type::now() - start_time);
562  }
563  else
564  {
565  m_journal.trace << "Skipping processTask ('" << data.name () << "')";
566  }
567 
568  {
569  ScopedLock lock (m_mutex);
570  finishJob (job, lock);
571  --m_processCount;
572  checkStopped (lock);
573  }
