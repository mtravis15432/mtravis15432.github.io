56 int InternalKeyComparator::Compare(const Slice& akey, const Slice& bkey) const {
57  // Order by:
58  // increasing user key (according to user-supplied comparator)
59  // decreasing sequence number
60  // decreasing type (though sequence# should be enough to disambiguate)
61  int r = user_comparator_->Compare(ExtractUserKey(akey), ExtractUserKey(bkey));
62  BumpPerfCount(&perf_context.user_key_comparison_count);
63  if (r == 0) {
64  const uint64_t anum = DecodeFixed64(akey.data() + akey.size() - 8);
65  const uint64_t bnum = DecodeFixed64(bkey.data() + bkey.size() - 8);
66  if (anum > bnum) {
67  r = -1;
68  } else if (anum < bnum) {
69  r = +1;
70  }
71  }
72  return r;
73 }

75 void InternalKeyComparator::FindShortestSeparator(
76  std::string* start,
77  const Slice& limit) const {
78  // Attempt to shorten the user portion of the key
79  Slice user_start = ExtractUserKey(*start);
80  Slice user_limit = ExtractUserKey(limit);
81  std::string tmp(user_start.data(), user_start.size());
82  user_comparator_->FindShortestSeparator(&tmp, user_limit);
83  if (tmp.size() < user_start.size() &&
84  user_comparator_->Compare(user_start, tmp) < 0) {
85  // User key has become shorter physically, but larger logically.
86  // Tack on the earliest possible number to the shortened user key.
87  PutFixed64(&tmp, PackSequenceAndType(kMaxSequenceNumber,kValueTypeForSeek));
88  assert(this->Compare(*start, tmp) < 0);
89  assert(this->Compare(tmp, limit) < 0);
90  start->swap(tmp);
91  }
92 }

128 LookupKey::LookupKey(const Slice& user_key, SequenceNumber s) {
129  size_t usize = user_key.size();
130  size_t needed = usize + 13; // A conservative estimate
131  char* dst;
132  if (needed <= sizeof(space_)) {
133  dst = space_;
134  } else {
135  dst = new char[needed];
136  }
137  start_ = dst;
138  dst = EncodeVarint32(dst, usize + 8);
139  kstart_ = dst;
140  memcpy(dst, user_key.data(), usize);
141  dst += usize;
142  EncodeFixed64(dst, PackSequenceAndType(s, kValueTypeForSeek));
143  dst += 8;
144  end_ = dst;
145 }
