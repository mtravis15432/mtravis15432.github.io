45 void startServer ()
46 {
47  //
48  // Execute start up rpc commands.
49  //
50  if (getConfig ().RPC_STARTUP.isArray ())
51  {
52  for (int i = 0; i != getConfig ().RPC_STARTUP.size (); ++i)
53  {
54  const Json::Value& jvCommand = getConfig ().RPC_STARTUP[i];
55 
56  if (!getConfig ().QUIET)
57  Log::out() << "Startup RPC: " << jvCommand;
58 
59  RPCHandler rhHandler (&getApp().getOPs ());
60 
61  Resource::Charge loadType = Resource::feeReferenceRPC;
62  Json::Value jvResult = rhHandler.doCommand (jvCommand, Config::ADMIN, loadType);
63 
64  if (!getConfig ().QUIET)
65  Log::out() << "Result: " << jvResult;
66  }
67  }
68 
69  getApp().run (); // Blocks till we get a stop RPC.
70 }

201  ("parameters", po::value< vector<string> > (), "Specify comma separated parameters.")
202  ("quiet,q", "Reduce diagnotics.")

207  ("start", "Start from a fresh Ledger.")

211  ("version", "Display the build version.")
212  ;
213 
214  // Interpret positional arguments as --parameters.
215  po::positional_options_description p;
216  p.add ("parameters", -1);

218  if (! RandomNumbers::getInstance ().initialize ())

259  return 0;

342  getConfig ().VALIDATION_QUORUM = 2;
343  }
344 
345  if (iResult == 0)
346  {
347  // These overrides must happen after the config file is loaded.
348 
349  // Override the RPC destination IP address
350  //
